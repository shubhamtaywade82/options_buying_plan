```ruby
module Api
  module V1
    class DashboardController < ApplicationController
      before_action :set_instrument, only: %i[regime option_chain liquidity]

      def overview
        render json: {
          timestamp: Time.current.iso8601,
          market_status: market_status,
          active_positions: active_positions_summary,
          today_pnl: today_pnl,
          open_risk: open_risk,
          system_health: system_health_snapshot
        }
      end

      def regime
        result = Engines::MarketRegime::MarketRegimeEngine.new(instrument: @instrument).call(
          context: { candles: recent_candles }
        )

        if result.success?
          render json: {
            instrument: @instrument.trading_symbol,
            current_regime: result.value,
            regime_history: regime_history,
            timestamp: Time.current.iso8601
          }
        else
          render json: { error: result.error.to_h }, status: :unprocessable_entity
        end
      end

      def option_chain
        chain_result = OptionChainService.new(underlying: @instrument).call

        if chain_result.success?
          render json: {
            underlying: @instrument.trading_symbol,
            spot_price: current_spot_price,
            expiry: nearest_expiry,
            chain: chain_result.value,
            atm_strike: calculate_atm,
            timestamp: Time.current.iso8601
          }
        else
          render json: { error: chain_result.error.to_h }, status: :unprocessable_entity
        end
      end

      def liquidity
        depth_result = MarketDepthService.new(instrument: @instrument).call

        if depth_result.success?
          liq_result = Engines::Liquidity::LiquidityEngine.new(instrument: @instrument).call(
            context: { depth: depth_result.value }
          )

          if liq_result.success?
            render json: {
              instrument: @instrument.trading_symbol,
              liquidity: liq_result.value,
              depth: depth_result.value,
              timestamp: Time.current.iso8601
            }
          else
            render json: { error: liq_result.error.to_h }, status: :unprocessable_entity
          end
        else
          render json: { error: depth_result.error.to_h }, status: :unprocessable_entity
        end
      end

      def current_trades
        positions = Trade.where(exit_price: nil)
                         .where("entry_time > ?", Date.current.beginning_of_day)
                         .order(entry_time: :desc)

        render json: {
          count: positions.count,
          positions: positions.map { |t| position_payload(t) },
          total_exposure: positions.sum { |t| t.position_value.to_f },
          timestamp: Time.current.iso8601
        }
      end

      def risk_metrics
        positions = Trade.where(exit_price: nil)
        calculator = Risk::PortfolioRiskCalculator.new(trades: positions)

        render json: {
          total_exposure: calculator.total_exposure,
          margin_utilization: calculator.margin_utilization,
          daily_loss_limit: daily_loss_limit,
          daily_loss_used: daily_loss_used,
          daily_loss_remaining: daily_loss_limit - daily_loss_used,
          max_drawdown_today: max_drawdown_today,
          var_95: calculator.value_at_risk(confidence: 0.95),
          var_99: calculator.value_at_risk(confidence: 0.99),
          concentration_risk: calculator.concentration_risk,
          greek_exposure: calculator.greek_exposure,
          timestamp: Time.current.iso8601
        }
      end

      def ai_analysis
        latest = AiAnalysisCache.recent.first(5)

        render json: {
          latest_insights: latest.map { |a| ai_payload(a) },
          agent_activity: agent_activity_summary,
          ai_gateway_health: ai_gateway_health,
          timestamp: Time.current.iso8601
        }
      end

      def performance
        lookback = params[:lookback]&.to_i || 1

        result = Engines::Performance::PerformanceDashboardData.new(
          lookback: lookback.days
        ).call

        render json: result.merge(timestamp: Time.current.iso8601)
      end

      def trade_journal
        trades = Trade.where("exit_time IS NOT NULL")
                      .where("exit_time > ?", 24.hours.ago)
                      .order(exit_time: :desc)

        render json: {
          count: trades.count,
          trades: trades.map { |t| journal_payload(t) },
          daily_summary: daily_summary(trades),
          timestamp: Time.current.iso8601
        }
      end

      def health
        render json: {
          status: :healthy,
          checks: health_checks,
          version: AppConfig.config.app_version,
          uptime: uptime,
          timestamp: Time.current.iso8601
        }
      end

      def alert_history
        alerts = AlertHistory.recent.limit(params[:limit]&.to_i || 20)

        render json: {
          count: alerts.count,
          alerts: alerts.map { |a| alert_payload(a) },
          unacknowledged: alerts.where(acknowledged: false).count,
          timestamp: Time.current.iso8601
        }
      end

      private

      def set_instrument
        @instrument = Instrument.find(params[:instrument_id])
      end

      def market_status
        calendar = ExchangeCalendar.for_exchange("NSE").find_by(date: Date.current)
        {
          trading_day: calendar&.trading_day || false,
          market_open: calendar&.market_open&.strftime("%H:%M"),
          market_close: calendar&.market_close&.strftime("%H:%M"),
          status: market_open_now?(calendar) ? :open : :closed,
          time_to_open: time_to_open(calendar),
          time_to_close: time_to_close(calendar)
        }
      end

      def market_open_now?(calendar)
        return false unless calendar&.trading_day
        now = Time.current
        open_time = Time.zone.parse(calendar.market_open.to_s)
        close_time = Time.zone.parse(calendar.market_close.to_s)
        now.between?(open_time, close_time)
      end

      def time_to_open(calendar)
        return nil unless calendar&.trading_day
        open_time = Time.zone.parse(calendar.market_open.to_s)
        diff = open_time - Time.current
        diff > 0 ? diff.to_i : 0
      end

      def time_to_close(calendar)
        return nil unless calendar&.trading_day
        close_time = Time.zone.parse(calendar.market_close.to_s)
        diff = close_time - Time.current
        diff > 0 ? diff.to_i : 0
      end

      def active_positions_summary
        positions = Trade.where(exit_price: nil)
        {
          count: positions.count,
          long_count: positions.where(direction: :long).count,
          short_count: positions.where(direction: :short).count,
          total_unrealized_pnl: positions.sum { |t| t.unrealized_pnl.to_f }.round(2)
        }
      end

      def today_pnl
        closed = Trade.where("exit_time > ?", Date.current.beginning_of_day)
        realized = closed.sum { |t| t.gross_pnl.to_f }
        open = Trade.where(exit_price: nil)
        unrealized = open.sum { |t| t.unrealized_pnl.to_f }

        {
          realized: realized.round(2),
          unrealized: unrealized.round(2),
          total: (realized + unrealized).round(2)
        }
      end

      def open_risk
        positions = Trade.where(exit_price: nil)
        at_risk = positions.sum do |t|
          next 0 unless t.stop_loss && t.entry_price
          (t.entry_price.to_d - t.stop_loss.to_d).abs.to_f * t.quantity.to_i
        end

        { total_at_risk: at_risk.round(2), position_count: positions.count }
      end

      def system_health_snapshot
        {
          data_freshness: data_freshness,
          websocket_status: websocket_status,
          broker_connection: broker_connection_status,
          ai_gateway: ai_gateway_health,
          last_tick_received: last_tick_time
        }
      end

      def data_freshness
        last = Tick.maximum(:timestamp)
        return { status: :unknown, seconds_ago: nil } unless last

        seconds = (Time.current - last).to_i
        {
          status: seconds < 10 ? :fresh : seconds < 60 ? :stale : :dead,
          seconds_ago: seconds,
          last_tick: last.iso8601
        }
      end

      def websocket_status
        DhanWebSocketManager.instance.connected? ? :connected : :disconnected
      rescue
        :unknown
      end

      def broker_connection_status
        DhanHQ::Client.new.health_check.success? ? :connected : :disconnected
      rescue
        :unknown
      end

      def ai_gateway_health
        AiGateway::AIGateway.new.health.values.any? { |h| h[:available] } ? :healthy : :degraded
      rescue
        :unknown
      end

      def last_tick_time
        Tick.maximum(:timestamp)&.iso8601
      end

      def recent_candles
        CandleRepository.ohlcv(
          instrument: @instrument,
          timeframe: "5m",
          from: 2.hours.ago,
          to: Time.current
        ).value || []
      end

      def regime_history
        TradeRegimeLog.where(instrument: @instrument)
                      .where("created_at > ?", 2.hours.ago)
                      .order(:created_at)
                      .pluck(:regime_type, :created_at)
                      .map { |type, time| { regime: type, at: time.iso8601 } }
      end

      def current_spot_price
        @instrument.ticks.order(:timestamp).last&.ltp
      end

      def nearest_expiry
        OptionExpiry.where("expiry_date >= ?", Date.current)
                    .order(:expiry_date)
                    .first&.expiry_date
      end

      def calculate_atm
        spot = current_spot_price
        return nil unless spot
        step = @instrument.trading_symbol.include?("BANK") ? 100 : 50
        (spot / step).round * step
      end

      def position_payload(trade)
        {
          id: trade.id,
          instrument: trade.instrument&.trading_symbol,
          direction: trade.direction,
          strategy: trade.setup_type,
          entry_price: trade.entry_price,
          current_price: trade.current_price,
          stop_loss: trade.stop_loss,
          take_profit: trade.take_profit,
          quantity: trade.quantity,
          unrealized_pnl: trade.unrealized_pnl.to_f.round(2),
          unrealized_pnl_percent: trade.unrealized_pnl_percent.to_f.round(2),
          mfe: trade.mfe_r_multiple,
          mae: trade.mae_r_multiple,
          entry_time: trade.entry_time.iso8601,
          holding_time_seconds: trade.holding_time_seconds,
          trade_score: trade.trade_score&.total_score
        }
      end

      def daily_loss_limit
        AppConfig.config.risk&.daily_loss_limit || 50_000
      end

      def daily_loss_used
        Trade.where("exit_time > ?", Date.current.beginning_of_day)
             .sum { |t| [t.gross_pnl.to_f, 0].min }.abs
      end

      def max_drawdown_today
        today_trades = Trade.where("exit_time > ?", Date.current.beginning_of_day)
                            .order(:exit_time)

        peak = 0
        max_dd = 0
        cumulative = 0

        today_trades.each do |t|
          cumulative += t.gross_pnl.to_f
          peak = [peak, cumulative].max
          max_dd = [max_dd, peak - cumulative].max
        end

        max_dd.round(2)
      end

      def ai_payload(analysis)
        {
          agent: analysis.agent_name,
          confidence: analysis.ai_confidence,
          summary: analysis.summary,
          created_at: analysis.created_at.iso8601
        }
      end

      def agent_activity_summary
        AiAnalysisLog.where("created_at > ?", 1.hour.ago)
                     .group(:agent_name)
                     .count
      end

      def journal_payload(trade)
        {
          id: trade.id,
          instrument: trade.instrument&.trading_symbol,
          direction: trade.direction,
          strategy: trade.setup_type,
          entry_price: trade.entry_price,
          exit_price: trade.exit_price,
          pnl: trade.gross_pnl.to_f.round(2),
          pnl_percent: trade.pnl_percent.to_f.round(2),
          exit_reason: trade.exit_reason,
          holding_time: trade.holding_time_seconds,
          exit_time: trade.exit_time&.iso8601,
          trade_score: trade.trade_score&.total_score
        }
      end

      def daily_summary(trades)
        wins = trades.select { |t| t.gross_pnl.to_f > 0 }
        {
          total: trades.count,
          wins: wins.count,
          losses: trades.count - wins.count,
          win_rate: trades.empty? ? 0 : (wins.count.to_f / trades.count * 100).round(2),
          gross_pnl: trades.sum { |t| t.gross_pnl.to_f }.round(2)
        }
      end

      def health_checks
        {
          database: database_check,
          redis: redis_check,
          websocket: websocket_check,
          broker_api: broker_api_check,
          data_freshness: data_freshness_check
        }
      end

      def database_check
        ActiveRecord::Base.connection.execute("SELECT 1")
        { status: :pass, latency_ms: 0 }
      rescue => e
        { status: :fail, error: e.message }
      end

      def redis_check
        RedisPool.general.with { |r| r.ping }
        { status: :pass }
      rescue => e
        { status: :fail, error: e.message }
      end

      def websocket_check
        status = websocket_status
        { status: status == :connected ? :pass : :warn, detail: status }
      end

      def broker_api_check
        status = broker_connection_status
        { status: status == :connected ? :pass : :warn, detail: status }
      end

      def data_freshness_check
        freshness = data_freshness
        { status: freshness[:status] == :fresh ? :pass : freshness[:status] == :stale ? :warn : :fail, detail: freshness }
      end

      def uptime
        # In production, would track from process start
        Process.clock_gettime(Process::CLOCK_MONOTONIC).to_i
      end

      def alert_payload(alert)
        {
          id: alert.id,
          severity: alert.severity,
          category: alert.category,
          message: alert.message,
          acknowledged: alert.acknowledged,
          created_at: alert.created_at.iso8601
        }
      end
    end
  end
end
```

```ruby
class DashboardChannel < ApplicationCable::Channel
  def subscribed
    stream_from "dashboard:global"
    stream_from "dashboard:user:#{current_user_id}" if current_user_id

    # Send initial snapshot
    transmit({
      type: "connection",
      message: "Subscribed to dashboard updates",
      timestamp: Time.current.iso8601
    })
  end

  def unsubscribed
    # Cleanup handled by ActionCable
  end

  def request_update(data)
    case data["stream"]
    when "positions"
      broadcast_positions
    when "pnl"
      broadcast_pnl
    when "regime"
      broadcast_regime(data["instrument_id"]) if data["instrument_id"]
    when "liquidity"
      broadcast_liquidity(data["instrument_id"]) if data["instrument_id"]
    end
  end

  private

  def current_user_id
    # Extract from connection or token
    connection.current_user&.id
  end

  def broadcast_positions
    positions = Trade.where(exit_price: nil)
                     .where("entry_time > ?", Date.current.beginning_of_day)

    ActionCable.server.broadcast("dashboard:global", {
      type: "positions_update",
      data: {
        count: positions.count,
        positions: positions.map { |t| position_summary(t) },
        total_exposure: positions.sum { |t| t.position_value.to_f }.round(2),
        total_unrealized_pnl: positions.sum { |t| t.unrealized_pnl.to_f }.round(2)
      },
      timestamp: Time.current.iso8601
    })
  end

  def broadcast_pnl
    closed = Trade.where("exit_time > ?", Date.current.beginning_of_day)
    open = Trade.where(exit_price: nil)

    ActionCable.server.broadcast("dashboard:global", {
      type: "pnl_update",
      data: {
        realized: closed.sum { |t| t.gross_pnl.to_f }.round(2),
        unrealized: open.sum { |t| t.unrealized_pnl.to_f }.round(2),
        total: (closed.sum { |t| t.gross_pnl.to_f } + open.sum { |t| t.unrealized_pnl.to_f }).round(2),
        trade_count: closed.count
      },
      timestamp: Time.current.iso8601
    })
  end

  def broadcast_regime(instrument_id)
    instrument = Instrument.find(instrument_id)
    result = Engines::MarketRegime::MarketRegimeEngine.new(instrument: instrument).call(
      context: { candles: recent_candles_for(instrument) }
    )

    return unless result.success?

    ActionCable.server.broadcast("dashboard:global", {
      type: "regime_update",
      instrument: instrument.trading_symbol,
      data: result.value,
      timestamp: Time.current.iso8601
    })
  end

  def broadcast_liquidity(instrument_id)
    instrument = Instrument.find(instrument_id)
    depth_result = MarketDepthService.new(instrument: instrument).call
    return unless depth_result.success?

    liq_result = Engines::Liquidity::LiquidityEngine.new(instrument: instrument).call(
      context: { depth: depth_result.value }
    )

    return unless liq_result.success?

    ActionCable.server.broadcast("dashboard:global", {
      type: "liquidity_update",
      instrument: instrument.trading_symbol,
      data: liq_result.value,
      timestamp: Time.current.iso8601
    })
  end

  def position_summary(trade)
    {
      id: trade.id,
      instrument: trade.instrument&.trading_symbol,
      direction: trade.direction,
      unrealized_pnl: trade.unrealized_pnl.to_f.round(2),
      unrealized_pnl_percent: trade.unrealized_pnl_percent.to_f.round(2),
      current_price: trade.current_price,
      entry_price: trade.entry_price
    }
  end

  def recent_candles_for(instrument)
    CandleRepository.ohlcv(
      instrument: instrument,
      timeframe: "5m",
      from: 2.hours.ago,
      to: Time.current
    ).value || []
  end
end
```

```ruby
class DashboardBroadcastJob < JobBase
  queue_as :dashboard

  BROADCAST_INTERVAL = 5  # seconds

  def perform
    loop do
      break unless should_broadcast?

      broadcast_tick_updates
      broadcast_position_updates
      broadcast_pnl_updates

      sleep BROADCAST_INTERVAL
    end
  end

  private

  def should_broadcast?
    # Check market hours
    calendar = ExchangeCalendar.for_exchange("NSE").find_by(date: Date.current)
    return false unless calendar&.trading_day

    now = Time.current
    open_time = Time.zone.parse(calendar.market_open.to_s)
    close_time = Time.zone.parse(calendar.market_close.to_s)

    now.between?(open_time, close_time)
  end

  def broadcast_tick_updates
    # Broadcast latest ticks for watched instruments
    watched = Trade.where(exit_price: nil).distinct.pluck(:instrument_id)
    watched.each do |instrument_id|
      instrument = Instrument.find(instrument_id)
      last_tick = instrument.ticks.order(:timestamp).last

      next unless last_tick

      ActionCable.server.broadcast("dashboard:global", {
        type: "tick_update",
        instrument: instrument.trading_symbol,
        data: {
          ltp: last_tick.ltp,
          change: last_tick.change,
          change_percent: last_tick.change_percent,
          volume: last_tick.volume,
          timestamp: last_tick.timestamp.iso8601
        },
        timestamp: Time.current.iso8601
      })
    end
  end

  def broadcast_position_updates
    positions = Trade.where(exit_price: nil)
    return if positions.empty?

    ActionCable.server.broadcast("dashboard:global", {
      type: "positions_snapshot",
      data: {
        count: positions.count,
        total_unrealized_pnl: positions.sum { |t| t.unrealized_pnl.to_f }.round(2),
        positions: positions.map { |t| quick_position(t) }
      },
      timestamp: Time.current.iso8601
    })
  end

  def broadcast_pnl_updates
    today = Date.current.beginning_of_day
    realized = Trade.where("exit_time > ?", today).sum { |t| t.gross_pnl.to_f }
    open = Trade.where(exit_price: nil).sum { |t| t.unrealized_pnl.to_f }

    ActionCable.server.broadcast("dashboard:global", {
      type: "pnl_snapshot",
      data: {
        realized: realized.round(2),
        unrealized: open.round(2),
        total: (realized + open).round(2)
      },
      timestamp: Time.current.iso8601
    })
  end

  def quick_position(trade)
    {
      id: trade.id,
      instrument: trade.instrument&.trading_symbol,
      direction: trade.direction,
      unrealized_pnl: trade.unrealized_pnl.to_f.round(2),
      unrealized_pnl_percent: trade.unrealized_pnl_percent.to_f.round(2),
      current_price: trade.current_price
    }
  end
end
```

```ruby
class AiAnalysisCache < ApplicationRecord
  belongs_to :analyzable, polymorphic: true, optional: true

  validates :agent_name, presence: true
  validates :ai_confidence, numericality: { greater_than_or_equal_to: 0, less_than_or_equal_to: 100 }, allow_nil: true
```

`# app/services/risk/portfolio_risk_calculator.rb`
`# frozen_string_literal: true`

```ruby
module Risk
  class PortfolioRiskCalculator
    def initialize(trades:)
      @trades = trades
    end

    def total_exposure
      @trades.sum { |t| t.position_value.to_f }.round(2)
    end

    def margin_utilization
      total = total_exposure
      limit = AppConfig.config.risk&.max_position_value || 1_000_000
      return 0 if limit.zero?
      (total / limit * 100).round(2)
    end

    def value_at_risk(confidence: 0.95)
      # Simplified VaR using historical simulation
      returns = @trades.map { |t| t.unrealized_pnl_percent.to_f }.compact
      return 0 if returns.empty?

      sorted = returns.sort
      index = ((1 - confidence) * sorted.length).floor
      at_risk = sorted[index] || sorted.first

      (at_risk * total_exposure / 100).abs.round(2)
    end

    def concentration_risk
      return {} if @trades.empty?

      by_instrument = @trades.group_by { |t| t.instrument_id }
      total = total_exposure

      by_instrument.transform_values do |positions|
        exposure = positions.sum { |t| t.position_value.to_f }
        {
          exposure: exposure.round(2),
          percent_of_portfolio: total.zero? ? 0 : (exposure / total * 100).round(2)
        }
      end
    end

    def greek_exposure
      {
        delta: @trades.sum { |t| t.delta_exposure.to_f }.round(4),
        gamma: @trades.sum { |t| t.gamma_exposure.to_f }.round(6),
        theta: @trades.sum { |t| t.theta_exposure.to_f }.round(4),
        vega: @trades.sum { |t| t.vega_exposure.to_f }.round(4)
      }
    end

    def correlation_risk
      # Simplified: flag if >50% in one instrument
      max_concentration = concentration_risk.values.map { |v| v[:percent_of_portfolio] }.max || 0
      {
        max_concentration: max_concentration,
        high_concentration: max_concentration > 50,
        diversification_score: [100 - max_concentration, 0].max.round(2)
      }
    end
  end
end
```

```ruby
def position_value
  return 0 unless entry_price && quantity

  return exit_price if exit_price.present?
  instrument&.ticks&.order(:timestamp)&.last&.ltp || entry_price

def unrealized_pnl
  return gross_pnl if exit_price.present?
  return 0 unless entry_price && current_price
  if long?
    (current_price.to_d - entry_price.to_d) * quantity.to_i
  else
    (entry_price.to_d - current_price.to_d) * quantity.to_i
  end
end

def unrealized_pnl_percent
  return 0 unless entry_price && entry_price.to_d > 0
  (unrealized_pnl / (entry_price.to_d * quantity.to_i) * 100).round(4)
end

def unrealized_pnl_percent
  return 0 unless entry_price && entry_price.to_d > 0
  (unrealized_pnl.to_f / (entry_price.to_d * quantity.to_i) * 100).round(4)
end

def delta_exposure
  return 0 unless delta_at_entry && quantity
  delta_at_entry.to_d * quantity.to_i
end

def gamma_exposure
  return 0 unless gamma_at_entry && quantity
  gamma_at_entry.to_d * quantity.to_i
end

  return 0 unless theta_at_entry && quantity
  theta_at_entry.to_d * quantity.to_i
end

  return 0 unless vega_at_entry && quantity
  vega_at_entry.to_d * quantity.to_i
end

  return 0 unless entry_time
  end_time = exit_time || Time.current
  (end_time - entry_time).to_i
end

  return 0 unless mfe_r_multiple && entry_price && stop_loss

  (mfe_r_multiple.to_f * risk / entry_price.to_d * 100).round(4)
```

```typescript
import { useEffect, useState } from 'react';

export function useDashboard() {
  const [positions, setPositions] = useState([]);

  }, []);

}
```
