module PaperTrading
  class PaperTradingBroker < BaseService
    attr_reader :order_book, :position_manager, :pnl_calculator

    def initialize
      @order_book = PaperOrderBook.new
      @position_manager = PaperPositionManager.new
      @pnl_calculator = PaperPnLCalculator.new(position_manager: @position_manager)
      @slippage = PaperSlippageSimulator.new
      @order_id_counter = 0
    end

    # Mimics DhanHQ::Client interface
    def place_order(params)
      order_id = generate_order_id

      order = {
        order_id: order_id,
        status: "PENDING",
        order_type: params[:order_type], # MARKET, LIMIT, SL, SL-M
        transaction_type: params[:transaction_type], # BUY, SELL
        exchange_segment: params[:exchange_segment],
        product_type: params[:product_type],
        quantity: params[:quantity].to_i,
        price: params[:price].to_d,
        trigger_price: params[:trigger_price]&.to_d,
        instrument_token: params[:instrument_token],
        trading_symbol: params[:trading_symbol],
        created_at: Time.current,
        paper: true
      }

      # Validate against paper limits
      validation = validate_order(order)
      return validation if validation.failure?

      # Simulate matching
      fill_result = @order_book.match(order, slippage: @slippage)

      if fill_result[:filled] > 0
        order[:status] = fill_result[:partial] ? "PARTIALLY_FILLED" : "COMPLETE"
        order[:average_price] = fill_result[:average_price]
        order[:filled_quantity] = fill_result[:filled]
        order[:filled_at] = Time.current

        # Update positions
        @position_manager.update(order)
        @pnl_calculator.recalculate(order[:trading_symbol])
      else
        order[:status] = "OPEN"
        @order_book.add_pending(order)
      end

      persist_order(order)

      emit(EventTypes::PAPER_ORDER_PLACED, {
        order_id: order_id,
        symbol: order[:trading_symbol],
        quantity: order[:quantity],
        status: order[:status]
      })

      Result.success(order.slice(:order_id, :status, :average_price, :filled_quantity))
    end

    def modify_order(order_id, params)
      order = find_order(order_id)
      return Result.failure(:order_not_found) unless order

      order[:price] = params[:price].to_d if params[:price]
      order[:trigger_price] = params[:trigger_price].to_d if params[:trigger_price]
      order[:quantity] = params[:quantity].to_i if params[:quantity]
      order[:modified_at] = Time.current

      # Re-evaluate pending orders
      if order[:status] == "OPEN"
        fill_result = @order_book.match(order, slippage: @slippage)
        if fill_result[:filled] > 0
          order[:status] = fill_result[:partial] ? "PARTIALLY_FILLED" : "COMPLETE"
          order[:average_price] = fill_result[:average_price]
          order[:filled_quantity] = fill_result[:filled]
          order[:filled_at] ||= Time.current

          @position_manager.update(order)
          @pnl_calculator.recalculate(order[:trading_symbol])
        end
      end

      persist_order(order)
      Result.success(order.slice(:order_id, :status, :average_price, :filled_quantity))
    end

    def cancel_order(order_id)
      order = find_order(order_id)
      return Result.failure(:order_not_found) unless order

      if %w[COMPLETE CANCELLED REJECTED].include?(order[:status])
        return Result.failure(:order_already_terminal)
      end

      @order_book.remove_pending(order_id) if order[:status] == "OPEN"
      order[:status] = "CANCELLED"
      order[:cancelled_at] = Time.current

      persist_order(order)
      Result.success(order_id: order_id, status: "CANCELLED")
    end

    def order_status(order_id)
      order = find_order(order_id)
      return Result.failure(:order_not_found) unless order

      Result.success(order.slice(:order_id, :status, :average_price, :filled_quantity, :created_at))
    end

    def positions
      positions = @position_manager.all.map do |symbol, pos|
        {
          trading_symbol: symbol,
          exchange_segment: pos[:exchange_segment],
          quantity: pos[:net_quantity],
          buy_average: pos[:buy_average],
          sell_average: pos[:sell_average],
          unrealized_pnl: pos[:unrealized_pnl],
          realized_pnl: pos[:realized_pnl]
        }
      end

      Result.success(positions)
    end

    def holdings
      # For paper trading, holdings = net long positions
      holdings = @position_manager.all.select { |_, pos| pos[:net_quantity] > 0 }.map do |symbol, pos|
        {
          trading_symbol: symbol,
          quantity: pos[:net_quantity],
          average_price: pos[:buy_average],
          last_price: pos[:last_price],
          pnl: pos[:unrealized_pnl]
        }
      end

      Result.success(holdings)
    end

    def funds
      # Simulated fund ledger
      ledger = PaperFundLedger.instance
      Result.success({
        available_balance: ledger.available,
        utilized_amount: ledger.utilized,
        total_balance: ledger.total
      })
    end

    def health_check
      Result.success(:healthy)
    end

    private

    def generate_order_id
      @order_id_counter += 1
      "PAPER#{Time.current.to_i}#{@order_id_counter}"
    end

    def validate_order(order)
      ledger = PaperFundLedger.instance

      # Check margin for new positions
      if order[:transaction_type] == "BUY"
        estimated_cost = order[:quantity] * (order[:price] > 0 ? order[:price] : 100)
        unless ledger.can_debit?(estimated_cost)
          return Result.failure(:insufficient_funds)
        end
      end

      # Check position limits
      if AppConfig.config.paper_trading&.max_position_value
        max_val = AppConfig.config.paper_trading.max_position_value
        current_val = @position_manager.total_exposure
        order_val = order[:quantity] * (order[:price] > 0 ? order[:price] : 100)

        if current_val + order_val > max_val
          return Result.failure(:position_limit_exceeded)
        end
      end

      Result.success(true)
    end

    def find_order(order_id)
      PaperOrderStore.find(order_id)
    end

    def persist_order(order)
      PaperOrderStore.save(order)
    end
  end
end

module PaperTrading
  class PaperOrderBook
    def initialize
      @pending = {}
      @trades = []
    end

    def add_pending(order)
      @pending[order[:order_id]] = order
    end

    def remove_pending(order_id)
      @pending.delete(order_id)
    end

    def match(order, slippage:)
      symbol = order[:trading_symbol]
      ltp = fetch_ltp(symbol)
      return { filled: 0, average_price: 0, partial: false } unless ltp

      case order[:order_type]
      when "MARKET"
        fill_price = slippage.simulate(
          base_price: ltp,
          quantity: order[:quantity],
          direction: order[:transaction_type] == "BUY" ? :buy : :sell,
          depth: fetch_depth(symbol)
        )
        fill(order, fill_price, order[:quantity])

      when "LIMIT"
        if limit_fillable?(order, ltp)
          fill_price = order[:price] # Limit orders fill at limit price (or better)
          fill_price = slippage.simulate(
            base_price: fill_price,
            quantity: order[:quantity],
            direction: order[:transaction_type] == "BUY" ? :buy : :sell,
            depth: fetch_depth(symbol),
            order_type: :limit
          )
          fill(order, fill_price, order[:quantity])
        else
          { filled: 0, average_price: 0, partial: false }
        end

      when "SL", "SL-M"
        if stop_loss_triggered?(order, ltp)
          fill_price = slippage.simulate(
            base_price: order[:trigger_price],
            quantity: order[:quantity],
            direction: order[:transaction_type] == "BUY" ? :buy : :sell,
            depth: fetch_depth(symbol),
            order_type: :stop_loss
          )
          fill(order, fill_price, order[:quantity])
        else
          { filled: 0, average_price: 0, partial: false }
        end

      else
        { filled: 0, average_price: 0, partial: false }
      end
    end

    def process_pending_ticks(symbol, ltp)
      @pending.values.select { |o| o[:trading_symbol] == symbol }.each do |order|
        result = match(order, slippage: PaperSlippageSimulator.new)
        if result[:filled] > 0
          order[:status] = result[:partial] ? "PARTIALLY_FILLED" : "COMPLETE"
          order[:average_price] = result[:average_price]
          order[:filled_quantity] = result[:filled]
          order[:filled_at] ||= Time.current

          PaperPositionManager.new.update(order)
          remove_pending(order[:order_id])
        end
      end
    end

    private

    def fill(order, price, quantity)
      filled = [quantity, order[:quantity]].min
      partial = filled < order[:quantity]

      @trades << {
        order_id: order[:order_id],
        price: price,
        quantity: filled,
        symbol: order[:trading_symbol],
        side: order[:transaction_type],
        timestamp: Time.current
      }

      { filled: filled, average_price: price, partial: partial }
    end

    def limit_fillable?(order, ltp)
      if order[:transaction_type] == "BUY"
        ltp <= order[:price] # Buy limit: market at or below limit
      else
        ltp >= order[:price] # Sell limit: market at or above limit
      end
    end

    def stop_loss_triggered?(order, ltp)
      if order[:transaction_type] == "BUY"
        # SL Buy: trigger when market goes above trigger (short covering)
        ltp >= order[:trigger_price]
      else
        # SL Sell: trigger when market goes below trigger (stop loss on long)
        ltp <= order[:trigger_price]
      end
    end

    def fetch_ltp(symbol)
      Instrument.find_by(trading_symbol: symbol)&.ticks&.order(:timestamp)&.last&.ltp
    end

    def fetch_depth(symbol)
      # Return cached depth or default
      MarketDepthService.new(instrument: Instrument.find_by(trading_symbol: symbol)).call.value rescue nil
    end
  end
end

module PaperTrading
  class PaperSlippageSimulator
    BASE_SLIPPAGE_BPS = 0.5  # 0.5 basis points for small orders
    VOLATILITY_MULTIPLIER = 0.02
    DEPTH_PENALTY_BPS = 2.0  # Additional per level of depth consumed

    def simulate(base_price:, quantity:, direction:, depth:, order_type: :market)
      return base_price if base_price.nil? || base_price.zero?

      slippage_bps = calculate_slippage_bps(quantity, depth, order_type)
      slippage_amount = base_price * slippage_bps / 10000.0

      # Direction: buys slip up, sells slip down
      if direction == :buy
        base_price + slippage_amount
      else
        base_price - slippage_amount
      end
    end

    private

    def calculate_slippage_bps(quantity, depth, order_type)
      base = BASE_SLIPPAGE_BPS

      # Order type adjustment
      base *= case order_type
              when :limit then 0.3   # Less slippage on limits
              when :stop_loss then 1.5 # More slippage on SL
              else 1.0
              end

      # Quantity impact (larger orders = more slippage)
      qty_factor = Math.log10([quantity, 10].max) / 3.0
      base += qty_factor

      # Depth impact
      if depth
        depth_penalty = calculate_depth_penalty(quantity, depth)
        base += depth_penalty
      else
        base += 5.0 # Unknown depth = assume thin market
      end

      # Volatility component (simplified: random noise)
      base += rand(-0.5..0.5)

      [base, 50.0].min # Cap at 50 bps
    end

    def calculate_depth_penalty(quantity, depth)
      bid_qty = depth[:total_bid_quantity].to_i
      ask_qty = depth[:total_ask_quantity].to_i
      total_depth = bid_qty + ask_qty

      return 10.0 if total_depth.zero?

      depth_ratio = quantity.to_f / total_depth
      depth_ratio * DEPTH_PENALTY_BPS * 100
    end
  end
end

module PaperTrading
  class PaperPnLCalculator
    def initialize(position_manager:)
      @position_manager = position_manager
    end

    def recalculate(symbol)
      position = @position_manager.find(symbol)
      return unless position

      ltp = fetch_ltp(symbol)
      return unless ltp

      # Unrealized P&L
      if position[:net_quantity] > 0
        position[:unrealized_pnl] = (ltp - position[:buy_average]) * position[:net_quantity]
      elsif position[:net_quantity] < 0
        position[:unrealized_pnl] = (position[:sell_average] - ltp) * position[:net_quantity].abs
      else
        position[:unrealized_pnl] = 0
      end

      position[:last_price] = ltp
      position[:last_updated] = Time.current

      @position_manager.persist(symbol, position)
    end

    def realized_pnl_for_trade(order)
      symbol = order[:trading_symbol]
      position = @position_manager.find(symbol)
      return 0 unless position

      # For closing trades
      if order[:transaction_type] == "SELL" && position[:buy_average] > 0
        (order[:average_price] - position[:buy_average]) * order[:filled_quantity]
      elsif order[:transaction_type] == "BUY" && position[:sell_average] > 0
        (position[:sell_average] - order[:average_price]) * order[:filled_quantity]
      else
        0
      end
    end

    def total_unrealized_pnl
      @position_manager.all.values.sum { |pos| pos[:unrealized_pnl].to_f }
    end

    def total_realized_pnl
      PaperTradeStore.realized_pnl
    end

    private

    def fetch_ltp(symbol)
      Instrument.find_by(trading_symbol: symbol)&.ticks&.order(:timestamp)&.last&.ltp
    end
  end
end

module PaperTrading
  class PaperPositionManager
    def initialize
      @positions = PaperPositionStore.all
    end

    def update(order)
      symbol = order[:trading_symbol]
      position = find(symbol) || empty_position(symbol, order[:exchange_segment])

      qty = order[:filled_quantity].to_i
      price = order[:average_price].to_d

      if order[:transaction_type] == "BUY"
        # Update buy side
        total_buy_qty = position[:buy_quantity] + qty
        position[:buy_average] = if total_buy_qty > 0
          ((position[:buy_average] * position[:buy_quantity]) + (price * qty)) / total_buy_qty
        else
          0
        end
        position[:buy_quantity] = total_buy_qty
      else
        # Update sell side
        total_sell_qty = position[:sell_quantity] + qty
        position[:sell_average] = if total_sell_qty > 0
          ((position[:sell_average] * position[:sell_quantity]) + (price * qty)) / total_sell_qty
        else
          0
        end
        position[:sell_quantity] = total_sell_qty
      end

      # Net quantity
      position[:net_quantity] = position[:buy_quantity] - position[:sell_quantity]

      # Realized P&L on closing
      if order[:transaction_type] == "SELL" && position[:buy_average] > 0
        realized = (price - position[:buy_average]) * qty
        position[:realized_pnl] += realized
        PaperTradeStore.add_realized(realized)
      elsif order[:transaction_type] == "BUY" && position[:sell_average] > 0
        realized = (position[:sell_average] - price) * qty
        position[:realized_pnl] += realized
        PaperTradeStore.add_realized(realized)
      end

      position[:last_updated] = Time.current
      persist(symbol, position)
    end

    def find(symbol)
      @positions[symbol] || PaperPositionStore.find(symbol)
    end

    def all
      @positions = PaperPositionStore.all
      @positions
    end

    def persist(symbol, position)
      PaperPositionStore.save(symbol, position)
    end

    def total_exposure
      all.values.sum do |pos|
        next 0 if pos[:net_quantity].zero?
        price = pos[:last_price] || pos[:buy_average] || pos[:sell_average]
        pos[:net_quantity].abs * price
      end
    end

    private

    def empty_position(symbol, exchange_segment)
      {
        symbol: symbol,
        exchange_segment: exchange_segment,
        buy_quantity: 0,
        sell_quantity: 0,
        net_quantity: 0,
        buy_average: 0,
        sell_average: 0,
        unrealized_pnl: 0,
        realized_pnl: 0,
        last_price: 0,
        last_updated: Time.current
      }
    end
  end
end

module PaperTrading
  class PaperFundLedger
    include Singleton

    PREFIX = "paper:funds".freeze
    DEFAULT_CAPITAL = 500_000

    def initialize
      @key = "#{PREFIX}:ledger"
    end

    def setup(initial_capital = nil)
      capital = initial_capital || AppConfig.config.paper_trading&.initial_capital || DEFAULT_CAPITAL
      RedisPool.general.with do |r|
        r.set(@key, capital.to_f)
        r.set("#{@key}:utilized", 0)
      end
    end

    def available
      RedisPool.general.with { |r| r.get(@key).to_f }
    end

    def utilized
      RedisPool.general.with { |r| r.get("#{@key}:utilized").to_f }
    end

    def total
      available + utilized
    end

    def debit(amount)
      return false if amount > available

      RedisPool.general.with do |r|
        r.decrbyfloat(@key, amount)
        r.incrbyfloat("#{@key}:utilized", amount)
      end
      true
    end

    def credit(amount)
      RedisPool.general.with do |r|
        r.incrbyfloat(@key, amount)
        r.decrbyfloat("#{@key}:utilized", amount)
      end
      true
    end

    def can_debit?(amount)
      available >= amount
    end

    def reset
      RedisPool.general.with do |r|
        r.del(@key, "#{@key}:utilized")
      end
    end
  end
end

module PaperTrading
  class PaperTradeRecorder
    def record(order, strategy: nil, trade_score: nil)
      instrument = Instrument.find_by(trading_symbol: order[:trading_symbol])

      trade = Trade.create!(
        instrument: instrument,
        direction: order[:transaction_type] == "BUY" ? :long : :short,
        setup_type: strategy || "paper_manual",
        quantity: order[:filled_quantity],
        entry_price: order[:average_price],
        entry_time: order[:filled_at],
        broker_order_id: order[:order_id],
        paper: true,
        trade_score: trade_score,
        stop_loss: order[:metadata]&.dig(:stop_loss),
        take_profit: order[:metadata]&.dig(:take_profit),
        delta_at_entry: order[:metadata]&.dig(:delta),
        gamma_at_entry: order[:metadata]&.dig(:gamma),
        theta_at_entry: order[:metadata]&.dig(:theta),
        regime_type: order[:metadata]&.dig(:regime_type)
      )

      PaperTradeStore.add_trade({
        trade_id: trade.id,
        order_id: order[:order_id],
        symbol: order[:trading_symbol],
        entry_price: order[:average_price],
        quantity: order[:filled_quantity],
        strategy: strategy,
        recorded_at: Time.current.iso8601
      })

      trade
    end

    def record_exit(trade, exit_order)
      trade.update!(
        exit_price: exit_order[:average_price],
        exit_time: exit_order[:filled_at],
        gross_pnl: calculate_pnl(trade, exit_order),
        exit_reason: exit_order[:metadata]&.dig(:exit_reason) || "paper_close",
        pnl_percent: calculate_pnl_percent(trade, exit_order)
      )

      # Queue learning analysis for paper trades too
      LearningAnalysisJob.perform_later(trade.id) if trade.exit_price.present?

      trade
    end

    private

    def calculate_pnl(trade, exit_order)
      if trade.long?
        (exit_order[:average_price].to_d - trade.entry_price.to_d) * trade.quantity
      else
        (trade.entry_price.to_d - exit_order[:average_price].to_d) * trade.quantity
      end
    end

    def calculate_pnl_percent(trade, exit_order)
      return 0 if trade.entry_price.to_d.zero?
      (calculate_pnl(trade, exit_order).to_f / (trade.entry_price.to_d * trade.quantity) * 100).round(4)
    end

module PaperTrading
  class PaperTradingDashboard
    def initialize
      @broker = PaperTradingBroker.new
      @ledger = PaperFundLedger.instance
    end

    def metrics
      trades = Trade.where(paper: true).where("entry_time > ?", 30.days.ago)
      closed = trades.where("exit_price IS NOT NULL")

      {
        mode: :paper,
        capital: {
          initial: initial_capital,
          available: @ledger.available,
          utilized: @ledger.utilized,
          total: @ledger.total
        },
        performance: {
          total_trades: trades.count,
          closed_trades: closed.count,
          open_positions: trades.where(exit_price: nil).count,
          win_rate: win_rate(closed),
          profit_factor: profit_factor(closed),
          net_pnl: closed.sum { |t| t.gross_pnl.to_f }.round(2),
          unrealized_pnl: unrealized_pnl,
          max_drawdown: max_drawdown(closed),
          sharpe_approx: sharpe_approx(closed)
        },
        positions: current_positions,
        daily_summary: daily_summary(closed),
        readiness_score: readiness_score(closed)
      }
    end

    def readiness_score(closed_trades)
      return 0 if closed_trades.count < 20

      score = 0
      score += 20 if win_rate(closed_trades) >= 50
      score += 20 if profit_factor(closed_trades) >= 1.2
      score += 20 if max_drawdown(closed_trades) < initial_capital * 0.05
      score += 20 if closed_trades.count >= 50
      score += 20 if sharpe_approx(closed_trades) > 0.5

      score
    end

    def gating_status
      score = readiness_score(Trade.where(paper: true).where("exit_price IS NOT NULL"))
      {
        ready_for_live: score >= 80,
        readiness_score: score,
        minimum_required: 80,
        trades_required: 50,
        current_trades: Trade.where(paper: true).where("exit_price IS NOT NULL").count,
        days_in_paper: days_in_paper,
        minimum_days: 14
      }
    end

    private

    def initial_capital
      AppConfig.config.paper_trading&.initial_capital || 500_000
    end

    def win_rate(trades)
      return 0 if trades.empty?
      wins = trades.count { |t| t.gross_pnl.to_f > 0 }
      (wins.to_f / trades.count * 100).round(2)
    end

    def profit_factor(trades)
      return 0 if trades.empty?
      gross_profit = trades.sum { |t| [t.gross_pnl.to_f, 0].max }
      gross_loss = trades.sum { |t| [t.gross_pnl.to_f, 0].min }.abs
      gross_loss.zero? ? 0 : (gross_profit / gross_loss).round(2)
    end

    def unrealized_pnl
      PaperPositionStore.all.values.sum { |pos| pos[:unrealized_pnl].to_f }.round(2)
    end

    def max_drawdown(trades)
      return 0 if trades.empty?
      sorted = trades.sort_by(&:exit_time)
      peak = 0
      max_dd = 0
      cumulative = 0

      sorted.each do |t|
        cumulative += t.gross_pnl.to_f
        peak = [peak, cumulative].max
        max_dd = [max_dd, peak - cumulative].max
      end

      max_dd.round(2)
    end

    def sharpe_approx(trades)
      returns = trades.group_by { |t| t.exit_time&.to_date }
                      .transform_values { |day| day.sum { |tr| tr.gross_pnl.to_f } }
                      .values
      return 0 if returns.length < 2

      mean = returns.sum.to_f / returns.length
      std_dev = Math.sqrt(returns.sum { |r| (r - mean)**2 } / returns.length)
      std_dev.zero? ? 0 : (mean / std_dev).round(2)
    end

    def current_positions
      PaperPositionStore.all.map do |symbol, pos|
        {
          symbol: symbol,
          quantity: pos[:net_quantity],
          avg_price: pos[:net_quantity] > 0 ? pos[:buy_average] : pos[:sell_average],
          unrealized_pnl: pos[:unrealized_pnl].to_f.round(2),
          last_price: pos[:last_price]
        }
      end
    end

    def daily_summary(trades)
      trades.group_by { |t| t.exit_time&.to_date || Date.current }
            .transform_values do |day_trades|
              {
                trades: day_trades.count,
                pnl: day_trades.sum { |t| t.gross_pnl.to_f }.round(2),
                wins: day_trades.count { |t| t.gross_pnl.to_f > 0 }
              }
            end
    end

    def days_in_paper
      first = Trade.where(paper: true).minimum(:entry_time)
      return 0 unless first
      (Date.current - first.to_date).to_i
    end
  end
end

module PaperTrading
  class PaperTradingValidator
    VALIDATION_RULES = [
      :minimum_trades,
      :minimum_days,
      :positive_expectancy,
      :controlled_drawdown,
      :order_fidelity,
      :data_integrity,
      :risk_compliance
    ].freeze

    def initialize
      @dashboard = PaperTradingDashboard.new
    end

    def validate
      results = VALIDATION_RULES.map do |rule|
        { rule: rule, passed: send(rule), details: send(:"#{rule}_details") }
      end

      all_passed = results.all? { |r| r[:passed] }

      {
        valid: all_passed,
        passed_rules: results.select { |r| r[:passed] }.map { |r| r[:rule] },
        failed_rules: results.reject { |r| r[:passed] }.map { |r| r[:rule] },
        details: results,
        gating: @dashboard.gating_status
      }
    end

    def validate!
      result = validate
      raise PaperTradingNotReady, format_errors(result[:failed_rules]) unless result[:valid]
      result
    end

    private

    def minimum_trades
      Trade.where(paper: true).where("exit_price IS NOT NULL").count >= 50
    end

    def minimum_trades_details
      count = Trade.where(paper: true).where("exit_price IS NOT NULL").count
      { required: 50, actual: count, remaining: [50 - count, 0].max }
    end

    def minimum_days
      first = Trade.where(paper: true).minimum(:entry_time)
      return false unless first
      (Date.current - first.to_date).to_i >= 14
    end

    def minimum_days_details
      first = Trade.where(paper: true).minimum(:entry_time)
      days = first ? (Date.current - first.to_date).to_i : 0
      { required: 14, actual: days, remaining: [14 - days, 0].max }
    end

    def positive_expectancy
      closed = Trade.where(paper: true).where("exit_price IS NOT NULL")
      return false if closed.empty?

      wins = closed.select { |t| t.gross_pnl.to_f > 0 }
      losses = closed.reject { |t| t.gross_pnl.to_f > 0 }

      avg_win = wins.empty? ? 0 : wins.sum { |t| t.gross_pnl.to_f } / wins.length
      avg_loss = losses.empty? ? 0 : losses.sum { |t| t.gross_pnl.to_f }.abs / losses.length
      win_rate = wins.length.to_f / closed.length

      expectancy = (win_rate * avg_win) - ((1 - win_rate) * avg_loss)
      expectancy > 0
    end

    def positive_expectancy_details
      closed = Trade.where(paper: true).where("exit_price IS NOT NULL")
      wins = closed.select { |t| t.gross_pnl.to_f > 0 }
      avg_win = wins.empty? ? 0 : wins.sum { |t| t.gross_pnl.to_f } / [wins.length, 1].max
      { avg_win: avg_win.round(2), required: "positive expectancy" }
    end

    def controlled_drawdown
      max_dd = @dashboard.metrics[:performance][:max_drawdown]
      capital = PaperFundLedger.instance.total
      return false if capital.zero?
      (max_dd / capital * 100) < 5
    end

    def controlled_drawdown_details
      max_dd = @dashboard.metrics[:performance][:max_drawdown]
      capital = PaperFundLedger.instance.total
      pct = capital.zero? ? 0 : (max_dd / capital * 100).round(2)
      { max_drawdown_percent: pct, limit: 5 }
    end

    def order_fidelity
      # Verify paper orders match expected behavior (no phantom fills)
      paper_orders = PaperOrderStore.all.values
      return false if paper_orders.empty?

      orphaned = paper_orders.count { |o| o[:status] == "OPEN" && o[:created_at] < 1.hour.ago }
      orphaned == 0
    end

    def order_fidelity_details
      paper_orders = PaperOrderStore.all.values
      orphaned = paper_orders.count { |o| o[:status] == "OPEN" && o[:created_at] < 1.hour.ago }
      { orphaned_orders: orphaned, limit: 0 }
    end

    def data_integrity
      # Check that paper trades have corresponding market data
      paper_trades = Trade.where(paper: true).where("entry_time > ?", 24.hours.ago)
      return true if paper_trades.empty?

      missing_data = paper_trades.count do |t|
        t.instrument&.ticks&.where("timestamp >= ?", t.entry_time)&.empty?
      end

      missing_data == 0
    end

    def data_integrity_details
      paper_trades = Trade.where(paper: true).where("entry_time > ?", 24.hours.ago)
      missing = paper_trades.count { |t| t.instrument&.ticks&.where("timestamp >= ?", t.entry_time)&.empty? }
      { trades_with_missing_data: missing, total: paper_trades.count }
    end

    def risk_compliance
      # Check no single loss exceeded daily limit
      daily_limit = AppConfig.config.risk&.daily_loss_limit || 50_000
      closed = Trade.where(paper: true).where("exit_price IS NOT NULL")
      max_loss = closed.map { |t| t.gross_pnl.to_f }.min || 0
      max_loss.abs < daily_limit
    end

    def risk_compliance_details
      daily_limit = AppConfig.config.risk&.daily_loss_limit || 50_000
      max_loss = Trade.where(paper: true).where("exit_price IS NOT NULL").map { |t| t.gross_pnl.to_f }.min || 0
      { max_single_loss: max_loss.abs.round(2), limit: daily_limit }
    end

    def format_errors(failed)
      failed.map { |f| "#{f}: FAILED" }.join(", ")
    end
  end

  class PaperTradingNotReady < StandardError; end
end

module PaperTrading
  class PaperTradingSwitch
    def self.enabled?
      AppConfig.config.paper_trading&.enabled != false
    end

    def self.live?
      !enabled?
    end

    def self.broker_client
      if enabled?
        PaperTradingBroker.new
      else
        DhanHQ::Client.new
      end
    end

    def self.mode
      enabled? ? :paper : :live
    end

    def self.activate_paper!
      AppConfig.config.paper_trading.enabled = true
      PaperFundLedger.instance.setup
      Rails.logger.info "[PaperTrading] Paper mode activated"
    end

    def self.deactivate_paper!(force: false)
      unless force
        validator = PaperTradingValidator.new
        result = validator.validate
        unless result[:valid]
          raise PaperTradingNotReady, "Cannot switch to live: #{result[:failed_rules].join(', ')}"
        end
      end

      AppConfig.config.paper_trading.enabled = false
      Rails.logger.info "[PaperTrading] Switched to LIVE trading"
    end

    def self.reset!
      PaperOrderStore.clear
      PaperPositionStore.clear
      PaperTradeStore.clear
      PaperFundLedger.instance.reset
      Trade.where(paper: true).destroy_all
      Rails.logger.info "[PaperTrading] Paper trading data reset"
    end
  end
end

## Overview

Paper trading simulates the full trading lifecycle without real capital. It uses:
- `PaperTradingBroker` (mimics DhanHQ API)

3. `PaperOrderBook` matches against live LTP with slippage
4. `PaperPositionManager` updates position

| Rule | Requirement | Purpose |
|------|-------------|---------|
| Minimum trades | 50 closed trades | Statistical significance |

| Positive expectancy | > 0 | Profitable edge |
| Controlled drawdown | < 5% of capital | Risk management |
| Order fidelity | 0 orphaned orders | System reliability |
| Data integrity | 0 missing tick data | Data quality |
| Risk compliance | No single loss > daily limit | Downside protection |

- 50+ trades: +20

- Profit factor ≥ 1.2: +20

- Sharpe > 0.5: +20

**Minimum 80/100 required to deactivate paper mode.**
