class DailyReviewJob < JobBase
  queue_as :reports

  def perform(date = Date.current)
    return unless ExchangeCalendar.for_exchange("NSE").find_by(date: date)&.trading_day

    trades = Trade.where("DATE(entry_time) = ?", date)
    return if trades.empty?

    report = {
      date: date,
      total_trades: trades.count,
      win_rate: calculate_win_rate(trades),
      pnl: trades.sum { |t| t.gross_pnl.to_f }.round(2),
      best_trade: trades.max_by { |t| t.gross_pnl.to_f }&.id,
      worst_trade: trades.min_by { |t| t.gross_pnl.to_f }&.id,
      regime_breakdown: regime_breakdown(trades),
      session_breakdown: session_breakdown(trades),
      ai_accuracy: ai_accuracy(trades),
      generated_at: Time.current.iso8601
    }

    # Store in Redis for quick access
    RedisPool.general.setex("daily_review:#{date}", 7.days.to_i, report.to_json)

    # Generate AI retrospective if enabled
    if AppConfig.config.ai&.daily_retrospective_enabled
      AiRetrospectiveJob.perform_later(date)
    end

    Rails.logger.info "[DailyReviewJob] Completed review for #{date}"
    report
  end

  private

  def calculate_win_rate(trades)
    wins = trades.count { |t| t.gross_pnl.to_f > 0 }
    (wins.to_f / trades.count * 100).round(2)
  end

  def regime_breakdown(trades)
    trades.group_by(&:regime_type).transform_values do |group|
      {
        count: group.count,
        win_rate: calculate_win_rate(group),
        avg_pnl: (group.sum { |t| t.gross_pnl.to_f } / group.count).round(2)
      }
    end
  end

  def session_breakdown(trades)
    trades.group_by { |t| session_name(t.entry_time) }.transform_values do |group|
      {
        count: group.count,
        win_rate: calculate_win_rate(group),
        avg_pnl: (group.sum { |t| t.gross_pnl.to_f } / group.count).round(2)
      }
    end
  end

  def session_name(time)
    return :unknown unless time
    hour = time.hour
    case hour
    when 9..10 then :opening
    when 10..12 then :morning
    when 12..14 then :afternoon
    when 14..15 then :closing
    else :after_hours
    end
  end

  def ai_accuracy(trades)
    ai_trades = trades.where(ai_validated: true)
    return nil if ai_trades.empty?
    ai_wins = ai_trades.count { |t| t.gross_pnl.to_f > 0 }
    (ai_wins.to_f / ai_trades.count * 100).round(2)
  end
end

module ContinuousImprovement
  class TradeClustering
    def initialize(trades:, features: default_features)
      @trades = trades
      @features = features
    end

    def call
      return { clusters: 0 } if @trades.count < 10

      # K-means style clustering using trade features
      clusters = perform_clustering

      {
        cluster_count: clusters.length,
        clusters: clusters.map { |c| summarize_cluster(c) },
        outliers: detect_outliers(clusters),
        recommendations: cluster_recommendations(clusters)
      }
    end

    private

    def default_features
      [:entry_hour, :regime_type, :setup_type, :delta_at_entry, :pnl_percent]
    end

    def perform_clustering
      # Simplified: group by regime + strategy + hour
      @trades.group_by do |t|
        [
          t.regime_type || :unknown,
          t.setup_type || :unknown,
          t.entry_time&.hour || 0
        ]
      end.values
    end

    def summarize_cluster(cluster)
      pnls = cluster.map { |t| t.gross_pnl.to_f }
      {
        size: cluster.length,
        win_rate: (cluster.count { |t| t.gross_pnl.to_f > 0 }.to_f / cluster.length * 100).round(2),
        avg_pnl: (pnls.sum / cluster.length).round(2),
        total_pnl: pnls.sum.round(2),
        avg_holding_time: (cluster.sum { |t| t.holding_time_seconds.to_i } / cluster.length).round(2),
        dominant_regime: cluster.group_by(&:regime_type).max_by { |_, v| v.length }&.first,
        dominant_strategy: cluster.group_by(&:setup_type).max_by { |_, v| v.length }&.first
      }
    end

    def detect_outliers(clusters)
      all_pnls = clusters.flatten.map { |t| t.gross_pnl.to_f }
      mean = all_pnls.sum / all_pnls.length
      std_dev = Math.sqrt(all_pnls.sum { |p| (p - mean)**2 } / all_pnls.length)

      clusters.flatten.select do |t|
        pnl = t.gross_pnl.to_f
        (pnl - mean).abs > 2 * std_dev
      end.map do |t|
        { trade_id: t.id, pnl: t.gross_pnl.to_f, z_score: ((t.gross_pnl.to_f - mean) / std_dev).round(2) }
      end
    end

    def cluster_recommendations(clusters)
      recommendations = []

      best = clusters.max_by { |c| summarize_cluster(c)[:avg_pnl] }
      if best
        rec = summarize_cluster(best)
        recommendations << "Focus on #{rec[:dominant_regime]} regime during #{rec[:dominant_strategy]} with #{rec[:win_rate]}% win rate"
      end

      worst = clusters.min_by { |c| summarize_cluster(c)[:avg_pnl] }
      if worst && summarize_cluster(worst)[:avg_pnl] < 0
        recommendations << "Avoid #{summarize_cluster(worst)[:dominant_regime]} + #{summarize_cluster(worst)[:dominant_strategy]} combination"
      end

      recommendations
    end
  end
end

module ContinuousImprovement
  class StrategyOptimizer
    def initialize(strategy:, lookback_days: 30)
      @strategy = strategy
      @lookback = lookback_days
    end

    def call
      trades = fetch_trades
      return { status: :insufficient_data } if trades.count < 10

      current_params = current_strategy_params
      optimized = optimize_parameters(trades, current_params)

      {
        strategy: @strategy,
        current_params: current_params,
        suggested_params: optimized[:params],
        expected_improvement: optimized[:improvement],
        confidence: optimized[:confidence],
        backtest_result: optimized[:backtest],
        recommendation: optimization_recommendation(optimized)
      }
    end

    private

    def fetch_trades
      Trade.where(setup_type: @strategy)
           .where("entry_time > ?", @lookback.days.ago)
           .where("exit_time IS NOT NULL")
           .order(:entry_time)
    end

    def current_strategy_params
      StrategyConfig.find_by(name: @strategy)&.params || {}
    end

    def optimize_parameters(trades, current)
      # Grid search over parameter space
      grid = parameter_grid(current)

      best = grid.max_by do |params|
        simulate_with_params(trades, params)[:score]
      end

      backtest = simulate_with_params(trades, best)
      current_score = simulate_with_params(trades, current)[:score]

      {
        params: best,
        improvement: ((backtest[:score] - current_score) / [current_score.abs, 1].max * 100).round(2),
        confidence: backtest[:trades] > 20 ? :high : :medium,
        backtest: backtest
      }
    end

    def parameter_grid(base)
      # Vary each parameter ±20%
      keys = base.keys
      return [base] if keys.empty?

      keys.map do |key|
        val = base[key].to_f
        [key, [val * 0.8, val, val * 1.2].map { |v| v.round(2) }]
      end.to_h.then { |h| h.values[0].product(*h.values[1..]).map { |combo| keys.zip(combo).to_h } }
    end

    def simulate_with_params(trades, params)
      # Simplified: filter trades that would have met stricter criteria
      filtered = trades.select { |t| trade_meets_params?(t, params) }
      pnls = filtered.map { |t| t.gross_pnl.to_f }

      {
        trades: filtered.length,
        score: pnls.sum,
        win_rate: filtered.empty? ? 0 : (filtered.count { |t| t.gross_pnl.to_f > 0 }.to_f / filtered.length * 100).round(2)
      }
    end

    def trade_meets_params?(trade, params)
      # Apply parameter filters
      return true if params.empty?
      # Implementation depends on strategy-specific parameters
      true
    end

    def optimization_recommendation(optimized)
      if optimized[:improvement] > 10 && optimized[:confidence] == :high
        :deploy_suggested
      elsif optimized[:improvement] > 5
        :test_in_paper
      else
        :keep_current
      end
    end
  end
end

module ContinuousImprovement
  class FeatureImportanceAnalyzer
    FEATURES = [
      :market_regime_score,
      :momentum_score,
      :liquidity_score,
      :option_flow_score,
      :greeks_score,
      :strike_quality_score,
      :time_of_day,
      :days_to_expiry
    ].freeze

    def initialize(trades:)
      @trades = trades
    end

    def call
      return { status: :insufficient_data } if @trades.count < 20

      importance = FEATURES.map { |f| [f, feature_correlation(f)] }.to_h
      sorted = importance.sort_by { |_, v| -v.abs }

      {
        feature_importance: sorted.to_h.transform_values { |v| v.round(4) },
        top_features: sorted.first(3).map(&:first),
        least_important: sorted.last(3).map(&:first),
        recommendation: feature_recommendation(sorted)
      }
    end

    private

    def feature_correlation(feature)
      # Point-biserial correlation between feature and win/loss
      wins = @trades.select { |t| t.gross_pnl.to_f > 0 }
      losses = @trades.reject { |t| t.gross_pnl.to_f > 0 }

      win_avg = extract_feature(wins, feature).sum.to_f / wins.length rescue 0
      loss_avg = extract_feature(losses, feature).sum.to_f / losses.length rescue 0

      overall_std = standard_deviation(extract_feature(@trades, feature))
      return 0 if overall_std.zero?

      (win_avg - loss_avg) / overall_std
    end

    def extract_feature(trades, feature)
      trades.map do |t|
        case feature
        when :market_regime_score
          t.trade_score&.breakdown&.dig("market_regime", "score") || 0
        when :momentum_score
          t.trade_score&.momentum_score || 0
        when :liquidity_score
          t.trade_score&.liquidity_score || 0
        when :option_flow_score
          t.trade_score&.breakdown&.dig("option_flow", "score") || 0
        when :greeks_score
          t.trade_score&.breakdown&.dig("greeks", "score") || 0
        when :strike_quality_score
          t.trade_score&.breakdown&.dig("strike_quality", "score") || 0
        when :time_of_day
          t.entry_time&.hour || 0
        when :days_to_expiry
          t.days_to_expiry || 0
        else
          0
        end
      end
    end

    def standard_deviation(values)
      return 0 if values.empty?
      mean = values.sum.to_f / values.length
      Math.sqrt(values.sum { |v| (v - mean)**2 } / values.length)
    end

    def feature_recommendation(sorted)
      top = sorted.first(3).map(&:first)
      recommendations = []

      if top.include?(:market_regime_score)
        recommendations << "Market regime is the strongest predictor. Increase weight in scoring."
      end

      if top.include?(:liquidity_score)
        recommendations << "Liquidity matters significantly. Avoid low-liquidity entries."
      end

      if sorted.last(3).map(&:first).include?(:time_of_day)
        recommendations << "Time of day shows low predictive power. Consider removing from filters."
      end

      recommendations
    end
  end
end

module ContinuousImprovement
  class RegimeAdaptation
    def initialize(lookback_days: 30)
      @lookback = lookback_days
    end

    def call
      regimes = analyze_regime_performance

      {
        current_weights: current_weights,
        suggested_weights: suggested_weights(regimes),
        regime_performance: regimes,
        adaptation_reasoning: reasoning(regimes),
        confidence: confidence(regimes)
      }
    end

    private

    def analyze_regime_performance
      trades = Trade.where("entry_time > ?", @lookback.days.ago)
                    .where("exit_time IS NOT NULL")

      trades.group_by(&:regime_type).transform_values do |group|
        wins = group.count { |t| t.gross_pnl.to_f > 0 }
        pnls = group.map { |t| t.gross_pnl.to_f }

        {
          trades: group.length,
          win_rate: (wins.to_f / group.length * 100).round(2),
          avg_pnl: (pnls.sum / group.length).round(2),
          total_pnl: pnls.sum.round(2),
          expectancy: ((wins.to_f / group.length) * (pnls.sum / group.length)).round(2)
        }
      end
    end

    def current_weights
      StrategyWeight.current
    end

    def suggested_weights(regimes)
      total_expectancy = regimes.values.sum { |r| [r[:expectancy], 0].max }

      if total_expectancy.zero?
        return regimes.transform_values { |_| 1.0 / regimes.length }
      end

      regimes.transform_values do |stats|
        positive_expectancy = [stats[:expectancy], 0].max
        (positive_expectancy / total_expectancy).round(4)
      end
    end

    def reasoning(regimes)
      best = regimes.max_by { |_, v| v[:expectancy] }
      worst = regimes.min_by { |_, v| v[:expectancy] }

      [
        "Best performing regime: #{best.first} (expectancy: #{best.last[:expectancy]})",
        "Worst performing regime: #{worst.first} (expectancy: #{worst.last[:expectancy]})",
        "Weight shifted toward #{best.first}, reduced for #{worst.first}"
      ]
    end

    def confidence(regimes)
      sample_sizes = regimes.values.map { |v| v[:trades] }
      return :low if sample_sizes.any? { |s| s < 5 }
      return :medium if sample_sizes.any? { |s| s < 10 }
      :high
    end
  end
end

module ContinuousImprovement
  class StrikeSelectionOptimizer
    DELTA_BUCKETS = [(0.0..0.3), (0.3..0.5), (0.5..0.7), (0.7..1.0)].freeze

    def initialize(strategy:, lookback_days: 60)
      @strategy = strategy
      @lookback = lookback_days
    end

    def call
      trades = fetch_trades
      return { status: :insufficient_data } if trades.count < 10

      by_delta = trades.group_by { |t| delta_bucket(t.delta_at_entry) }

      {
        delta_performance: by_delta.transform_values { |g| performance_stats(g) },
        optimal_delta_range: optimal_range(by_delta),
        dte_performance: dte_performance(trades),
        recommendation: strike_recommendation(by_delta)
      }
    end

    private

    def fetch_trades
      Trade.where(setup_type: @strategy)
           .where("entry_time > ?", @lookback.days.ago)
           .where("exit_time IS NOT NULL")
           .where("delta_at_entry IS NOT NULL")
    end

    def delta_bucket(delta)
      return :unknown unless delta
      DELTA_BUCKETS.each_with_index do |range, i|
        return "bucket_#{i}" if range.cover?(delta.to_f.abs)
      end
      :extreme
    end

    def performance_stats(trades)
      pnls = trades.map { |t| t.gross_pnl.to_f }
      wins = trades.count { |t| t.gross_pnl.to_f > 0 }

      {
        count: trades.length,
        win_rate: (wins.to_f / trades.length * 100).round(2),
        avg_pnl: (pnls.sum / trades.length).round(2),
        total_pnl: pnls.sum.round(2)
      }
    end

    def optimal_range(by_delta)
      best = by_delta.max_by { |_, trades| performance_stats(trades)[:avg_pnl] }
      best&.first
    end

    def dte_performance(trades)
      by_dte = trades.group_by do |t|
        dte = t.days_to_expiry
        case dte
        when 0..2 then :same_day
        when 3..7 then :week
        when 8..14 then :two_weeks
        else :month
        end
      end

      by_dte.transform_values { |g| performance_stats(g) }
    end

    def strike_recommendation(by_delta)
      optimal = optimal_range(by_delta)
      return :insufficient_data unless optimal

      {
        preferred_delta_range: optimal,
        avoid_ranges: by_delta.select { |k, _| performance_stats(by_delta[k])[:avg_pnl] < 0 }.keys
      }
    end
  end
end

# app/services/continuous_improvement/risk_parameter_tuning.rb

module ContinuousImprovement
  class RiskParameterTuning
    def initialize(lookback_days: 30)
      @lookback = lookback_days
    end

    def call
      drawdown_stats = analyze_drawdowns
      win_rate_stats = analyze_win_rate
      mae_stats = analyze_mae

      {
        current_limits: current_limits,
        suggested_limits: suggested_limits(drawdown_stats, mae_stats),
        drawdown_analysis: drawdown_stats,
        stop_loss_recommendation: stop_loss_recommendation(mae_stats),
        position_sizing: position_sizing_recommendation(drawdown_stats),
        confidence: :medium
      }
    end

    private

    def analyze_drawdowns
      trades = Trade.where("entry_time > ?", @lookback.days.ago)
                    .where("exit_time IS NOT NULL")
                    .order(:entry_time)

      return {} if trades.empty?

      cumulative = 0
      peak = 0
      max_dd = 0
      dd_periods = []

      trades.each do |t|
        cumulative += t.gross_pnl.to_f
        if cumulative > peak
          peak = cumulative
        else
          dd = peak - cumulative
          max_dd = dd if dd > max_dd
        end
      end

      {
        max_drawdown: max_dd.round(2),
        max_drawdown_percent: (peak.zero? ? 0 : (max_dd / peak * 100)).round(2),
        consecutive_losses: consecutive_losses(trades),
        avg_loss: trades.select { |t| t.gross_pnl.to_f < 0 }.map { |t| t.gross_pnl.to_f }.sum.fdiv(trades.count { |t| t.gross_pnl.to_f < 0 }).round(2) rescue 0
      }
    end

    def consecutive_losses(trades)
      max_streak = 0
      current = 0
      trades.each do |t|
        if t.gross_pnl.to_f < 0
          current += 1
          max_streak = [max_streak, current].max
        else
          current = 0
        end
      end
      max_streak
    end

    def analyze_win_rate
      trades = Trade.where("entry_time > ?", @lookback.days.ago).where("exit_time IS NOT NULL")
      wins = trades.count { |t| t.gross_pnl.to_f > 0 }
      { rate: trades.empty? ? 0 : (wins.to_f / trades.count * 100).round(2) }
    end

    def analyze_mae
      trades = Trade.where("entry_time > ?", @lookback.days.ago)
                    .where("exit_time IS NOT NULL")
                    .where("mae_r_multiple IS NOT NULL")

      return {} if trades.empty?

      maes = trades.map { |t| t.mae_r_multiple.to_f }
      {
        avg_mae_r: (maes.sum / maes.length).round(2),
        max_mae_r: maes.max.round(2),
        recommended_stop_r: (maes.max * 1.2).round(2)
      }
    end

    def current_limits
      {
        daily_loss: AppConfig.config.risk&.daily_loss_limit,
        max_position: AppConfig.config.risk&.max_position_value,
        max_positions: AppConfig.config.risk&.max_concurrent_positions
      }
    end

    def suggested_limits(drawdown, mae)
      # Tighten if drawdowns are large
      current = current_limits
      multiplier = drawdown[:max_drawdown_percent].to_f > 10 ? 0.8 : 1.0

      {
        daily_loss: (current[:daily_loss].to_f * multiplier).round(2),
        max_position: current[:max_position],
        stop_loss_r: mae[:recommended_stop_r]
      }
    end

    def stop_loss_recommendation(mae)
      return :insufficient_data if mae.empty?
      "Set stop loss at #{mae[:recommended_stop_r]}R based on max observed MAE"
    end

    def position_sizing_recommendation(drawdown)
      if drawdown[:consecutive_losses].to_i > 3
        :reduce_size_25_percent
      else
        :maintain_current
      end
    end
  end
end

# app/services/continuous_improvement/ai_prompt_optimization.rb
# frozen_string_literal: true

module ContinuousImprovement
  class AIPromptOptimization
    def initialize(agent_type:, lookback_days: 30)
      @agent_type = agent_type
      @lookback = lookback_days
    end

    def call
      analyses = fetch_analyses
      return { status: :insufficient_data } if analyses.count < 5

      accuracy = calculate_accuracy(analyses)
      confidence_distribution = confidence_distribution(analyses)

      {
        agent: @agent_type,
        current_template: current_template,
        accuracy: accuracy,
        confidence_distribution: confidence_distribution,
        prompt_adjustments: suggest_adjustments(accuracy, confidence_distribution),
        test_recommendation: test_recommendation(accuracy)
      }
    end

    private

    def fetch_analyses
      AiAnalysisCache.where(agent_name: @agent_type)
                     .where("created_at > ?", @lookback.days.ago)
                     .where.not(analyzable: nil)
    end

    def calculate_accuracy(analyses)
      # Compare AI verdict with actual outcome
      correct = analyses.count do |analysis|
        actual = analysis.analyzable
        next false unless actual.respond_to?(:gross_pnl)

        ai_verdict = analysis.full_response&.dig("verdict")
        actual_win = actual.gross_pnl.to_f > 0

        (ai_verdict == "approve" && actual_win) ||
          (ai_verdict == "reject" && !actual_win)
      end

      (correct.to_f / analyses.count * 100).round(2)
    end

    def confidence_distribution(analyses)
      analyses.group_by { |a| confidence_bucket(a.ai_confidence) }.transform_values(&:count)
    end

    def confidence_bucket(score)
      case score.to_f
      when 0..50 then :low
      when 50..75 then :medium
      else :high
      end
    end

    def current_template
      template_path = Rails.root.join("app/prompts/#{@agent_type}.md.erb")
      File.exist?(template_path) ? File.read(template_path) : "Template not found"
    end

    def suggest_adjustments(accuracy, distribution)
      adjustments = []

      if accuracy < 60
        adjustments << "Add more specific examples of #{@agent_type} failures in prompt"
      end

      if (distribution[:low] || 0) > (distribution[:high] || 0)
        adjustments << "Simplify prompt structure to increase AI confidence"
      end

      if accuracy > 75 && (distribution[:high] || 0) > 0
        adjustments << "Consider lowering temperature for more deterministic responses"
      end

      adjustments
    end

    def test_recommendation(accuracy)
      case accuracy
      when 0..50 then :major_revision_required
      when 50..65 then :prompt_tuning_recommended
      when 65..80 then :minor_adjustments
      else :maintain_current
      end
    end
  end

class WeeklyRetrospectiveGenerator < JobBase
  queue_as :reports

  def perform(week_ending = Date.current)
    start_date = week_ending - 7.days

    trades = Trade.where("entry_time BETWEEN ? AND ?", start_date, week_ending)
                  .where("exit_time IS NOT NULL")

    clustering = ContinuousImprovement::TradeClustering.new(trades: trades).call
    feature_analysis = ContinuousImprovement::FeatureImportanceAnalyzer.new(trades: trades).call
    regime_adaptation = ContinuousImprovement::RegimeAdaptation.new(lookback_days: 7).call
    risk_tuning = ContinuousImprovement::RiskParameterTuning.new(lookback_days: 7).call

    report = {
      week: "#{start_date} to #{week_ending}",
      summary: weekly_summary(trades),
      clustering: clustering,
      feature_importance: feature_analysis,
      regime_adaptation: regime_adaptation,
      risk_tuning: risk_tuning,
      action_items: generate_action_items(trades, clustering, feature_analysis),
      generated_at: Time.current.iso8601
    }

    # Store and optionally send
    RedisPool.general.setex("weekly_retrospective:#{week_ending}", 30.days.to_i, report.to_json)

    # AI-generated narrative if enabled
    if AppConfig.config.ai&.weekly_retrospective_enabled
      AiWeeklyRetrospectiveJob.perform_later(week_ending)
    end

    report
  end

  private

  def weekly_summary(trades)
    pnls = trades.map { |t| t.gross_pnl.to_f }
    wins = trades.count { |t| t.gross_pnl.to_f > 0 }

    {
      total_trades: trades.count,
      win_rate: trades.empty? ? 0 : (wins.to_f / trades.count * 100).round(2),
      total_pnl: pnls.sum.round(2),
      avg_pnl: trades.empty? ? 0 : (pnls.sum / trades.count).round(2),
      best_day: best_day(trades),
      worst_day: worst_day(trades)
    }
  end

  def best_day(trades)
    trades.group_by { |t| t.entry_time&.to_date }
          .transform_values { |g| g.sum { |t| t.gross_pnl.to_f } }
          .max_by { |_, v| v }
  end

  def worst_day(trades)
    trades.group_by { |t| t.entry_time&.to_date }
          .transform_values { |g| g.sum { |t| t.gross_pnl.to_f } }
          .min_by { |_, v| v }
  end

  def generate_action_items(trades, clustering, features)
    actions = []

    if clustering[:recommendations]
      actions.concat(clustering[:recommendations])
    end

    if features[:recommendation]
      actions.concat(features[:recommendation])
    end

    if trades.count > 10
      actions << "Review top 3 feature predictors for next week's scoring"
    end

    actions
  end
end

# app/jobs/monthly_strategy_review.rb
# frozen_string_literal: true

class MonthlyStrategyReview < JobBase
  queue_as :reports

  def perform(month = Date.current.month, year = Date.current.year)
    start_date = Date.new(year, month, 1)
    end_date = start_date.end_of_month

    strategies = Trade.where("entry_time BETWEEN ? AND ?", start_date, end_date)
                      .where("exit_time IS NOT NULL")
                      .distinct
                      .pluck(:setup_type)

    strategy_reviews = strategies.map do |strategy|
      review_strategy(strategy, start_date, end_date)
    end

    {
      month: "#{year}-#{month.to_s.rjust(2, '0')}",
      strategy_reviews: strategy_reviews,
      overall_recommendation: overall_recommendation(strategy_reviews),
      capital_allocation: suggested_allocation(strategy_reviews),
      generated_at: Time.current.iso8601
    }
  end

  private

  def review_strategy(strategy, start_date, end_date)
    trades = Trade.where(setup_type: strategy)
                  .where("entry_time BETWEEN ? AND ?", start_date, end_date)
                  .where("exit_time IS NOT NULL")

    pnls = trades.map { |t| t.gross_pnl.to_f }
    wins = trades.count { |t| t.gross_pnl.to_f > 0 }

    expectancy = Engines::Learning::StrategyExpectancyCalculator.new(
      strategy: strategy,
      lookback: (end_date - start_date).to_i.days
    ).call

    {
      strategy: strategy,
      trades: trades.count,
      win_rate: trades.empty? ? 0 : (wins.to_f / trades.count * 100).round(2),
      total_pnl: pnls.sum.round(2),
      expectancy: expectancy,
      recommendation: strategy_recommendation(expectancy, trades.count)
    }
  end

  def strategy_recommendation(expectancy, trade_count)
    return :insufficient_data if trade_count < 10

    if expectancy[:expectancy].to_f > 1000 && expectancy[:win_rate].to_f > 50
      :increase_allocation
    elsif expectancy[:expectancy].to_f > 0
      :maintain
    elsif expectancy[:expectancy].to_f > -500
      :reduce_allocation
    else
      :deprecate
    end
  end

  def overall_recommendation(reviews)
    positive = reviews.count { |r| r[:recommendation] == :increase_allocation }
    negative = reviews.count { |r| r[:recommendation] == :deprecate }

    if positive > negative
      :expand_winning_strategies
    elsif negative > positive
      :consolidate_and_cut_losers
    else
      :steady_state
    end
  end

  def suggested_allocation(reviews)
    total_expectancy = reviews.sum { |r| [r[:expectancy][:expectancy].to_f, 0].max }
    return reviews.map { |r| [r[:strategy], 1.0 / reviews.length] }.to_h if total_expectancy.zero?

    reviews.map do |r|
      weight = [r[:expectancy][:expectancy].to_f, 0].max / total_expectancy
      [r[:strategy], weight.round(4)]
    end.to_h
  end
end

# app/jobs/backtest_automation_job.rb
# frozen_string_literal: true

class BacktestAutomationJob < JobBase
  queue_as :replay

  def perform(strategy:, from:, to:, params: {})
    instrument = Instrument.find_by(trading_symbol: params[:instrument] || "NIFTY")

    # Run replay with new parameters
    replay = Engines::Replay::HistoricalReplayEngine.new(
      strategy_engine: strategy_engine_for(strategy, params),
      instrument: instrument,
      from: Time.parse(from),
      to: Time.parse(to)
    )

    result = replay.call(context: { replay_config: { speed: :instant } })

    if result.success?
      # Compare to current production parameters
      current = Engines::Replay::HistoricalReplayEngine.new(
        strategy_engine: strategy_engine_for(strategy, {}),
        instrument: instrument,
        from: Time.parse(from),
        to: Time.parse(to)
      ).call(context: { replay_config: { speed: :instant } })

      comparison = if current.success?
        {
          new_params_pnl: result.value[:metrics][:gross_pnl],
          current_params_pnl: current.value[:metrics][:gross_pnl],
          improvement: (result.value[:metrics][:gross_pnl] - current.value[:metrics][:gross_pnl]).round(2)
        }
      else
        { error: "Current params replay failed" }
      end

      {
        strategy: strategy,
        params: params,
        backtest: result.value,
        comparison: comparison,
        recommendation: comparison[:improvement].to_f > 0 ? :test_in_paper : :discard
      }
    else
      { error: result.error.to_h }
    end
  end

  private

  def strategy_engine_for(strategy, params)
    klass = "Engines::Strategy::#{strategy.camelize}Engine".constantize
    engine = klass.new
    engine.apply_params(params) if engine.respond_to?(:apply_params)
    engine
  rescue
    Engines::Strategy::BaseStrategyEngine.new
  end
end
