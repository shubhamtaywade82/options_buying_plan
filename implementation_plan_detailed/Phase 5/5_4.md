```ruby
# app/engines/trade_scoring/trade_scoring_engine.rb
# frozen_string_literal: true

module Engines
  module TradeScoring
    class TradeScoringEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[
        market_context
        market_regime
        market_structure
        momentum
        liquidity
        option_flow
        greeks
        strike_quality
      ].freeze

      DEFAULT_THRESHOLD = 80.0
      MINIMUM_COMPONENT_SCORE = 30.0  # No component can be below this

      # Institutional framework weights
      DEFAULT_WEIGHTS = {
        market_context: 0.10,
        market_regime: 0.20,
        market_structure: 0.20,
        momentum: 0.10,
        liquidity: 0.10,
        option_flow: 0.15,
        greeks: 0.10,
        strike_quality: 0.05
      }.freeze

      def initialize(instrument:, direction:)
        @instrument = instrument
        @direction = direction  # :long or :short
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        # Validate no component is catastrophically low
        component_validation = validate_minimum_scores(context)
        return component_validation if component_validation.failure?

        # Calculate individual component scores
        scores = calculate_component_scores(context)

        # Apply dynamic weight adjustment based on regime
        weights = adjust_weights_for_regime(context[:market_regime], DEFAULT_WEIGHTS.dup)

        # Calculate weighted total
        total_score = calculate_weighted_total(scores, weights)

        # Determine pass/fail
        threshold = context[:threshold] || DEFAULT_THRESHOLD
        passed = total_score >= threshold && all_critical_components_pass?(scores)

        # Generate explanation
        explanation = generate_explanation(scores, weights, total_score, passed)

        # Build breakdown
        breakdown = build_score_breakdown(scores, weights, total_score)

        emit(EventTypes::TRADE_SCORED, {
          instrument_id: @instrument.id,
          direction: @direction,
          total_score: total_score,
          threshold: threshold,
          passed: passed,
          dominant_factors: dominant_factors(scores)
        })

        success(
          total_score: total_score.round(2),
          threshold: threshold,
          passed: passed,
          component_scores: scores,
          weights_used: weights,
          breakdown: breakdown,
          explanation: explanation,
          confidence: calculate_overall_confidence(scores),
          recommendation: trade_recommendation(total_score, passed, scores)
        )
      end

      private

      def validate_minimum_scores(context)
        critical_components = %i[market_regime market_structure liquidity option_flow]

        critical_components.each do |component|
          score = extract_score(context[component])
          if score && score < MINIMUM_COMPONENT_SCORE
            return Result.failure(
              RiskError.new(
                component: component,
                score: score,
                minimum_required: MINIMUM_COMPONENT_SCORE,
                reason: "Critical component below minimum threshold"
              )
            )
          end
        end

        Result.success(true)
      end

      def calculate_component_scores(context)
        {
          market_context: score_market_context(context[:market_context]),
          market_regime: score_market_regime(context[:market_regime]),
          market_structure: score_market_structure(context[:market_structure]),
          momentum: score_momentum(context[:momentum]),
          liquidity: score_liquidity(context[:liquidity]),
          option_flow: score_option_flow(context[:option_flow]),
          greeks: score_greeks(context[:greeks]),
          strike_quality: score_strike_quality(context[:strike_quality])
        }
      end

      def extract_score(component_data)
        return nil unless component_data.is_a?(Hash)
        component_data[:score] || component_data[:regime_score] || component_data[:liquidity_score]
      end

      def adjust_weights_for_regime(regime_data, weights)
        return weights unless regime_data.is_a?(Hash)

        regime_type = regime_data[:type] || regime_data[:regime_type]

        case regime_type
        when :trending_up, :trending_down
          # Increase momentum and structure weight in trends
          weights[:momentum] += 0.05
          weights[:market_structure] += 0.05
          weights[:market_context] -= 0.05
          weights[:option_flow] -= 0.05
        when :range, :compression
          # Increase option flow and Greeks in ranging markets
          weights[:option_flow] += 0.05
          weights[:greeks] += 0.05
          weights[:momentum] -= 0.05
          weights[:market_structure] -= 0.05
        when :expansion, :volatile
          # Increase liquidity and context in volatile markets
          weights[:liquidity] += 0.05
          weights[:market_context] += 0.05
          weights[:strike_quality] -= 0.05
          weights[:greeks] -= 0.05
        end

        # Normalize to ensure sum = 1.0
        total = weights.values.sum
        weights.transform_values { |v| (v / total).round(4) }
      end

      def calculate_weighted_total(scores, weights)
        scores.sum { |component, data| data[:score] * weights[component] }.round(2)
      end

      def all_critical_components_pass?(scores)
        critical = %i[market_regime market_structure liquidity]
        critical.all? { |c| scores[c][:score] >= MINIMUM_COMPONENT_SCORE }
      end

      def dominant_factors(scores)
        scores.sort_by { |_, data| -data[:score] }.first(3).to_h
      end

      def generate_explanation(scores, weights, total, passed)
        parts = []

        if passed
          parts << "Trade PASSED with score #{total}/100 (threshold: #{DEFAULT_THRESHOLD})"
        else
          parts << "Trade FAILED with score #{total}/100 (threshold: #{DEFAULT_THRESHOLD})"
        end

        # Top contributing factors
        top = scores.max_by { |_, data| data[:score] }
        parts << "Strongest factor: #{top[0]} (#{top[1][:score]}pts)"

        # Weakest factor
        bottom = scores.min_by { |_, data| data[:score] }
        parts << "Weakest factor: #{bottom[0]} (#{bottom[1][:score]}pts)"

        # Direction alignment
        aligned = scores.count { |_, data| data[:direction_aligned] == true }
        parts << "Direction alignment: #{aligned}/#{scores.length} components aligned"

        # Risk assessment
        risk_factors = scores.select { |_, data| data[:risk_flag] }.keys
        parts << "Risk flags: #{risk_factors.any? ? risk_factors.join(', ') : 'none'}" if risk_factors.any?

        parts.join(". ")
      end

      def build_score_breakdown(scores, weights, total)
        {
          components: scores.transform_values { |data| data[:score] },
          weighted_contributions: scores.transform_values { |data| data[:score] }.to_h do |component, score|
            [component, (score * weights[component]).round(2)]
          end,
          total: total,
          grade: score_grade(total),
          threshold: DEFAULT_THRESHOLD,
          margin_above_threshold: (total - DEFAULT_THRESHOLD).round(2)
        }
      end

      def score_grade(score)
        case score
        when 90..100 then :excellent
        when 80..90 then :strong
        when 70..80 then :acceptable
        when 60..70 then :marginal
        when 50..60 then :weak
        else :reject
        end
      end

      def calculate_overall_confidence(scores)
        # Higher confidence when components agree
        score_values = scores.values.map { |d| d[:score] }
        mean = score_values.sum.to_f / score_values.length
        variance = score_values.sum { |s| (s - mean) ** 2 } / score_values.length
        std_dev = Math.sqrt(variance)

        # Low variance = high confidence (components agree)
        agreement_score = [100 - std_dev * 2, 0].max

        # Weight by number of data-rich components
        data_quality = scores.count { |_, d| d[:data_quality] == :high }.to_f / scores.length * 100

        ((agreement_score + data_quality) / 2).round(2)
      end

      def trade_recommendation(total, passed, scores)
        return :reject unless passed

        if total >= 90
          :immediate_entry_full_size
        elsif total >= 85
          :favorable_entry_standard_size
        elsif total >= 80
          :conditional_entry_reduced_size
        else
          :observe_wait
        end
      end

      # Component scoring methods
      def score_market_context(context_data)
        return { score: 50, direction_aligned: true, risk_flag: false, data_quality: :low } unless context_data

        allow = context_data[:allow_trading] == true
        score = context_data[:context_score] || 50

        {
          score: allow ? score : 0,
          context: context_data[:context] || :unknown,
          direction_aligned: true,  # Context is direction-agnostic
          risk_flag: !allow,
          data_quality: :high
        }
      end

      def score_market_regime(regime_data)
        return { score: 50, direction_aligned: false, risk_flag: true, data_quality: :low } unless regime_data

        regime_score = regime_data[:regime_score] || 50
        regime_type = regime_data[:regime_type] || regime_data[:type]

        aligned = direction_aligned_with_regime?(regime_type)

        {
          score: aligned ? regime_score : regime_score * 0.3,
          regime_type: regime_type,
          direction_aligned: aligned,
          risk_flag: regime_score < 40,
          data_quality: :high
        }
      end

      def score_market_structure(structure_data)
        return { score: 50, direction_aligned: false, risk_flag: true, data_quality: :low } unless structure_data

        structure_score = structure_data[:structure_score] || structure_data[:score] || 50
        structure_type = structure_data[:structure_type] || structure_data[:type]

        aligned = direction_aligned_with_structure?(structure_type)

        {
          score: aligned ? structure_score : structure_score * 0.3,
          structure_type: structure_type,
          direction_aligned: aligned,
          risk_flag: structure_score < 40,
          data_quality: :high
        }
      end

      def score_momentum(momentum_data)
        return { score: 50, direction_aligned: false, risk_flag: true, data_quality: :low } unless momentum_data

        momentum_score = momentum_data[:momentum_score] || momentum_data[:score] || 50
        momentum_direction = momentum_data[:direction] || momentum_data[:momentum_direction]

        aligned = direction_aligned_with_momentum?(momentum_direction)

        {
          score: aligned ? momentum_score : momentum_score * 0.3,
          momentum_direction: momentum_direction,
          direction_aligned: aligned,
          risk_flag: momentum_data[:deceleration]&.dig(:overall) == true,
          data_quality: :high
        }
      end

      def score_liquidity(liquidity_data)
        return { score: 50, direction_aligned: true, risk_flag: true, data_quality: :low } unless liquidity_data

        liquidity_score = liquidity_data[:score] || liquidity_data[:liquidity_score] || 50
        tradable = liquidity_data[:tradable] != false

        {
          score: tradable ? liquidity_score : 0,
          tradable: tradable,
          spread_percent: liquidity_data.dig(:spread, :percent),
          direction_aligned: true,  # Liquidity is direction-agnostic
          risk_flag: !tradable || liquidity_score < 40,
          data_quality: :high
        }
      end

      def score_option_flow(option_data)
        return { score: 50, direction_aligned: false, risk_flag: true, data_quality: :low } unless option_data

        flow_score = option_data[:composite_score] || option_data[:score] || 50
        sentiment = option_data[:sentiment] || option_data[:oi_flow]&.dig(:sentiment)

        aligned = direction_aligned_with_flow?(sentiment)

        {
          score: aligned ? flow_score : flow_score * 0.3,
          sentiment: sentiment,
          direction_aligned: aligned,
          risk_flag: flow_score < 40,
          data_quality: :high
        }
      end

      def score_greeks(greeks_data)
        return { score: 50, direction_aligned: true, risk_flag: false, data_quality: :low } unless greeks_data

        # Ideal Greeks for naked options buying
        delta_score = greeks_data[:delta_score] || 50
        gamma_score = greeks_data[:gamma_score] || 50
        theta_score = greeks_data[:theta_score] || 50
        vega_score = greeks_data[:vega_score] || 50

        # For buyers: want positive delta, high gamma, low theta, positive vega
        composite = (delta_score + gamma_score + (100 - theta_score) + vega_score) / 4.0

        {
          score: composite.round(2),
          delta: greeks_data[:delta],
          gamma: greeks_data[:gamma],
          theta: greeks_data[:theta],
          direction_aligned: true,
          risk_flag: theta_score > 70,  # High theta = bad for buyers
          data_quality: :high
        }
      end

      def score_strike_quality(quality_data)
        return { score: 50, direction_aligned: true, risk_flag: false, data_quality: :low } unless quality_data

        quality_score = quality_data[:total_score] || quality_data[:score] || 50
        passed = quality_data[:passed] != false

        {
          score: passed ? quality_score : quality_score * 0.5,
          strike: quality_data[:strike],
          grade: quality_data[:grade],
          direction_aligned: true,
          risk_flag: !passed,
          data_quality: :high
        }
      end

      # Direction alignment helpers
      def direction_aligned_with_regime?(regime_type)
        return true unless regime_type

        case @direction
        when :long
          %i[trending_up bullish expansion].include?(regime_type)
        when :short
          %i[trending_down bearish expansion].include?(regime_type)
        else
          true
        end
      end

      def direction_aligned_with_structure?(structure_type)
        return true unless structure_type

        case @direction
        when :long
          %w[HL HH BOS].include?(structure_type.to_s)
        when :short
          %w[LH LL CHOCH].include?(structure_type.to_s)
        else
          true
        end
      end

      def direction_aligned_with_momentum?(momentum_direction)
        return true unless momentum_direction

        case @direction
        when :long
          momentum_direction.to_s.include?('bullish')
        when :short
          momentum_direction.to_s.include?('bearish')
        else
          true
        end
      end

      def direction_aligned_with_flow?(sentiment)
        return true unless sentiment

        case @direction
        when :long
          %i[bullish consolidating].include?(sentiment)
        when :short
          %i[bearish consolidating].include?(sentiment)
        else
          true
        end
      end
    end
  end
end
```

```ruby
module Engines
  module TradeScoring
    class ScoreThreshold
      DEFAULT = 80.0

      THRESHOLDS = {
        entry: 80.0,
        add_to_position: 85.0,
        reduce_position: 70.0,
        emergency_exit: 60.0,
        skip: 75.0
      }.freeze

      attr_reader :threshold, :action

      def initialize(threshold = DEFAULT, action: :entry)
        @threshold = threshold.to_f
        @action = action
      end

      def passed?(score)
        score >= @threshold
      end

      def margin(score)
        score - @threshold
      end

      def grade(score)
        case score
        when 0..50 then :reject
        when 50..60 then :weak
        when 60..70 then :marginal
        when 70..80 then :acceptable
        when 80..90 then :strong
        else :excellent
        end
      end

      def to_h
        {
          threshold: @threshold,
          action: @action,
          description: threshold_description
        }
      end

      private

      def threshold_description
        case @action
        when :entry then "Minimum score for new position entry"
        when :add_to_position then "Score required to add to existing position"
        when :reduce_position then "Score below which to reduce exposure"
        when :emergency_exit then "Immediate exit threshold"
        else "Standard entry threshold"
        end
      end
    end
  end
end
```

```ruby
module Engines
  module TradeScoring
    class ScoreValidator
      VALIDATION_RULES = [
        :minimum_total_score,
        :no_critical_failures,
        :direction_alignment_majority,
        :liquidity_minimum,
        :risk_reward_minimum,
        :time_of_day_valid,
        :position_limit_check
      ].freeze

      def initialize(score_result:, position: nil, portfolio: nil)
        @score_result = score_result
        @position = position
        @portfolio = portfolio
      end

      def call
        validations = VALIDATION_RULES.map do |rule|
          { rule: rule, passed: send(rule), message: send(:"#{rule}_message") }
        end

        all_passed = validations.all? { |v| v[:passed] }

        {
          valid: all_passed,
          validations: validations,
          failed_rules: validations.reject { |v| v[:passed] }.map { |v| v[:rule] }
        }
      end

      private

      def minimum_total_score
        @score_result[:total_score] >= TradeScoringEngine::DEFAULT_THRESHOLD
      end

      def minimum_total_score_message
        "Total score #{@score_result[:total_score]} must be >= #{TradeScoringEngine::DEFAULT_THRESHOLD}"
      end

      def no_critical_failures
        critical = %i[market_regime market_structure liquidity]
        critical.all? { |c| @score_result[:component_scores][c][:score] >= 30 }
      end

      def no_critical_failures_message
        "Critical components (regime, structure, liquidity) must score >= 30"
      end

      def direction_alignment_majority
        aligned = @score_result[:component_scores].count { |_, d| d[:direction_aligned] }
        aligned >= @score_result[:component_scores].length * 0.6  # 60% majority
      end

      def direction_alignment_majority_message
        "Majority of components must align with trade direction"
      end

      def liquidity_minimum
        @score_result[:component_scores][:liquidity][:score] >= 40
      end

      def liquidity_minimum_message
        "Liquidity score must be >= 40"
      end

      def risk_reward_minimum
        # Check if metadata contains R/R
        rr = @score_result.dig(:breakdown, :risk_reward) || 2.0
        rr >= 1.5
      end

      def risk_reward_minimum_message
        "Risk/Reward ratio must be >= 1.5:1"
      end

      def time_of_day_valid
        # No trades in first 5 minutes or last 15 minutes
        now = Time.current
        market_open = ExchangeCalendar.market_hours(exchange: "NSE")[:open]
        market_close = ExchangeCalendar.market_hours(exchange: "NSE")[:close]

        return true unless market_open && market_close

        open_time = Time.zone.parse(market_open.to_s)
        close_time = Time.zone.parse(market_close.to_s)

        now > open_time + 5.minutes && now < close_time - 15.minutes
      end

      def time_of_day_valid_message
        "Trading restricted to 09:20-15:15"
      end

      def position_limit_check
        return true unless @portfolio
        @portfolio[:open_positions].to_i < AppConfig.config.risk.max_open_positions
      end

      def position_limit_check_message
        "Maximum open positions reached"
      end
    end
  end
end
```

```ruby
module Engines
  module TradeScoring
    class ScoreHistory
      def initialize(instrument:)
        @instrument = instrument
        @key = "score_history:#{instrument.id}"
      end

      def record(score_result)
        entry = {
          timestamp: Time.current,
          total_score: score_result[:total_score],
          passed: score_result[:passed],
          component_scores: score_result[:component_scores].transform_values { |d| d[:score] },
          direction: score_result[:direction]
        }

        history = fetch_history
        history << entry
        history = history.last(100)  # Keep last 100

        CacheStore.write(@key, history, ttl: 1.day)
      end

      def average_score(period: 1.hour)
        history = fetch_history
        cutoff = Time.current - period
        recent = history.select { |h| h[:timestamp] > cutoff }

        return 0 if recent.empty?
        recent.sum { |h| h[:total_score] } / recent.length.to_f
      end

      def pass_rate(period: 1.day)
        history = fetch_history
        cutoff = Time.current - period
        recent = history.select { |h| h[:timestamp] > cutoff }

        return 0 if recent.empty?
        (recent.count { |h| h[:passed] }.to_f / recent.length * 100).round(2)
      end

      def trend
        history = fetch_history
        return :flat if history.length < 10

        first_half = history.first(history.length / 2).sum { |h| h[:total_score] } / (history.length / 2.0)
        second_half = history.last(history.length / 2).sum { |h| h[:total_score] } / (history.length / 2.0)

        return :improving if second_half > first_half * 1.1
        return :declining if second_half < first_half * 0.9
        :stable
      end

      private

      def fetch_history
        CacheStore.fetch(@key) || []
      end
    end
  end
end
```

```ruby
# app/engines/trade_scoring/score_explanation.rb
# frozen_string_literal: true

module Engines
  module TradeScoring
    class ScoreExplanation
      def initialize(score_result)
        @result = score_result
      end

      def generate
        sections = [
          executive_summary,
          component_analysis,
          risk_assessment,
          recommendation,
          ai_prompt_context
        ]

        sections.join("\n\n")
      end

      def to_h
        {
          summary: executive_summary,
          components: component_analysis_h,
          risks: risk_assessment_h,
          recommendation: recommendation,
          ai_context: ai_prompt_context
        }
      end

      private

      def executive_summary
        score = @result[:total_score]
        threshold = @result[:threshold]
        passed = @result[:passed]

        status = passed ? "✓ PASSED" : "✗ FAILED"
        margin = (score - threshold).round(2)

        <<~SUMMARY
          TRADE SCORE: #{score}/100 | Threshold: #{threshold} | #{status}
          Margin: #{margin > 0 ? "+#{margin}" : margin} points
          Confidence: #{@result[:confidence]}%
          Recommendation: #{@result[:recommendation]}
        SUMMARY
      end

      def component_analysis
        components = @result[:component_scores]

        lines = ["COMPONENT BREAKDOWN:"]
        components.each do |name, data|
          score = data[:score]
          aligned = data[:direction_aligned] ? "✓" : "✗"
          flag = data[:risk_flag] ? "⚠ RISK" : ""

          bar = score_bar(score)
          lines << "  #{aligned} #{name.to_s.upcase}: #{score}/100 #{bar} #{flag}"
        end

        lines.join("\n")
      end

      def component_analysis_h
        @result[:component_scores].transform_values do |data|
          {
            score: data[:score],
            aligned: data[:direction_aligned],
            risk_flag: data[:risk_flag]
          }
        end
      end

      def risk_assessment
        risks = @result[:component_scores].select { |_, d| d[:risk_flag] }

        lines = ["RISK ASSESSMENT:"]
        if risks.any?
          risks.each do |name, data|
            lines << "  ⚠ #{name}: #{risk_description(name, data)}"
          end
        else
          lines << "  ✓ No risk flags detected"
        end

        lines.join("\n")
      end

      def risk_assessment_h
        @result[:component_scores].select { |_, d| d[:risk_flag] }.transform_values do |data|
          { risk_type: data[:risk_flag], description: risk_description(nil, data) }
        end

      def risk_description(name, data)
        case name
        when :market_context then "Trading conditions unfavorable"
        when :market_regime then "Regime direction opposes trade"
        when :market_structure then "Structure signals against position"
        when :momentum then "Momentum decelerating"
        when :liquidity then "Insufficient liquidity for execution"
        when :option_flow then "OI flow contradicts direction"
        when :greeks then "Theta decay risk elevated"
        when :strike_quality then "Selected strike below quality threshold"
        else "General risk condition"
        end
      end

      def recommendation
        rec = @result[:recommendation]
        case rec
        when :immediate_entry_full_size
          "EXECUTE: Full position size immediately"
        when :favorable_entry_standard_size
          "EXECUTE: Standard position size"
        when :conditional_entry_reduced_size
          "EXECUTE: Reduced size (50%), monitor closely"
        when :observe_wait
          "HOLD: Wait for score improvement"
        else
          "REJECT: Do not execute"
        end
      end

      def ai_prompt_context
        <<~AI_CONTEXT
          AI ANALYSIS CONTEXT:
          - Trade Direction: #{@result[:direction]}
          - Composite Score: #{@result[:total_score]}/100
          - Dominant Factor: #{dominant_factor}

          - Structure: #{@result[:component_scores][:market_structure][:structure_type]}
          - Sentiment: #{@result[:component_scores][:option_flow][:sentiment]}

      end

      def score_bar(score)
        filled = (score / 5).round  # 20 chars = 100
        empty = 20 - filled
        "█" * filled + "░" * empty
      end

      def dominant_factor
        @result[:component_scores].max_by { |_, d| d[:score] }.first
      end

      def weakest_factor
        @result[:component_scores].min_by { |_, d| d[:score] }.first
```

```ruby
module Engines
  module TradeScoring
    class ScoreConfidence
      DATA_QUALITY_MULTIPLIERS = {
        high: 1.0,
        medium: 0.85,
        low: 0.70,
        stale: 0.50
      }.freeze

      def initialize(score_result, data_freshness: {})
        @result = score_result
        @freshness = data_freshness
      end

      def apply
        base_confidence = @result[:confidence]

        # Adjust for data freshness
        freshness_penalty = calculate_freshness_penalty

        # Adjust for component agreement
        agreement_bonus = calculate_agreement_bonus

        # Adjust for historical performance
        historical_adjustment = calculate_historical_adjustment

        adjusted = base_confidence * freshness_penalty + agreement_bonus + historical_adjustment
        [[adjusted, 100].min, 0].max.round(2)
      end

      def data_quality_report
        @freshness.transform_values do |timestamp|
          age = Time.current - timestamp
          case age
          when 0..5 then :high
          when 5..30 then :medium
          when 30..300 then :low
          else :stale
          end
        end
      end

      private

      def calculate_freshness_penalty
        qualities = data_quality_report.values
        return 1.0 if qualities.empty?

        avg_quality = qualities.sum { |q| DATA_QUALITY_MULTIPLIERS[q] } / qualities.length
        avg_quality
      end

      def calculate_agreement_bonus
        scores = @result[:component_scores].values.map { |d| d[:score] }
        return 0 if scores.length < 2

        mean = scores.sum.to_f / scores.length
        variance = scores.sum { |s| (s - mean) ** 2 } / scores.length

        # Low variance = high agreement = bonus
        if variance < 100
          10
        elsif variance < 400
          5
        else
          0
        end
      end

      def calculate_historical_adjustment
        # Would query ScoreHistory for this instrument
        0
      end
    end
  end
```
