### Day 1 — Strike Selection Engine Foundation & Scorer Interface

**Task 1.1: Create `StrikeSelectionEngine`**

```ruby
# app/engines/strike_selection/strike_selection_engine.rb
# frozen_string_literal: true

module Engines
  module StrikeSelection
    class StrikeSelectionEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[
        option_chain
        spot_price
        underlying
      ].freeze

      DEFAULT_RANGE = 3  # ATM ± 3 strikes
      MIN_LIQUIDITY_SCORE = 40  # Reject strikes below this

      def initialize(underlying:)
        @underlying = underlying
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        spot_price = context[:spot_price].to_d
        option_chain = context[:option_chain]
        direction = context[:direction] || :long  # long = CE for bullish, PE for bearish

        # Step 1: Identify ATM and candidate strikes
        atm_strike = calculate_atm_strike(spot_price)
        candidate_strikes = generate_candidate_strikes(atm_strike, spot_price)

        # Step 2: Fetch option data for candidates
        strikes_data = fetch_strikes_data(candidate_strikes, option_chain, direction)

        # Step 3: Score each strike
        scored_strikes = score_strikes(strikes_data, context)

        # Step 4: Select optimal with fallback
        selection = select_optimal(scored_strikes)

        if selection
          emit(EventTypes::STRIKE_SELECTED, {
            underlying_id: @underlying.id,
            selected_strike: selection[:strike],
            score: selection[:total_score],
            direction: direction
          })
        end

        success(
          selected_strike: selection,
          all_scores: scored_strikes.sort_by { |s| -s[:total_score] },
          atm_strike: atm_strike,
          direction: direction
        )
      end

      private

      def calculate_atm_strike(spot_price)
        step = strike_step
        (spot_price / step).round * step
      end

      def generate_candidate_strikes(atm_strike, spot_price)
        step = strike_step
        range = DEFAULT_RANGE

        (-range..range).map do |offset|
          atm_strike + (offset * step)
        end.sort
      end

      def strike_step
        case @underlying.trading_symbol
        when /NIFTY/ then 50
        when /BANKNIFTY/ then 100
        when /SENSEX/ then 100
        when /FINNIFTY/ then 50
        else 50
        end
      end

      def fetch_strikes_data(strikes, option_chain, direction)
        option_type = direction == :long ? "CE" : "PE"

        strikes.filter_map do |strike|
          strike_data = option_chain.find { |s| s[:strike].to_d == strike }
          next unless strike_data

          option = strike_data[option_type.downcase.to_sym]
          next unless option && option[:ltp].to_d > 0

          {
            strike: strike,
            option_type: option_type,
            ltp: option[:ltp].to_d,
            bid: option[:bid].to_d,
            ask: option[:ask].to_d,
            spread: option[:spread].to_d,
            volume: option[:volume].to_i,
            oi: option[:oi].to_i,
            oi_change: option[:oi_change].to_i,
            iv: option[:iv].to_d,
            delta: option[:delta].to_d,
            gamma: option[:gamma].to_d,
            theta: option[:theta].to_d,
            vega: option[:vega].to_d
          }
        end
      end

      def score_strikes(strikes_data, context)
        strikes_data.map do |strike_data|
          scores = {
            delta: DeltaScorer.new(strike_data).call,
            liquidity: LiquidityScorer.new(strike_data).call,
            gamma: GammaScorer.new(strike_data).call,
            oi: OIScorer.new(strike_data).call,
            iv: IVScorer.new(strike_data, context[:iv_history]).call,
            volume: VolumeScorer.new(strike_data).call,
            theta: ThetaScorer.new(strike_data).call,
            spread: SpreadScorer.new(strike_data).call
          }

          total = calculate_weighted_total(scores)

          strike_data.merge(
            component_scores: scores,
            total_score: total,
            passed: total >= 60 && scores[:liquidity][:score] >= MIN_LIQUIDITY_SCORE
          )
        end
      end

      def calculate_weighted_total(scores)
        weights = {
          delta: 0.20,
          liquidity: 0.20,
          gamma: 0.15,
          oi: 0.15,
          iv: 0.10,
          volume: 0.10,
          theta: 0.05,
          spread: 0.05
        }

        scores.sum { |component, data| data[:score] * weights[component] }.round(2)
      end

      def select_optimal(scored_strikes)
        # Filter tradable strikes
        tradable = scored_strikes.select { |s| s[:passed] }

        if tradable.any?
          # Return highest scoring tradable strike
          tradable.max_by { |s| s[:total_score] }
        else
          # Fallback: return highest scoring regardless, with warning
          best = scored_strikes.max_by { |s| s[:total_score] }
          best&.merge(fallback: true, warning: "Liquidity below threshold")
        end
      end
    end
  end
end
```

```ruby
module Engines
  module StrikeSelection
    class StrikeQualityScore
      GRADE_THRESHOLDS = {
        excellent: 85,
        good: 70,
        acceptable: 60,
        poor: 40,
        reject: 0
      }.freeze

      attr_reader :total, :components, :grade, :recommendation

      def initialize(total:, components:)
        @total = total
        @components = components
        @grade = calculate_grade
        @recommendation = calculate_recommendation
      end

      def excellent?
        @grade == :excellent
      end

      def acceptable?
        %i[excellent good acceptable].include?(@grade)
      end

      def reject?
        %i[poor reject].include?(@grade)
      end

      def to_h
        {
          total: @total,
          grade: @grade,
          recommendation: @recommendation,
          components: @components.transform_values { |v| v[:score] }
        }
      end

      private

      def calculate_grade
        GRADE_THRESHOLDS.find { |_, threshold| @total >= threshold }&.first || :reject
      end

      def calculate_recommendation
        case @grade
        when :excellent then "Immediate entry - ideal strike"
        when :good then "Favorable conditions - proceed with standard sizing"
        when :acceptable then "Marginal - reduce position size by 25%"
        when :poor then "Avoid - wait for better strike or skip trade"
        else "Reject - do not trade"
        end
      end
    end
  end
end
```

```ruby
module Engines
  module StrikeSelection
    class DeltaScorer
      IDEAL_DELTA_LONG = 0.50..0.65   # For long options (buying)
      IDEAL_DELTA_SHORT = 0.35..0.50  # For short options (selling)
      MIN_DELTA = 0.20
      MAX_DELTA = 0.80

      def initialize(strike_data, direction: :long)
        @strike_data = strike_data
        @direction = direction
        @delta = strike_data[:delta].to_d.abs
      end

      def call
        {
          score: calculate_score,
          delta: @delta,
          ideal_range: ideal_range,
          distance_from_ideal: distance_from_ideal,
          assessment: assessment
        }
      end

      private

      def calculate_score
        return 0 if @delta < MIN_DELTA || @delta > MAX_DELTA

        ideal = ideal_range
        center = (ideal.begin + ideal.end) / 2.0

        if ideal.cover?(@delta)
          # Inside ideal range: perfect score at center, decreasing toward edges
          100 - ((@delta - center).abs / (ideal.end - center) * 20).round(2)
        else
          # Outside ideal range: linear penalty
          distance = [@delta - ideal.begin, @delta - ideal.end].min_by(&:abs)
          [100 - (distance.abs * 200), 0].max
        end
      end

      def ideal_range
        @direction == :long ? IDEAL_DELTA_LONG : IDEAL_DELTA_SHORT
      end

      def distance_from_ideal
        ideal = ideal_range
        center = (ideal.begin + ideal.end) / 2.0
        (@delta - center).abs.round(4)
      end

      def assessment
        case calculate_score
        when 90..100 then :ideal
        when 70..90 then :favorable
        when 50..70 then :marginal
        when 0..50 then :unfavorable
        else :invalid
        end
      end
    end
  end
end
```

```ruby
module Engines
  module StrikeSelection
    class LiquidityScorer
      MIN_SPREAD_PERCENT = 0.30  # 0.30% max spread
      MIN_VOLUME = 100
      MIN_OI = 10_000

      def initialize(strike_data)
        @strike_data = strike_data
      end

      def call
        {
          score: calculate_score,
          spread_percent: spread_percent,
          spread_score: spread_score,
          volume_score: volume_score,
          oi_score: oi_score,
          depth_score: depth_score,
          assessment: assessment
        }
      end

      private

      def calculate_score
        weights = {
          spread: 0.35,
          volume: 0.25,
          oi: 0.25,
          depth: 0.15
        }

        (spread_score * weights[:spread] +
         volume_score * weights[:volume] +
         oi_score * weights[:oi] +
         depth_score * weights[:depth]).round(2)
      end

      def spread_percent
        ltp = @strike_data[:ltp].to_d
        spread = @strike_data[:spread].to_d
        return 0 if ltp.zero?
        (spread / ltp * 100).round(4)
      end

      def spread_score
        sp = spread_percent
        return 100 if sp <= 0.10
        return 90 if sp <= 0.20
        return 75 if sp <= MIN_SPREAD_PERCENT
        return 50 if sp <= 0.50
        return 25 if sp <= 1.0
        0
      end

      def volume_score
        vol = @strike_data[:volume].to_i
        return 100 if vol >= 10_000
        return 80 if vol >= 5_000
        return 60 if vol >= 2_000
        return 40 if vol >= MIN_VOLUME
        return 20 if vol >= 50
        0
      end

      def oi_score
        oi = @strike_data[:oi].to_i
        return 100 if oi >= 100_000
        return 85 if oi >= 50_000
        return 70 if oi >= 20_000
        return 50 if oi >= MIN_OI
        return 30 if oi >= 5_000
        0
      end

      def depth_score
        # Placeholder: would require market depth data
        50
      end

      def assessment
        case calculate_score
        when 80..100 then :excellent
        when 60..80 then :good
        when 40..60 then :adequate
```

```ruby
module Engines
  module StrikeSelection
    class GammaScorer
      IDEAL_GAMMA = 0.003..0.008  # Responsive but not excessive
      MAX_GAMMA = 0.020

      def initialize(strike_data)
        @strike_data = strike_data
        @gamma = strike_data[:gamma].to_d.abs
      end

      def call
        {
          score: calculate_score,
          gamma: @gamma,
          responsiveness: responsiveness,
          pin_risk: pin_risk,
          assessment: assessment
        }
      end

      private

      def calculate_score
        return 0 if @gamma > MAX_GAMMA || @gamma.zero?

        if IDEAL_GAMMA.cover?(@gamma)
          # Ideal range: higher gamma = more responsive
          70 + ((@gamma - IDEAL_GAMMA.begin) / (IDEAL_GAMMA.end - IDEAL_GAMMA.begin) * 30).round(2)
        elsif @gamma < IDEAL_GAMMA.begin
          # Too low: less responsive
          (@gamma / IDEAL_GAMMA.begin * 70).round(2)
        else
          # Above ideal but below max: penalty for pin risk
          [100 - ((@gamma - IDEAL_GAMMA.end) / (MAX_GAMMA - IDEAL_GAMMA.end) * 50), 50].max
        end
      end

      def responsiveness
        case @gamma
        when 0..0.001 then :low
        when 0.001..0.003 then :moderate
        when 0.003..0.008 then :high
        when 0.008..0.015 then :very_high
        else :extreme
        end
      end

      def pin_risk
        @gamma > 0.015 ? :high : @gamma > 0.008 ? :moderate : :low
      end

      def assessment
        case calculate_score
        when 80..100 then :ideal
        when 60..80 then :favorable
        when 40..60 then :acceptable
        else :unfavorable
        end
      end
    end
  end
end
```

```ruby
module Engines
  module StrikeSelection
    class OIScorer
      MIN_OI = 10_000
      IDEAL_OI_CHANGE_RATIO = 0.05..0.30  # 5-30% OI change

      def initialize(strike_data)
        @strike_data = strike_data
      end

      def call
        {
          score: calculate_score,
          oi: @strike_data[:oi].to_i,
          oi_change: @strike_data[:oi_change].to_i,
          oi_change_ratio: oi_change_ratio,
          flow_classification: flow_classification,
          assessment: assessment
        }
      end

      private

      def calculate_score
        oi = @strike_data[:oi].to_i
        return 0 if oi < MIN_OI

        base_score = case oi
                     when 0..10_000 then 0
                     when 10_000..30_000 then 40
                     when 30_000..50_000 then 60
                     when 50_000..100_000 then 80
                     else 90
                     end

        # Bonus for OI build-up (institutional interest)
        build_bonus = case oi_change_ratio
                      when 0.05..0.15 then 10
                      when 0.15..0.30 then 15
                      when 0.30..1.0 then 5  # Too aggressive, possible trap
                      else 0
                      end

        [base_score + build_bonus, 100].min
      end

      def oi_change_ratio
        oi = @strike_data[:oi].to_i
        return 0 if oi.zero?
        (@strike_data[:oi_change].to_f / oi).abs.round(4)
      end

      def flow_classification
        change = @strike_data[:oi_change].to_i
        price_change = @strike_data[:ltp].to_d - @strike_data[:bid].to_d  # Simplified

        if change > 0 && price_change > 0
          :long_buildup
        elsif change > 0 && price_change < 0
          :short_buildup
        elsif change < 0 && price_change < 0
          :long_unwinding
        elsif change < 0 && price_change > 0
          :short_covering
        else
          :neutral
        end
      end

      def assessment
        case calculate_score
        when 80..100 then :strong_institutional_interest
        when 60..80 then :moderate_interest
        when 40..60 then :limited_interest
        else :weak_interest
        end
      end
    end
```

```ruby
# app/engines/strike_selection/iv_scorer.rb
# frozen_string_literal: true

module Engines
  module StrikeSelection
    class IVScorer
      IDEAL_IV_RANK = 30..70  # Middle of range
      MIN_IV = 10
      MAX_IV = 60

      def initialize(strike_data, iv_history = [])
        @strike_data = strike_data
        @iv_history = iv_history
        @iv = strike_data[:iv].to_d
      end

      def call
        {
          score: calculate_score,
          iv: @iv,
          iv_rank: iv_rank,
          iv_percentile: iv_percentile,
          iv_trend: iv_trend,
          assessment: assessment
        }
      end

      private

      def calculate_score
        return 0 if @iv < MIN_IV || @iv > MAX_IV

        rank = iv_rank
        trend = iv_trend

        # Base score from IV rank
        base = if IDEAL_IV_RANK.cover?(rank)
                 80 + (20 - (rank - 50).abs * 0.5)
               elsif rank < 30
                 rank * 2.5  # Low IV = lower score
               else
                 (100 - rank) * 2.5  # High IV = lower score
               end

        # Trend adjustment
        trend_bonus = case trend
                      when :rising then 5   # IV rising, good for buyers
                      when :falling then -5  # IV falling, bad for buyers
                      else 0
                      end

        [[base + trend_bonus, 100].min, 0].max.round(2)
      end

      def iv_rank
        return 50 if @iv_history.empty?
        min = @iv_history.min
        max = @iv_history.max
        return 50 if max == min
        ((@iv - min) / (max - min) * 100).round(2)
      end

      def iv_percentile
        return 50 if @iv_history.empty?
        below = @iv_history.count { |h| h < @iv }
        (below.to_f / @iv_history.length * 100).round(2)
      end

      def iv_trend
        return :unknown if @iv_history.length < 5
        recent = @iv_history.last(5)
        recent.last > recent.first ? :rising : :falling
      end

      def assessment
        case iv_rank
        when 0..20 then :low_iv_favorable_for_buyers
        when 20..40 then :below_average
        when 40..60 then :moderate_ideal
        when 60..80 then :above_average
        else :high_iv_caution
        end
      end
    end
  end
end
```

```ruby
      def initialize(strike_data, days_to_expiry: nil)
        @strike_data = strike_data

          score: calculate_score,
          theta: @theta,
          daily_decay_percent: daily_decay_percent,
          time_risk: time_risk,
          assessment: assessment
        }

      private

      def calculate_score
        return 100 if @theta.zero? || @ltp.zero?

        decay = daily_decay_percent
        return 0 if decay > MAX_DAILY_THETA_PERCENT * 2

        if decay <= 1.0
          90 + (10 - decay * 10)
        elsif decay <= 2.0
          70 + (20 - (decay - 1) * 20)
        elsif decay <= MAX_DAILY_THETA_PERCENT
          50 - ((decay - 2) / (MAX_DAILY_THETA_PERCENT - 2) * 50)
        else
          0
        end
      end

      def daily_decay_percent
        return 0 if @ltp.zero?
        (@theta / @ltp * 100).round(4)
      end

      def time_risk
        case daily_decay_percent
        when 0..1.0 then :low
        when 1.0..2.5 then :moderate
        when 2.5..5.0 then :high
        else :extreme
        end
      end

      def assessment
        case calculate_score
        when 80..100 then :minimal_time_decay
        when 60..80 then :acceptable_decay
        when 40..60 then :noticeable_decay
        else :severe_decay
        end
      end
```

**Task 4.1: Create `StrikeSelector` with Fallback Logic**

```ruby
# app/engines/strike_selection/strike_selector.rb
# frozen_string_literal: true

module Engines
  module StrikeSelection
    class StrikeSelector
      FALLBACK_STRATEGIES = [
        :nearest_atm,
        :wider_range,
        :adjacent_expiry,
        :skip_trade
      ].freeze

      def initialize(scored_strikes:, min_score: 60, min_liquidity: 40)
        @scored_strikes = scored_strikes
        @min_score = min_score
        @min_liquidity = min_liquidity
      end

      def call
        # Primary: Best tradable strike
        selection = select_best_tradable
        return selection if selection

        # Fallback 1: Nearest ATM with any score
        selection = select_nearest_atm
        return selection_with_fallback(selection, :nearest_atm) if selection

        # Fallback 2: Wider range (ATM ± 5)
        selection = select_wider_range
        return selection_with_fallback(selection, :wider_range) if selection

        # Fallback 3: Adjacent expiry series
        selection = select_adjacent_expiry
        return selection_with_fallback(selection, :adjacent_expiry) if selection

        # Final: Skip trade
        {
          strike: nil,
          action: :skip_trade,
          reason: "No suitable strike found after all fallbacks",
          best_attempt: @scored_strikes.max_by { |s| s[:total_score] }
        }
      end

      private

      def select_best_tradable
        tradable = @scored_strikes.select do |s|
          s[:passed] == true &&
            s[:total_score] >= @min_score &&
```

```ruby
class StrikeSelectionJob < JobBase
  def perform(underlying_id, direction, spot_price)
    underlying = Instrument.find(underlying_id)
    return unless underlying&.index?

    # Fetch latest option chain
    chain_result = OptionChainService.new(underlying: underlying).call
    return if chain_result.failure?

    # Fetch IV history for scoring
    iv_history = OptionChainRepository.iv_history(
      underlying: underlying,
      strike: calculate_atm_strike(underlying, spot_price),
      option_type: direction == :long ? "CE" : "PE",
      from: 6.months.ago,
      to: Time.current
    ).value&.map(&:last) || []

    result = Engines::StrikeSelection::StrikeSelectionEngine.new(underlying: underlying).call(
      context: {
        spot_price: spot_price,
        option_chain: chain_result.value,
        direction: direction,
        iv_history: iv_history
      }
    )

    return if result.failure?

    selection = result.value[:selected_strike]

    # Cache selection for trade execution
    CacheStore.write(
      "strike_selection:#{underlying.id}:#{direction}",
      selection,
      ttl: 60
    )

    EventBus.publish(EventTypes::STRIKE_SELECTED, {
      underlying: underlying.trading_symbol,
      selected_strike: selection[:strike],
      score: selection[:total_score],
      fallback: selection[:fallback]
    })
  end

  private

  def calculate_atm_strike(underlying, spot_price)
    step = case underlying.trading_symbol
           when /NIFTY/ then 50
           when /BANKNIFTY/ then 100
           else 50
           end
    (spot_price / step).round * step
  end
end
```

```ruby
module StrikeSelectionCache
  CACHE_TTL = 30  # seconds

  def self.key(underlying_id, direction, spot_price)
    # Round spot price to nearest strike step for cache stability
    rounded_price = (spot_price / 50).round * 50
    "strike_selection:#{underlying_id}:#{direction}:#{rounded_price}"
  end

  def self.fetch(underlying_id, direction, spot_price)
    CacheStore.fetch(key(underlying_id, direction, spot_price))
  end

  def self.store(underlying_id, direction, spot_price, result)
    CacheStore.write(
      key(underlying_id, direction, spot_price),
      result,
      ttl: CACHE_TTL
    )
  end

  def self.invalidate(underlying_id)
    # Invalidate all cached selections for underlying
    keys = CacheStore.keys("strike_selection:#{underlying_id}:*")
    keys.each { |k| CacheStore.delete(k) }
  end
end
```

```ruby
module Engines
  module StrikeSelection
    class StrikeSelectionOptimizer
      def initialize(learning_records:)
        @learning_records = learning_records
      end

      def optimize_weights
        return default_weights if @learning_records.empty?

        # Analyze which components correlated with winning trades
        winning_scores = @learning_records.select(&:win?).map(&:component_scores)
        losing_scores = @learning_records.reject(&:win?).map(&:component_scores)

        return default_weights if winning_scores.empty?

        optimized = {}

        %i[delta liquidity gamma oi iv volume theta spread].each do |component|
          win_avg = winning_scores.map { |s| s[component][:score] }.sum.to_f / winning_scores.length
          loss_avg = losing_scores.map { |s| s[component][:score] }.sum.to_f / losing_scores.length

          # Increase weight for components that differentiate wins from losses
          differentiation = (win_avg - loss_avg).abs
          optimized[component] = [differentiation * 2, 0.30].min  # Cap at 30%
        end

        # Normalize to sum to 1.0
        total = optimized.values.sum
        optimized.transform_values { |v| (v / total).round(4) }
      end

      def default_weights
        {
          delta: 0.20,
          liquidity: 0.20,
          gamma: 0.15,
          oi: 0.15,
          iv: 0.10,
          volume: 0.10,
          theta: 0.05,
          spread: 0.05
        }
      end
    end
  end
end
```
