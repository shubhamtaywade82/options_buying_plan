```ruby
# app/engines/position_management/position_management_engine.rb
# frozen_string_literal: true

module Engines
  module PositionManagement
    class PositionManagementEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[position market_data].freeze

      MONITORING_INTERVAL = 5  # seconds between checks

      def initialize(position:)
        @position = position
        @instrument = position.instrument
        @user = position.user
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        market_data = context[:market_data]

        # Update position metrics
        update_position_metrics(market_data)

        # Check exit conditions
        exit_check = evaluate_exit_conditions(market_data)
        return exit_check if exit_check && exit_check[:action] != :hold

        # Check stop-loss adjustments
        adjust_stop_loss(market_data) if trailing_stop_enabled?

        # Check partial exit opportunities
        partial_exit = evaluate_partial_exit(market_data)

        # Monitor Greeks changes
        greeks_check = monitor_greeks(market_data)

        # Emit monitoring event
        emit(EventTypes::POSITION_MONITORED, {
          position_id: @position.id,
          unrealized_pnl: @position.unrealized_pnl,
          mfe: @position.mfe,
          mae: @position.mae,
          exit_signal: exit_check&.dig(:action)
        })

        success(
          position: @position,
          metrics: position_metrics,
          exit_recommendation: exit_check,
          partial_exit: partial_exit,
          greeks_alert: greeks_check,
          holding_time: holding_time
        )
      end

      private

      def update_position_metrics(market_data)
        current_price = market_data[:last_price].to_d
        entry = @position.entry_price.to_d
        quantity = @position.quantity.to_i

        # Calculate P&L
        if @position.long?
          pnl = (current_price - entry) * quantity
        else
          pnl = (entry - current_price) * quantity
        end

        @position.update!(
          current_price: current_price,
          unrealized_pnl: pnl,
          mfe: [@position.mfe.to_d, pnl].max,
          mae: [@position.mae.to_d, pnl].min
        )
      end

      def evaluate_exit_conditions(market_data)
        current_price = market_data[:last_price].to_d

        # Stop loss hit
        if stop_loss_hit?(current_price)
          return {
            action: :exit_stop_loss,
            reason: :stop_loss_triggered,
            price: current_price,
            urgency: :immediate
          }
        end

        # Target hit
        if target_hit?(current_price)
          return {
            action: :exit_target,
            reason: :target_achieved,
            price: current_price,
            urgency: :immediate
          }
        end

        # Trailing stop hit
        if trailing_stop_hit?(current_price)
          return {
            action: :exit_trailing_stop,
            reason: :trailing_stop_triggered,
            price: current_price,
            urgency: :immediate
          }
        end

        # Time decay exit
        if time_decay_exit_needed?
          return {
            action: :exit_time_decay,
            reason: :theta_acceleration,
            price: current_price,
            urgency: :gradual
          }
        end

        # Structure break
        if structure_broken?(market_data)
          return {
            action: :exit_structure_break,
            reason: :market_structure_changed,
            price: current_price,
            urgency: :high
          }
        end

        # Catastrophic move
        if catastrophic_move?(market_data)
          return {
            action: :exit_emergency,
            reason: :catastrophic_price_move,
            price: current_price,
            urgency: :immediate
          }
        end

        { action: :hold }
      end

      def stop_loss_hit?(current_price)
        return false unless @position.stop_loss.to_d > 0

        if @position.long?
          current_price <= @position.stop_loss.to_d
        else
          current_price >= @position.stop_loss.to_d
        end
      end

      def target_hit?(current_price)
        return false unless @position.take_profit.to_d > 0

        if @position.long?
          current_price >= @position.take_profit.to_d
        else
          current_price <= @position.take_profit.to_d
        end
      end

      def trailing_stop_hit?(current_price)
        return false unless @position.trailing_stop.to_d > 0

        if @position.long?
          current_price <= @position.trailing_stop.to_d
        else
          current_price >= @position.trailing_stop.to_d
        end
      end

      def time_decay_exit_needed?
        return false unless @position.instrument.option?

        # Check if theta decay exceeds expected move
        # Simplified: exit if holding time > 20 minutes for options
        holding_time > 20.minutes
      end

      def structure_broken?(market_data)
        return false unless market_data[:structure]

        structure = market_data[:structure]
        if @position.long?
          structure[:bearish] == true || %w[LH LL CHOCH].include?(structure[:type])
        else
          structure[:bullish] == true || %w[HH HL BOS].include?(structure[:type])
        end
      end

      def catastrophic_move?(market_data)
        return false unless @position.entry_price.to_d > 0

        move_percent = ((market_data[:last_price].to_d - @position.entry_price.to_d).abs / @position.entry_price.to_d * 100).round(2)

        # Catastrophic: >3% move against position in 1 minute
        if @position.long?
          market_data[:last_price].to_d < @position.entry_price.to_d && move_percent > 3.0
        else
          market_data[:last_price].to_d > @position.entry_price.to_d && move_percent > 3.0
        end
      end

      def adjust_stop_loss(market_data)
        return unless trailing_stop_enabled?

        current_price = market_data[:last_price].to_d
        atr = market_data[:atr]&.dig(:current).to_d

        return if atr <= 0

        # ATR-based trailing stop
        if @position.long?
          new_stop = current_price - (atr * 2)
          if new_stop > @position.trailing_stop.to_d
            @position.update!(trailing_stop: new_stop)
            log_stop_adjustment(:trailing_stop_moved, new_stop)
          end
        else
          new_stop = current_price + (atr * 2)
          if new_stop < @position.trailing_stop.to_d || @position.trailing_stop.to_d == 0
            @position.update!(trailing_stop: new_stop)
            log_stop_adjustment(:trailing_stop_moved, new_stop)
          end
        end
      end

      def trailing_stop_enabled?
        AppConfig.config.position_management.trailing_stop_enabled != false
      end

      def evaluate_partial_exit(market_data)
        return nil unless partial_exit_enabled?

        current_price = market_data[:last_price].to_d
        entry = @position.entry_price.to_d

        return nil if entry <= 0

        profit_percent = ((current_price - entry).abs / entry * 100).round(2)

        # Scale out at 1R, 2R, 3R
        scale_levels = AppConfig.config.position_management.partial_exit_levels || [1.0, 2.0, 3.0]

        scale_levels.each do |level|
          next if partial_exit_already_done?(level)

          if profit_percent >= level
            return {
              action: :partial_exit,
              level: level,
              percent_to_close: 25,  # Close 25% at each level
              price: current_price
            }
          end
        end

        nil
      end

      def partial_exit_enabled?
        AppConfig.config.position_management.partial_exits_enabled != false
      end

      def partial_exit_already_done?(level)
        @position.management_log.any? { |log| log["type"] == "partial_exit" && log["level"] == level }
      end

      def monitor_greeks(market_data)
        return nil unless @position.instrument.option?
        return nil unless market_data[:greeks]

        greeks = market_data[:greeks]
        alerts = []

        # Delta drift check
        if greeks[:delta].to_d.abs < 0.30
          alerts << {
            type: :delta_drift,
            message: "Delta decreased to #{greeks[:delta]}, position becoming less responsive",
            severity: :warning
          }
        end

        # Gamma acceleration
        if greeks[:gamma].to_d > 0.015
          alerts << {
            type: :gamma_pin_risk,
            message: "High gamma #{greeks[:gamma]}, pin risk near expiry",
            severity: :high
          }
        end

        # Theta acceleration
        if greeks[:theta].to_d.abs > @position.entry_price.to_d * 0.02
          alerts << {
            type: :theta_acceleration,
            message: "Theta decay accelerating, time erosion high",
            severity: :warning
          }
        end

        # IV crush
        if market_data[:iv_change].to_d < -5
          alerts << {
            type: :iv_crush,
            message: "IV dropping rapidly, option value eroding",
            severity: :high
          }
        end

        alerts.any? ? alerts : nil
      end

      def position_metrics
        {
          unrealized_pnl: @position.unrealized_pnl.to_d,
          pnl_percent: pnl_percent,
          mfe: @position.mfe.to_d,
          mae: @position.mae.to_d,
          holding_time: holding_time,
          current_price: @position.current_price.to_d,
          stop_loss: @position.stop_loss.to_d,
          trailing_stop: @position.trailing_stop.to_d,
          take_profit: @position.take_profit.to_d
        }
      end

      def pnl_percent
        return 0 if @position.entry_price.to_d.zero?

        if @position.long?
          ((@position.current_price.to_d - @position.entry_price.to_d) / @position.entry_price.to_d * 100).round(2)
        else
          ((@position.entry_price.to_d - @position.current_price.to_d) / @position.entry_price.to_d * 100).round(2)
        end
      end

      def holding_time
        return 0 unless @position.opened_at
        Time.current - @position.opened_at
      end

      def log_stop_adjustment(type, new_value)
        @position.update!(
          management_log: @position.management_log + [{
            type: type,
            from: @position.trailing_stop_was,
            to: new_value,
            at: Time.current.iso8601
          }]
        )
      end
    end
  end
end
```

```ruby
# app/engines/position_management/stop_loss_manager.rb
# frozen_string_literal: true

module Engines
  module PositionManagement
    class StopLossManager
      STOP_TYPES = %i[fixed atr_based percent_based structure_based].freeze

      def initialize(position:)
        @position = position
      end

      def calculate_initial_stop(entry_price:, stop_type: :fixed, **params)
        case stop_type
        when :fixed
          fixed_stop(entry_price, params[:stop_price])
        when :atr_based
          atr_stop(entry_price, params[:atr], params[:multiplier])
        when :percent_based
          percent_stop(entry_price, params[:percent])
        when :structure_based
          structure_stop(entry_price, params[:structure_level])
        else
          fixed_stop(entry_price, entry_price * 0.99)
        end
      end

      def update_stop(new_stop:, reason:)
        return false if new_stop <= 0

        # Validate: stop must be worse than entry for initial, can improve for trailing
        if @position.trailing_stop.to_d > 0
          # Trailing stop: can only move in favorable direction
          return false unless valid_trailing_movement?(new_stop)
        end

        old_stop = @position.stop_loss.to_d
        @position.update!(
          stop_loss: new_stop,
          management_log: @position.management_log + [{
            type: "stop_loss_updated",
            from: old_stop,
            to: new_stop,
            reason: reason,
            at: Time.current.iso8601
          }]
        )

        true
      end

      def breakeven_stop
        return false unless @position.unrealized_pnl.to_d > 0

        entry = @position.entry_price.to_d
        # Move stop to entry + small buffer
        buffer = entry * 0.001  # 0.1% buffer

        if @position.long?
          new_stop = entry + buffer
        else
          new_stop = entry - buffer
        end

        update_stop(new_stop: new_stop, reason: :move_to_breakeven)
      end

      private

      def fixed_stop(entry_price, stop_price)
        stop_price.to_d
      end

      def atr_stop(entry_price, atr, multiplier = 2)
        return entry_price * 0.99 if atr.to_d <= 0

        if @position.long?
          entry_price - (atr.to_d * multiplier)
        else
          entry_price + (atr.to_d * multiplier)
        end
      end

      def percent_stop(entry_price, percent = 1)
        if @position.long?
          entry_price * (1 - percent / 100.0)
        else
          entry_price * (1 + percent / 100.0)
        end
      end

      def structure_stop(entry_price, structure_level)
        structure_level.to_d
      end

      def valid_trailing_movement?(new_stop)
        if @position.long?
          new_stop >= @position.trailing_stop.to_d
        else
          new_stop <= @position.trailing_stop.to_d
        end
      end
    end
  end
end
```

```ruby
# app/engines/position_management/trailing_stop_manager.rb
# frozen_string_literal: true

module Engines
  module PositionManagement
    class TrailingStopManager
      TRAILING_METHODS = %i[atr_based percent_based step_based chandelier].freeze

      def initialize(position:)
        @position = position
      end

      def activate(method: :atr_based, **params)
        return false unless @position.status == "open"

        initial_stop = calculate_trailing_stop(method, params)
        return false unless initial_stop > 0

        @position.update!(
          trailing_stop: initial_stop,
          trailing_method: method.to_s,
          management_log: @position.management_log + [{
            type: "trailing_stop_activated",
            method: method,
            initial_value: initial_stop,
            at: Time.current.iso8601
          }]
        )

        true
      end

      def update(current_price:, **market_data)
        return false unless @position.trailing_stop.to_d > 0

        new_stop = calculate_new_trailing_stop(current_price, market_data)

        return false unless new_stop > 0
        return false unless valid_movement?(new_stop)

        old_stop = @position.trailing_stop.to_d
        return true if (new_stop - old_stop).abs < 0.01  # No significant change

        @position.update!(
          trailing_stop: new_stop,
          management_log: @position.management_log + [{
            type: "trailing_stop_updated",
            from: old_stop,
            to: new_stop,
            at: Time.current.iso8601
          }]
        )

        true
      end

      def deactivate
        @position.update!(
          trailing_stop: 0,
          management_log: @position.management_log + [{
            type: "trailing_stop_deactivated",
            at: Time.current.iso8601
          }]
        )
      end

      private

      def calculate_trailing_stop(method, params)
        case method
        when :atr_based
          atr_based_stop(params[:atr], params[:multiplier] || 2)
        when :percent_based
          percent_based_stop(params[:percent] || 1.5)
        when :step_based
          step_based_stop(params[:step] || 0.5)
        when :chandelier
          chandelier_stop(params[:high], params[:low], params[:period] || 22)
        else
          0
        end
      end

      def calculate_new_trailing_stop(current_price, market_data)
        case @position.trailing_method
        when "atr_based"
          atr_based_update(current_price, market_data[:atr])
        when "percent_based"
          percent_based_update(current_price)
        when "step_based"
          step_based_update(current_price)
        when "chandelier"
          chandelier_update(current_price, market_data[:high], market_data[:period])
        else
          0
        end
      end

      def atr_based_stop(atr, multiplier)
        return 0 if atr.to_d <= 0

        if @position.long?
          @position.entry_price.to_d - (atr.to_d * multiplier)
        else
          @position.entry_price.to_d + (atr.to_d * multiplier)
        end
      end

      def atr_based_update(current_price, atr)
        return @position.trailing_stop.to_d if atr.to_d <= 0

        if @position.long?
          [current_price - (atr.to_d * 2), @position.trailing_stop.to_d].max
        else
          [current_price + (atr.to_d * 2), @position.trailing_stop.to_d].min
        end
      end

      def percent_based_stop(percent)
        if @position.long?
          @position.entry_price.to_d * (1 - percent / 100.0)
        else
          @position.entry_price.to_d * (1 + percent / 100.0)
        end
      end

      def percent_based_update(current_price)
        percent = @position.trailing_params&.dig("percent") || 1.5

        if @position.long?
          current_price * (1 - percent / 100.0)
        else
          current_price * (1 + percent / 100.0)
        end
      end

      def step_based_stop(step)
        entry = @position.entry_price.to_d
        if @position.long?
          entry - step
        else
          entry + step
        end
      end

      def step_based_update(current_price)
        step = @position.trailing_params&.dig("step") || 0.5
        current_level = ((current_price - @position.entry_price.to_d) / step).floor

        if @position.long?
          @position.entry_price.to_d + (current_level * step) - step
        else
          @position.entry_price.to_d - (current_level * step) + step
        end
      end

      def chandelier_stop(high, low, period)
        # Simplified: use highest high - 3*ATR
        0
      end

      def chandelier_update(current_price, high, period)
        0
      end

      def valid_movement?(new_stop)
        if @position.long?
          new_stop >= @position.trailing_stop.to_d && new_stop < @position.current_price.to_d
        else
          new_stop <= @position.trailing_stop.to_d && new_stop > @position.current_price.to_d
        end
      end
    end
  end
end
```

### Day 3 — Partial Exit & Time Decay Monitors

**Task 3.1: Create `PartialExitManager`**

```ruby
# app/engines/position_management/partial_exit_manager.rb
# frozen_string_literal: true

module Engines
  module PositionManagement
    class PartialExitManager
      DEFAULT_SCALE_LEVELS = [
        { r_multiple: 1.0, percent: 25 },
        { r_multiple: 2.0, percent: 25 },
        { r_multiple: 3.0, percent: 25 }
      ].freeze

      def initialize(position:)
        @position = position
      end

      def evaluate(current_price:)
        return nil unless @position.status == "open"

        entry = @position.entry_price.to_d
        stop = @position.stop_loss.to_d
        current = current_price.to_d

        return nil if entry <= 0 || stop <= 0

        # Calculate R-multiple
        risk = (entry - stop).abs
        profit = @position.long? ? (current - entry) : (entry - current)
        r_multiple = risk.zero? ? 0 : (profit / risk).round(2)

        # Check if any scale level reached
        scale_levels = AppConfig.config.position_management.partial_exit_levels || DEFAULT_SCALE_LEVELS

        scale_levels.each do |level|
          next if already_scaled?(level[:r_multiple])

          if r_multiple >= level[:r_multiple]
            return {
              action: :partial_exit,
              r_multiple: r_multiple,
              scale_level: level[:r_multiple],
              percent_to_close: level[:percent],
              quantity_to_close: (@position.quantity * level[:percent] / 100.0).floor,
              reason: "R#{level[:r_multiple]} reached"
            }
          end
        end

        nil
      end

      def execute(scale_level:, quantity_to_close:)
        return Result.failure(:invalid_scale) unless scale_level > 0
        return Result.failure(:invalid_quantity) if quantity_to_close <= 0

        # Place closing order for partial quantity
        side = @position.long? ? "sell" : "buy"

        order_result = OrderService.new(user: @position.user).place(
          instrument: @position.instrument,
          side: side,
          quantity: quantity_to_close,
          order_type: "market",
          product_type: @position.order.product_type
        )

        return order_result if order_result.failure?

        order = order_result.value[:order]

        # Update position
        new_quantity = @position.quantity - quantity_to_close
        new_status = new_quantity <= 0 ? "closed" : "partial_exit"

        @position.update!(
          quantity: new_quantity,
          status: new_status,
          realized_pnl: @position.realized_pnl.to_d + calculate_realized_pnl(order),
          management_log: @position.management_log + [{
            type: "partial_exit",
            scale_level: scale_level,
            quantity_closed: quantity_to_close,
            remaining_quantity: new_quantity,
            exit_price: order.average_price,
            at: Time.current.iso8601
          }]
        )

        # Move stop to breakeven after first scale
        if first_scale?(scale_level)
          StopLossManager.new(position: @position).breakeven_stop
        end

        Result.success(
          closed_quantity: quantity_to_close,
          remaining_quantity: new_quantity,
          status: new_status
        )
      end

      private

      def already_scaled?(r_multiple)
        @position.management_log.any? { |log| log["type"] == "partial_exit" && log["scale_level"] == r_multiple }
      end

      def first_scale?(current_level)
        scales = @position.management_log.select { |log| log["type"] == "partial_exit" }
        scales.length == 1 && scales.first["scale_level"] == current_level
      end

      def calculate_realized_pnl(order)
        entry = @position.entry_price.to_d
        exit_price = order.average_price.to_d
        quantity = order.filled_quantity.to_i

        if @position.long?
          (exit_price - entry) * quantity
        else
          (entry - exit_price) * quantity
        end
      end
    end
  end
end
```

```ruby
# app/engines/position_management/time_decay_monitor.rb
# frozen_string_literal: true

module Engines
  module PositionManagement
    class TimeDecayMonitor
      THETA_EXIT_THRESHOLD = 2.0  # Exit if theta > 2% of position value per day

      def initialize(position:)
        @position = position
      end

      def evaluate(greeks_data:)
        return nil unless @position.instrument.option?
        return nil unless greeks_data

        theta = greeks_data[:theta].to_d.abs
        position_value = @position.current_price.to_d * @position.quantity.to_i

        return nil if position_value <= 0

        daily_decay_percent = (theta / position_value * 100).round(2)

        # Time to expiry
        days_to_expiry = calculate_days_to_expiry

        {
          theta: theta,
          daily_decay_percent: daily_decay_percent,
          days_to_expiry: days_to_expiry,
          decay_acceleration: decay_acceleration(days_to_expiry),
          recommendation: time_decay_recommendation(daily_decay_percent, days_to_expiry)
        }
      end

      def should_exit?(greeks_data:)
        evaluation = evaluate(greeks_data: greeks_data)
        return false unless evaluation

        evaluation[:recommendation] == :exit_immediately ||
          evaluation[:recommendation] == :exit_gradually
      end

      private

      def calculate_days_to_expiry
        return 0 unless @position.instrument.expiry
        (@position.instrument.expiry - Date.current).to_i
      end

      def decay_acceleration(days_to_expiry)
        # Theta accelerates as expiry approaches
        case days_to_expiry
        when 0..1 then :extreme    # 10x normal
        when 1..3 then :very_high  # 5x normal
        when 3..7 then :high       # 2x normal
        when 7..14 then :moderate
        else :normal
        end
      end

      def time_decay_recommendation(daily_decay, days_to_expiry)
        if daily_decay > THETA_EXIT_THRESHOLD * 2 || days_to_expiry <= 1
          :exit_immediately
        elsif daily_decay > THETA_EXIT_THRESHOLD || days_to_expiry <= 3
          :exit_gradually
        elsif daily_decay > THETA_EXIT_THRESHOLD * 0.5
          :reduce_size
        else
          :hold
        end
      end
    end
  end
end
```

```ruby
# app/engines/position_management/emergency_exit_manager.rb
# frozen_string_literal: true

module Engines
  module PositionManagement
    class EmergencyExitManager
      EMERGENCY_TYPES = %i[
        catastrophic_move
        broker_disconnect
        system_failure
        margin_call
        manual_trigger
        circuit_breaker
      ].freeze

      def initialize(position:)
        @position = position
      end

      def trigger(emergency_type:, reason:)
        return Result.failure(:already_closed) if @position.status == "closed"

        unless EMERGENCY_TYPES.include?(emergency_type)
          return Result.failure(:invalid_emergency_type)
        end

        # Log emergency
        log_emergency(emergency_type, reason)

        # Attempt immediate market exit
        exit_result = execute_market_exit

        if exit_result.success?
          close_position(exit_result.value, emergency_type, reason)
          notify_emergency_resolved(emergency_type, exit_result.value)
        else
          # If market exit fails, try limit order at current price
          fallback_result = execute_fallback_exit
          if fallback_result.success?
            close_position(fallback_result.value, emergency_type, reason)
          else
            # Critical: could not exit
            notify_critical_failure(emergency_type, exit_result.error)
          end
        end

        Result.success(
          emergency_type: emergency_type,
          reason: reason,
          exit_executed: exit_result.success? || fallback_result&.success?,
          position_status: @position.status
        )
      end

      private

      def execute_market_exit
        side = @position.long? ? "sell" : "buy"

        OrderService.new(user: @position.user).place(
          instrument: @position.instrument,
          side: side,
          quantity: @position.quantity,
          order_type: "market",
          product_type: @position.order.product_type
        )
      end

      def execute_fallback_exit
        side = @position.long? ? "sell" : "buy"
        current_price = @position.current_price.to_d

        OrderService.new(user: @position.user).place(
          instrument: @position.instrument,
          side: side,
          quantity: @position.quantity,
          order_type: "limit",
          price: current_price * 0.99,  # Slightly worse for guaranteed fill
          product_type: @position.order.product_type
        )
      end

      def close_position(order, emergency_type, reason)
        realized = calculate_realized_pnl(order)

        @position.update!(
          status: "closed",
          closed_at: Time.current,
          realized_pnl: @position.realized_pnl.to_d + realized,
          management_log: @position.management_log + [{
            type: "emergency_exit",
            emergency_type: emergency_type,
            reason: reason,
            exit_price: order.average_price,
            realized_pnl: realized,
            at: Time.current.iso8601
          }]
        )

        # Create trade record
        Trade.create!(
          position: @position,
          instrument: @position.instrument,
          user: @position.user,
          entry_price: @position.entry_price,
          exit_price: order.average_price,
          quantity: @position.quantity,
          gross_pnl: realized,
          outcome: realized > 0 ? "win" : "loss",
          exit_reason: "emergency_#{emergency_type}",
          entry_time: @position.opened_at,
          exit_time: Time.current
        )
      end

      def calculate_realized_pnl(order)
        entry = @position.entry_price.to_d
        exit_price = order.average_price.to_d
        qty = order.filled_quantity.to_i

        if @position.long?
          (exit_price - entry) * qty
        else
          (entry - exit_price) * qty
        end
      end

      def log_emergency(emergency_type, reason)
        RiskEvent.log!(
          user: @position.user,
          instrument: @position.instrument,
          category: :emergency_exit,
          severity: :critical,
          message: "Emergency exit triggered: #{emergency_type} - #{reason}",
          metadata: {
            position_id: @position.id,
            emergency_type: emergency_type,
            reason: reason
          }
        )
      end

      def notify_emergency_resolved(emergency_type, order)
        AlertService.notify(
          level: :critical,
          message: "Emergency exit completed: #{emergency_type} on #{@position.instrument.trading_symbol}",
          metadata: {
            position_id: @position.id,
            exit_price: order.average_price,
            realized_pnl: @position.realized_pnl
          }
        )
      end

      def notify_critical_failure(emergency_type, error)
        AlertService.notify(
          level: :critical,
          message: "EMERGENCY EXIT FAILED: #{emergency_type} on #{@position.instrument.trading_symbol}",
          metadata: {
            position_id: @position.id,
            error: error.to_h,
            action_required: "Manual intervention required"
          }
        )
      end
    end
  end
end
```

```ruby
# app/services/position_alert_service.rb

class PositionAlertService < BaseService
  ALERT_THRESHOLDS = {
    pnl_percent: { warning: -2.0, critical: -5.0 },
    mae_percent: { warning: -3.0, critical: -6.0 },
    holding_time: { warning: 30, critical: 60 }  # minutes
  }.freeze

  def initialize(position:)
    @position = position
  end

  def check_and_notify
    alerts = []

    # P&L alert
    pnl_alert = check_pnl_alert
    alerts << pnl_alert if pnl_alert

    # MAE alert
    mae_alert = check_mae_alert
    alerts << mae_alert if mae_alert

    # Holding time alert
    time_alert = check_holding_time
    alerts << time_alert if time_alert

    # Greeks alert
    greeks_alert = check_greeks_alert
    alerts << greeks_alert if greeks_alert

    # Send consolidated notification
    if alerts.any?
      send_position_alert(alerts)
    end

    Result.success(alerts: alerts)
  end

  private

  def check_pnl_alert
    pnl_percent = calculate_pnl_percent
        return nil unless pnl_percent

    if pnl_percent <= ALERT_THRESHOLDS[:pnl_percent][:critical]
      {
        level: :critical,
        type: :pnl_loss,
        message: "Critical loss: #{pnl_percent}% on #{@position.instrument.trading_symbol}",
        value: pnl_percent
      }
    elsif pnl_percent <= ALERT_THRESHOLDS[:pnl_percent][:warning]
      {
        level: :warning,
        type: :pnl_loss,
        message: "Warning: Loss at #{pnl_percent}% on #{@position.instrument.trading_symbol}",
        value: pnl_percent
      }
    end
  end

  def check_mae_alert
    return nil unless @position.mae.to_d < 0

    mae_percent = (@position.mae.to_d / (@position.entry_price.to_d * @position.quantity) * 100).round(2)

    if mae_percent <= ALERT_THRESHOLDS[:mae_percent][:critical]
      {
        level: :critical,
        type: :mae_extreme,
        message: "Extreme adverse excursion: #{mae_percent}% on #{@position.instrument.trading_symbol}",
        value: mae_percent
      }
    elsif mae_percent <= ALERT_THRESHOLDS[:mae_percent][:warning]
      {
        level: :warning,
        type: :mae_extreme,
        message: "High adverse excursion: #{mae_percent}% on #{@position.instrument.trading_symbol}",
        value: mae_percent
      }
    end
  end

  def check_holding_time
    return nil unless @position.opened_at

    minutes = ((Time.current - @position.opened_at) / 60).round

    if minutes >= ALERT_THRESHOLDS[:holding_time][:critical]
      {
        level: :warning,
        type: :holding_time,
        message: "Position held #{minutes} minutes on #{@position.instrument.trading_symbol}",
        value: minutes
      }
    elsif minutes >= ALERT_THRESHOLDS[:holding_time][:warning]
      {
        level: :info,
        type: :holding_time,
        message: "Position held #{minutes} minutes on #{@position.instrument.trading_symbol}",
        value: minutes
      }
    end
  end

  def check_greeks_alert
    return nil unless @position.instrument.option?

    # Would fetch latest Greeks from option chain
    # Simplified check
    nil
  end

  def calculate_pnl_percent
    return nil if @position.entry_price.to_d.zero?

    if @position.long?
      ((@position.current_price.to_d - @position.entry_price.to_d) / @position.entry_price.to_d * 100).round(2)
    else
      ((@position.entry_price.to_d - @position.current_price.to_d) / @position.entry_price.to_d * 100).round(2)
    end
  end

  def send_position_alert(alerts)
    critical = alerts.select { |a| a[:level] == :critical }
    warnings = alerts.select { |a| a[:level] == :warning }

    message = ["📊 POSITION ALERT: #{@position.instrument.trading_symbol}"]

    critical.each { |a| message << "🔴 #{a[:message]}" }
    warnings.each { |a| message << "🟡 #{a[:message]}" }

    AlertService.notify(
      level: critical.any? ? :critical : :warning,
      message: message.join("\n"),
      metadata: {
        position_id: @position.id,
        alerts: alerts
      }
    )
  end
end
```

```ruby
# app/jobs/position_reconciliation_job.rb
# frozen_string_literal: true

class PositionReconciliationJob < JobBase
  RECONCILE_INTERVAL = 30  # seconds

  def perform
    User.active.find_each do |user|
      reconcile_user_positions(user)
    end
  end

  private

  def reconcile_user_positions(user)
    # Fetch broker positions
    broker_result = PositionService.new(user: user).call

    if broker_result.failure?
      log_reconciliation_error(user, broker_result.error)
      return
    end

    broker_positions = broker_result.value
    local_positions = Position.where(user: user, status: "open")

    # Reconcile each broker position
    broker_positions.each do |bp|
      reconcile_broker_position(user, bp, local_positions)
    end

    # Check for orphaned local positions
    local_positions.each do |local|
      unless broker_positions.any? { |bp| bp[:position_id] == local.position_id }
        handle_orphaned_position(user, local)
      end
    end

    # Emit reconciliation complete
    EventBus.publish(EventTypes::POSITION_RECONCILED, {
      user_id: user.id,
      broker_positions: broker_positions.length,
      local_positions: local_positions.count,
      timestamp: Time.current
    })
  end

  def reconcile_broker_position(user, broker_pos, local_positions)
    local = local_positions.find_by(position_id: broker_pos[:position_id])

    if local
      # Update local position with broker data
      local.update!(
        current_price: broker_pos[:current_price],
        unrealized_pnl: broker_pos[:unrealized_pnl],
        realized_pnl: broker_pos[:realized_pnl],
        quantity: broker_pos[:quantity],
        last_reconciled_at: Time.current
      )

      # Check for discrepancies
      discrepancy = detect_discrepancy(local, broker_pos)
      if discrepancy
        log_discrepancy(user, local, discrepancy)
      end
    else
      # Create new local position from broker data
      create_local_from_broker(user, broker_pos)
    end
  end

  def detect_discrepancy(local, broker_pos)
    discrepancies = []

    if (local.quantity.to_i - broker_pos[:quantity].to_i).abs > 0
      discrepancies << {
        field: :quantity,
        local: local.quantity,
        broker: broker_pos[:quantity]
      }
    end

    if (local.unrealized_pnl.to_d - broker_pos[:unrealized_pnl].to_d).abs > 1
      discrepancies << {
        field: :unrealized_pnl,
        local: local.unrealized_pnl,
        broker: broker_pos[:unrealized_pnl]
      }
    end

    discrepancies.any? ? discrepancies : nil
  end

  def handle_orphaned_position(user, local)
    # Position exists locally but not at broker (closed externally?)
    local.update!(
      status: "reconciled_closed",
      management_log: local.management_log + [{
        type: "reconciled_closed",
        reason: "Position not found at broker during reconciliation",
        at: Time.current.iso8601
      }]
    )

    RiskEvent.log!(
      user: user,
      instrument: local.instrument,
      category: :correlation,
      severity: :warning,
      message: "Orphaned position reconciled: #{local.position_id}",
      metadata: { position_id: local.position_id }
    )
  end

  def create_local_from_broker(user, broker_pos)
    instrument = Instrument.find_by(
      trading_symbol: broker_pos[:trading_symbol],
      exchange_segment: broker_pos[:exchange_segment]
    )

    return unless instrument

    order = Order.find_by(broker_order_id: broker_pos[:order_id])

    Position.create!(
      user: user,
      instrument: instrument,
      order: order,
      position_id: broker_pos[:position_id],
      side: broker_pos[:side],
      quantity: broker_pos[:quantity],
      entry_price: broker_pos[:entry_price],
      current_price: broker_pos[:current_price],
      unrealized_pnl: broker_pos[:unrealized_pnl],
      realized_pnl: broker_pos[:realized_pnl],
      status: "open",
      opened_at: broker_pos[:opened_at] || Time.current,
      management_log: [{
        type: "created_from_reconciliation",
        broker_position_id: broker_pos[:position_id],
        at: Time.current.iso8601
      }]
    )
  end

  def log_reconciliation_error(user, error)
    RiskEvent.log!(
      user: user,
      category: :system_error,
      severity: :high,
      message: "Position reconciliation failed: #{error.message}",
      metadata: { error: error.to_h }
    )
  end

  def log_discrepancy(user, position, discrepancies)
    RiskEvent.log!(
      user: user,
      instrument: position.instrument,
      category: :correlation,
      severity: :warning,
      message: "Position discrepancy detected: #{position.position_id}",
      metadata: {
        position_id: position.position_id,
        discrepancies: discrepancies
      }
    )
  end
end
```

```ruby
# app/jobs/position_management_job.rb
# frozen_string_literal: true

class PositionManagementJob < JobBase
  def perform(trade_setup_id)
    setup = TradeSetup.find(trade_setup_id)
    position = setup.order&.position

    return unless position&.status == "open"

    # Fetch latest market data
    market_data = fetch_market_data(position.instrument)

    # Run position management
    result = Engines::PositionManagement::PositionManagementEngine.new(
      position: position
    ).call(context: { market_data: market_data })

    if result.success?
      # Check for exit signals
      exit_rec = result.value[:exit_recommendation]

      if exit_rec && exit_rec[:action] != :hold
        execute_exit(position, exit_rec)
      end

      # Check for partial exits
      partial = result.value[:partial_exit]
      if partial
        execute_partial_exit(position, partial)
      end

      # Check alerts
      PositionAlertService.new(position: position).check_and_notify
    end

    # Reschedule if position still open
    if position.reload.status == "open"
      PositionManagementJob.set(wait: 5.seconds).perform_later(trade_setup_id)
    end
  end

  private

  def fetch_market_data(instrument)
    quote_result = MarketQuoteService.new(instrument: instrument).call
    depth_result = MarketDepthService.new(instrument: instrument).call

    {
      last_price: quote_result.success? ? quote_result.value[:ltp] : 0,
      bid: depth_result.success? ? depth_result.value[:bid_levels]&.first&.dig(:price) : 0,
      ask: depth_result.success? ? depth_result.value[:ask_levels]&.first&.dig(:price) : 0,
      volume: quote_result.success? ? quote_result.value[:volume] : 0,
      structure: fetch_latest_structure(instrument),
      greeks: fetch_latest_greeks(instrument)
    }
  end

  def fetch_latest_structure(instrument)
    MarketStructure.where(instrument: instrument).recent.first&.to_h
  end

  def fetch_latest_greeks(instrument)
    OptionChainSnapshot.where(instrument: instrument).recent.first&.to_h
  end

  def execute_exit(position, exit_rec)
    EmergencyExitManager.new(position: position).trigger(
      emergency_type: map_exit_reason(exit_rec[:reason]),
      reason: exit_rec[:reason]
    )
  end

  def execute_partial_exit(position, partial)
    PartialExitManager.new(position: position).execute(
      scale_level: partial[:level],
      quantity_to_close: partial[:quantity_to_close]
    )
  end

  def map_exit_reason(reason)
    case reason
    when :stop_loss_triggered then :catastrophic_move
    when :target_achieved then :manual_trigger
    when :trailing_stop_triggered then :manual_trigger
    when :theta_acceleration then :manual_trigger
    when :market_structure_changed then :manual_trigger
    when :catastrophic_price_move then :catastrophic_move
    else :manual_trigger
    end
  end
end
```

```ruby
module Api
  module V1
    class PositionsController < ApplicationController
      def index
        positions = Position.where(user: current_user)
                            .order(opened_at: :desc)

        render json: {
          open: positions.open.map(&:to_h),
          closed: positions.closed.map(&:to_h),
          total_pnl: positions.sum(:realized_pnl),
          open_pnl: positions.open.sum(:unrealized_pnl)
        }
      end

      def show
        position = Position.find(params[:id])
        return render json: { error: "Not found" }, status: :not_found unless position.user == current_user

        render json: position.to_h.merge(
          metrics: {
            pnl_percent: ((position.current_price - position.entry_price) / position.entry_price * 100).round(2),
            holding_time: ((Time.current - position.opened_at) / 60).round(2),
            mfe: position.mfe,
            mae: position.mae
          },
          management_log: position.management_log
        )
      end

      def update_stop
        position = Position.find(params[:id])
        return render json: { error: "Not found" }, status: :not_found unless position.user == current_user

        manager = StopLossManager.new(position: position)
        success = manager.update_stop(
          new_stop: params[:stop_loss],
          reason: params[:reason] || :manual_update
        )

        if success
          render json: { updated: true, new_stop: position.stop_loss }
        else
          render json: { updated: false, error: "Invalid stop loss movement" }, status: :unprocessable_entity
        end
      end

      def activate_trailing
        position = Position.find(params[:id])
        return render json: { error: "Not found" }, status: :not_found unless position.user == current_user

        manager = TrailingStopManager.new(position: position)
        success = manager.activate(
          method: params[:method]&.to_sym || :atr_based,
          atr: params[:atr],
          multiplier: params[:multiplier]
        )

        if success
          render json: { activated: true, trailing_stop: position.trailing_stop }
        else
          render json: { activated: false, error: "Could not activate trailing stop" }, status: :unprocessable_entity
        end
      end

      def emergency_exit
        position = Position.find(params[:id])
        return render json: { error: "Not found" }, status: :not_found unless position.user == current_user

        result = EmergencyExitManager.new(position: position).trigger(
          emergency_type: params[:reason]&.to_sym || :manual_trigger,
          reason: params[:detail] || "Manual emergency exit"
        )

        if result.success?
          render json: { exited: true, emergency_type: result.value[:emergency_type] }
        else
          render json: { exited: false, error: result.error.to_h }, status: :unprocessable_entity
        end
      end
    end
  end
```

```markdown
# Position Management Engine

## Overview

The Position Management Engine provides real-time monitoring and adaptive management of open positions with automated exit triggers, trailing stops, partial exits, and emergency protocols.

## Monitoring Loop

Runs every 5 seconds for each open position:
1. Update P&L, MFE, MAE
2. Evaluate exit conditions
3. Adjust trailing stop if enabled
4. Check partial exit opportunities
5. Monitor Greeks changes
6. Emit monitoring event

## Exit Conditions

| Condition | Trigger | Action | Urgency |
|-----------|---------|--------|---------|
| Stop Loss | Price hits stop | Market exit | Immediate |
| Target | Price hits target | Market exit | Immediate |
| Trailing Stop | Price reverses from peak | Market exit | Immediate |
| Time Decay | Theta > 2% daily | Gradual exit | Gradual |
| Structure Break | BOS/CHOCH against position | Market exit | High |
| Catastrophic Move | >3% adverse move in 1 min | Emergency exit | Immediate |

## Trailing Stop Methods

| Method | Calculation | Best For |
|--------|-------------|----------|
| ATR-based | Current price ± 2×ATR | Volatile markets |
| Percent-based | Current price ± 1.5% | Trending markets |
| Step-based | Entry + (R × step size) | Systematic scaling |
| Chandelier | Highest high - 3×ATR | Long-term trends |

## Partial Exit (Scale Out)

Default scale levels:
- 1R: Close 25%
- 2R: Close 25% (move stop to breakeven)
- 3R: Close 25%

After first scale: stop automatically moves to breakeven.

## Emergency Exit

Triggered by:
- Catastrophic price move
- Broker disconnect
- System failure
- Margin call
- Manual trigger
- Circuit breaker

Protocol:
1. Attempt market order exit
2. If fails: limit order at current price - 1%
3. If fails: alert for manual intervention
4. Log critical risk event

## Position Reconciliation

Every 30 seconds:
1. Fetch broker positions
2. Update local positions
3. Detect discrepancies
4. Handle orphaned positions
5. Log reconciliation results
```
