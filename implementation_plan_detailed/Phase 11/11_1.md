```ruby
# app/services/live_trading/live_trading_guard.rb
# frozen_string_literal: true

module LiveTrading
  class LiveTradingGuard
    class TradingBlocked < StandardError; end

    def initialize(trader:)
      @trader = trader
      @capital_limit = AppConfig.config.live_trading&.max_daily_capital || 500_000
      @max_positions = AppConfig.config.live_trading&.max_concurrent_positions || 5
      @max_order_value = AppConfig.config.live_trading&.max_order_value || 100_000
    end

    def permit_order?(order)
      checks = [
        capital_check(order),
        position_check,
        order_size_check(order),
        trading_hours_check,
        data_freshness_check,
        daily_loss_check,
        connection_check
      ]

      failed = checks.select { |c| !c[:passed] }
      if failed.any?
        block_reason = failed.map { |f| f[:reason] }.join("; ")
        Rails.logger.error "[LiveTradingGuard] BLOCKED: #{block_reason}"
        emit(EventTypes::TRADING_BLOCKED, { reason: block_reason, order: order })
        return Result.failure(TradingBlocked.new(block_reason))
      end

      Result.success(passed: true)
    end

    private

    def capital_check(order)
      used = Trade.where("created_at > ?", Date.current.beginning_of_day)
                  .sum { |t| t.position_value.to_f }
      remaining = @capital_limit - used
      order_value = order[:quantity].to_i * order[:price].to_f

      {
        passed: order_value <= remaining,
        reason: "Capital limit exceeded (#{used}/#{@capital_limit})"
      }
    end

    def position_check
      current = Trade.where(exit_price: nil).count
      {
        passed: current < @max_positions,
        reason: "Max positions reached (#{current}/#{@max_positions})"
      }
    end

    def order_size_check(order)
      value = order[:quantity].to_i * order[:price].to_f
      {
        passed: value <= @max_order_value,
        reason: "Order value too large (#{value}/#{@max_order_value})"
      }
    end

    def trading_hours_check
      enforcer = TradingHoursEnforcer.new
      {
        passed: enforcer.trading_allowed?,
        reason: "Outside trading hours"
      }
    end

    def data_freshness_check
      last_tick = Tick.maximum(:timestamp)
      return { passed: false, reason: "No tick data" } unless last_tick

      fresh = (Time.current - last_tick) < 30.seconds
      {
        passed: fresh,
        reason: "Stale tick data (#{Time.current - last_tick}s old)"
      }
    end

    def daily_loss_check
      limit = AppConfig.config.risk&.daily_loss_limit || 50_000
      used = Trade.where("exit_time > ?", Date.current.beginning_of_day)
                  .sum { |t| [t.gross_pnl.to_f, 0].min }
      remaining = limit - used.abs

      {
        passed: remaining > 0,
        reason: "Daily loss limit reached (#{used.abs}/#{limit})"
      }
    end

    def connection_check
      healthy = DhanHQ::Client.new.health_check.success? rescue false
      {
        passed: healthy,
        reason: "Broker connection unhealthy"
      }
    end
  end
end
```

```ruby
module LiveTrading
  class TradingHoursEnforcer
    MARKET_OPEN = "09:15".freeze
    MARKET_CLOSE = "15:30".freeze
    PRE_MARKET = "09:00".freeze
    COOL_OFF_START = "15:15".freeze

    def trading_allowed?
      return false unless trading_day?

      now = Time.current
      market_open = Time.zone.parse("#{Date.current} #{MARKET_OPEN}")
      market_close = Time.zone.parse("#{Date.current} #{MARKET_CLOSE}")

      now.between?(market_open, market_close)
    end

    def pre_market?
      return false unless trading_day?

      now = Time.current
      pre = Time.zone.parse("#{Date.current} #{PRE_MARKET}")
      open = Time.zone.parse("#{Date.current} #{MARKET_OPEN}")

      now.between?(pre, open)
    end

    def cool_off_period?
      return false unless trading_day?

      now = Time.current
      cool = Time.zone.parse("#{Date.current} #{COOL_OFF_START}")
      close = Time.zone.parse("#{Date.current} #{MARKET_CLOSE}")

      now.between?(cool, close)
    end

    def time_to_close_seconds
      return nil unless trading_day?
      close_time = Time.zone.parse("#{Date.current} #{MARKET_CLOSE}")
      [(close_time - Time.current).to_i, 0].max
    end

    private

    def trading_day?
      calendar = ExchangeCalendar.for_exchange("NSE").find_by(date: Date.current)
      calendar&.trading_day || false
    end
  end
end
```

```ruby
module LiveTrading
  class BrokerConnectionMonitor
    CHECK_INTERVAL = 30 # seconds
    MAX_RECONNECT_ATTEMPTS = 5

    def initialize
      @client = DhanHQ::Client.new
      @healthy = false
      @consecutive_failures = 0
      @reconnect_attempts = 0
    end

    def start
      @timer = Concurrent::TimerTask.new(execution_interval: CHECK_INTERVAL) do
        check_health
      end
      @timer.execute
      Rails.logger.info "[BrokerConnectionMonitor] Started monitoring"
    end

    def stop
      @timer&.shutdown
    end

    def healthy?
      @healthy
    end

    private

    def check_health
      result = @client.health_check
      if result.success?
        @healthy = true
        @consecutive_failures = 0
        @reconnect_attempts = 0
      else
        @consecutive_failures += 1
        Rails.logger.warn "[BrokerConnectionMonitor] Health check failed (#{@consecutive_failures})"

        if @consecutive_failures >= 3
          @healthy = false
          attempt_reconnect
        end
      end
    end

    def attempt_reconnect
      if @reconnect_attempts >= MAX_RECONNECT_ATTEMPTS
        Rails.logger.error "[BrokerConnectionMonitor] Max reconnect attempts reached. Trading halted."
        emit(EventTypes::BROKER_DISCONNECTED, { attempts: @reconnect_attempts })
        return
      end

      @reconnect_attempts += 1
      Rails.logger.info "[BrokerConnectionMonitor] Reconnect attempt #{@reconnect_attempts}/#{MAX_RECONNECT_ATTEMPTS}"

      begin
        @client = DhanHQ::Client.new  # Reinitialize
        @client.health_check
        @healthy = true
        @consecutive_failures = 0
        Rails.logger.info "[BrokerConnectionMonitor] Reconnected successfully"
      rescue => e
        Rails.logger.error "[BrokerConnectionMonitor] Reconnect failed: #{e.message}"
      end
    end
  end
end
```

```ruby
# frozen_string_literal: true

module LiveTrading
  class DataQualityGate
    MAX_TICK_AGE_SECONDS = 30
    MAX_CANDLE_GAP_MINUTES = 10
    MIN_TICKS_PER_MINUTE = 10

    def initialize(instrument:)
      @instrument = instrument
    end

    def pass?
      checks = [
        tick_freshness,
        candle_continuity,
        tick_density,
        price_sanity
      ]

      failed = checks.reject { |c| c[:passed] }
      if failed.any?
        Rails.logger.error "[DataQualityGate] FAILED for #{@instrument.trading_symbol}: #{failed.map { |f| f[:reason] }.join(', ')}"
        false
      else
        true
      end
    end

    def status
      {
        tick_freshness: tick_freshness,
        candle_continuity: candle_continuity,
        tick_density: tick_density,
        price_sanity: price_sanity,
        instrument: @instrument.trading_symbol
      }
    end

    private

    def tick_freshness
      last = @instrument.ticks.order(:timestamp).last
      return { passed: false, reason: "no_ticks" } unless last

      age = (Time.current - last.timestamp).to_i
      {
        passed: age <= MAX_TICK_AGE_SECONDS,
        reason: "stale_ticks_#{age}s",
        age: age
      }
    end

    def candle_continuity
      recent = Candle.where(instrument: @instrument)
                     .where("timestamp > ?", 30.minutes.ago)
                     .order(:timestamp)
                     .pluck(:timestamp)

      return { passed: false, reason: "no_candles" } if recent.empty?

      gaps = 0
      recent.each_cons(2) do |a, b|
        gap = (b - a) / 60.0
        gaps += 1 if gap > MAX_CANDLE_GAP_MINUTES
      end

      {
        passed: gaps == 0,
        reason: "candle_gaps_#{gaps}",
        gaps: gaps
      }
    end

    def tick_density
      count = @instrument.ticks.where("timestamp > ?", 1.minute.ago).count
      {
        passed: count >= MIN_TICKS_PER_MINUTE,
        reason: "low_tick_density_#{count}",
        count: count
      }
    end

    def price_sanity
      last_tick = @instrument.ticks.order(:timestamp).last
      return { passed: false, reason: "no_price" } unless last_tick

      last_close = @instrument.candles.order(:timestamp).last&.close
      return { passed: true, reason: "no_reference" } unless last_close

      change = ((last_tick.ltp - last_close.to_f) / last_close.to_f).abs
      {
        passed: change < 0.10,  # 10% move from last close
        reason: "extreme_price_move_#{(change * 100).round(2)}%",
        change: change
      }
    end
  end
end
```

```ruby
# app/services/live_trading/risk_circuit_breaker.rb
# frozen_string_literal: true

module LiveTrading
  class RiskCircuitBreaker
    STATES = %i[closed open half_open].freeze

    def initialize
      @state = :closed
      @failure_count = 0
      @success_count = 0
      @last_failure_time = nil
      @threshold = AppConfig.config.risk&.circuit_breaker_threshold || 3
      @timeout = AppConfig.config.risk&.circuit_breaker_timeout || 300
      @daily_loss_limit = AppConfig.config.risk&.daily_loss_limit || 50_000
    end

    def check
      case @state
      when :open
        if ready_to_half_open?
          transition_to(:half_open)
          :half_open
        else
          :blocked
        end
      when :half_open
        :cautious
      else
        :allowed
      end
    end

    def record_success
      @success_count += 1
      if @state == :half_open && @success_count >= 2
        transition_to(:closed)
      end
    end

    def record_failure(type: :general)
      @failure_count += 1
      @last_failure_time = Time.current

      if type == :daily_loss_limit
        Rails.logger.error "[CircuitBreaker] DAILY LOSS LIMIT HIT. Opening circuit."
        transition_to(:open)
      elsif @failure_count >= @threshold
        Rails.logger.error "[CircuitBreaker] Threshold reached (#{@failure_count}/#{@threshold}). Opening circuit."
        transition_to(:open)
      end
    end

    def daily_loss_check
      today_loss = Trade.where("exit_time > ?", Date.current.beginning_of_day)
                        .sum { |t| [t.gross_pnl.to_f, 0].min }
                        .abs

      if today_loss >= @daily_loss_limit
        record_failure(type: :daily_loss_limit)
        false
      else
        true
      end
    end

    def state
      @state
    end

    private

    def ready_to_half_open?
      return false unless @last_failure_time
      (Time.current - @last_failure_time) > @timeout
    end

    def transition_to(new_state)
      old_state = @state
      @state = new_state
      @failure_count = 0 if new_state == :closed
      @success_count = 0 if new_state == :open

      emit(EventTypes::CIRCUIT_BREAKER_STATE_CHANGE, {
        from: old_state,
        to: new_state,
        timestamp: Time.current.iso8601
      })

      Rails.logger.info "[CircuitBreaker] #{old_state} -> #{new_state}"
    end
  end
end
```

```ruby
module Api
  module V1
    class EmergencyStopController < ApplicationController
      def trigger
        reason = params[:reason] || "manual_trigger"
        user = current_user&.email || "unknown"

        # 1. Block all new orders
        RedisPool.general.set("emergency_stop:active", "1", ex: 3600)
        RedisPool.general.set("emergency_stop:reason", reason, ex: 3600)
        RedisPool.general.set("emergency_stop:triggered_by", user, ex: 3600)
        RedisPool.general.set("emergency_stop:triggered_at", Time.current.iso8601, ex: 3600)

        # 2. Cancel all pending orders
        cancel_pending_orders

        # 3. Flatten all positions (market orders)
        flatten_positions

        # 4. Notify
        emit(EventTypes::EMERGENCY_STOP_TRIGGERED, {
          reason: reason,
          triggered_by: user,
          positions_closed: Trade.where(exit_price: nil).count
        })

        # 5. Alert
        AlertHistory.create!(
          severity: "critical",
          category: "risk",
          message: "EMERGENCY STOP triggered by #{user}: #{reason}",
          metadata: { positions_closed: Trade.where(exit_price: nil).count }
        )

        render json: {
          status: :triggered,
          reason: reason,
          triggered_by: user,
          timestamp: Time.current.iso8601,
          message: "All trading halted. Positions flattened."
        }
      end

      def status
        active = RedisPool.general.get("emergency_stop:active") == "1"
        render json: {
          active: active,
          reason: RedisPool.general.get("emergency_stop:reason"),
          triggered_by: RedisPool.general.get("emergency_stop:triggered_by"),
          triggered_at: RedisPool.general.get("emergency_stop:triggered_at")
        }
      end

      def reset
        # Requires admin
        unless current_user&.admin?
          return render json: { error: "Unauthorized" }, status: :forbidden
        end

        RedisPool.general.del("emergency_stop:active")
        RedisPool.general.del("emergency_stop:reason")
        RedisPool.general.del("emergency_stop:triggered_by")
        RedisPool.general.del("emergency_stop:triggered_at")

        AlertHistory.create!(
          severity: "warning",
          category: "system",
          message: "Emergency stop reset by #{current_user.email}"
        )

        render json: { status: :reset, timestamp: Time.current.iso8601 }
      end

      private

      def cancel_pending_orders
        # Interface with broker to cancel all open orders
        DhanHQ::Client.new.cancel_all_orders rescue nil
      end

      def flatten_positions
        Trade.where(exit_price: nil).find_each do |trade|
          begin
            direction = trade.direction == :long ? :sell : :buy
            DhanHQ::Client.new.place_order(
              instrument: trade.instrument,
              transaction_type: direction,
              quantity: trade.quantity,
              order_type: :market
            )
            trade.update!(exit_price: trade.current_price, exit_reason: "emergency_stop")
          rescue => e
            Rails.logger.error "[EmergencyStop] Failed to flatten trade #{trade.id}: #{e.message}"
          end
        end
      end
    end
  end
```

**Task 4.1: Create** `GracefulShutdown`

```ruby
# app/services/live_trading/graceful_shutdown.rb
# frozen_string_literal: true

module LiveTrading
  class GracefulShutdown
    SHUTDOWN_TIMEOUT = 60 # seconds

    def initialize
      @shutting_down = false
    end

    def initiate(reason: "scheduled")
      return if @shutting_down

      @shutting_down = true
      Rails.logger.info "[GracefulShutdown] Initiated: #{reason}"

      # 1. Stop accepting new signals
      RedisPool.general.set("system:accepting_signals", "false", ex: 300)

      # 2. Wait for active engines to complete
      wait_for_active_engines

      # 3. Reconcile positions
      reconcile_positions

      # 4. Persist state
      persist_state

      # 5. Disconnect broker
      disconnect_broker

      emit(EventTypes::SYSTEM_SHUTDOWN, {
        reason: reason,
        positions_reconciled: true,
        timestamp: Time.current.iso8601
      })

      Rails.logger.info "[GracefulShutdown] Complete"
    end

    def shutting_down?
      @shutting_down
    end

    private

    def wait_for_active_engines
      start = Time.current
      loop do
        active = SolidQueue::Job.where(finished_at: nil).count
        break if active == 0 || (Time.current - start) > SHUTDOWN_TIMEOUT
        Rails.logger.info "[GracefulShutdown] Waiting for #{active} active jobs..."
        sleep 1
      end
    end

    def reconcile_positions
      broker_positions = fetch_broker_positions
      local_positions = Trade.where(exit_price: nil)

      local_positions.find_each do |trade|
        broker_match = broker_positions.find { |bp| bp[:instrument_id] == trade.instrument_id }
        if broker_match
          Rails.logger.info "[GracefulShutdown] Position #{trade.id} reconciled with broker"
        else
          Rails.logger.warn "[GracefulShutdown] Position #{trade.id} not found in broker. Marking as orphaned."
          trade.update!(notes: "Orphaned during shutdown")
        end
      end
    end

    def fetch_broker_positions
      DhanHQ::Client.new.positions rescue []
    end

    def persist_state
      RedisPool.general.set("shutdown:last_time", Time.current.iso8601)
      RedisPool.general.set("shutdown:positions_count", Trade.where(exit_price: nil).count)
    end

    def disconnect_broker
      DhanWebSocketManager.instance.disconnect rescue nil
      Rails.logger.info "[GracefulShutdown] Broker disconnected"
    end
  end
end
```

```ruby
module LiveTrading
  class DeploymentChecklist
    CHECKS = [
      :database_connectivity,
      :redis_connectivity,
      :pgvector_extension,
      :broker_api_health,
      :websocket_connectivity,
      :market_hours_configured,

      :ai_gateway_healthy,
      :solid_queue_running
    ].freeze

    def run

      all_passed = results.all? { |_, r| r[:status] == :pass }

      {
        deployment_ready: all_passed,
        checks: results.to_h,
        timestamp: Time.current.iso8601,
        version: AppConfig.config.app_version
      }
    end

    private

    def database_connectivity
      ActiveRecord::Base.connection.execute("SELECT 1")
      { status: :pass, message: "Connected" }
    rescue => e
      { status: :fail, message: e.message }
    end

    def redis_connectivity
      RedisPool.general.ping
      { status: :pass, message: "Connected" }
    rescue => e
      { status: :fail, message: e.message }
    end

    def pgvector_extension
      result = ActiveRecord::Base.connection.execute("SELECT 1 FROM pg_extension WHERE extname = 'vector'")
      if result.ntuples > 0
        { status: :pass, message: "Active" }
      else
        { status: :fail, message: "Not installed" }
      end
    end

    def broker_api_health
      DhanHQ::Client.new.health_check.success? ?
        { status: :pass, message: "Healthy" } :
        { status: :fail, message: "Unhealthy" }
    rescue => e
      { status: :fail, message: e.message }
    end

    def websocket_connectivity
      DhanWebSocketManager.instance.connected? ?
        { status: :pass, message: "Connected" } :
        { status: :warn, message: "Disconnected" }
    rescue
      { status: :warn, message: "Unknown" }
    end

    def market_hours_configured
      cal = ExchangeCalendar.for_exchange("NSE").find_by(date: Date.current)
      cal ? { status: :pass, message: cal.market_open } : { status: :fail, message: "Not configured" }
    end

      { status: :fail, message: e.message }

      healthy = health.values.any? { |h| h[:available] }

    rescue => e
      { status: :warn, message: e.message }
    end

    def solid_queue_running
      # Check if Solid Queue supervisor is active
      { status: :pass, message: "Assumed running" }
    end
```

```ruby
# app/services/live_trading/incident_response.rb
# frozen_string_literal: true

module LiveTrading
  class IncidentResponse
    SEVERITY_ACTIONS = {
      critical: [:emergency_stop, :page_oncall, :send_telegram],
      warning: [:send_telegram, :log_detail],
      info: [:log_detail]
    }.freeze

    def initialize(alert)
      @alert = alert
      @severity = alert.severity.to_sym
    end

    def respond
      actions = SEVERITY_ACTIONS[@severity] || [:log_detail]
      actions.each { |action| send(action) }
    end

    private

    def emergency_stop
      return unless @severity == :critical
      LiveTrading::GracefulShutdown.new.initiate(reason: "critical_alert: #{@alert.message}")
    end

    def page_oncall
      # Integration with PagerDuty/Opsgenie
      Rails.logger.error "[IncidentResponse] PAGING ONCALL: #{@alert.message}"
    end

    def send_telegram
      TelegramBot.send_message(
```

**Task 6.1: Create** `PostTradeReconciliation`

```ruby
# app/services/live_trading/post_trade_reconciliation.rb
# frozen_string_literal: true

module LiveTrading
  class PostTradeReconciliation
    def initialize(date: Date.current)
      @date = date
      @discrepancies = []
    end

    def call
      broker_trades = fetch_broker_trades
      local_trades = fetch_local_trades

      reconcile(broker_trades, local_trades)

      {
        date: @date,
        broker_count: broker_trades.length,
        local_count: local_trades.length,
        matched: @discrepancies.empty?,
        discrepancies: @discrepancies,
        timestamp: Time.current.iso8601
      }
    end

    private

    def fetch_broker_trades
      DhanHQ::Client.new.trade_history(
        from: @date.beginning_of_day,
        to: @date.end_of_day
      ).value || []
    rescue => e
      Rails.logger.error "[Reconciliation] Broker fetch failed: #{e.message}"
      []
    end

    def fetch_local_trades
      Trade.where("created_at BETWEEN ? AND ?", @date.beginning_of_day, @date.end_of_day)
           .order(:created_at)
    end

    def reconcile(broker_trades, local_trades)
      local_trades.each do |local|
        match = broker_trades.find { |bt| bt[:order_id] == local.broker_order_id }

        if match.nil?
          @discrepancies << {
            type: :missing_in_broker,
            local_trade_id: local.id,
            broker_order_id: local.broker_order_id
          }
        else
          verify_fields(local, match)
        end
      end

      broker_trades.each do |bt|
        match = local_trades.find { |lt| lt.broker_order_id == bt[:order_id] }
        if match.nil?
          @discrepancies << {
            type: :missing_in_local,
            broker_order_id: bt[:order_id],
            broker_details: bt
          }
        end
      end
    end

    def verify_fields(local, broker)
      checks = {
        quantity: local.quantity == broker[:quantity],
        price: (local.entry_price.to_f - broker[:price].to_f).abs < 0.01,
        direction: local.direction.to_s == broker[:transaction_type].to_s.downcase
      }

      checks.each do |field, match|
        unless match
          @discrepancies << {
            type: :field_mismatch,
            field: field,
            local: local.send(field),
            broker: broker[field == :price ? :price : field],
            trade_id: local.id
          }
        end
      end
    end
  end
end
```
