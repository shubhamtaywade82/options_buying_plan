### Day 1 — Risk Validation Engine Foundation & Rule Interface

**Task 1.1: Create `RiskValidationEngine`**

```ruby
# app/engines/risk/risk_validation_engine.rb
# frozen_string_literal: true

module Engines
  module Risk
    class RiskValidationEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[
        trade_score
        portfolio
        market_context
      ].freeze

      RULES = [
        DailyLossLimitChecker,
        WeeklyLossLimitChecker,
        MonthlyLossLimitChecker,
        MaxOpenPositionsChecker,
        MarginAvailabilityChecker,
        RiskPerTradeCalculator,
        PositionSizingCalculator,
        MaxExposureChecker,
        TimeFilterChecker,
        ExpiryRuleChecker,
        RewardRiskRatioChecker,
        CorrelationChecker
      ].freeze

      def initialize(user:, instrument:)
        @user = user
        @instrument = instrument
        @rule_results = []
        @violations = []
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        trade_score = context[:trade_score]
        portfolio = context[:portfolio]
        market_context = context[:market_context]

        # Run all risk rules
        RULES.each do |rule_class|
          checker = rule_class.new(
            user: @user,
            instrument: @instrument,
            trade_score: trade_score,
            portfolio: portfolio,
            market_context: market_context
          )

          rule_result = checker.call

          @rule_results << {
            rule: rule_class.name.demodulize,
            passed: rule_result[:passed],
            message: rule_result[:message],
            metadata: rule_result[:metadata]
          }

          unless rule_result[:passed]
            @violations << {
              rule: rule_class.name.demodulize,
              severity: rule_result[:severity] || :high,
              message: rule_result[:message],
              metadata: rule_result[:metadata]
            }
          end
        end

        # Determine overall validation
        passed = @violations.none? { |v| v[:severity] == :critical } &&
                 @violations.count { |v| v[:severity] == :high } <= 1  # Allow 1 high-severity violation with warning

        # Check for emergency stop condition
        if emergency_stop_triggered?
          trigger_emergency_stop!
          return Result.failure(
            RiskError.new(
              code: :emergency_stop,
              message: "Emergency stop triggered - catastrophic loss detected",
              violations: @violations
            )
          )
        end

        emit(EventTypes::RISK_VALIDATED, {
          user_id: @user.id,
          instrument_id: @instrument.id,
          passed: passed,
          violations: @violations.length,
          rule_results: @rule_results
        })

        if passed
          success(
            passed: true,
            rule_results: @rule_results,
            position_size: calculate_approved_position_size,
            max_loss: calculate_max_loss,
            warnings: @violations.select { |v| v[:severity] == :medium }
          )
        else
          Result.failure(
            RiskError.new(
              code: :risk_validation_failed,
              message: "Trade failed #{@violations.length} risk rules",
              violations: @violations,
              rule_results: @rule_results
            )
          )
        end
      end

      private

      def emergency_stop_triggered?
        # Check if daily loss exceeds catastrophic threshold
        daily_pnl = calculate_daily_pnl
        catastrophic_limit = AppConfig.config.risk.daily_catastrophic_loss_limit

        daily_pnl <= -catastrophic_limit
      end

      def trigger_emergency_stop!
        # Set global emergency stop flag
        RedisPool.general.with do |redis|
          redis.set("emergency_stop:#{@user.id}", Time.current.iso8601, ex: 1.day)
        end

        # Alert all systems
        EventBus.publish(EventTypes::SYSTEM_ERROR, {
          type: :emergency_stop,
          user_id: @user.id,
          daily_pnl: calculate_daily_pnl,
          triggered_at: Time.current
        })
      end

      def calculate_daily_pnl
        # Sum realized + unrealized P&L for today
        today_trades = Trade.where(user: @user, entry_time: Date.current.all_day)
        realized = today_trades.sum(:gross_pnl)

        open_positions = Position.where(user: @user, status: "open")
        unrealized = open_positions.sum(:unrealized_pnl)

        realized + unrealized
      end

      def calculate_approved_position_size
        # Return the minimum approved size across all rules
        sizing_rules = @rule_results.select { |r| r[:rule].include?("PositionSizing") || r[:rule].include?("RiskPerTrade") }
        sizing_rules.map { |r| r[:metadata][:approved_size] }.compact.min || 0
      end

      def calculate_max_loss
        approved_size = calculate_approved_position_size
        stop_loss_distance = @rule_results.find { |r| r[:rule] == "RewardRiskRatioChecker" }&.dig(:metadata, :stop_loss_distance) || 0
        approved_size * stop_loss_distance
      end
    end
  end
end
```

```ruby
module Engines
  module Risk
    class BaseRiskChecker
      attr_reader :user, :instrument, :trade_score, :portfolio, :market_context

      def initialize(user:, instrument:, trade_score:, portfolio:, market_context:)
        @user = user
        @instrument = instrument
        @trade_score = trade_score
        @portfolio = portfolio
        @market_context = market_context
      end

      def call
        raise NotImplementedError
      end

      protected

      def passed(message: "Passed", metadata: {})
        {
          passed: true,
          severity: :none,
          message: message,
          metadata: metadata
        }
      end

      def failed(message:, severity: :high, metadata: {})
        {
          passed: false,
          severity: severity,
          message: message,
          metadata: metadata
        }
      end

      def warning(message:, metadata: {})
        {
          passed: true,
          severity: :medium,
          message: message,
          metadata: metadata
        }
      end

      def current_equity
        @portfolio[:total_equity].to_d
      end

      def open_positions_count
        @portfolio[:open_positions].to_i
      end

      def today_trades
        @today_trades ||= Trade.where(user: @user, entry_time: Date.current.all_day)
      end

      def today_pnl
        today_trades.sum(:gross_pnl).to_d
      end

      def realized_pnl_today
        today_trades.where.not(exit_time: nil).sum(:gross_pnl).to_d
      end

      def unrealized_pnl
        Position.where(user: @user, status: "open").sum(:unrealized_pnl).to_d
      end
    end
  end
```

```ruby
module Engines
  module Risk
    class DailyLossLimitChecker < BaseRiskChecker
      def call
        limit = AppConfig.config.risk.daily_loss_limit.to_d
        current_loss = calculate_current_loss.abs

        if current_loss >= limit
          failed(
            message: "Daily loss limit exceeded: ₹#{current_loss.round(2)} / ₹#{limit}",
            severity: :critical,
            metadata: {
              current_loss: current_loss,
              limit: limit,
              remaining: 0,
              trades_today: today_trades.count
            }
          )
        elsif current_loss >= limit * 0.8
          warning(
            message: "Daily loss at 80% of limit: ₹#{current_loss.round(2)} / ₹#{limit}",
            metadata: {
              current_loss: current_loss,
              limit: limit,
              remaining: limit - current_loss,
              percent_used: (current_loss / limit * 100).round(2)
            }
          )
        else
          passed(
            message: "Daily loss within limit: ₹#{current_loss.round(2)} / ₹#{limit}",
            metadata: {
              current_loss: current_loss,
              limit: limit,
              remaining: limit - current_loss,
              percent_used: (current_loss / limit * 100).round(2)
            }
          )
        end
      end

      private

      def calculate_current_loss
        realized = today_trades.where.not(exit_time: nil).sum(:gross_pnl).to_d
        unrealized = Position.where(user: @user, status: "open").sum(:unrealized_pnl).to_d

        # Losses are negative numbers
        total = realized + unrealized
        total < 0 ? total : 0
      end
    end
  end
end
```

```ruby
module Engines
  module Risk
    class MaxOpenPositionsChecker < BaseRiskChecker
      def call
        max_total = AppConfig.config.risk.max_open_positions_total.to_i
        max_per_index = AppConfig.config.risk.max_open_positions_per_index.to_i

        current_total = total_open_positions
        current_per_index = open_positions_for_index

        violations = []

        if current_total >= max_total
          violations << {
            type: :total_limit,
            current: current_total,
            limit: max_total
          }
        end

        if current_per_index >= max_per_index
          violations << {
            type: :per_index_limit,
            index: @instrument.trading_symbol,
            current: current_per_index,
            limit: max_per_index
          }
        end

        if violations.any?
          failed(
            message: "Position limit exceeded: #{violations.map { |v| "#{v[:type]} #{v[:current]}/#{v[:limit]}" }.join(', ')}",
            severity: :high,
            metadata: {
              violations: violations,
              current_total: current_total,
              max_total: max_total,
              current_per_index: current_per_index,
              max_per_index: max_per_index
            }
          )
        elsif current_total >= max_total * 0.8 || current_per_index >= max_per_index * 0.8
          warning(
            message: "Position limits approaching: Total #{current_total}/#{max_total}, #{@instrument.trading_symbol} #{current_per_index}/#{max_per_index}",
            metadata: {
              current_total: current_total,
              max_total: max_total,
              current_per_index: current_per_index,
              max_per_index: max_per_index
            }
          )
        else
          passed(
            message: "Position limits OK: Total #{current_total}/#{max_total}, #{@instrument.trading_symbol} #{current_per_index}/#{max_per_index}",
            metadata: {
              current_total: current_total,
              max_total: max_total,
              current_per_index: current_per_index,
              max_per_index: max_per_index
            }
          )
        end
      end

      private

      def total_open_positions
        Position.where(user: @user, status: "open").count
      end

      def open_positions_for_index
        Position.where(user: @user, status: "open")
                  .joins(:instrument)
                  .where(instruments: { underlying: @instrument.trading_symbol })
                  .count
      end
    end
  end
```

```ruby
# app/engines/risk/margin_availability_checker.rb
# frozen_string_literal: true

module Engines
  module Risk
    class MarginAvailabilityChecker < BaseRiskChecker
      MARGIN_BUFFER = 1.2  # 20% buffer above required margin

      def call
        # Fetch from broker
        funds_result = FundsService.new(user: @user).call
        return warning(message: "Could not verify margin - proceeding with caution", metadata: {}) if funds_result.failure?

        available = funds_result.value[:available_balance].to_d
        utilized = funds_result.value[:utilized_amount].to_d
        total = funds_result.value[:total_balance].to_d

        # Estimate required margin for this trade
        required_margin = estimate_required_margin
        required_with_buffer = required_margin * MARGIN_BUFFER

        if available < required_with_buffer
          failed(
            message: "Insufficient margin: Available ₹#{available.round(2)}, Required ₹#{required_with_buffer.round(2)} (with #{((MARGIN_BUFFER - 1) * 100).round(0)}% buffer)",
            severity: :critical,
            metadata: {
              available: available,
              required: required_margin,
              required_with_buffer: required_with_buffer,
              buffer_percent: (MARGIN_BUFFER - 1) * 100,
              utilization_percent: total.zero? ? 0 : (utilized / total * 100).round(2)
            }
          )
        elsif available < required_with_buffer * 1.5
          warning(
            message: "Margin tight: Available ₹#{available.round(2)}, Required ₹#{required_with_buffer.round(2)}",
            metadata: {
              available: available,
              required: required_with_buffer,
              utilization_percent: total.zero? ? 0 : (utilized / total * 100).round(2)
            }
          )
        else
          passed(
            message: "Margin sufficient: Available ₹#{available.round(2)}, Required ₹#{required_with_buffer.round(2)}",
            metadata: {
              available: available,
              required: required_with_buffer,
              utilization_percent: total.zero? ? 0 : (utilized / total * 100).round(2)
            }
          )
        end
      end

      private

      def estimate_required_margin
        # Use margin calculator or estimate based on position size
        # Simplified: ~15% of notional for naked options
        lot_size = @instrument.lot_size || 50
        price = @trade_score.dig(:entry_price) || 100
        quantity = lot_size

        # Rough estimate: 15% margin
        price * quantity * 0.15
      end
    end
  end
end
```

```ruby
# app/engines/risk/max_exposure_checker.rb
# frozen_string_literal: true

module Engines
  module Risk
    class MaxExposureChecker < BaseRiskChecker
      def call
        max_total_exposure = AppConfig.config.risk.max_total_exposure_percent.to_d
        max_per_index_exposure = AppConfig.config.risk.max_per_index_exposure_percent.to_d

        total_equity = current_equity
        return failed(message: "Cannot determine equity", severity: :high) if total_equity <= 0

        current_total_exposure = calculate_total_exposure
        current_index_exposure = calculate_index_exposure

        total_percent = (current_total_exposure / total_equity * 100).round(2)
        index_percent = (current_index_exposure / total_equity * 100).round(2)

        # Add proposed trade exposure
        proposed_trade_exposure = estimate_trade_exposure
        projected_total = ((current_total_exposure + proposed_trade_exposure) / total_equity * 100).round(2)
        projected_index = ((current_index_exposure + proposed_trade_exposure) / total_equity * 100).round(2)

        violations = []

        if projected_total > max_total_exposure
          violations << {
            type: :total_exposure,
            projected: projected_total,
            limit: max_total_exposure,
            current: total_percent
          }
        end

        if projected_index > max_per_index_exposure
          violations << {
            type: :index_exposure,
            index: @instrument.trading_symbol,
            projected: projected_index,
            limit: max_per_index_exposure,
            current: index_percent
          }
        end

        if violations.any?
          failed(
            message: "Exposure limit would be exceeded: #{violations.map { |v| "#{v[:type]} #{v[:projected]}%/#{v[:limit]}%" }.join(', ')}",
            severity: :high,
            metadata: {
              violations: violations,
              total_equity: total_equity,
              current_total_exposure: current_total_exposure,
              current_index_exposure: current_index_exposure,
              proposed_trade_exposure: proposed_trade_exposure
            }
          )
        elsif projected_total > max_total_exposure * 0.8 || projected_index > max_per_index_exposure * 0.8
          warning(
            message: "Exposure approaching limits: Total #{projected_total}%/#{max_total_exposure}%, #{@instrument.trading_symbol} #{projected_index}%/#{max_per_index_exposure}%",
            metadata: {
              projected_total: projected_total,
              max_total: max_total_exposure,
              projected_index: projected_index,
              max_per_index: max_per_index_exposure
            }
          )
        else
          passed(
            message: "Exposure within limits: Total #{projected_total}%/#{max_total_exposure}%, #{@instrument.trading_symbol} #{projected_index}%/#{max_per_index_exposure}%",
            metadata: {
              projected_total: projected_total,
              max_total: max_total_exposure,
              projected_index: projected_index,
              max_per_index: max_per_index_exposure
            }
          )
        end
      end

      private

      def calculate_total_exposure
        # Sum of all open position notional values
        Position.where(user: @user, status: "open").sum do |pos|
          pos.quantity.to_d * pos.current_price.to_d
        end
      end

      def calculate_index_exposure
        Position.where(user: @user, status: "open")
                .joins(:instrument)
                .where(instruments: { underlying: @instrument.trading_symbol })
                .sum do |pos|
          pos.quantity.to_d * pos.current_price.to_d
        end
      end

      def estimate_trade_exposure
        # Estimate notional of proposed trade
        lot_size = @instrument.lot_size || 50
        entry_price = @trade_score.dig(:entry_price) || 100
        lot_size * entry_price
      end
    end
  end
end
```

```ruby
# app/engines/risk/risk_per_trade_calculator.rb
# frozen_string_literal: true

module Engines
  module Risk
    class RiskPerTradeCalculator < BaseRiskChecker
      DEFAULT_RISK_PERCENT = 1.0  # 1% of capital per trade

      def call
        risk_percent = AppConfig.config.risk.risk_per_trade_percent.to_d || DEFAULT_RISK_PERCENT
        max_risk_rupees = AppConfig.config.risk.max_risk_per_trade_rupees.to_d

        total_equity = current_equity
        return failed(message: "Cannot determine equity for risk calculation", severity: :high) if total_equity <= 0

        # Calculate risk amount
        risk_amount = total_equity * (risk_percent / 100)
        risk_amount = [risk_amount, max_risk_rupees].min if max_risk_rupees > 0

        # Get stop loss distance from trade score
        stop_loss = @trade_score.dig(:stop_loss)
        entry_price = @trade_score.dig(:entry_price)

        if stop_loss && entry_price && stop_loss.to_d > 0 && entry_price.to_d > 0
          stop_distance = (entry_price.to_d - stop_loss.to_d).abs
          risk_per_unit = stop_distance

          if risk_per_unit > 0
            approved_size = (risk_amount / risk_per_unit).floor
            approved_size = [approved_size, 1].max  # Minimum 1 lot
          else
            approved_size = 0
          end
        else
          # Fallback: fixed rupee risk without stop distance
          approved_size = estimate_size_without_stop(risk_amount)
        end

        passed(
          message: "Risk per trade: ₹#{risk_amount.round(2)} (#{risk_percent}% of equity), Approved size: #{approved_size} units",
          metadata: {
            risk_percent: risk_percent,
            risk_amount: risk_amount,
            max_risk_rupees: max_risk_rupees,
            total_equity: total_equity,
            stop_loss: stop_loss,
            entry_price: entry_price,
            stop_distance: stop_distance,
            approved_size: approved_size
          }
        )
      end

      private

      def estimate_size_without_stop(risk_amount)
        # Estimate based on ATR or fixed percentage
        lot_size = @instrument.lot_size || 50
        price = @trade_score.dig(:entry_price) || 100

        # Assume 1% move as stop distance
        estimated_stop_distance = price * 0.01
        (risk_amount / estimated_stop_distance / lot_size).floor * lot_size
      end
    end
  end
end
```

```ruby
# app/engines/risk/position_sizing_calculator.rb
# frozen_string_literal: true

module Engines
  module Risk
    class PositionSizingCalculator < BaseRiskChecker
      def call
        # Kelly Criterion variant for sizing
        win_rate = calculate_win_rate
        avg_win = calculate_avg_win
        avg_loss = calculate_avg_loss

        kelly = Calculations::Risk.kelly_criterion(win_rate, avg_win, avg_loss)
        half_kelly = Calculations::Risk.half_kelly(win_rate, avg_win, avg_loss)

        # Apply practical constraints
        equity = current_equity
        max_position_size = equity * 0.10  # Max 10% of equity in one position

        # Kelly-based size
        kelly_size = equity * kelly
        half_kelly_size = equity * half_kelly

        # Constrain to max position size
        approved_size = [half_kelly_size, max_position_size].min

        # Further constrain by lot size
        lot_size = @instrument.lot_size || 50
        approved_lots = (approved_size / lot_size).floor
        approved_size = approved_lots * lot_size

        passed(
          message: "Position sizing: Kelly #{kelly.round(4)}, Half-Kelly #{half_kelly.round(4)}, Approved: #{approved_lots} lots (₹#{approved_size.round(2)})",
          metadata: {
            win_rate: win_rate,
            avg_win: avg_win,
            avg_loss: avg_loss,
            kelly_fraction: kelly,
            half_kelly: half_kelly,
            max_position_size: max_position_size,
            approved_lots: approved_lots,
            approved_size: approved_size,
            lot_size: lot_size
          }
        )
      end

      private

      def calculate_win_rate
        recent_trades = Trade.where(user: @user).where("entry_time > ?", 30.days.ago)
        return 0.5 if recent_trades.empty?

        wins = recent_trades.where("gross_pnl > 0").count
        wins.to_f / recent_trades.count
      end

      def calculate_avg_win
        wins = Trade.where(user: @user)
                    .where("entry_time > ?", 30.days.ago)
                    .where("gross_pnl > 0")
        return 0 if wins.empty?

        wins.average(:gross_pnl).to_f
      end

      def calculate_avg_loss
        losses = Trade.where(user: @user)
                      .where("entry_time > ?", 30.days.ago)
                      .where("gross_pnl < 0")
        return 1 if losses.empty?  # Avoid division by zero

        losses.average(:gross_pnl).to_f.abs
      end
    end
```

```ruby
  module Risk
    class TimeFilterChecker < BaseRiskChecker
      NO_TRADE_START_MINUTES = 5   # No trades in first 5 minutes
      NO_TRADE_END_MINUTES = 15    # No trades in last 15 minutes
      LUNCH_BREAK_START = "12:00"  # Optional lunch break
      LUNCH_BREAK_END = "13:00"

      def call
        now = Time.current
        market_hours = ExchangeCalendar.market_hours(exchange: "NSE")

        return failed(message: "Market hours unavailable", severity: :high) unless market_hours

        open_time = Time.zone.parse(market_hours[:open].to_s)
        close_time = Time.zone.parse(market_hours[:close].to_s)

        # Check if market is open
        unless now.between?(open_time, close_time)
          return failed(
            message: "Market closed: Current #{now.strftime('%H:%M')}, Market #{open_time.strftime('%H:%M')}-#{close_time.strftime('%H:%M')}",
            severity: :critical
          )
        end

        # Check opening blackout
        if now < open_time + NO_TRADE_START_MINUTES.minutes
          return failed(
            message: "Opening blackout: No trades until #{open_time + NO_TRADE_START_MINUTES.minutes}",
            severity: :high,
            metadata: {
              blackout_end: open_time + NO_TRADE_START_MINUTES.minutes
            }
          )
        end

        # Check closing blackout
        if now > close_time - NO_TRADE_END_MINUTES.minutes
          return failed(
            message: "Closing blackout: No trades after #{close_time - NO_TRADE_END_MINUTES.minutes}",
            severity: :high,
            metadata: {
              blackout_start: close_time - NO_TRADE_END_MINUTES.minutes
            }
          )
        end

        # Check lunch break (if configured)
        if AppConfig.config.risk.enforce_lunch_break != false
          lunch_start = Time.zone.parse(LUNCH_BREAK_START)
          lunch_end = Time.zone.parse(LUNCH_BREAK_END)

          if now.between?(lunch_start, lunch_end)
            return warning(
              message: "Lunch break period: Reduced liquidity expected #{LUNCH_BREAK_START}-#{LUNCH_BREAK_END}",
              metadata: {
                lunch_start: lunch_start,
                lunch_end: lunch_end
              }
            )
          end
        end

        passed(
          message: "Time filter passed: #{now.strftime('%H:%M')} within trading window",
          metadata: {
            market_open: open_time,
            market_close: close_time,
            effective_start: open_time + NO_TRADE_START_MINUTES.minutes,
            effective_end: close_time - NO_TRADE_END_MINUTES.minutes
```

```ruby
# app/engines/risk/expiry_rule_checker.rb
# frozen_string_literal: true

module Engines
  module Risk
    class ExpiryRuleChecker < BaseRiskChecker
      def call
        expiry_date = @instrument.expiry
        return passed(message: "No expiry date - not an option instrument", metadata: {}) unless expiry_date

        days_to_expiry = (expiry_date - Date.current).to_i

        if days_to_expiry < 0
          return failed(
            message: "Instrument expired on #{expiry_date}",
            severity: :critical,
            metadata: { expiry_date: expiry_date }
          )
        end

        # Same-day expiry rules
        if days_to_expiry == 0
          return check_expiry_day_rules
        end

        # Next-day expiry rules
        if days_to_expiry == 1
          return check_pre_expiry_rules
        end

        # Standard expiry rules (2+ days)
        if days_to_expiry < 3
          return warning(
            message: "Near expiry: #{days_to_expiry} days remaining. Theta decay accelerated.",
            metadata: {
              days_to_expiry: days_to_expiry,
              expiry_date: expiry_date,
              theta_risk: :elevated
            }
          )
        end

        passed(
          message: "Expiry OK: #{days_to_expiry} days to expiry",
          metadata: {
            days_to_expiry: days_to_expiry,
            expiry_date: expiry_date
          }
        )
      end

      private

      def check_expiry_day_rules
        now = Time.current
        market_close = Time.zone.parse(ExchangeCalendar.market_hours(exchange: "NSE")[:close].to_s)

        # No overnight positions on expiry day
        if now > market_close - 2.hours
          return failed(
            message: "Expiry day: No new positions after 13:30",
            severity: :high,
            metadata: {
              expiry_date: @instrument.expiry,
              cutoff_time: market_close - 2.hours
            }
          )
        end

        # Only MIS product on expiry day
        if @trade_score.dig(:product_type) != "MIS"
          return failed(
            message: "Expiry day: Only MIS (intraday) product allowed",
            severity: :high,
            metadata: {
              requested_product: @trade_score.dig(:product_type),
              allowed_product: "MIS"
            }
          )
        end

        warning(
          message: "Expiry day trading: Position must close by #{market_close.strftime('%H:%M')}",
          metadata: {
            expiry_date: @instrument.expiry,
            must_close_by: market_close
          }
        )
      end

      def check_pre_expiry_rules
        # No carry forward on day before expiry for certain instruments
        if @instrument.trading_symbol.include?("BANKNIFTY") && @trade_score.dig(:product_type) != "MIS"
          return warning(
            message: "Pre-expiry: Consider MIS only for BANKNIFTY",
            metadata: {
              days_to_expiry: 1,
              recommended_product: "MIS"
            }
          )
        end

        passed(
          message: "Pre-expiry: 1 day remaining, standard rules apply",
          metadata: { days_to_expiry: 1 }
        )
      end
    end
  end
```

```ruby
module Engines
  module Risk
    class RewardRiskRatioChecker < BaseRiskChecker
      MINIMUM_RR = 2.0  # Minimum 2:1 reward-risk

      def call
        entry = @trade_score.dig(:entry_price).to_d
        stop = @trade_score.dig(:stop_loss).to_d
        target = @trade_score.dig(:target).to_d

        return failed(message: "Missing entry/stop/target prices for R/R calculation", severity: :high) if entry <= 0 || stop <= 0

        risk = (entry - stop).abs
        reward = (target - entry).abs

        return failed(message: "Invalid stop loss: Stop must be below entry for long, above for short", severity: :high) if risk <= 0

        rr_ratio = (reward / risk).round(2)

        if rr_ratio < MINIMUM_RR
          failed(
            message: "Reward/Risk ratio too low: #{rr_ratio}:1 (minimum #{MINIMUM_RR}:1)",
            severity: :high,
            metadata: {
              entry_price: entry,
              stop_loss: stop,
              target: target,
              risk: risk,
              reward: reward,
              rr_ratio: rr_ratio,
              minimum_rr: MINIMUM_RR,
              required_target: entry + (risk * MINIMUM_RR)
            }
          )
        elsif rr_ratio < MINIMUM_RR * 1.5
          warning(
            message: "Reward/Risk ratio marginal: #{rr_ratio}:1 (ideal >= #{MINIMUM_RR * 1.5}:1)",
            metadata: {
              rr_ratio: rr_ratio,
              minimum_rr: MINIMUM_RR,
              ideal_rr: MINIMUM_RR * 1.5
            }
          )
        else
          passed(
            message: "Reward/Risk ratio excellent: #{rr_ratio}:1",
            metadata: {
              entry_price: entry,
              stop_loss: stop,
              target: target,
              risk: risk,
              reward: reward,
              rr_ratio: rr_ratio,
              stop_loss_distance: risk
            }
          )
        end
      end
    end
  end
end
```

```ruby
              same_direction: same_direction,
```

```ruby
      total_exposure: Position.where(user: setup.user, status: "open").sum { |p| p.quantity.to_d * p.current_price.to_d }

      instrument: setup.instrument
    ).call(
      context: {
        trade_score: trade_score.to_h,
        portfolio: portfolio,
        market_context: market_context
      }
    )

    if result.success?
      setup.update!(
        status: "risk_validated",
        approved_size: result.value[:position_size],
        max_loss: result.value[:max_loss]

      ExecutionJob.perform_later(setup.id)
    else
      setup.update!(
        status: "risk_rejected",
        rejection_reason: result.error.message
      )

      # Log risk event
      RiskEventLogger.new(user: setup.user, instrument: setup.instrument).log(
        category: :risk_validation,
        severity: :high,
        message: "Trade rejected: #{result.error.message}",
        metadata: result.error.metadata
      )
    end
  end
end
```
