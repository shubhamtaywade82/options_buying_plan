  ```ruby
  # frozen_string_literal: true

  module Engines
    class MarketContextEngine < BaseEngine
      OUTPUTS = %w[allow_trading reduced_risk no_trade].freeze

      def self.engine_name
        "market_context"
      end

      def self.required_features
        %w[
          market_open_gap
          india_vix
          expiry_proximity
          event_risk
          previous_day_trend
          market_session
        ]
      end

      def initialize(instrument:)
        super()
        @instrument = instrument
      end

      def analyze
        validate_inputs!(build_inputs)

        checks = {
          gap: gap_check,
          vix: vix_check,
          expiry: expiry_check,
          events: event_check,
          previous_trend: previous_trend_check,
          session: session_check,
          time_of_day: time_of_day_check
        }

        context = determine_context(checks)
        score = ContextScoreCalculator.new(checks: checks).call

        emit(EventTypes::REGIME_CHANGED, {
          engine: engine_name,
          instrument_id: @instrument.id,
          context: context,
          score: score.value,
          checks: checks
        })

        success(
          context: context,
          score: score.value,
          checks: checks,
          timestamp: Time.current
        )
      end

      private

      def build_inputs
        {
          instrument: @instrument,
          current_time: Time.current,
          current_price: fetch_current_price,
          previous_close: fetch_previous_close,
          india_vix: fetch_india_vix,
          today: Date.current
        }
      end

      def determine_context(checks)
        # Hard no-trade conditions
        return :no_trade if checks[:session] == :closed
        return :no_trade if checks[:events] == :high_impact
        return :no_trade if checks[:gap] == :extreme

        # Reduced risk conditions
        return :reduced_risk if checks[:expiry] == :expiry_day
        return :reduced_risk if checks[:vix] == :spike
        return :reduced_risk if checks[:time_of_day] == :low_liquidity
        return :reduced_risk if checks[:gap] == :large

        :allow_trading
      end

      def fetch_current_price
        CacheStore.fetch("tick:latest:#{@instrument.id}")&.dig(:last_price) ||
          MarketQuoteService.new(instrument: @instrument).ltp.value
      end

      def fetch_previous_close
        Candle.for_instrument(@instrument)
              .for_timeframe("1d")
              .where(is_complete: true)
              .order(timestamp: :desc)
              .first&.close
      end

      def fetch_india_vix
        # India VIX is instrument ID specific or fetched from dedicated source
        # Placeholder for VIX instrument lookup
        vix_instrument = Instrument.find_by(trading_symbol: "INDIA VIX")
        return nil unless vix_instrument

        CacheStore.fetch("tick:latest:#{vix_instrument.id}")&.dig(:last_price) ||
          MarketQuoteService.new(instrument: vix_instrument).ltp.value
      end
    end
  end
  ```
- **Deliverable:** Base engine with 7 check dimensions
- **Acceptance Criteria:** `MarketContextEngine.new(instrument: nifty).analyze` returns context + score

### Task 1.2: Create `GapAnalyzer`
- **File:** `app/engines/market_context/gap_analyzer.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketContext
      class GapAnalyzer
        GAP_THRESHOLDS = {
          normal: 0.5,      # < 0.5%
          moderate: 1.0,    # 0.5% - 1.0%
          large: 2.0,       # 1.0% - 2.0%
          extreme: 2.0      # > 2.0%
        }.freeze

        def initialize(instrument:, current_price:, previous_close:)
          @instrument = instrument
          @current_price = current_price.to_d
          @previous_close = previous_close.to_d
        end

        def call
          return { status: :unknown, gap_percent: 0 } if @previous_close.zero?

          gap_percent = ((@current_price - @previous_close) / @previous_close * 100).round(4)

          status = classify(gap_percent)

          {
            status: status,
            gap_percent: gap_percent,
            gap_points: (@current_price - @previous_close).round(4),
            direction: gap_direction(gap_percent),
            recommendation: recommendation(status)
          }
        end

        private

        def classify(gap_percent)
          abs_gap = gap_percent.abs

          if abs_gap >= GAP_THRESHOLDS[:extreme]
            :extreme
          elsif abs_gap >= GAP_THRESHOLDS[:large]
            :large
          elsif abs_gap >= GAP_THRESHOLDS[:moderate]
            :moderate
          else
            :normal
          end
        end

        def gap_direction(gap_percent)
          if gap_percent > 0
            :gap_up
          elsif gap_percent < 0
            :gap_down
          else
            :flat
          end
        end

        def recommendation(status)
          case status
          when :extreme then :no_trade
          when :large then :reduced_risk
          else :normal
          end
        end
      end
    end
  end
  ```
- **Deliverable:** Gap classification with direction
- **Acceptance Criteria:** `GapAnalyzer.new(instrument: nifty, current_price: 22700, previous_close: 22500).call` returns `:moderate` (0.89%)

### Task 1.3: Create `MarketOpenAnalyzer`
- **File:** `app/engines/market_context/market_open_analyzer.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketContext
      class MarketOpenAnalyzer
        def initialize(instrument:, current_time:)
          @instrument = instrument
          @current_time = current_time
        end

        def call
          market_open = market_open_time
          market_close = market_close_time

          return { status: :closed } unless market_hours?

          minutes_since_open = ((@current_time - market_open) / 60).round
          minutes_to_close = ((market_close - @current_time) / 60).round

          {
            status: :open,
            minutes_since_open: minutes_since_open,
            minutes_to_close: minutes_to_close,
            session_phase: session_phase(minutes_since_open, minutes_to_close),
            volatility_phase: volatility_phase(minutes_since_open)
          }
        end

        private

        def market_hours?
          calendar = ExchangeCalendar.for_exchange(exchange: exchange)
          return false unless calendar && calendar[:trading_day]

          @current_time.between?(
            Time.parse(calendar[:open].to_s),
            Time.parse(calendar[:close].to_s)
          )
        end

        def market_open_time
          Time.parse("#{Date.current} #{AppConfig.config.trading.market_open} IST").utc
        end

        def market_close_time
          Time.parse("#{Date.current} #{AppConfig.config.trading.market_close} IST").utc
        end

        def exchange
          @instrument.exchange_segment.start_with?("NSE") ? "NSE" : "BSE"
        end

        def session_phase(minutes_since_open, minutes_to_close)
          if minutes_since_open <= 15
            :opening
          elsif minutes_since_open <= 60
            :early
          elsif minutes_to_close <= 30
            :closing
          elsif minutes_to_close <= 60
            :late
          else
            :mid
          end
        end

        def volatility_phase(minutes_since_open)
          if minutes_since_open <= 15
            :high
          elsif minutes_since_open <= 30
            :elevated
          else
            :normal
          end
        end
      end
    end
  end
  ```
- **Deliverable:** Market session phase detection
- **Acceptance Criteria:** `MarketOpenAnalyzer.new(instrument: nifty, current_time: Time.parse("09:20")).call[:session_phase]` returns `:opening`

  # frozen_string_literal: true

  module Engines
    module MarketContext
      class IndiaVIXAnalyzer
        THRESHOLDS = {
          low: 15.0,
          moderate: 20.0,
          elevated: 25.0,
          high: 30.0
        }.freeze

        def initialize(current_vix:, previous_vix: nil)
          @current_vix = current_vix.to_d
          @previous_vix = previous_vix&.to_d
        end

        def call
          {
            current: @current_vix.round(2),
            previous: @previous_vix&.round(2),
            change: vix_change,
            change_percent: vix_change_percent,
            regime: regime,
            trend: trend,
            risk_adjustment: risk_adjustment
          }
        end

        private

        def vix_change
          return nil unless @previous_vix
          (@current_vix - @previous_vix).round(2)
        end

        def vix_change_percent
          return nil unless @previous_vix && @previous_vix > 0
          ((@current_vix - @previous_vix) / @previous_vix * 100).round(2)
        end

        def regime
          if @current_vix >= THRESHOLDS[:high]
            :extreme
          elsif @current_vix >= THRESHOLDS[:elevated]
            :high
          elsif @current_vix >= THRESHOLDS[:moderate]
            :elevated
          elsif @current_vix >= THRESHOLDS[:low]
            :moderate
          else
            :low
          end
        end

        def trend
          return :unknown unless @previous_vix
          change = @current_vix - @previous_vix

          if change > 2
            :spiking
          elsif change > 0.5
            :rising
          elsif change < -2
            :crashing
          elsif change < -0.5
            :falling
          else
            :stable
          end
        end

        def risk_adjustment
          case regime
          when :extreme then :no_trade
          when :high then :reduced_risk
          when :elevated then :reduced_risk
          else :normal
          end
        end
      end
    end
  end
  ```
- **Deliverable:** VIX regime and trend classification
- **Acceptance Criteria:** `IndiaVIXAnalyzer.new(current_vix: 28).call[:regime]` returns `:high`

  # frozen_string_literal: true

  module Engines
    module MarketContext
      class PreviousDayTrendAnalyzer
        def initialize(instrument:)
          @instrument = instrument
        end

        def call
          yesterday = yesterday_candles
          return { status: :no_data } if yesterday.empty?

          open = yesterday.first.open
          close = yesterday.last.close
          high = yesterday.maximum(:high)
          low = yesterday.minimum(:low)

          change = close.to_d - open.to_d
          change_percent = open.zero? ? 0 : (change / open * 100).round(4)

          range = high.to_d - low.to_d
          body = (close - open).abs

          {
            open: open,
            high: high,
            low: low,
            close: close,
            change: change.round(4),
            change_percent: change_percent,
            trend: trend(change),
            strength: strength(range, body),
            overnight_bias: overnight_bias(change_percent)
          }
        end

        private

        def yesterday_candles
          yesterday = Date.current - 1
          market_open = Time.parse("#{yesterday} #{AppConfig.config.trading.market_open} IST").utc
          market_close = Time.parse("#{yesterday} #{AppConfig.config.trading.market_close} IST").utc

          Candle.for_instrument(@instrument)
                .for_timeframe("5m")
                .where(timestamp: market_open..market_close)
                .where(is_complete: true)
                .order(timestamp: :asc)
        end

        def trend(change)
          if change > 0
            :bullish
          elsif change < 0
            :bearish
          else
            :neutral
          end
        end

        def strength(range, body)
          return :unknown if range.zero?
          ratio = body / range
          if ratio > 0.8
            :strong
          elsif ratio > 0.5
            :moderate
          else
            :weak
          end
        end

        def overnight_bias(change_percent)
          if change_percent > 1.5
            :strong_bullish_overnight
          elsif change_percent > 0.5
            :mild_bullish_overnight
          elsif change_percent < -1.5
            :strong_bearish_overnight
          elsif change_percent < -0.5
            :mild_bearish_overnight
          else
            :neutral
          end
        end
      end
    end
  end
  ```
- **Deliverable:** Previous day trend with overnight bias
- **Acceptance Criteria:** `PreviousDayTrendAnalyzer.new(instrument: nifty).call[:overnight_bias]` returns `:strong_bullish_overnight` for >1.5% up day

  # frozen_string_literal: true

  module Engines
    module MarketContext
      class ExpiryChecker
        def initialize(instrument:, today: Date.current)
          @instrument = instrument
          @today = today
        end

        def call
          next_expiry = next_expiry_date
          return { status: :unknown, days_to_expiry: nil } unless next_expiry

          days = (next_expiry - @today).to_i

          {
            status: classify(days),
            days_to_expiry: days,
            next_expiry: next_expiry,
            is_expiry_day: days == 0,
            is_week_before: days.between?(1, 7),
            time_decay_risk: time_decay_risk(days)
          }
        end

        private

        def next_expiry_date
          # Find next Thursday (standard weekly expiry for NIFTY/BANKNIFTY)
          # Or use instrument-specific expiry
          return @instrument.expiry if @instrument.expiry

          # Default: next Thursday
          date = @today
          date += 1 until date.thursday?
          date
        end

        def classify(days)
          if days == 0
            :expiry_day
          elsif days <= 2
            :very_near
          elsif days <= 5
            :near
          elsif days <= 10
            :moderate
          else
            :far
          end
        end

        def time_decay_risk(days)
          if days == 0
            :extreme
          elsif days <= 2
            :high
          elsif days <= 5
          :moderate
          else
            :low
          end
        end
      end
    end
  end
  ```
- **Deliverable:** Expiry proximity classification

  # frozen_string_literal: true

  class EventCalendar < BaseService
    HIGH_IMPACT_EVENTS = [
      { name: "RBI Policy", type: :rbi, months: [2, 4, 6, 8, 10, 12], day: :first_friday },
      { name: "Fed FOMC", type: :fed, months: [1, 3, 5, 6, 7, 9, 11, 12], day: :wednesday_week },
      { name: "Union Budget", type: :budget, month: 2, day: 1 },  # Approximate
      { name: "GDP Release", type: :gdp, months: [3, 6, 9, 12], day: :last_friday }
    ].freeze

    def initialize(date: Date.current)
      @date = date
    end

    def call
      events = today_events
      next_events = upcoming_events(7)

      success(
        today: events,
        upcoming: next_events,
        risk_level: risk_level(events),
        recommendation: recommendation(events)
      )
    end

    def self.today_risk_level(date: Date.current)
      new(date: date).call.value[:risk_level]
    end

    private

    def today_events
      HIGH_IMPACT_EVENTS.filter_map do |event|
        if event_matches?(event)
          {
            name: event[:name],
            type: event[:type],
            impact: :high
          }
        end
      end
    end

    def upcoming_events(days)
      (1..days).filter_map do |i|
        future_date = @date + i.days
        events = HIGH_IMPACT_EVENTS.filter { |e| event_matches?(e, date: future_date) }
        events.map { |e| { date: future_date, name: e[:name], type: e[:type] } }
      end.flatten
    end

    def event_matches?(event, date: @date)
      case event[:day]
      when :first_friday
        date.friday? && date.day <= 7 && event[:months].include?(date.month)
      when :wednesday_week
        # Second Wednesday of month
        date.wednesday? && date.day.between?(8, 14) && event[:months].include?(date.month)
      when Integer
        date.month == event[:month] && date.day == event[:day]
      else
        false
      end
    end

    def risk_level(events)
      return :low if events.empty?
      return :high if events.any? { |e| e[:type] == :rbi || e[:type] == :fed }
      :moderate
    end

    def recommendation(events)
      return :normal if events.empty?
      return :no_trade if events.any? { |e| e[:type] == :rbi || e[:type] == :fed }
      :reduced_risk
    end
  end
  ```
- **Deliverable:** Event risk calendar with RBI/Fed detection
- **Acceptance Criteria:** `EventCalendar.new(date: Date.new(2026, 6, 5)).call` detects RBI policy day

### Task 4.1: Create `ContextScoreCalculator`
- **File:** `app/engines/market_context/context_score_calculator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketContext
      class ContextScoreCalculator
        WEIGHTS = {
          gap: 20,
          vix: 20,
          expiry: 15,
          events: 20,
          previous_trend: 10,
          session: 10,
          time_of_day: 5
        }.freeze

        def initialize(checks:)
          @checks = checks
        end

        def call
          scores = {
            gap: gap_score,
            vix: vix_score,
            expiry: expiry_score,
            events: events_score,
            previous_trend: previous_trend_score,
            session: session_score,
            time_of_day: time_of_day_score
          }

          total = scores.sum { |k, v| v * WEIGHTS[k] } / 100.0

          {
            total: total.round(2),
            components: scores,
            context: determine_context(total),
            risk_multiplier: risk_multiplier(total)
          }
        end

        private

        def gap_score
          case @checks[:gap][:status]
          when :normal then 100
          when :moderate then 70
          when :large then 40
          when :extreme then 0
          else 50
          end
        end

        def vix_score
          case @checks[:vix][:regime]
          when :low then 100
          when :moderate then 80
          when :elevated then 50
          when :high then 20
          when :extreme then 0
          else 50
          end
        end

        def expiry_score
          case @checks[:expiry][:status]
          when :far then 100
          when :moderate then 80
          when :near then 50
          when :very_near then 20
          when :expiry_day then 0
          else 50
          end
        end

        def events_score
          case @checks[:events][:risk_level]
          when :low then 100
          when :moderate then 60
          when :high then 0
          else 50
          end
        end

        def previous_trend_score
          return 50 unless @checks[:previous_trend]
          case @checks[:previous_trend][:trend]
          when :bullish then 80
          when :bearish then 80
          when :neutral then 60
          else 50
          end
        end

        def session_score
          return 0 unless @checks[:session][:status] == :open
          100
        end

        def time_of_day_score
          return 0 unless @checks[:time_of_day]
          case @checks[:time_of_day][:session_phase]
          when :opening then 70
          when :early then 90
          when :mid then 100
          when :late then 80
          when :closing then 60
          else 50
          end
        end

        def determine_context(total)
          if total >= 80
            :allow_trading
          elsif total >= 50
            :reduced_risk
          else
            :no_trade
          end
        end

        def risk_multiplier(total)
          case determine_context(total)
          when :allow_trading then 1.0
          when :reduced_risk then 0.5
          when :no_trade then 0.0
          end
        end
      end
    end
  end
  ```
- **Deliverable:** Weighted 0-100 context score with risk multiplier
- **Acceptance Criteria:** `ContextScoreCalculator.new(checks: checks).call[:total]` returns 0-100

## Day 5 — Context Validation & Risk Adjustment

### Task 5.1: Create `ContextValidationService`
- **File:** `app/services/context_validation_service.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class ContextValidationService < BaseService
    def initialize(context_result:)
      @context = context_result
    end

    def call
      return failure(:invalid_context) unless @context.is_a?(Hash)

      errors = []

      errors << :missing_context unless @context[:context].present?
      errors << :missing_score unless @context[:score].present?
      errors << :invalid_score unless @context[:score].is_a?(Numeric) && @context[:score].between?(0, 100)
      errors << :missing_checks unless @context[:checks].is_a?(Hash)
      errors << :stale_data if stale?

      valid = errors.empty?

      success(
        valid: valid,
        errors: errors,
        context: @context[:context],
        score: @context[:score],
        can_trade: can_trade?,
        risk_multiplier: risk_multiplier
      )
    end

    def can_trade?
      @context[:context] == :allow_trading || @context[:context] == :reduced_risk
    end

    def risk_multiplier
      @context[:score].to_f / 100.0
    end

    private

    def stale?
      timestamp = @context[:timestamp]
      return true unless timestamp
      Time.current - Time.parse(timestamp.to_s) > 5.minutes
    end
  end
  ```
- **Deliverable:** Context result validation
- **Acceptance Criteria:** `ContextValidationService.new(context_result: result).call[:can_trade?]` returns boolean

### Task 5.2: Create `RiskAdjustmentService`
- **File:** `app/services/risk_adjustment_service.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class RiskAdjustmentService < BaseService
    def initialize(context_result:, base_risk:)
      @context = context_result
      @base_risk = base_risk
    end

    def call
      multiplier = @context[:score].to_f / 100.0

      adjusted = {
        position_size_multiplier: multiplier,
        max_open_positions: (@base_risk[:max_open_positions] * multiplier).to_i,
        daily_loss_limit: (@base_risk[:daily_loss_limit] * multiplier).to_i,
        per_trade_risk_percent: @base_risk[:per_trade_risk_percent] * multiplier,
        stop_loss_multiplier: stop_loss_multiplier,
        allow_overnight: @context[:context] != :reduced_risk,
        allow_expiry_day: @context[:checks][:expiry][:status] != :expiry_day
      }

      success(adjusted)
    end

    private

    def stop_loss_multiplier
      case @context[:context]
      when :allow_trading then 1.0
      when :reduced_risk then 1.5
      when :no_trade then 0.0
      else 1.0
      end
    end
  end
  ```
- **Deliverable:** Dynamic risk parameter adjustment based on context
- **Acceptance Criteria:** `RiskAdjustmentService.new(context_result: ctx, base_risk: { daily_loss_limit: 50000 }).call` halves limits in reduced risk

### Task 5.3: Create `ContextGate`
- **File:** `app/services/context_gate.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class ContextGate < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      cached = MarketContextCache.read(@instrument.id)

      if cached && fresh?(cached)
        result = cached
      else
        engine = Engines::MarketContextEngine.new(instrument: @instrument)
        result = engine.analyze.value
        MarketContextCache.write(@instrument.id, result)
      end

      validator = ContextValidationService.new(context_result: result)
      validation = validator.call

      return failure(:context_blocked, validation.value) unless validation.value[:can_trade]

      success(
        context: result[:context],
        score: result[:score],
        risk_multiplier: validation.value[:risk_multiplier],
        checks: result[:checks]
      )
    end

    def pass?
      call.success?
    end

    private

    def fresh?(cached)
      cached[:timestamp] && Time.current - Time.parse(cached[:timestamp].to_s) < 2.minutes
    end
  end
  ```
- **Deliverable:** Trading gate that blocks on bad context
- **Acceptance Criteria:** `ContextGate.new(instrument: nifty).pass?` returns `false` when VIX extreme
