```ruby
module AiGateway
  class AIGateway < BaseService
    DEFAULT_TIMEOUT = 30
    MAX_RETRIES = 3

    def initialize
      @provider_pool = ProviderPool.new
      @router = Router.new(pool: @provider_pool)
      @failover = AutomaticFailover.new(pool: @provider_pool)
      @metrics = AIGatewayMetrics.new
      @timeout = RequestTimeout.new(default: DEFAULT_TIMEOUT)
    end

    # Main entry point for all AI requests
    def call(prompt:, model: default_model, **options)
      start_time = Time.current

      # Route to best provider
      provider = @router.select(model: model)
      return Result.failure(:no_provider_available) unless provider

      # Execute with timeout and failover
      result = @timeout.wrap do
        execute_with_failover(provider, prompt, model, options)
      end

      # Record metrics
      duration = ((Time.current - start_time) * 1000).round(2)
      @metrics.record_request(
        provider: provider.name,
        model: model,
        latency: duration,
        success: result.success?
      )

      # Emit event
      emit(EventTypes::AI_REQUEST_COMPLETED, {
        provider: provider.name,
        model: model,
        latency: duration,
        success: result.success?
      })

      result
    rescue => e
      @metrics.record_request(
        provider: provider&.name || "unknown",
        model: model,
        latency: ((Time.current - start_time) * 1000).round(2),
        success: false
      )
      Result.failure(AIError.new(message: e.message, provider: provider&.name))
    end

    def health
      @provider_pool.health_summary
    end

    private

    def execute_with_failover(provider, prompt, model, options)
      attempts = 0
      current_provider = provider

      loop do
        attempts += 1
        result = current_provider.generate(prompt: prompt, model: model, **options)

        if result.success?
          return result
        elsif failover_eligible?(result) && attempts < MAX_RETRIES
          current_provider = @failover.next_provider(current_provider)
          return Result.failure(:failover_exhausted) unless current_provider
        else
          return result
        end
      end
    end

    def failover_eligible?(result)
      return false unless result.failure?
      error = result.error
      error.is_a?(HTTPError) && [429, 500, 502, 503, 504].include?(error.code)
    end

    def default_model
      AppConfig.config.ai.default_model || "llama3.2"
    end
  end
end
```

```ruby
module AiGateway
  class ProviderPool
    TIERS = %i[primary secondary emergency local].freeze

    def initialize
      @providers = Hash.new { |h, k| h[k] = [] }
      @health_monitor = HealthMonitor.new
      load_providers
    end

    def register(provider, tier: :primary)
      raise ArgumentError, "Invalid tier: #{tier}" unless TIERS.include?(tier)
      @providers[tier] << provider
      @health_monitor.register(provider)
    end

    def providers(tier: nil)
      return @providers.values.flatten if tier.nil?
      @providers[tier] || []
    end

    def available(tier: nil)
      candidates = tier ? @providers[tier] : @providers.values.flatten
      candidates.select(&:available?)
    end

    def best_available
      TIERS.each do |tier|
        candidates = available(tier: tier)
        return candidates.min_by(&:latency) if candidates.any?
      end
      nil
    end

    def health_summary
      @providers.values.flatten.to_h do |p|
        [p.name, {
          tier: p.tier,
          available: p.available?,
          latency: p.latency,
          failures: p.consecutive_failures,
          cooldown_until: p.cooldown_until
        }]
      end
    end

    private

    def load_providers
      # Load from configuration
      if AppConfig.config.ai.ollama_cloud_enabled
        register(
          Providers::OllamaCloudProvider.new(
            endpoint: AppConfig.config.ai.ollama_cloud_endpoint,
            api_key: AppConfig.config.ai.ollama_cloud_key
          ),
          tier: :primary
        )
      end

      if AppConfig.config.ai.ollama_local_enabled
        register(
          Providers::OllamaLocalProvider.new(
            endpoint: AppConfig.config.ai.ollama_local_endpoint || "http://localhost:11434"
          ),
          tier: :local
        )
      end
    end
  end
end
```

```ruby
  module Providers
    class BaseProvider
      attr_reader :name, :tier, :endpoint

      def initialize(endpoint:, name:, tier: :primary)
        @endpoint = endpoint
        @name = name
        @tier = tier
        @latency_tracker = LatencyTracker.new(provider: name)
        @failure_tracker = FailureTracker.new(provider: name)
        @cooldown_manager = CooldownManager.new(provider: name)
        @rate_limit_tracker = RateLimitTracker.new(provider: name)
      end

      def generate(prompt:, model:, **options)
        return Result.failure(:provider_on_cooldown) if on_cooldown?
        return Result.failure(:rate_limit_exceeded) if rate_limited?

        start_time = Time.current
        response = execute_request(prompt, model, options)
        duration = ((Time.current - start_time) * 1000).round(2)

        @latency_tracker.record(duration)
        @failure_tracker.record_success
        @rate_limit_tracker.record_request

        Result.success(parse_response(response))
      rescue => e
        @failure_tracker.record_failure
        handle_error(e)
      end

      def available?
        !on_cooldown? && !rate_limited? && healthy?
      end

      def latency
        @latency_tracker.average
      end

      def consecutive_failures
        @failure_tracker.consecutive_count
      end

      def cooldown_until
        @cooldown_manager.cooldown_until
      end

      private

      def execute_request(prompt, model, options)
        raise NotImplementedError
      end

      def parse_response(response)
        raise NotImplementedError
      end

      def handle_error(error)
        if error.is_a?(Faraday::Error)
          code = error.response&.fetch(:status, 500) || 500
          Result.failure(HTTPError.new(code: code, message: error.message, provider: name))
        else
          Result.failure(ProviderError.new(message: error.message, provider: name))
        end
      end

      def on_cooldown?
        @cooldown_manager.on_cooldown?
      end

      def rate_limited?
        @rate_limit_tracker.limited?
      end

      def healthy?
        @failure_tracker.consecutive_count < 5
      end
    end
  end
end
```

```ruby
module AiGateway
  class KeyManager
    KEY_PREFIX = "ai_gateway:keys".freeze

    def initialize
      @redis = RedisPool.general
    end

    def register_key(provider_name, key:, tier: :primary, daily_limit: 10_000, minute_limit: 100)
      key_data = {
        key: key,
        tier: tier,
        daily_limit: daily_limit,
        minute_limit: minute_limit,
        created_at: Time.current.iso8601,
        active: true
      }
      @redis.hset("#{KEY_PREFIX}:#{provider_name}", key, key_data.to_json)
    end

    def get_key(provider_name)
      keys = all_keys(provider_name).select { |_, data| data[:active] }
      return nil if keys.empty?

      # Select key with lowest usage in current window
      keys.min_by { |key, data| daily_usage(provider_name, key) }&.first
    end

    def increment_usage(provider_name, key)
      minute_key = "#{KEY_PREFIX}:#{provider_name}:#{key}:minute:#{Time.current.strftime('%Y%m%d%H%M')}"
      daily_key = "#{KEY_PREFIX}:#{provider_name}:#{key}:daily:#{Time.current.strftime('%Y%m%d')}"

      @redis.multi do |pipeline|
        pipeline.incr(minute_key)
        pipeline.expire(minute_key, 120)
        pipeline.incr(daily_key)
        pipeline.expire(daily_key, 86_400)
      end
    end

    def deactivate_key(provider_name, key)
      data = key_data(provider_name, key)
      return unless data

      data[:active] = false
      @redis.hset("#{KEY_PREFIX}:#{provider_name}", key, data.to_json)
    end

    def usage_stats(provider_name, key)
      {
        minute: minute_usage(provider_name, key),
        daily: daily_usage(provider_name, key),
        limit_minute: key_data(provider_name, key)&.dig(:minute_limit),
        limit_daily: key_data(provider_name, key)&.dig(:daily_limit)
      }
    end

    private

    def all_keys(provider_name)
      @redis.hgetall("#{KEY_PREFIX}:#{provider_name}").transform_values do |json|
        JSON.parse(json, symbolize_names: true)
      end
    end

    def key_data(provider_name, key)
      json = @redis.hget("#{KEY_PREFIX}:#{provider_name}", key)
      JSON.parse(json, symbolize_names: true) if json
    end

    def minute_usage(provider_name, key)
      minute_key = "#{KEY_PREFIX}:#{provider_name}:#{key}:minute:#{Time.current.strftime('%Y%m%d%H%M')}"
      @redis.get(minute_key).to_i
    end

    def daily_usage(provider_name, key)
      daily_key = "#{KEY_PREFIX}:#{provider_name}:#{key}:daily:#{Time.current.strftime('%Y%m%d')}"
      @redis.get(daily_key).to_i
    end
  end
end
```

```ruby
module AiGateway
  class LatencyTracker
    WINDOW_SIZE = 50
    PREFIX = "ai_gateway:latency".freeze

    def initialize(provider:)
      @provider = provider
      @redis = RedisPool.general
    end

    def record(latency_ms)
      key = "#{PREFIX}:#{@provider}"
      @redis.lpush(key, latency_ms)
      @redis.ltrim(key, 0, WINDOW_SIZE - 1)
      @redis.expire(key, 86_400)
    end

    def average
      values = recent
      return 0 if values.empty?
      (values.sum.to_f / values.length).round(2)
    end

    def percentile(p = 95)
      values = recent.sort
      return 0 if values.empty?
      index = (values.length * (p / 100.0)).ceil - 1
      values[[index, 0].max]
    end

    def recent
      key = "#{PREFIX}:#{@provider}"
      @redis.lrange(key, 0, WINDOW_SIZE - 1).map(&:to_f)
    end
  end
end
```

```ruby
module AiGateway
  class FailureTracker
    MAX_CONSECUTIVE = 5
    PREFIX = "ai_gateway:failures".freeze

    def initialize(provider:)
      @provider = provider
      @redis = RedisPool.general
    end

    def record_success

    def record_failure
      key = "#{PREFIX}:#{@provider}:consecutive"
      count = @redis.incr(key)
      @redis.expire(key, 3_600)

      # Trigger cooldown if threshold reached
      if count >= MAX_CONSECUTIVE

      end

      count
    end

    def consecutive_count
      @redis.get("#{PREFIX}:#{@provider}:consecutive").to_i
    end

    def healthy?
      consecutive_count < MAX_CONSECUTIVE
    end
  end
end
```

```ruby
module AiGateway
  class CooldownManager
    DEFAULT_DURATION = 300  # 5 minutes
    PREFIX = "ai_gateway:cooldown".freeze

    def initialize(provider:)
      @provider = provider
      @redis = RedisPool.general
    end

    def trigger_cooldown(duration: DEFAULT_DURATION)
      key = "#{PREFIX}:#{@provider}"
      @redis.setex(key, duration, Time.current.iso8601)
      Rails.logger.warn "[AIGateway] Provider #{@provider} placed on cooldown for #{duration}s"
    end

    def on_cooldown?
      @redis.exists?("#{PREFIX}:#{@provider}")
    end

    def cooldown_until
      ttl = @redis.ttl("#{PREFIX}:#{@provider}")
      return nil if ttl <= 0
      Time.current + ttl.seconds
    end

    def remaining_seconds
      @redis.ttl("#{PREFIX}:#{@provider}")
    end

    def clear
      @redis.del("#{PREFIX}:#{@provider}")
    end
  end
end
```

```ruby
      @timer = Concurrent::TimerTask.new(execution_interval: CHECK_INTERVAL) do
        check_all
      end
      @timer.execute

      @timer&.shutdown
```

```ruby
module AiGateway
  class Router
    def initialize(pool:)
      @pool = pool
    end

    # Select best provider based on:
    # 1. Tier priority (primary > secondary > emergency > local)
    # 2. Availability (not on cooldown, not rate limited)
    # 3. Lowest latency
    # 4. Fewest consecutive failures
    def select(model:)
      # Try each tier in priority order
      ProviderPool::TIERS.each do |tier|
        candidates = @pool.available(tier: tier)
        next if candidates.empty?

        # Score candidates: lower is better
        scored = candidates.map do |p|
          score = calculate_score(p)
          [p, score]
        end

        return scored.min_by { |_, score| score }&.first
      end

      nil
    end

    private

    def calculate_score(provider)
      latency_weight = 0.5
      failure_weight = 0.3
      load_weight = 0.2

      latency_score = provider.latency
      failure_score = provider.consecutive_failures * 1000
      load_score = provider.instance_variable_get(:@rate_limit_tracker)&.stats&.dig(:minute).to_i * 10

      (latency_score * latency_weight) +
        (failure_score * failure_weight) +
        (load_score * load_weight)
    end
  end
end
```

```ruby
module AiGateway
  class AutomaticFailover
    RETRYABLE_CODES = [429, 500, 502, 503, 504].freeze

    def initialize(pool:)
      @pool = pool
    end

    def next_provider(current_provider)
      current_tier = current_provider.tier
      tier_index = ProviderPool::TIERS.index(current_tier)

      # Try next tier
      ((tier_index + 1)...ProviderPool::TIERS.length).each do |i|
        tier = ProviderPool::TIERS[i]
        candidates = @pool.available(tier: tier)
        return candidates.min_by(&:latency) if candidates.any?
      end

      # Try same tier, different provider
      same_tier = @pool.available(tier: current_tier).reject { |p| p.name == current_provider.name }
      return same_tier.min_by(&:latency) if same_tier.any?

      nil
    end

    def should_failover?(error)
      return false unless error.is_a?(HTTPError)
      RETRYABLE_CODES.include?(error.code)
    end
  end
end
```

```ruby
    def initialize(default: 30)

    end

      Concurrent::Promise.execute do
        yield
      end.then { |result| result }.rescue { |e| Result.failure(e) }.value(timeout)

      Result.failure(TimeoutError.new("Request timed out after #{timeout}s"))
```

```ruby
module AiGateway
  class AIGatewayMetrics
    PREFIX = "ai_gateway:metrics".freeze

    def initialize
      @redis = RedisPool.general
    end

    def record_request(provider:, model:, latency:, success:)
      minute = Time.current.strftime('%Y%m%d%H%M')

      # Latency histogram
      @redis.lpush("#{PREFIX}:latency:#{provider}:#{minute}", latency)
      @redis.expire("#{PREFIX}:latency:#{provider}:#{minute}", 3_600)

      # Success/failure counters
      if success
        @redis.incr("#{PREFIX}:success:#{provider}:#{minute}")
      else
        @redis.incr("#{PREFIX}:failure:#{provider}:#{minute}")
      end
      @redis.expire("#{PREFIX}:success:#{provider}:#{minute}", 3_600)
      @redis.expire("#{PREFIX}:failure:#{provider}:#{minute}", 3_600)

      # Model usage
      @redis.hincrby("#{PREFIX}:model_usage:#{minute}", model, 1)
      @redis.expire("#{PREFIX}:model_usage:#{minute}", 3_600)
    end

    def summary(period: 1.hour)
      cutoff = (Time.current - period).strftime('%Y%m%d%H%M')
      now = Time.current.strftime('%Y%m%d%H%M')

      providers = @redis.hkeys("ai_gateway:health").uniq

      providers.map do |provider|
        {
          provider: provider,
          total_requests: total_requests(provider, cutoff, now),
          success_rate: success_rate(provider, cutoff, now),
          avg_latency: avg_latency(provider, cutoff, now),
          p95_latency: p95_latency(provider, cutoff, now)
        }
      end
    end

    private

    def total_requests(provider, cutoff, now)
      success = count_range("#{PREFIX}:success:#{provider}", cutoff, now)
      failure = count_range("#{PREFIX}:failure:#{provider}", cutoff, now)
      success + failure
    end

    def success_rate(provider, cutoff, now)
      success = count_range("#{PREFIX}:success:#{provider}", cutoff, now)
      total = total_requests(provider, cutoff, now)
      return 0 if total.zero?
      (success.to_f / total * 100).round(2)
    end

    def avg_latency(provider, cutoff, now)
      latencies = fetch_latencies(provider, cutoff, now)
      return 0 if latencies.empty?
      (latencies.sum / latencies.length).round(2)
    end

    def p95_latency(provider, cutoff, now)
      latencies = fetch_latencies(provider, cutoff, now).sort
      return 0 if latencies.empty?
      index = (latencies.length * 0.95).ceil - 1
      latencies[[index, 0].max]
    end

    def count_range(key_prefix, cutoff, now)
      # Simplified: scan keys in range
      keys = @redis.keys("#{key_prefix}:*").select do |k|
        minute = k.split(':').last.to_i
        minute >= cutoff.to_i && minute <= now.to_i
      end

      keys.sum { |k| @redis.get(k).to_i }
    end

    def fetch_latencies(provider, cutoff, now)
      keys = @redis.keys("#{PREFIX}:latency:#{provider}:*").select do |k|
        minute = k.split(':').last.to_i
        minute >= cutoff.to_i && minute <= now.to_i
      end

      keys.flat_map { |k| @redis.lrange(k, 0, -1).map(&:to_f) }
    end
  end
end
```

```ruby
module Api
  module V1
    class AIGatewayController < ApplicationController
      def generate
        prompt = params[:prompt]
        model = params[:model]
        options = params[:options] || {}

        return render json: { error: 'Prompt required' }, status: :unprocessable_entity if prompt.blank?

        gateway = AiGateway::AIGateway.new
        result = gateway.call(prompt: prompt, model: model, **options.symbolize_keys)

        if result.success?
          render json: {
            response: result.value[:text],
            model: result.value[:model],
            provider: result.value[:provider],
            metadata: result.value.slice(:total_duration, :eval_count)
          }
        else
          render json: { error: result.error.to_h }, status: :service_unavailable
        end
      end

      def health
        gateway = AiGateway::AIGateway.new
        render json: {
          providers: gateway.health,
          timestamp: Time.current.iso8601
        }
      end

      def metrics
        metrics = AiGateway::AIGatewayMetrics.new
        render json: {
          summary: metrics.summary(period: params[:period]&.to_i&.hours || 1.hour)
        }
      end
    end
  end
end
```
