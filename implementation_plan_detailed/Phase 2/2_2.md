## Day 1 — Candle Builder Core (1-Minute)

### Task 1.1: Create `CandleBuilderEngine`
- **File:** `app/engines/candle_builder_engine.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    class CandleBuilderEngine < BaseEngine
      TIMEFRAMES = {
        "1m" => 60,
        "5m" => 300,
        "15m" => 900,
        "30m" => 1800
      }.freeze

      def initialize(instrument:, timeframe:)
        super()
        @instrument = instrument
        @timeframe = timeframe
        @seconds = TIMEFRAMES[timeframe]
        raise ArgumentError, "Invalid timeframe: #{timeframe}" unless @seconds
      end

      def build_from_ticks(ticks)
        return success([]) if ticks.empty?

        candles = aggregate(ticks)
        persist(candles)
        success(candles)
      end

      def build_incremental(tick)
        bucket = bucket_time(tick[:timestamp])
        candle = find_or_initialize(bucket)

        if candle.new_record?
          initialize_candle(candle, tick)
        else
          update_candle(candle, tick)
        end

        candle.save!
        success(candle)
      end

      def close_bucket!(timestamp)
        bucket = bucket_time(timestamp)
        candle = Candle.for_instrument(@instrument)
                      .for_timeframe(@timeframe)
                      .find_by(timestamp: bucket)

        return failure(:no_candle) unless candle

        candle.update!(is_complete: true)
        emit(EventTypes::CANDLE_CLOSED, {
          instrument_id: @instrument.id,
          timeframe: @timeframe,
          timestamp: bucket,
          candle: candle.attributes
        })

        success(candle)
      end

      private

      def aggregate(ticks)
        ticks.group_by { |t| bucket_time(t[:timestamp]) }.map do |bucket, group|
          {
            instrument_id: @instrument.id,
            timeframe: @timeframe,
            timestamp: bucket,
            open: group.first[:last_price],
            high: group.map { |t| t[:last_price] }.compact.max,
            low: group.map { |t| t[:last_price] }.compact.min,
            close: group.last[:last_price],
            volume: group.sum { |t| t[:volume].to_i },
            oi: group.last[:oi],
            vwap: calculate_vwap(group),
            is_complete: true
          }
        end
      end

      def persist(candles)
        return 0 if candles.empty?

        Candle.upsert_all(
          candles,
          unique_by: [:instrument_id, :timeframe, :timestamp],
          returning: false
        )

        candles.length
      end

      def find_or_initialize(bucket)
        Candle.for_instrument(@instrument)
              .for_timeframe(@timeframe)
              .find_by(timestamp: bucket) ||
          Candle.new(
            instrument: @instrument,
            timeframe: @timeframe,
            timestamp: bucket
          )
      end

      def initialize_candle(candle, tick)
        candle.assign_attributes(
          open: tick[:last_price],
          high: tick[:last_price],
          low: tick[:last_price],
          close: tick[:last_price],
          volume: tick[:volume].to_i,
          oi: tick[:oi],
          vwap: tick[:last_price],
          is_complete: false
        )
      end

      def update_candle(candle, tick)
        candle.assign_attributes(
          high: [candle.high, tick[:last_price]].compact.max,
          low: [candle.low, tick[:last_price]].compact.min,
          close: tick[:last_price],
          volume: candle.volume + tick[:volume].to_i,
          oi: tick[:oi],
          vwap: recalculate_vwap(candle, tick)
        )
      end

      def bucket_time(timestamp)
        epoch = timestamp.to_i
        Time.at((epoch / @seconds) * @seconds).utc
      end

      def calculate_vwap(ticks)
        total_value = ticks.sum { |t| t[:last_price].to_f * t[:volume].to_i }
        total_volume = ticks.sum { |t| t[:volume].to_i }
        total_volume.zero? ? nil : (total_value / total_volume).round(4)
      end

      def recalculate_vwap(candle, tick)
        old_value = candle.vwap.to_f * candle.volume
        new_value = tick[:last_price].to_f * tick[:volume].to_i
        total_volume = candle.volume + tick[:volume].to_i
        total_volume.zero? ? candle.vwap : ((old_value + new_value) / total_volume).round(4)
      end
    end
  end
  ```
- **Deliverable:** Core candle builder with incremental and batch modes
- **Acceptance Criteria:** 60 ticks → 1 complete 1m candle with correct OHLCV

### Task 1.2: Create `CandleBuildCoordinator`
- **File:** `app/services/candle_build_coordinator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class CandleBuildCoordinator < BaseService
    TIMEFRAMES = %w[1m 5m 15m 30m].freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def process_tick(tick)
      results = {}

      TIMEFRAMES.each do |tf|
        builder = Engines::CandleBuilderEngine.new(
          instrument: @instrument,
          timeframe: tf
        )

        result = builder.build_incremental(tick)
        results[tf] = result

        # Check if bucket is complete for this timeframe
        if bucket_complete?(tf, tick[:timestamp])
          close_result = builder.close_bucket!(tick[:timestamp])
          results["#{tf}_closed"] = close_result
        end
      end

      success(results)
    end

    def rebuild_range(from:, to:)
      # Batch rebuild for historical backfill
      TIMEFRAMES.each do |tf|
        ticks = TickRepository.for_range(
          instrument: @instrument,
          from: from,
          to: to
        ).value

        builder = Engines::CandleBuilderEngine.new(
          instrument: @instrument,
          timeframe: tf
        )

        builder.build_from_ticks(ticks)
      end

      success(rebuilt: true)
    end

    private

    def bucket_complete?(timeframe, timestamp)
      seconds = Engines::CandleBuilderEngine::TIMEFRAMES[timeframe]
      bucket = Time.at((timestamp.to_i / seconds) * seconds).utc
      next_bucket = bucket + seconds.seconds
      Time.current >= next_bucket
    end
  end
  ```
- **Deliverable:** Coordinates all timeframes from a single tick
- **Acceptance Criteria:** One tick update triggers incremental builds across 1m/5m/15m/30m

### Task 1.3: Create `CandleBuildJob` (Refactored)
- **File:** `app/jobs/candle_build_job.rb` (update)
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class CandleBuildJob < JobBase
    def perform(instrument_id, tick_data = nil)
      instrument = Instrument.find(instrument_id)
      return unless instrument

      coordinator = CandleBuildCoordinator.new(instrument: instrument)

      if tick_data
        # Real-time incremental build
        coordinator.process_tick(tick_data.symbolize_keys)
      else
        # Scheduled close check for all timeframes
        close_open_buckets(instrument)
      end
    end

    private

    def close_open_buckets(instrument)
      %w[1m 5m 15m 30m].each do |tf|
        open_candle = Candle.for_instrument(instrument)
                           .for_timeframe(tf)
                           .where(is_complete: false)
                           .order(timestamp: :desc)
                           .first

        next unless open_candle

        seconds = Engines::CandleBuilderEngine::TIMEFRAMES[tf]
        next_bucket = open_candle.timestamp + seconds.seconds

        if Time.current >= next_bucket
          builder = Engines::CandleBuilderEngine.new(
            instrument: instrument,
            timeframe: tf
          )
          builder.close_bucket!(open_candle.timestamp)
        end
      end
    end
  end
  ```
- **Deliverable:** Background job for candle building and closing
- **Acceptance Criteria:** Incomplete candles auto-close when next bucket starts

### Task 2.1: Create `VWAPCalculator`
- **File:** `app/lib/calculations/vwap_calculator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Calculations
    class VWAPCalculator
      def self.calculate(ticks)
        return nil if ticks.empty?

        total_value = ticks.sum { |t| t[:last_price].to_f * t[:volume].to_i }
        total_volume = ticks.sum { |t| t[:volume].to_i }

        total_volume.zero? ? nil : (total_value / total_volume).round(4)
      end

      def self.rolling(candles, period: 20)
        return [] if candles.length < period

        candles.each_cons(period).map do |window|
          total_value = window.sum { |c| c.vwap.to_f * c.volume.to_i }
          total_volume = window.sum(&:volume)
          total_volume.zero? ? nil : (total_value / total_volume).round(4)
        end
      end

      def self.std_dev(candles, vwap)
        return 0 if candles.length < 2

        prices = candles.map(&:close)
        mean = prices.sum.to_f / prices.length
        variance = prices.sum { |p| (p - mean) ** 2 } / prices.length
        Math.sqrt(variance).round(4)
      end

      def self.bands(candles, period: 20, multiplier: 1.0)
        vwaps = rolling(candles, period: period)
        return [] if vwaps.empty?

        std_devs = candles.each_cons(period).map do |window|
          std_dev(window, vwaps.first)
        end

        vwaps.zip(std_devs).map do |vwap, sd|
          {
            vwap: vwap,
            upper: vwap + (multiplier * sd),
            lower: vwap - (multiplier * sd)
          }
        end
      end
    end
  end
  ```
- **Deliverable:** VWAP with rolling bands (standard deviation)
- **Acceptance Criteria:** `VWAPCalculator.bands(candles, period: 20)` returns upper/lower bands

  ```ruby
  # frozen_string_literal: true

  class CandleGapDetector < BaseService
    GAP_THRESHOLD_SECONDS = 5  # Allow 5s drift

    def initialize(instrument:, timeframe:)
      @instrument = instrument
      @timeframe = timeframe
      @seconds = Engines::CandleBuilderEngine::TIMEFRAMES[timeframe]
    end

    def call(from:, to:)
      expected = expected_buckets(from, to)
      actual = actual_buckets(from, to)

      gaps = expected - actual

      success(
        instrument: @instrument.trading_symbol,
        timeframe: @timeframe,
        expected_count: expected.length,
        actual_count: actual.length,
        gap_count: gaps.length,
        gaps: gaps.map(&:iso8601),
        coverage_percent: coverage_percent(expected.length, actual.length),
        first_gap: gaps.first,
        last_gap: gaps.last
      )
    end

    def has_gaps?(from:, to:)
      result = call(from: from, to: to)
      result.success? && result.value[:gap_count] > 0
    end

    private

    def expected_buckets(from, to)
      buckets = []
      current = align_bucket(from)
      while current <= to
        buckets << current
        current += @seconds.seconds
      end
      buckets
    end

    def actual_buckets(from, to)
      Candle.for_instrument(@instrument)
          .for_timeframe(@timeframe)
          .where(timestamp: from..to)
          .where(is_complete: true)
          .pluck(:timestamp)
          .map(&:utc)
    end

    def align_bucket(time)
      epoch = time.to_i
      Time.at((epoch / @seconds) * @seconds).utc
    end

    def coverage_percent(expected, actual)
      return 0 if expected.zero?
      (actual.to_f / expected * 100).round(2)
    end
  end
  ```
- **Deliverable:** Precise gap detection with coverage metrics
- **Acceptance Criteria:** Missing 5m interval detected with exact gap timestamps

### Task 3.2: Create `CandleGapFiller` (Production)
- **File:** `app/services/candle_gap_filler.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class CandleGapFiller < BaseService
    def initialize(instrument:, timeframe:)
      @instrument = instrument
      @timeframe = timeframe
    end

    def call(from:, to:)
      detector = CandleGapDetector.new(instrument: @instrument, timeframe: @timeframe)
      gap_result = detector.call(from: from, to: to)

      return gap_result if gap_result.failure?
      return success(filled: 0) if gap_result.value[:gap_count].zero?

      gaps = gap_result.value[:gaps].map { |t| Time.parse(t) }
      filled_count = 0

      gaps.each do |gap_time|
        fill_result = fill_single_gap(gap_time)
        filled_count += 1 if fill_result.success?
      end

      success(
        filled: filled_count,
        gaps_found: gaps.length,
        instrument: @instrument.trading_symbol,
        timeframe: @timeframe
      )
    end

    private

    def fill_single_gap(gap_time)
      # Strategy 1: Try to build from raw ticks in buffer/DB
      tick_result = build_from_ticks(gap_time)
      return tick_result if tick_result.success?

      # Strategy 2: Fetch from DhanHQ historical API
      api_result = fetch_from_api(gap_time)
      return api_result if api_result.success?

      # Strategy 3: Interpolate from adjacent candles
      interpolate_result = interpolate(gap_time)
      return interpolate_result if interpolate_result.success?

      failure(:unfillable_gap, { gap_time: gap_time })
    end

    def build_from_ticks(gap_time)
      seconds = Engines::CandleBuilderEngine::TIMEFRAMES[@timeframe]
      from = gap_time
      to = gap_time + seconds.seconds

      ticks = TickRepository.for_range(
        instrument: @instrument,
        from: from,
        to: to
      ).value

      return failure(:no_ticks) if ticks.empty?

      builder = Engines::CandleBuilderEngine.new(
        instrument: @instrument,
        timeframe: @timeframe
      )

      result = builder.build_from_ticks(ticks)
      success(filled: result.value.length) if result.success?
    end

    def fetch_from_api(gap_time)
      result = HistoricalDataService.new(instrument: @instrument).call(
        timeframe: @timeframe,
        from: gap_time,
        to: gap_time + Engines::CandleBuilderEngine::TIMEFRAMES[@timeframe].seconds
      )

      return result if result.failure?

      candles = result.value.map do |c|
        {
          instrument_id: @instrument.id,
          timeframe: @timeframe,
          timestamp: c[:timestamp],
          open: c[:open],
          high: c[:high],
          low: c[:low],
          close: c[:close],
          volume: c[:volume],
          oi: c[:oi],
          vwap: c[:vwap],
          is_complete: true
        }
      end

      Candle.upsert_all(candles, unique_by: [:instrument_id, :timeframe, :timestamp])
      success(filled: candles.length)
    end

    def interpolate(gap_time)
      prev = Candle.for_instrument(@instrument)
                  .for_timeframe(@timeframe)
                  .where("timestamp < ?", gap_time)
                  .order(timestamp: :desc)
                  .first

      next_c = Candle.for_instrument(@instrument)
                   .for_timeframe(@timeframe)
                   .where("timestamp > ?", gap_time)
                   .order(timestamp: :asc)
                   .first

      return failure(:no_adjacent_candles) unless prev && next_c

      # Linear interpolation
      Candle.create!(
        instrument: @instrument,
        timeframe: @timeframe,
        timestamp: gap_time,
        open: prev.close,
        high: [prev.close, next_c.open].max,
        low: [prev.close, next_c.open].min,
        close: ((prev.close + next_c.open) / 2).round(4),
        volume: 0,
        oi: prev.oi,
        vwap: ((prev.vwap.to_f + next_c.vwap.to_f) / 2).round(4),
        is_complete: true,
        interpolated: true
      )

      success(filled: 1, method: :interpolated)
    end
  end
  ```
- **Deliverable:** 3-strategy gap filling (ticks → API → interpolate)
- **Acceptance Criteria:** Missing candle filled from ticks first, API second, interpolation last

  class OpeningRangeCalculator < BaseService
    RANGES = {
      "15m" => 15.minutes,
      "30m" => 30.minutes,
      "60m" => 60.minutes
    }.freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def call(range_name: "15m", date: Date.current)
      duration = RANGES[range_name]
      return failure(:invalid_range) unless duration

      market_open = Time.parse("#{date} #{AppConfig.config.trading.market_open} IST")
      market_open = market_open.utc

      candles = Candle.for_instrument(@instrument)
                     .for_timeframe("1m")
                     .where(timestamp: market_open..(market_open + duration))
                     .where(is_complete: true)
                     .order(timestamp: :asc)

      return failure(:no_candles) if candles.empty?

      high = candles.maximum(:high)
      low = candles.minimum(:low)
      close = candles.last&.close

      success(
        range: range_name,
        date: date,
        market_open: market_open,
        high: high,
        low: low,
        range: high - low,
        close: close,
        breakout_high: close > high,
        breakdown_low: close < low,
        midpoint: ((high + low) / 2).round(4)
      )
    end

    def breakout_level(range_name: "15m", date: Date.current)
      result = call(range_name: range_name, date: date)
      return result if result.failure?

      success(
        high: result.value[:high],
        low: result.value[:low],
        midpoint: result.value[:midpoint]
      )
    end
  end
  ```
- **Deliverable:** Opening range (15m/30m/60m) calculation
- **Acceptance Criteria:** `OpeningRangeCalculator.new(instrument: nifty).call(range_name: "15m")` returns high/low/midpoint

  class PreviousHighLowTracker < BaseService
    PERIODS = {
      "day" => 1.day,
      "week" => 1.week,
      "month" => 1.month
    }.freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def call(period: "day", date: Date.current)
      duration = PERIODS[period]
      return failure(:invalid_period) unless duration

      from = date.to_time.utc - duration
      to = date.to_time.utc

      candles = Candle.for_instrument(@instrument)
                     .for_timeframe("5m")
                     .where(timestamp: from..to)
                     .where(is_complete: true)

      return failure(:no_candles) if candles.empty?

      high = candles.maximum(:high)
      low = candles.minimum(:low)
      high_time = candles.where(high: high).order(timestamp: :asc).first&.timestamp
      low_time = candles.where(low: low).order(timestamp: :asc).first&.timestamp

      success(
        period: period,
        high: high,
        high_time: high_time,
        low: low,
        low_time: low_time,
        range: high - low,
        pivot: ((high + low) / 2).round(4)
      )
    end

    def levels(date: Date.current)
      results = {}

      PERIODS.keys.each do |period|
        result = call(period: period, date: date)
        results[period] = result.success? ? result.value : nil
      end

      success(results)
    end
  end
  ```
- **Deliverable:** Daily/weekly/monthly high-low tracking
- **Acceptance Criteria:** `PreviousHighLowTracker.new(instrument: nifty).levels` returns all period highs/lows

    class OIChangeCalculator
      def self.calculate(current_oi:, previous_oi:)

        }

        return nil unless previous

        calculate(current_oi: candle.oi, previous_oi: previous.oi)
      end

      def self.oi_trend(instrument:, timeframe:, periods: 5)
        candles = Candle.for_instrument(instrument)
                       .for_timeframe(timeframe)

        return nil if ois.length < 2

        slope = Calculations::Statistics.slope(ois.reverse, ois.length)

          increasing: slope > 0,
          accelerating: slope > 0 && slope > (ois[-1] - ois[-2])
        }
      end

  # frozen_string_literal: true

  class HistoricalBackfillService < BaseService
    TIMEFRAMES = %w[1m 5m 15m 30m].freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def call(days: 30, force: false)
      to = Date.today
      from = to - days.days

      results = {}

      TIMEFRAMES.each do |tf|
        # Check existing coverage
        unless force
          coverage = gap_coverage(tf, from, to)
          if coverage >= 99
            results[tf] = { status: :skipped, coverage: coverage }
            next
          end
        end

        # Fetch from API
        api_result = fetch_and_persist(tf, from, to)
        results[tf] = api_result.success? ?
          { status: :filled, candles: api_result.value } :
          { status: :failed, error: api_result.error }

        # Fill any remaining gaps
        if api_result.success?
          filler = CandleGapFiller.new(instrument: @instrument, timeframe: tf)
          filler.call(from: from, to: to)
        end
      end

      success(results)
    end

    def backfill_missing_only
      to = Date.today
      from = to - 30.days

      results = {}

      TIMEFRAMES.each do |tf|
        detector = CandleGapDetector.new(instrument: @instrument, timeframe: tf)
        gap_result = detector.call(from: from, to: to)

        if gap_result.success? && gap_result.value[:gap_count] > 0
          filler = CandleGapFiller.new(instrument: @instrument, timeframe: tf)
          fill_result = filler.call(from: from, to: to)
          results[tf] = { gaps: gap_result.value[:gap_count], filled: fill_result.value[:filled] }
        else
          results[tf] = { gaps: 0, filled: 0 }
        end
      end

      success(results)
    end

    private

    def gap_coverage(timeframe, from, to)
      detector = CandleGapDetector.new(instrument: @instrument, timeframe: timeframe)
      result = detector.call(from: from, to: to)
      return 0 if result.failure?
      result.value[:coverage_percent]
    end

    def fetch_and_persist(timeframe, from, to)
      result = HistoricalDataService.new(instrument: @instrument).call(
        timeframe: timeframe,
        from: from,
        to: to
      )

      return result if result.failure?

      candles = result.value.map do |c|
        {
          instrument_id: @instrument.id,
          timeframe: timeframe,
          timestamp: c[:timestamp],
          open: c[:open],
          high: c[:high],
          low: c[:low],
          close: c[:close],
          volume: c[:volume],
          oi: c[:oi],
          vwap: c[:vwap] || ((c[:open] + c[:high] + c[:low] + c[:close]) / 4),
          is_complete: true
        }
      end

      Candle.upsert_all(candles, unique_by: [:instrument_id, :timeframe, :timestamp])
      success(candles.length)
    end
  end
  ```
- **Deliverable:** Automated historical backfill with coverage check
- **Acceptance Criteria:** `HistoricalBackfillService.new(instrument: nifty).call(days: 7)` fills all timeframes

      RULES = {
        high_low: ->(c) { c.high >= c.low },
        open_in_range: ->(c) { c.open >= c.low && c.open <= c.high },
        close_in_range: ->(c) { c.close >= c.low && c.close <= c.high },

          seconds = Engines::CandleBuilderEngine::TIMEFRAMES[c.timeframe]

          (epoch % seconds).zero?
        }
      }.freeze

      def self.validate(candle)
        errors = []

        RULES.each do |name, rule|
          errors << name unless rule.call(candle)
        end

        {
          valid: errors.empty?,
          errors: errors,
          candle_id: "#{candle.instrument_id}:#{candle.timeframe}:#{candle.timestamp}"
        }
      end

      def self.validate_batch(candles)
        candles.map { |c| validate(c) }
      end
    end

  class CandleQualityChecker < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def call(timeframe: "5m", from: 1.hour.ago, to: Time.current)
      candles = Candle.for_instrument(@instrument)
                     .for_timeframe(timeframe)
                     .where(timestamp: from..to)
                     .where(is_complete: true)
                     .order(timestamp: :asc)

      return failure(:no_candles) if candles.empty?

      validations = Validators::CandleValidator.validate_batch(candles)
      invalid = validations.reject { |v| v[:valid] }

      gap_detector = CandleGapDetector.new(instrument: @instrument, timeframe: timeframe)
      gap_result = gap_detector.call(from: from, to: to)

      success(
        instrument: @instrument.trading_symbol,
        timeframe: timeframe,
        candle_count: candles.count,
        invalid_count: invalid.length,
        invalid_candles: invalid.first(5),
        gap_count: gap_result.success? ? gap_result.value[:gap_count] : 0,
        coverage: gap_result.success? ? gap_result.value[:coverage_percent] : 0,
        quality_score: calculate_quality_score(candles.count, invalid.length, gap_result)
      )
    end

    private

    def calculate_quality_score(total, invalid, gap_result)
      gap_penalty = gap_result.success? ? (100 - gap_result.value[:coverage_percent]) : 100
      invalid_penalty = total.zero? ? 0 : (invalid.to_f / total * 100)
      [100 - gap_penalty - invalid_penalty, 0].max.round(2)
    end
  end
  ```
- **Deliverable:** Composite candle quality assessment
- **Acceptance Criteria:** `CandleQualityChecker.new(instrument: nifty).call` returns quality score 0-100
