### Task 1.2: Create `DhanHQ::RestClient` Wrapper
- **File:** `app/gateway/dhanhq/rest_client.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Gateway
    module DhanHQ
      class RestClient < Gateway::Broker
        HTTP_METHODS = %i[get post put delete].freeze

        class << self
          def request(method, endpoint, payload: nil, headers: {})
            with_circuit_breaker do
              with_metrics(endpoint) do
                with_retry do
                  execute(method, endpoint, payload, headers)
                end
              end
            end
          end

          def get(endpoint, params: {})
            query = params.any? ? "?#{params.to_query}" : ""
            request(:get, "#{endpoint}#{query}")
          end

          def post(endpoint, payload:)
            request(:post, endpoint, payload: payload)
          end

          def put(endpoint, payload:)
            request(:put, endpoint, payload: payload)
          end

          def delete(endpoint)
            request(:delete, endpoint)
          end

          private

          def execute(method, endpoint, payload, headers)
            client = ::DhanHQ::Client.new(
              client_id: AppConfig.config.broker.client_id,
              access_token: Rails.application.credentials.broker[:access_token]
            )

            response = case method
                       when :get then client.get(endpoint)
                       when :post then client.post(endpoint, payload)
                       when :put then client.put(endpoint, payload)
                       when :delete then client.delete(endpoint)
                       end

            Result.success(normalize_response(response))
          rescue ::DhanHQ::Error => e
            Result.failure(BrokerError.new(
              code: e.code,
              message: e.message,
              endpoint: endpoint
            ))
          rescue Timeout::Error, Net::OpenTimeout => e
            Result.failure(BrokerError.new(
              code: :timeout,
              message: e.message,
              endpoint: endpoint
            ))
          rescue => e
            Result.failure(BrokerError.new(
              code: :unknown,
              message: e.message,
              endpoint: endpoint
            ))
          end

          def normalize_response(response)
            return {} if response.nil?
            response.is_a?(Hash) ? response.with_indifferent_access : response
          end

          def with_retry(max_retries: AppConfig.config.broker.max_retries, base_delay: 1.0)
            retries = 0
            begin
              yield
            rescue => e
              retries += 1
              if retries <= max_retries
                sleep(base_delay * (2 ** retries))  # Exponential backoff
                retry
              end
              raise
            end
          end

          def with_metrics(endpoint)
            start = Time.current
            result = yield
            latency = ((Time.current - start) * 1000).round

            MetricsCollector.record(
              name: "dhanhq.request",
              labels: { endpoint: endpoint.split("?").first, status: result.success? ? "success" : "failure" },
              value: latency,
              type: :histogram
            )

            result
          end

          def with_circuit_breaker
            # Circuit breaker state stored in Redis
            state = RedisPool.general.with { |r| r.get("cb:dhanhq") }

            if state == "open"
              return Result.failure(BrokerError.new(
                code: :circuit_open,
                message: "DhanHQ circuit breaker is open"
              ))
            end

            result = yield

            if result.failure?
              record_failure
            else
              clear_failures
            end

            result
          end

          def record_failure
            key = "cb:dhanhq:failures"
            RedisPool.general.with do |r|
              r.incr(key)
              r.expire(key, 60)
              count = r.get(key).to_i
              r.set("cb:dhanhq", "open", ex: 30) if count >= 5
            end
          end

          def clear_failures
            RedisPool.general.with { |r| r.del("cb:dhanhq:failures") }
          end
        end
      end
    end
  end
  ```
- **Deliverable:** Centralized REST client with retry, metrics, circuit breaker
- **Acceptance Criteria:** `Gateway::DhanHQ::RestClient.get("/fundLimit")` returns `Result`; 5 failures open circuit for 30s

### Task 1.3: Create `BrokerResponseNormalizer`
- **File:** `app/gateway/dhanhq/response_normalizer.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Gateway
    module DhanHQ
      class ResponseNormalizer
        def self.normalize(data, type: :generic)
          return {} if data.nil?

          case type
          when :historical then normalize_historical(data)
          when :option_chain then normalize_option_chain(data)
          when :market_quote then normalize_quote(data)
          when :order then normalize_order(data)
          when :position then normalize_position(data)
          when :funds then normalize_funds(data)
          else data.with_indifferent_access
          end
        end

        def self.normalize_historical(data)
          data.map do |candle|
            {
              timestamp: Time.zone.parse(candle["timestamp"]),
              open: candle["open"].to_d,
              high: candle["high"].to_d,
              low: candle["low"].to_d,
              close: candle["close"].to_d,
              volume: candle["volume"].to_i,
              oi: candle["oi"]&.to_i
            }
          end
        end

        def self.normalize_option_chain(data)
          data.map do |strike|
            {
              strike: strike["strikePrice"].to_d,
              expiry: Date.parse(strike["expiryDate"]),
              ce: normalize_option(strike["CE"]),
              pe: normalize_option(strike["PE"])
            }
          end
        end

        private

        def self.normalize_option(opt)
          return nil if opt.nil?
          {
            security_id: opt["securityId"],
            trading_symbol: opt["tradingSymbol"],
            ltp: opt["lastPrice"]&.to_d,
            bid: opt["bidPrice"]&.to_d,
            ask: opt["askPrice"]&.to_d,
            volume: opt["volume"]&.to_i,
            oi: opt["openInterest"]&.to_i,
            oi_change: opt["openInterestChange"]&.to_i,
            iv: opt["impliedVolatility"]&.to_d,
            delta: opt["delta"]&.to_d,
            gamma: opt["gamma"]&.to_d,
            theta: opt["theta"]&.to_d,
            vega: opt["vega"]&.to_d
          }
        end

        def self.normalize_quote(data)
          {
            security_id: data["securityId"],
            trading_symbol: data["tradingSymbol"],
            ltp: data["lastPrice"]&.to_d,
            open: data["openPrice"]&.to_d,
            high: data["highPrice"]&.to_d,
            low: data["lowPrice"]&.to_d,
            close: data["closePrice"]&.to_d,
            prev_close: data["previousClosePrice"]&.to_d,
            volume: data["volume"]&.to_i,
            oi: data["openInterest"]&.to_i,
            bid: data["bidPrice"]&.to_d,
            ask: data["askPrice"]&.to_d
          }
        end

        def self.normalize_order(data)
          {
            broker_order_id: data["orderId"] || data["dhanOrderId"],
            status: map_order_status(data["orderStatus"]),
            side: data["transactionType"]&.downcase,
            quantity: data["quantity"]&.to_i,
            price: data["price"]&.to_d,
            average_price: data["averagePrice"]&.to_d,
            order_type: map_order_type(data["orderType"]),
            product_type: data["productType"],
            instrument: data["tradingSymbol"],
            placed_at: data["createTime"] ? Time.zone.parse(data["createTime"]) : nil
          }
        end

        def self.normalize_position(data)
          {
            position_id: data["positionId"],
            security_id: data["securityId"],
            trading_symbol: data["tradingSymbol"],
            side: data["positionType"] == "LONG" ? "long" : "short",
            quantity: data["netQuantity"]&.to_i,
            entry_price: data["buyAvgPrice"]&.to_d || data["sellAvgPrice"]&.to_d,
            current_price: data["lastPrice"]&.to_d,
            unrealized_pnl: data["unrealizedProfit"]&.to_d,
            realized_pnl: data["realizedProfit"]&.to_d,
            product_type: data["productType"]
          }
        end

        def self.normalize_funds(data)
          {
            available_balance: data["availableBalance"]&.to_d,
            utilized_amount: data["utilizedAmount"]&.to_d,
            pay_in_amount: data["payInAmount"]&.to_d,
            collateral_amount: data["collateralAmount"]&.to_d,
            total_balance: data["totalBalance"]&.to_d
          }
        end

        def self.map_order_status(status)
          {
            "PENDING" => "pending",
            "OPEN" => "open",
            "PARTIAL" => "partial",
            "COMPLETE" => "completed",
            "CANCELLED" => "cancelled",
            "REJECTED" => "rejected"
          }[status] || status&.downcase
        end

        def self.map_order_type(type)
          {
            "LIMIT" => "limit",
            "MARKET" => "market",
            "SL" => "sl",
            "SL-M" => "sl_m"
          }[type] || type&.downcase
        end
      end
    end
  end
  ```
- **Deliverable:** DhanHQ response → domain object normalization
- **Acceptance Criteria:** `ResponseNormalizer.normalize([{"timestamp":"2024-01-01","open":100}], type: :historical)` returns array of normalized hashes

### Task 2.3: Create `MarketDepthService`
- **File:** `app/services/market_depth_service.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class MarketDepthService < BaseService
    LEVELS = 5

    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      result = Gateway::DhanHQ::RestClient.get(
        "/market/v1/depth/#{CGI.escape(@instrument.exchange_segment)}/#{CGI.escape(@instrument.trading_symbol)}"
      )

      return result if result.failure?

      data = result.value
      success(
        bid_levels: normalize_levels(data["bids"] || []),
        ask_levels: normalize_levels(data["asks"] || []),
        total_bid_quantity: data["bids"]&.sum { |b| b["quantity"].to_i } || 0,
        total_ask_quantity: data["asks"]&.sum { |a| a["quantity"].to_i } || 0,
        spread: calculate_spread(data)
      )
    end

    private

    def normalize_levels(levels)
      levels.first(LEVELS).map do |level|
        {
          price: level["price"]&.to_d,
          quantity: level["quantity"]&.to_i,
          orders: level["orders"]&.to_i
        }
      end
    end

    def calculate_spread(data)
      bids = data["bids"] || []
      asks = data["asks"] || []
      return nil if bids.empty? || asks.empty?

      asks.first["price"].to_d - bids.first["price"].to_d
    end
  end
  ```
- **Deliverable:** 5-level market depth with spread calculation
- **Acceptance Criteria:** `MarketDepthService.new(instrument: nifty).call` returns bid/ask levels array

### Task 3.1: Create `OptionChainService`
- **File:** `app/services/option_chain_service.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class OptionChainService < BaseService
    def initialize(underlying:)
      @underlying = underlying
    end

    def call(expiry: nil)
      params = {
        underlyingSymbol: @underlying.trading_symbol,
        exchangeSegment: @underlying.exchange_segment
      }
      params[:expiry] = expiry.strftime("%Y-%m-%d") if expiry

      result = Gateway::DhanHQ::RestClient.get(
        "/market/v1/option-chain",
        params: params
      )

      return result if result.failure?

      normalized = Gateway::DhanHQ::ResponseNormalizer.normalize(
        result.value,
        type: :option_chain
      )

      success(normalized)
    end

    def atm_chain(spot_price:, range: 5)
      result = call
      return result if result.failure?

      chain = result.value
      atm_strike = find_atm_strike(chain, spot_price)
      step = detect_step(chain)

      min_strike = atm_strike - range * step
      max_strike = atm_strike + range * step

      filtered = chain.select do |strike_data|
        strike_data[:strike] >= min_strike && strike_data[:strike] <= max_strike
      end

      success(
        atm_strike: atm_strike,
        step: step,
        strikes: filtered.sort_by { |s| s[:strike] }
      )
    end

    def greeks_for(strike:, option_type:, expiry: nil)
      result = call(expiry: expiry)
      return result if result.failure?

      chain = result.value
      strike_data = chain.find { |s| s[:strike] == strike }
      return failure(:strike_not_found) unless strike_data

      option = strike_data[option_type.downcase.to_sym]
      return failure(:option_not_found) unless option

      success(option)
    end

    private

    def find_atm_strike(chain, spot_price)
      chain.min_by { |s| (s[:strike] - spot_price).abs }&.fetch(:strike)
    end

    def detect_step(chain)
      return 50 if chain.size < 2
      (chain[1][:strike] - chain[0][:strike]).abs
    end
  end
  ```
- **Deliverable:** Option chain with ATM filtering and Greeks extraction
- **Acceptance Criteria:** `OptionChainService.new(underlying: nifty).atm_chain(spot_price: 22500, range: 2)` returns 5 strikes

  ```ruby
  class OptionChainSnapshotJob < JobBase
    def perform(underlying_id)
      underlying = Instrument.find(underlying_id)
      return unless underlying&.index?

      result = OptionChainService.new(underlying: underlying).call

            underlying: underlying,
            snapshot_time: snapshot_time,
            strike: strike_data[:strike],
            option_type: option_type.to_s.upcase,

            ask: option[:ask],

            iv: option[:iv],
            delta: option[:delta],
            gamma: option[:gamma],
            theta: option[:theta],
            vega: option[:vega]
          )
        end
      end

      EventBus.publish(EventTypes::OPTION_CHAIN_UPDATED, {
        underlying: underlying.trading_symbol,
  ```

  ```ruby
  # frozen_string_literal: true

  class OrderService < BaseService
    def initialize(user:)
      @user = user
    end

    def place(instrument:, side:, quantity:, order_type:, product_type:, price: nil, trigger_price: nil)
      payload = {
        dhanClientId: AppConfig.config.broker.client_id,
        correlationId: SecureRandom.uuid,
        transactionType: side.upcase,
        exchangeSegment: instrument.exchange_segment,
        productType: product_type,
        orderType: order_type.upcase,
        validity: "DAY",
        tradingSymbol: instrument.trading_symbol,
        securityId: instrument.security_id,
        quantity: quantity,
        price: price&.to_f || 0,
        triggerPrice: trigger_price&.to_f || 0,
        afterMarket: false
      }

      result = Gateway::DhanHQ::RestClient.post("/orders", payload: payload)
      return result if result.failure?

      normalized = Gateway::DhanHQ::ResponseNormalizer.normalize(
        result.value,
        type: :order
      )

      # Persist order
      order = Order.create!(
        user: @user,
        instrument: instrument,
        broker_order_id: normalized[:broker_order_id],
        side: normalized[:side],
        quantity: quantity,
        price: price,
        order_type: order_type,
        product_type: product_type,
        status: normalized[:status],
        placed_at: normalized[:placed_at]
      )

      EventBus.publish(EventTypes::ORDER_PLACED, {
        order_id: order.id,
        broker_order_id: order.broker_order_id,
        status: order.status
      })

      success(order: order, broker_response: normalized)
    end

    def cancel(order:)
      return failure(:already_filled) if order.filled?

      result = Gateway::DhanHQ::RestClient.delete("/orders/#{order.broker_order_id}")
      return result if result.failure?

      order.update!(status: "cancelled", cancelled_at: Time.current)

      EventBus.publish(EventTypes::ORDER_CANCELLED, {
        order_id: order.id,
        broker_order_id: order.broker_order_id
      })

      success(order: order)
    end

    def modify(order:, params:)
      payload = {}
      payload[:quantity] = params[:quantity] if params[:quantity]
      payload[:price] = params[:price]&.to_f if params[:price]
      payload[:triggerPrice] = params[:trigger_price]&.to_f if params[:trigger_price]

      result = Gateway::DhanHQ::RestClient.put(
        "/orders/#{order.broker_order_id}",
        payload: payload
      )

      return result if result.failure?

      order.update!(params.slice(:quantity, :price, :trigger_price))
      success(order: order)
    end
  end
  ```
- **Deliverable:** Order placement, cancellation, modification
- **Acceptance Criteria:** `OrderService.new(user: admin).place(...)` creates `Order` record and returns broker order ID

  ```ruby
            unrealized_pnl: bp[:unrealized_pnl],

          )
        end
      end

      local_positions.each do |local|
        found = broker_positions.any? { |bp| bp[:position_id] == local.position_id }
        EventBus.publish(EventTypes::SYSTEM_ERROR, {
          type: :orphan_position,
          position_id: local.position_id
  ```

  ```ruby
    state = RedisPool.general.with { |r| r.get("cb:dhanhq") }

    case state
    when "open"
      half_open_at = RedisPool.general.with { |r| r.get("cb:dhanhq:half_open_at") }
      if half_open_at && Time.current.to_i >= half_open_at.to_i

    if result.failure?
      record_failure

        RedisPool.general.with { |r| r.set("cb:dhanhq", "open", ex: 60) }
      end
    else
      clear_failures
      RedisPool.general.with { |r| r.del("cb:dhanhq") } if state == "half_open"
    end

    result
  end
  ```

- **File:** `app/services/instrument_manager.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class InstrumentManager < BaseService
    INDICES = %w[NIFTY BANKNIFTY SENSEX FINNIFTY].freeze
    STRIKE_RANGE = 5
    ROLL_THRESHOLD = 0.5  # Roll ATM when spot moves > 50% of step

    def initialize(underlying:)
      @underlying = underlying
      @step = step_for(underlying)
    end

    def call
      refresh_chain
      track_atm
      success(
        underlying: @underlying.trading_symbol,
        atm_strike: current_atm,
        tracked_strikes: tracked_strikes,
        chain_valid: chain_fresh?
      )
    end

    def atm_strike
      current_atm
    end

    def strikes_to_track
      atm = current_atm
      return [] unless atm

      ((-STRIKE_RANGE..STRIKE_RANGE).to_a.map { |i| atm + i * @step }).sort
    end

    def should_roll?(new_spot_price)
      return false unless current_atm
      (new_spot_price - current_atm).abs > @step * ROLL_THRESHOLD
    end

    def roll!(new_spot_price:)
      return success(rolled: false) unless should_roll?(new_spot_price)

      old_atm = current_atm
      new_atm = ((new_spot_price / @step).round * @step)

      # Update cache
      CacheStore.write(atm_key, new_atm, ttl: 1.day)

      # Log roll event
      EventBus.publish(EventTypes::OPTION_CHAIN_UPDATED, {
        event: :atm_rolled,
        underlying: @underlying.trading_symbol,
        from: old_atm,
        to: new_atm,
        spot_price: new_spot_price
      })

      success(rolled: true, from: old_atm, to: new_atm)
    end

    private

    def current_atm
      cached = CacheStore.fetch(atm_key)
      return cached if cached

      # Fetch from broker if not cached
      result = MarketQuoteService.new(instrument: @underlying).ltp
      return nil if result.failure?

      spot = result.value
      atm = ((spot / @step).round * @step)
      CacheStore.write(atm_key, atm, ttl: 60)
      atm
    end

    def tracked_strikes
      strikes_to_track
    end

    def refresh_chain
      # Trigger option chain fetch if stale
      last_fetch = CacheStore.fetch("chain:#{@underlying.trading_symbol}:last_fetch")
      return if last_fetch && last_fetch > 30.seconds.ago

      OptionChainSnapshotJob.perform_later(@underlying.id)
      CacheStore.write("chain:#{@underlying.trading_symbol}:last_fetch", Time.current, ttl: 60)
    end

    def chain_fresh?
      last = OptionChainSnapshot.for_underlying(@underlying).order(snapshot_time: :desc).first
      last.present? && last.snapshot_time > 2.minutes.ago
    end

    def step_for(underlying)
      case underlying.trading_symbol
      when /NIFTY/ then 50
      when /BANKNIFTY/ then 100
      when /SENSEX/ then 100
      when /FINNIFTY/ then 50
      else 50
      end
    end

    def atm_key
      "instrument_manager:#{@underlying.trading_symbol}:atm"
    end
  end
  ```
- **Deliverable:** Dynamic ATM tracking with automatic rolling
- **Acceptance Criteria:** `InstrumentManager.new(underlying: nifty).strikes_to_track` returns 11 strikes (ATM ±5); `should_roll?` triggers when spot moves > 25 points

### Task 8.3: Create `StrikeSelector`
- **File:** `app/services/strike_selector.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class StrikeSelector < BaseService
    def initialize(underlying:, spot_price:)
      @underlying = underlying
      @spot_price = spot_price
      @manager = InstrumentManager.new(underlying: underlying)
    end

    def call(criteria: {})
      result = @manager.call
      return result if result.failure?

      strikes = @manager.strikes_to_track

      # Fetch option chain for these strikes
      chain_result = OptionChainService.new(underlying: @underlying).atm_chain(
        spot_price: @spot_price,
        range: InstrumentManager::STRIKE_RANGE
      )
      return chain_result if chain_result.failure?

      chain = chain_result.value[:strikes]

      # Score each strike
      scored = chain.map do |strike_data|
        score = score_strike(strike_data, criteria)
        strike_data.merge(score: score)
      end

      best = scored.max_by { |s| s[:score] }
      success(
        best_strike: best,
        all_scores: scored.sort_by { |s| -s[:score] }
      )
    end

    private

    def score_strike(strike_data, criteria)
      score = 0
      weights = {
        delta: 0.25,
        liquidity: 0.25,
        gamma: 0.15,
        oi: 0.15,
        iv: 0.10,
        spread: 0.10
      }

      [:ce, :pe].each do |type|
        option = strike_data[type]
        next unless option

        score += delta_score(option[:delta]) * weights[:delta]
        score += liquidity_score(option) * weights[:liquidity]
        score += gamma_score(option[:gamma]) * weights[:gamma]
        score += oi_score(option[:oi]) * weights[:oi]
        score += iv_score(option[:iv]) * weights[:iv]
        score += spread_score(option) * weights[:spread]
      end

      [score, 100].min
    end

    def delta_score(delta)
      return 0 unless delta
      ideal = 0.525  # midpoint of 0.40-0.65
      [100 - (delta - ideal).abs * 200, 0].max
    end

    def liquidity_score(option)
      return 0 unless option[:volume] && option[:oi]
      volume_score = [option[:volume] / 1000.0, 100].min
      oi_score = [option[:oi] / 5000.0, 100].min
      (volume_score + oi_score) / 2
    end

    def gamma_score(gamma)
      return 0 unless gamma
      [gamma * 10000, 100].min
    end

    def oi_score(oi)
      return 0 unless oi
      [oi / 5000.0, 100].min
    end

    def iv_score(iv)
      return 50 unless iv
      # Prefer moderate IV, not extreme
      [100 - (iv - 20).abs * 2, 0].max
    end

    def spread_score(option)
      return 0 unless option[:bid] && option[:ask] && option[:ltp] && option[:ltp] > 0
      spread = ((option[:ask] - option[:bid]) / option[:ltp] * 100)
      spread <= 0.3 ? 100 : [100 - (spread - 0.3) * 50, 0].max
    end
  end
  ```
- **Deliverable:** Strike scoring and selection
- **Acceptance Criteria:** `StrikeSelector.new(underlying: nifty, spot_price: 22500).call` returns best strike with score > 0
