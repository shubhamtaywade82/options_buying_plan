module AiGateway
  class PromptTemplate
    class MissingTemplate < StandardError; end

    def initialize(template_name)
      @template_name = template_name
      @path = Rails.root.join("app/prompts/#{template_name}.md.erb")
    end

    def render(context = {})
      raise MissingTemplate, "Template not found: #{@path}" unless File.exist?(@path)

      template = ERB.new(File.read(@path), trim_mode: '-')
      ctx = PromptContext.new(context)
      template.result(ctx.get_binding)
    end

    private

    class PromptContext < OpenStruct
      def get_binding
        binding
      end
    end
  end
end

module AiGateway
  module Agents
    class BaseAgent
      attr_reader :name, :version

      DEFAULT_MODEL = "llama3.2".freeze
      DEFAULT_TIMEOUT = 45

      def initialize(gateway: nil, model: nil)
        @gateway = gateway || AiGateway::AIGateway.new
        @model = model || DEFAULT_MODEL
        @cache = AiGateway::AIAnalysisCache.new
        @parser = AiGateway::ResponseParser.new
        @confidence_scorer = AiGateway::AIConfidenceScorer.new
        @name = self.class.name.demodulize.underscore
        @version = "1.0.0"
      end

      def call(context: {})
        raise NotImplementedError, "#{self.class} must implement #call"
      end

      private

      def render_prompt(template_name, context)
        PromptTemplate.new(template_name).render(context)
      end

      def generate(prompt)
        cache_key = build_cache_key(prompt)

        cached = @cache.fetch(cache_key)
        return Result.success(cached) if cached

        result = @gateway.call(prompt: prompt, model: @model, max_tokens: 2048, temperature: 0.3)
        return result if result.failure?

        raw_text = result.value[:text]
        parsed = @parser.parse(raw_text)

        return Result.failure(:parse_error) if parsed.nil?

        confidence = @confidence_scorer.score(parsed, raw_text)
        enriched = parsed.merge(
          ai_confidence: confidence,
          ai_provider: result.value[:provider],
          ai_model: result.value[:model],
          ai_latency_ms: result.value[:total_duration],
          agent: @name,
          version: @version,
          generated_at: Time.current.iso8601
        )

        @cache.store(cache_key, enriched)
        emit(EventTypes::AI_AGENT_COMPLETED, { agent: @name, confidence: confidence })

        Result.success(enriched)
      end

      def build_cache_key(prompt)
        "ai_agent:#{@name}:#{Digest::MD5.hexdigest(prompt)}"
      end
    end
  end
end
```

module AiGateway
  class ResponseParser
    def parse(text)
      return nil if text.blank?

      # Extract JSON from markdown code blocks
      json = extract_json(text)
      return nil if json.nil?

      parsed = JSON.parse(json, symbolize_names: true)
      normalize(parsed)
    rescue JSON::ParserError => e
      Rails.logger.warn "[ResponseParser] JSON parse error: #{e.message}"
      nil
    end

    private

    def extract_json(text)
      # Try fenced code block first
      if text =~ /```(?:json)?\s*(.*?)```/m
        return $1.strip
      end

      # Try raw JSON object/array
      if text =~ /(\{.*\}|\[.*\])/m
        return $1.strip
      end

      nil
    end

    def normalize(parsed)
      # Ensure consistent key types
      parsed.deep_transform_keys do |key|
        key.to_s.underscore.to_sym
      end
    end
  end
end

module AiGateway
  module Agents
    class SetupValidatorAgent < BaseAgent
      VERSION = "1.0.0".freeze

      def initialize(gateway: nil, model: nil)
        super
        @version = VERSION
      end

      def call(context: {})
        required = %i[setup trade_score direction instrument strategy]
        missing = required - context.keys
        return Result.failure(ValidationError.new(missing: missing)) if missing.any?

        prompt = render_prompt("setup", build_context(context))
        result = generate(prompt)

        return result if result.failure?

        validated = apply_validation_rules(result.value, context[:trade_score])
        Result.success(validated)
      end

      private

      def build_context(context)
        score = context[:trade_score]
        {
          instrument: context[:instrument],
          direction: context[:direction],
          strategy: context[:strategy],
          market_context: score.dig(:component_scores, :market_context, :score) || 0,
          market_regime: score.dig(:component_scores, :market_regime, :score) || 0,
          market_structure: score.dig(:component_scores, :market_structure, :score) || 0,
          momentum: score.dig(:component_scores, :momentum, :score) || 0,
          liquidity: score.dig(:component_scores, :liquidity, :score) || 0,
          option_flow: score.dig(:component_scores, :option_flow, :score) || 0,
          greeks: score.dig(:component_scores, :greeks, :score) || 0,
          strike_quality: score.dig(:component_scores, :strike_quality, :score) || 0,
          total_score: score[:total_score],
          threshold: score[:threshold],
          metadata: score[:breakdown]&.to_json || "{}"
        }
      end

      def apply_validation_rules(ai_response, original_score)
        ai_score = ai_response[:ai_validation_score].to_f

        # AI cannot approve below deterministic threshold - 10
        if original_score[:total_score] < original_score[:threshold] - 10
          ai_response[:verdict] = "reject"
          ai_response[:reasoning] = "Deterministic score too low. #{ai_response[:reasoning]}"
        end

        # Cap AI score at deterministic score + 5
        capped = [ai_score, original_score[:total_score] + 5].min
        ai_response[:ai_validation_score] = capped.round(2)

        ai_response
      end
    end
  end
end

```ruby
module AiGateway
  class AIConfidenceScorer
    def score(parsed_response, raw_text)
      scores = [
        structure_score(parsed_response),
        completeness_score(parsed_response),
        consistency_score(parsed_response),
        length_score(raw_text)
      ]

      (scores.sum / scores.length).round(2)
    end

    private

    def structure_score(parsed)
      return 0 unless parsed.is_a?(Hash)
      return 100 if parsed.keys.length >= 3
      50
    end

    def completeness_score(parsed)
      return 0 unless parsed.is_a?(Hash)
      non_nil = parsed.values.compact.reject { |v| v == "" }.length
      total = parsed.keys.length
      return 0 if total.zero?
      ((non_nil.to_f / total) * 100).round(2)
    end

    def consistency_score(parsed)
      return 0 unless parsed.is_a?(Hash)

      checks = 0
      passes = 0

      # Check numeric ranges
      if parsed.key?(:ai_validation_score)
        checks += 1
        passes += 1 if parsed[:ai_validation_score].to_f.between?(0, 100)
      end

      if parsed.key?(:confidence)
        checks += 1
        passes += 1 if %w[high medium low].include?(parsed[:confidence].to_s.downcase)
      end

      if parsed.key?(:verdict)
        checks += 1
        passes += 1 if %w[approve caution reject].include?(parsed[:verdict].to_s.downcase)
      end

      return 100 if checks.zero?
      ((passes.to_f / checks) * 100).round(2)
    end

    def length_score(raw_text)
      return 0 if raw_text.blank?
      chars = raw_text.length
      return 100 if chars.between?(50, 2000)
      return 50 if chars.between?(20, 50) || chars.between?(2000, 5000)
      20
    end
  end
end
```

module AiGateway
  class AIAnalysisCache
    DEFAULT_TTL = 300  # 5 minutes

    def initialize
      @redis = RedisPool.general
    end

    def fetch(key)
      json = @redis.get(cache_key(key))
      return nil unless json

      JSON.parse(json, symbolize_names: true)
    rescue JSON::ParserError
      nil
    end

    def store(key, value, ttl: DEFAULT_TTL)
      @redis.setex(cache_key(key), ttl, value.to_json)
    end

    def invalidate(key)
      @redis.del(cache_key(key))
    end

    def invalidate_pattern(pattern)
      keys = @redis.keys(cache_key(pattern))
      @redis.del(*keys) if keys.any?
    end

    private

    def cache_key(key)
      "ai_analysis_cache:#{key}"
    end
  end
end

module AiGateway
  class AgentOrchestrator
    AGENT_REGISTRY = {
      setup_validator: Agents::SetupValidatorAgent,
      market_analyst: Agents::MarketAnalystAgent,
      trade_reviewer: Agents::TradeReviewerAgent,
      journal_writer: Agents::JournalWriterAgent,
      strategy_researcher: Agents::StrategyResearcherAgent
    }.freeze

    def initialize(gateway: nil)
      @gateway = gateway || AiGateway::AIGateway.new
      @instances = {}
    end

    def call(agent_type, context: {})
      agent_class = AGENT_REGISTRY[agent_type.to_sym]
      return Result.failure(:unknown_agent) unless agent_class

      agent = get_or_create(agent_class)
      result = agent.call(context: context)

      emit(EventTypes::AI_AGENT_ORCHESTRATED, {
        agent: agent_type,
        success: result.success?,
        confidence: result.success? ? result.value[:ai_confidence] : nil
      })

      result
    end

    def available_agents
      AGENT_REGISTRY.keys
    end

    def agent_info(agent_type)
      klass = AGENT_REGISTRY[agent_type.to_sym]
      return nil unless klass

      instance = get_or_create(klass)
      {
        name: instance.name,
        version: instance.version,
        model: instance.instance_variable_get(:@model)
      }
    end

    private

    def get_or_create(agent_class)
      @instances[agent_class] ||= agent_class.new(gateway: @gateway)
    end
  end
end

module Api
  module V1
    class AiAgentsController < ApplicationController
      def run
        agent_type = params[:agent]&.to_sym
        context = params[:context] || {}

        unless AiGateway::AgentOrchestrator::AGENT_REGISTRY.key?(agent_type)
          return render json: { error: "Unknown agent: #{agent_type}" }, status: :not_found
        end

        orchestrator = AiGateway::AgentOrchestrator.new
        result = orchestrator.call(agent_type, context: context.symbolize_keys)

        if result.success?
          render json: {
            agent: agent_type,
            result: result.value,
            cached: result.value[:generated_at].present? && result.value[:generated_at] < 1.minute.ago
          }
        else
          render json: { error: result.error.to_h }, status: :unprocessable_entity
        end
      end

      def index
        orchestrator = AiGateway::AgentOrchestrator.new
        render json: {
          agents: orchestrator.available_agents.map do |key|
            {
              name: key,
              info: orchestrator.agent_info(key)
            }
          end
        }
      end

      def validate_setup
        # Convenience endpoint for setup validation
        orchestrator = AiGateway::AgentOrchestrator.new
        result = orchestrator.call(:setup_validator, context: {
          setup: params[:setup],
          trade_score: params[:trade_score],
          direction: params[:direction],
          instrument: params[:instrument],
          strategy: params[:strategy]
        }.compact)

        if result.success?
          render json: {
            ai_validation_score: result.value[:ai_validation_score],
            verdict: result.value[:verdict],
            confidence: result.value[:ai_confidence],
            reasoning: result.value[:reasoning]
          }
        else
          render json: { error: result.error.to_h }, status: :unprocessable_entity
        end
      end

      def market_narrative
        orchestrator = AiGateway::AgentOrchestrator.new
        result = orchestrator.call(:market_analyst, context: {
          instrument: params[:instrument],
          spot_price: params[:spot_price],
          timeframe: params[:timeframe] || "5m",
          regime: params[:regime],
          structure: params[:structure],
          momentum: params[:momentum],
          liquidity: params[:liquidity],
          option_flow: params[:option_flow]
        }.compact)

        if result.success?
          render json: {
            narrative: result.value[:narrative],
            bias: result.value[:bias],
            confidence: result.value[:confidence],
            key_levels: result.value[:key_levels],
            outlook: result.value[:outlook]
          }
        else
          render json: { error: result.error.to_h }, status: :unprocessable_entity
        end
      end
    end
  end
end

# AI Agents

## Architecture

All agents inherit from `AiGateway::Agents::BaseAgent` and use:
- `PromptTemplate` for ERB-based prompt rendering
- `AIGateway` for LLM generation with failover
- `ResponseParser` for structured JSON extraction
- `AIConfidenceScorer` for response reliability
- `AIAnalysisCache` for 5-minute result caching

## Agents

| Agent | Purpose | Cache Key |
|-------|---------|-----------|
| `setup_validator` | Validates deterministic trade scores | `ai_agent:setup_validator:md5` |
| `market_analyst` | Generates 5-minute market narrative | `ai_agent:market_analyst:md5` |
| `trade_reviewer` | Post-trade performance analysis | `ai_agent:trade_reviewer:md5` |
| `journal_writer` | Structured trade journaling | `ai_agent:journal_writer:md5` |
| `strategy_researcher` | Historical pattern comparison | `ai_agent:strategy_researcher:md5` |

## Prompt Templates

Located in `app/prompts/*.md.erb`. Each template expects specific context variables and must return JSON.

## Safety Rules

1. **SetupValidator** cannot approve trades scoring below `threshold - 10`
2. **SetupValidator** caps AI score at `deterministic_score + 5`
3. All agents return `ai_confidence` (0-100) for downstream filtering
4. Responses are cached for 5 minutes to reduce LLM costs

## API

```bash
# Run any agent
POST /api/v1/ai_agents/run
{ "agent": "setup_validator", "context": { ... } }

# Validate setup (convenience)
POST /api/v1/ai_agents/validate_setup
{ "trade_score": { ... }, "direction": "long", "instrument": "NIFTY" }

# Market narrative (convenience)
GET /api/v1/ai_agents/market_narrative?instrument=NIFTY&spot_price=22500
```

Rails.application.config.after_initialize do
  # Verify prompt templates exist
  required_templates = %w[setup market_analyst review journal strategy_researcher]
  missing = required_templates.reject do |t|
    File.exist?(Rails.root.join("app/prompts/#{t}.md.erb"))
  end

  if missing.any?
    Rails.logger.error "[AIAgents] Missing prompt templates: #{missing.join(', ')}"
  else
    Rails.logger.info "[AIAgents] All #{required_templates.length} prompt templates loaded"
  end

  # Warm up local Ollama if available
  if AppConfig.config.ai.ollama_local_enabled
    begin
      gateway = AiGateway::AIGateway.new
      health = gateway.health
      Rails.logger.info "[AIAgents] AI Gateway health: #{health}"
    rescue => e
      Rails.logger.warn "[AIAgents] Could not reach AI Gateway: #{e.message}"
    end
  end
end
