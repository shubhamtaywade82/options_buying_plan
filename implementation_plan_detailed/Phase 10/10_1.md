```ruby
module CandleFixtureBuilder
  def build_candles(count: 20, instrument: nil, timeframe: "5m", pattern: :random, base_price: 22500)
    instrument ||= create(:instrument, :nifty)
    candles = []
    price = base_price.to_d

    count.times do |i|
      candle = case pattern
      when :uptrend
        build_uptrend_candle(price, i)
      when :downtrend
        build_downtrend_candle(price, i)
      when :range
        build_range_candle(price, i)
      when :breakout
        build_breakout_candle(price, i, count)
      when :reversal
        build_reversal_candle(price, i, count)
      else
        build_random_candle(price, i)
      end

      candle.merge!(
        instrument_id: instrument.id,
        timeframe: timeframe,
        timestamp: (Time.current - (count - i) * 5.minutes).iso8601
      )

      candles << candle
      price = candle[:close]
    end

    candles
  end

  def create_candles_in_db(count: 20, instrument: nil, **options)
    build_candles(count: count, instrument: instrument, **options).map do |attrs|
      create(:candle, attrs)
    end
  end

  private

  def build_uptrend_candle(price, _index)
    open_p = price
    close_p = price + rand(5..15)
    {
      open: open_p,
      high: close_p + rand(0..3),
      low: open_p - rand(0..3),
      close: close_p,
      volume: rand(5000..20000)
    }
  end

  def build_downtrend_candle(price, _index)
    open_p = price
    close_p = price - rand(5..15)
    {
      open: open_p,
      high: open_p + rand(0..3),
      low: close_p - rand(0..3),
      close: close_p,
      volume: rand(5000..20000)
    }
  end

  def build_range_candle(price, _index)
    move = rand(-10..10)
    {
      open: price,
      high: price + rand(5..15),
      low: price - rand(5..15),
      close: price + move,
      volume: rand(3000..15000)
    }
  end

  def build_breakout_candle(price, index, total)
    if index < total * 0.7
      build_range_candle(price, index)
    else
      {
        open: price,
        high: price + 50 + rand(0..10),
        low: price - rand(0..5),
        close: price + 40 + rand(0..10),
        volume: rand(25000..50000)
      }
    end
  end

  def build_reversal_candle(price, index, total)
    mid = total / 2
    if index < mid
      build_uptrend_candle(price, index)
    else
      build_downtrend_candle(price, index)
    end
  end

  def build_random_candle(price, _index)
    move = rand(-15..15)
    {
      open: price,
      high: price + rand(5..20),
      low: price - rand(5..20),
      close: price + move,
      volume: rand(1000..25000)
    }
  end
end
```

```ruby
  class MockServer
    attr_reader :messages, :connected

    def initialize
      @messages = []
      @connected = false
      @handlers = {}
    end

    def connect
      @connected = true
      @handlers[:on_open]&.call
    end

    def disconnect
      @connected = false
      @handlers[:on_close]&.call
    end

    def on(event, &block)
      @handlers[event] = block
    end

    def send(message)
      @messages << message
    end

    def simulate_incoming(data)
      @handlers[:on_message]&.call(data.to_json)
    end

    def simulate_tick(instrument_id, ltp, volume: 1000)

        type: "tick",
        instrument_id: instrument_id,
        ltp: ltp,

        instrument_id: instrument_id,
        data: depth_data,
        timestamp: Time.current.iso8601
      })
    end

    def last_message
      @messages.last
    end

    def message_count
      @messages.length
    end
  end
```

```ruby
module StrategyBacktestRunner
  class Backtest
    attr_reader :trades, :equity_curve, :metrics

    def initialize(strategy_class:, instrument:, candles:, initial_capital: 500_000)
      @strategy_class = strategy_class
      @instrument = instrument
      @candles = candles
      @initial_capital = initial_capital
      @capital = initial_capital
      @equity_curve = []
      @trades = []
      @position = nil
      @metrics = {}
    end

    def run
      @candles.each_with_index do |candle, i|
        context = build_context(candle, i)
        strategy = @strategy_class.new(instrument: @instrument, context: context)

        if @position
          check_exit(strategy, candle, i)
        else
          check_entry(strategy, candle, i)
        end

        record_equity(candle[:timestamp])
      end

      calculate_metrics
      self
    end

    def win_rate
      return 0 if @trades.empty?
      wins = @trades.count { |t| t[:pnl] > 0 }
      (wins.to_f / @trades.length * 100).round(2)
    end

    def profit_factor
      return 0 if @trades.empty?
      gross_profit = @trades.sum { |t| [t[:pnl], 0].max }
      gross_loss = @trades.sum { |t| [t[:pnl], 0].min }.abs
      gross_loss.zero? ? 0 : (gross_profit / gross_loss).round(2)
    end

    def max_drawdown
      peak = 0
      max_dd = 0
      @equity_curve.each do |point|
        peak = [peak, point[:equity]].max
        dd = peak - point[:equity]
        max_dd = [max_dd, dd].max
      end
      max_dd
    end

    private

    def build_context(candle, index)
      recent_candles = @candles[[0, index - 20].max..index]
      {
        spot_price: candle[:close],
        candles: recent_candles,
        timeframe: "5m",
        volume: { current: candle[:volume], average: 10000, accelerating: candle[:volume] > 15000 },
        regime: { type: :trending_up, direction: :bullish },
        momentum: { direction: :strong_bullish, score: 75 }
      }
    end

    def check_entry(strategy, candle, index)
      result = strategy.call
      return unless result.success?

      signal = result.value[:signal]
      return unless signal == :enter

      @position = {
        entry_price: candle[:close],
        entry_index: index,
        direction: result.value[:direction],
        stop_loss: result.value[:stop_loss] || (candle[:close] * 0.995),
        take_profit: result.value[:target] || (candle[:close] * 1.015),
        quantity: 1
      }
    end

    def check_exit(strategy, candle, index)
      return unless @position

      # Check stop loss
      if @position[:direction] == :long && candle[:low] <= @position[:stop_loss]
        close_trade(@position[:stop_loss], :stop_loss, index)
      elsif @position[:direction] == :short && candle[:high] >= @position[:stop_loss]
        close_trade(@position[:stop_loss], :stop_loss, index)
      # Check target
      elsif @position[:direction] == :long && candle[:high] >= @position[:take_profit]
        close_trade(@position[:take_profit], :target, index)
      elsif @position[:direction] == :short && candle[:low] <= @position[:take_profit]
        close_trade(@position[:take_profit], :target, index)
      # Time-based exit (max 10 candles)
      elsif index - @position[:entry_index] >= 10
        close_trade(candle[:close], :time_exit, index)
      end
    end

    def close_trade(exit_price, reason, index)
      pnl = if @position[:direction] == :long
        (exit_price - @position[:entry_price]) * @position[:quantity]
      else
        (@position[:entry_price] - exit_price) * @position[:quantity]
      end

      @capital += pnl

      @trades << {
        entry_price: @position[:entry_price],
        exit_price: exit_price,
        direction: @position[:direction],
        pnl: pnl,
        reason: reason,
        holding_candles: index - @position[:entry_index]
      }

      @position = nil
    end

    def record_equity(timestamp)
      @equity_curve << { timestamp: timestamp, equity: @capital }
    end

    def calculate_metrics
      @metrics = {
        total_trades: @trades.length,
        win_rate: win_rate,
        profit_factor: profit_factor,
        net_pnl: @capital - @initial_capital,
        max_drawdown: max_drawdown,
        final_capital: @capital,
        return_percent: ((@capital - @initial_capital) / @initial_capital * 100).round(2)
      }
    end
```
