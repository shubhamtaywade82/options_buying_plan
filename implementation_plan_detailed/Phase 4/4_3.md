### Task 1.1: Create `MarketStructureEngine`
- **File:** `app/engines/market_structure_engine.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    class MarketStructureEngine < BaseEngine
      STRUCTURE_TYPES = %w[HH HL LH LL BOS CHOCH liquidity_sweep fvg order_block].freeze
      TIMEFRAMES = %w[1m 5m 15m].freeze

      def self.engine_name
        "market_structure"
      end

      def self.required_features
        %w[high low close timestamp]
      end

      def initialize(instrument:, timeframe: "15m")
        super()
        @instrument = instrument
        @timeframe = timeframe
      end

      def analyze
        validate_inputs!(build_inputs)

        candles = fetch_candles
        return failure(:insufficient_data) if candles.length < 5

        swing_points = detect_swing_points(candles)
        structures = detect_structures(candles, swing_points)

        # Score the overall structure
        score = StructureScoreCalculator.new(structures: structures).call

        # Persist
        persist_structures(structures)

        emit(EventTypes::STRUCTURE_DETECTED, {
          engine: engine_name,
          instrument_id: @instrument.id,
          timeframe: @timeframe,
          structures: structures.map(&:to_h),
          score: score.value,
          timestamp: Time.current
        })

        success(
          instrument: @instrument.trading_symbol,
          timeframe: @timeframe,
          swing_points: swing_points,
          structures: structures,
          structure_score: score.value,
          bullish_structure: bullish_structure?(structures),
          bearish_structure: bearish_structure?(structures),
          timestamp: Time.current
        )
      end

      private

      def build_inputs
        { instrument: @instrument, timeframe: @timeframe }
      end

      def fetch_candles
        Candle.for_instrument(@instrument)
            .for_timeframe(@timeframe)
            .where(is_complete: true)
            .order(timestamp: :asc)
            .last(100)
      end

      def detect_swing_points(candles)
        MarketStructure::SwingPointDetector.new(candles: candles).call
      end

      def detect_structures(candles, swing_points)
        structures = []

        structures.concat(MarketStructure::HigherHighDetector.new(candles: candles, swing_points: swing_points).call)
        structures.concat(MarketStructure::HigherLowDetector.new(candles: candles, swing_points: swing_points).call)
        structures.concat(MarketStructure::LowerHighDetector.new(candles: candles, swing_points: swing_points).call)
        structures.concat(MarketStructure::LowerLowDetector.new(candles: candles, swing_points: swing_points).call)
        structures.concat(MarketStructure::BreakOfStructureDetector.new(candles: candles, swing_points: swing_points).call)
        structures.concat(MarketStructure::ChangeOfCharacterDetector.new(candles: candles, swing_points: swing_points).call)
        structures.concat(MarketStructure::LiquiditySweepDetector.new(candles: candles, swing_points: swing_points).call)
        structures.concat(MarketStructure::FairValueGapDetector.new(candles: candles).call)
        structures.concat(MarketStructure::OrderBlockDetector.new(candles: candles).call)

        structures
      end

      def persist_structures(structures)
        records = structures.map(&:to_h).map do |s|
          s.merge(
            instrument_id: @instrument.id,
            timeframe: @timeframe,
            detected_at: Time.current
          )
        end

        MarketStructure.insert_all(records, returning: false) if records.any?
      end

      def bullish_structure?(structures)
        bos = structures.any? { |s| s.type == "BOS" && s.bullish }
        hh = structures.any? { |s| s.type == "HH" }
        hl = structures.any? { |s| s.type == "HL" }
        bos && hh && hl
      end

      def bearish_structure?(structures)
        bos = structures.any? { |s| s.type == "BOS" && s.bearish }
        lh = structures.any? { |s| s.type == "LH" }
        ll = structures.any? { |s| s.type == "LL" }
        bos && lh && ll
      end
    end
  end
  ```
- **Deliverable:** Base engine orchestrating all structure detectors
- **Acceptance Criteria:** `MarketStructureEngine.new(instrument: nifty).analyze` returns structures array

### Task 1.2: Create `SwingPointDetector`
- **File:** `app/engines/market_structure/swing_point_detector.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketStructure
      class SwingPointDetector
        DEFAULT_LEFT_BARS = 3
        DEFAULT_RIGHT_BARS = 2

        def initialize(candles:, left_bars: DEFAULT_LEFT_BARS, right_bars: DEFAULT_RIGHT_BARS)
          @candles = candles
          @left = left_bars
          @right = right_bars
        end

        def call
          return [] if @candles.length < (@left + @right + 1)

          swings = []

          (@left..(@candles.length - @right - 1)).each do |i|
          candle = @candles[i]

          if swing_high?(i)
            swings << SwingPoint.new(
              type: :high,
              timestamp: candle.timestamp,
              price: candle.high.to_d,
              index: i,
              candle: candle
            )
          elsif swing_low?(i)
            swings << SwingPoint.new(
              type: :low,
              timestamp: candle.timestamp,
              price: candle.low.to_d,
              index: i,
              candle: candle
            )
          end
          end

          # Remove consecutive same-type swings (keep highest high / lowest low)
          deduplicate(swings)
        end

        private

        def swing_high?(i)
          pivot = @candles[i]
          left_range = @candles[i - @left...i]
          right_range = @candles[i + 1..i + @right]

          left_range.all? { |c| c.high.to_d <= pivot.high.to_d } &&
            right_range.all? { |c| c.high.to_d <= pivot.high.to_d }
        end

        def swing_low?(i)
          pivot = @candles[i]
          left_range = @candles[i - @left...i]
          right_range = @candles[i + 1..i + @right]

          left_range.all? { |c| c.low.to_d >= pivot.low.to_d } &&
            right_range.all? { |c| c.low.to_d >= pivot.low.to_d }
        end

        def deduplicate(swings)
          result = []
          swings.each do |swing|
            if result.empty? || result.last.type != swing.type
              result << swing
            else
              # Keep the more extreme one
              if swing.type == :high && swing.price > result.last.price
                result[-1] = swing
              elsif swing.type == :low && swing.price < result.last.price
                result[-1] = swing
              end
            end
          end
          result
        end
      end

      class SwingPoint < Struct.new(:type, :timestamp, :price, :index, :candle, keyword_init: true)
        def high?
          type == :high
        end

        def low?
          type == :low
        end
      end
    end
  end
  ```
- **Deliverable:** Swing high/low detection with deduplication
- **Acceptance Criteria:** 3 left + 2 right bars confirm swing; consecutive same-type merged

          type == "HL" || type == "LL"
        end
      end
    end
  end

- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketStructure
      class HigherHighDetector
        def initialize(candles:, swing_points:)
          @candles = candles
          @swings = swing_points.select(&:high?)
        end

        def call
          return [] if @swings.length < 2

          structures = []

          @swings.each_cons(2) do |prev, curr|
            if curr.price > prev.price
              structures << Structure.new(
                type: "HH",
                timestamp: curr.timestamp,
                price_level: curr.price,
                bullish: true,
                bearish: false,
                context: {
                  previous_high: prev.price,
                  breakout_amount: (curr.price - prev.price).round(4)
                },
                timeframe: @candles.first.timeframe
              )
            end
          end

          structures
        end
      end
    end
  end
  ```
- **Deliverable:** Higher High detection
- **Acceptance Criteria:** `HigherHighDetector.new(...).call` returns HH when current swing > previous

## Day 3 — BOS & CHOCH Detectors

### Task 3.1: Create `BreakOfStructureDetector` (BOS)
- **File:** `app/engines/market_structure/break_of_structure_detector.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketStructure
      class BreakOfStructureDetector
        def initialize(candles:, swing_points:)
          @candles = candles
          @swings = swing_points
        end

        def call
          return [] if @swings.length < 3

          structures = []

          @swings.each_cons(3) do |a, b, c|
            next unless a.high? && b.low? && c.high?

            # Bullish BOS: c breaks above a
            if c.price > a.price
              structures << Structure.new(
                type: "BOS",
                timestamp: c.timestamp,
                price_level: c.price,
                bullish: true,
                bearish: false,
                context: {
                  previous_structure_high: a.price,
                  breakout_amount: (c.price - a.price).round(4),
                  swing_low: b.price,
                  bos_type: :bullish_breakout
                },
                timeframe: @candles.first.timeframe
              )
            end
          end

          @swings.each_cons(3) do |a, b, c|
            next unless a.low? && b.high? && c.low?

            # Bearish BOS: c breaks below a
            if c.price < a.price
              structures << Structure.new(
                type: "BOS",
                timestamp: c.timestamp,
                price_level: c.price,
                bullish: false,
                bearish: true,
                context: {
                  previous_structure_low: a.price,
                  breakout_amount: (a.price - c.price).round(4),
                  swing_high: b.price,
                  bos_type: :bearish_breakdown
                },
                timeframe: @candles.first.timeframe
              )
            end
          end

          structures
        end
      end
    end
  end
  ```
- **Deliverable:** BOS detection for bullish breakout and bearish breakdown
- **Acceptance Criteria:** Bullish BOS when HH breaks previous structure high; Bearish BOS when LL breaks previous structure low

### Task 3.2: Create `ChangeOfCharacterDetector` (CHOCH)
- **File:** `app/engines/market_structure/change_of_character_detector.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketStructure
      class ChangeOfCharacterDetector
        def initialize(candles:, swing_points:)
          @candles = candles
          @swings = swing_points
        end

        def call
          return [] if @swings.length < 4

          structures = []

          # Bullish CHOCH: after LL, price takes out previous HL
          @swings.each_cons(4) do |a, b, c, d|
            next unless a.high? && b.low? && c.high? && d.low?

            # Bearish structure: LH then LL (a=H, b=L, c=H, d=L where d < b)
            next unless c.price < a.price && d.price < b.price

            # CHOCH: next high after d takes out c
            # Simplified: check if c was a lower high and subsequent move breaks above c
            # For this detector, we look for the pattern: HH -> HL -> LH -> LL -> CHOCH
            # Actually CHOCH is when internal structure breaks against larger trend
            # Bullish CHOCH: price makes LL then breaks above previous LH
            structures << Structure.new(
              type: "CHOCH",
              timestamp: d.timestamp,
              price_level: d.price,
              bullish: true,
              bearish: false,
              context: {
                previous_trend: :bearish,
                internal_low: d.price,
                previous_internal_high: c.price,
                choch_type: :bullish_reversal
              },
              timeframe: @candles.first.timeframe
            ) if d.price < b.price
          end

          # Bearish CHOCH: after HH, price takes out previous LH
          @swings.each_cons(4) do |a, b, c, d|
            next unless a.low? && b.high? && c.low? && d.high?

            next unless c.price > a.price && d.price > b.price

            structures << Structure.new(
              type: "CHOCH",
              timestamp: d.timestamp,
              price_level: d.price,
              bullish: false,
              bearish: true,
              context: {
                previous_trend: :bullish,
                internal_high: d.price,
                previous_internal_low: c.price,
                choch_type: :bearish_reversal
              },
              timeframe: @candles.first.timeframe
            ) if d.price > b.price
          end

          structures
        end
      end
    end
  end
  ```
- **Deliverable:** CHOCH detection for trend reversals
- **Acceptance Criteria:** Bullish CHOCH after LL breaks previous LH; Bearish CHOCH after HH breaks previous HL

### Task 3.3: Create `StructureValidator`
- **File:** `app/validators/structure_validator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Validators
    class StructureValidator
      RULES = {
        price_positive: ->(s) { s.price_level.to_f > 0 },
        timestamp_present: ->(s) { s.timestamp.present? },
        type_valid: ->(s) { Engines::MarketStructureEngine::STRUCTURE_TYPES.include?(s.type) },
        direction_consistent: ->(s) { !(s.bullish && s.bearish) }
      }.freeze

      def self.validate(structure)
        errors = []
        RULES.each do |name, rule|
          errors << name unless rule.call(structure)
        end

        {
          valid: errors.empty?,
          errors: errors,
          type: structure.type
        }
      end
    end
  end
  ```
- **Deliverable:** Structure integrity validation
- **Acceptance Criteria:** `StructureValidator.validate(structure)` catches invalid type or conflicting direction

### Task 4.1: Create `LiquiditySweepDetector`
- **File:** `app/engines/market_structure/liquidity_sweep_detector.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketStructure
      class LiquiditySweepDetector
        def initialize(candles:, swing_points:)
          @candles = candles
          @swings = swing_points
        end

        def call
          return [] if @candles.length < 10 || @swings.length < 3

          structures = []

          # Bullish sweep: wick below previous swing low, close back above
          @swings.select(&:low?).each_cons(2) do |prev_low, curr_low|
            next unless curr_low.price < prev_low.price

            # Look for candle between prev_low and curr_low that wicked below prev_low but closed above
            range = @candles[prev_low.index..curr_low.index]
            sweep = range.find do |c|
              c.low.to_d < prev_low.price && c.close.to_d > prev_low.price
            end

            if sweep
              structures << Structure.new(
                type: "liquidity_sweep",
                timestamp: sweep.timestamp,
                price_level: sweep.low.to_d,
                bullish: true,
                bearish: false,
                context: {
                  sweep_level: prev_low.price,
                  close_after_sweep: sweep.close.to_d,
                  wick_size: (prev_low.price - sweep.low.to_d).round(4),
                  sweep_type: :bullish_stop_hunt
                },
                timeframe: @candles.first.timeframe
              )
            end
          end

          # Bearish sweep: wick above previous swing high, close back below
          @swings.select(&:high?).each_cons(2) do |prev_high, curr_high|
            next unless curr_high.price > prev_high.price

            range = @candles[prev_high.index..curr_high.index]
            sweep = range.find do |c|
              c.high.to_d > prev_high.price && c.close.to_d < prev_high.price
            end

            if sweep
              structures << Structure.new(
                type: "liquidity_sweep",
                timestamp: sweep.timestamp,
                price_level: sweep.high.to_d,
                bullish: false,
                bearish: true,
                context: {
                  sweep_level: prev_high.price,
                  close_after_sweep: sweep.close.to_d,
                  wick_size: (sweep.high.to_d - prev_high.price).round(4),
                  sweep_type: :bearish_stop_hunt
                },
                timeframe: @candles.first.timeframe
              )
            end
          end

          structures
        end
      end
    end
  end
  ```
- **Deliverable:** Stop hunt / liquidity sweep detection
- **Acceptance Criteria:** Bullish sweep when wick below swing low + close above; Bearish sweep when wick above swing high + close below

### Task 4.3: Create `OrderBlockDetector`
- **File:** `app/engines/market_structure/order_block_detector.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketStructure
      class OrderBlockDetector
        def initialize(candles:)
          @candles = candles
        end

        def call
          return [] if @candles.length < 5

          structures = []

          # Bullish OB: last bearish candle before strong bullish move
          @candles.each_cons(5).with_index do |window, i|
            a, b, c, d, e = window

            # Strong bullish move: c, d, e consecutive up with increasing volume
            bullish_move = c.close > c.open && d.close > d.open && e.close > e.open &&
                         e.close.to_d > a.high.to_d

            if bullish_move
              # Find the last bearish candle before the move (a or b)
              ob = [a, b].reverse.find { |cnd| cnd.close < cnd.open }
              if ob
                structures << Structure.new(
                  type: "order_block",
                  timestamp: ob.timestamp,
                  price_level: ob.close.to_d,
                  bullish: true,
                  bearish: false,
                  context: {
                    ob_type: :bullish,
                    high: ob.high.to_d,
                    low: ob.low.to_d,
                    close: ob.close.to_d,
                    volume: ob.volume,
                    subsequent_move: (e.close.to_d - ob.close.to_d).round(4)
                  },
                  timeframe: @candles.first.timeframe
                )
              end
            end

            # Bearish OB: last bullish candle before strong bearish move
            bearish_move = c.close < c.open && d.close < d.open && e.close < e.open &&
                           e.close.to_d < a.low.to_d

            if bearish_move
              ob = [a, b].reverse.find { |cnd| cnd.close > cnd.open }
              if ob
                structures << Structure.new(
                  type: "order_block",

  module Engines
    module MarketStructure
      class StructureScoreCalculator
        WEIGHTS = {
          BOS: 25,
          CHOCH: 20,
          HH: 10,

          clamped = [[normalized, 0].max, 100].min

        end

  # frozen_string_literal: true

  class MultiTimeframeStructureAggregator < BaseService
    TIMEFRAMES = %w[1m 5m 15m].freeze
    WEIGHTS = { "1m" => 0.2, "5m" => 0.3, "15m" => 0.5 }.freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      tf_structures = TIMEFRAMES.to_h do |tf|
        cached = CacheStore.fetch("structure:#{@instrument.id}:#{tf}")
        [tf, cached]
      end

      # Check alignment: all bullish or all bearish
      scores = tf_structures.transform_values { |s| s&.dig(:structure_score) || 50 }
      bullish = tf_structures.values.count { |s| s&.dig(:bullish_structure) }
      bearish = tf_structures.values.count { |s| s&.dig(:bearish_structure) }

      weighted_score = scores.sum { |tf, score| (score || 50) * WEIGHTS[tf] }

      aligned_bullish = bullish == TIMEFRAMES.length
      aligned_bearish = bearish == TIMEFRAMES.length

      success(
        timeframes: tf_structures,
        weighted_score: weighted_score.round(2),
        aligned_bullish: aligned_bullish,
        aligned_bearish: aligned_bearish,
        mixed: !aligned_bullish && !aligned_bearish,
        recommendation: recommendation(aligned_bullish, aligned_bearish, weighted_score)
      )
    end

    private

    def recommendation(bullish, bearish, score)
      if bullish && score > 70
        :strong_long
      elsif bullish && score > 50
        :moderate_long
      elsif bearish && score < 30
        :strong_short
      elsif bearish && score < 50
        :moderate_short
      else
        :neutral
      end
    end
  end
  ```
- **Deliverable:** Multi-timeframe structure alignment
- **Acceptance Criteria:** Returns `:strong_long` when all 3 timeframes bullish and score > 70

### Task 6.2: Create `StructureConfluenceDetector`

  class StructureConfluenceDetector < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      # Fetch structures from all engines
      structures = fetch_recent_structures

      confluences = []

      # BOS + FVG confluence
      bos_list = structures.select { |s| s[:type] == "BOS" }
      fvg_list = structures.select { |s| s[:type] == "fvg" }

      bos_list.each do |bos|
        nearby_fvg = fvg_list.find { |fvg| (fvg[:price_level].to_d - bos[:price_level].to_d).abs < 10 }
        if nearby_fvg
          confluences << {
            type: :bos_fvg,
            direction: bos[:bullish] ? :bullish : :bearish,
            level: bos[:price_level],
            strength: 85
          }
        end
      end

      # Liquidity sweep + Order block confluence
      sweep_list = structures.select { |s| s[:type] == "liquidity_sweep" }
      ob_list = structures.select { |s| s[:type] == "order_block" }

      sweep_list.each do |sweep|
        nearby_ob = ob_list.find { |ob| (ob[:price_level].to_d - sweep[:price_level].to_d).abs < 15 }
        if nearby_ob
          confluences << {
            type: :sweep_ob,
            direction: sweep[:bullish] ? :bullish : :bearish,
            level: sweep[:price_level],
            strength: 90
          }
        end
      end

      # CHOCH + BOS confluence (strong trend change)
      choch_list = structures.select { |s| s[:type] == "CHOCH" }
      choch_list.each do |choch|
        follow_up_bos = bos_list.find { |bos| bos[:timestamp] > choch[:timestamp] }
        if follow_up_bos
          confluences << {
            type: :choch_bos,
            direction: choch[:bullish] ? :bullish : :bearish,
            level: choch[:price_level],
            strength: 95
          }
        end
      end

      best = confluences.max_by { |c| c[:strength] }

      success(
        confluences: confluences,
        best_confluence: best,
        confluence_count: confluences.length,
        high_confidence: confluences.any? { |c| c[:strength] >= 90 }
      )
    end

    private

    def fetch_recent_structures
      %w[1m 5m 15m].flat_map do |tf|
        cached = CacheStore.fetch("structure:#{@instrument.id}:#{tf}")
        cached&.dig(:structures) || []
      end
    end
  end

### Task 6.3: Create `StructureBasedEntryTrigger`
- **File:** `app/services/structure_based_entry_trigger.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class StructureBasedEntryTrigger < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      confluence = StructureConfluenceDetector.new(instrument: @instrument).call
      return failure(:no_confluence) if confluence.failure? || confluence.value[:confluences].empty?

      best = confluence.value[:best_confluence]
      return failure(:weak_confluence) unless best && best[:strength] >= 80

      # Verify with current price
      current = MarketQuoteService.new(instrument: @instrument).ltp.value
      return failure(:no_price) unless current

      direction = best[:direction]
      level = best[:level].to_d

      # Price must be near the confluence level (within 0.3%)
      distance = ((current - level) / level * 100).abs
      return failure(:price_far_from_level) if distance > 0.3

      success(
        direction: direction,
        entry_level: level,
        trigger_type: best[:type],
        strength: best[:strength],
        current_price: current,
        distance_percent: distance.round(4)
      )
    end
  end
  ```
- **Deliverable:** Entry trigger based on structure confluence
- **Acceptance Criteria:** Returns trigger when price within 0.3% of confluence level and strength >= 80
