### Task 1.1: Create `OptionChainIngestionPipeline`
- **File:** `app/pipelines/option_chain_ingestion_pipeline.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Pipelines
    class OptionChainIngestionPipeline
      def initialize(underlying:)
        @underlying = underlying
      end

      def ingest
        result = OptionChainService.new(underlying: @underlying).call
        return result if result.failure?

        chain = result.value
        snapshot_time = Time.current

        records = build_records(chain, snapshot_time)
        return Result.failure(:no_records) if records.empty?

        # Bulk upsert
        OptionChainSnapshot.upsert_all(
          records,
          unique_by: [:instrument_id, :snapshot_time],
          returning: false
        )

        # Cache latest
        CacheStore.write(
          "option_chain:latest:#{@underlying.id}",
          { snapshot_time: snapshot_time, records: records.length },
          ttl: 30
        )

        # Emit event
        EventBus.publish(EventTypes::OPTION_CHAIN_UPDATED, {
          underlying: @underlying.trading_symbol,
          snapshot_time: snapshot_time,
          strikes_captured: chain.length
        })

        Result.success(records.length)
      end

      private

      def build_records(chain, snapshot_time)
        records = []

        chain.each do |strike_data|
          [:ce, :pe].each do |option_type|
            option = strike_data[option_type]
            next unless option

            instrument = Instrument.find_by(
              trading_symbol: option[:trading_symbol],
              exchange_segment: @underlying.exchange_segment
            )
            next unless instrument

            records << {
              instrument_id: instrument.id,
              underlying_id: @underlying.id,
              snapshot_time: snapshot_time,
              strike: strike_data[:strike],
              option_type: option_type.to_s.upcase,
              ltp: option[:ltp],
              bid: option[:bid],
              ask: option[:ask],
              spread: option[:ask].to_d - option[:bid].to_d,
              volume: option[:volume],
              volume_change: calculate_volume_change(instrument, option[:volume]),
              oi: option[:oi],
              oi_change: option[:oi_change],
              iv: option[:iv],
              iv_change: calculate_iv_change(instrument, option[:iv]),
              delta: option[:delta],
              gamma: option[:gamma],
              theta: option[:theta],
              vega: option[:vega],
              iv_rank: calculate_iv_rank(instrument, option[:iv]),
              iv_percentile: calculate_iv_percentile(instrument, option[:iv]),
              gamma_velocity: calculate_gamma_velocity(instrument, option[:gamma]),
              liquidity_score: calculate_liquidity_score(option),
              oi_flow_classification: classify_oi_flow(option, option_type)
            }
          end
        end

        records
      end

      def calculate_volume_change(instrument, current_volume)
        last = OptionChainSnapshot.where(instrument: instrument)
                                 .order(snapshot_time: :desc)
                                 .first
        return 0 unless last && last.volume
        current_volume.to_i - last.volume
      end

      def calculate_iv_change(instrument, current_iv)
        last = OptionChainSnapshot.where(instrument: instrument)
                                 .order(snapshot_time: :desc)
                                 .first
        return 0 unless last && last.iv
        current_iv.to_d - last.iv
      end

      def calculate_iv_rank(instrument, current_iv)
        return nil unless current_iv

        history = OptionChainSnapshot.where(instrument: instrument)
                                    .where("snapshot_time > ?", 52.weeks.ago)
                                    .where.not(iv: nil)
                                    .order(snapshot_time: :asc)
                                    .pluck(:iv)

        return 50 if history.empty?
        history.push(current_iv.to_d)
        sorted = history.sort
        rank = sorted.index(current_iv.to_d).to_f / (sorted.length - 1) * 100
        rank.round(2)
      end

      def calculate_iv_percentile(instrument, current_iv)
        return nil unless current_iv

        history = OptionChainSnapshot.where(instrument: instrument)
                                    .where("snapshot_time > ?", 52.weeks.ago)
                                    .where.not(iv: nil)
                                    .pluck(:iv)

        return 50 if history.empty?
        Calculations::Statistics.percentile(history, current_iv.to_d).round(2)
      end

      def calculate_gamma_velocity(instrument, current_gamma)
        return 0 unless current_gamma

        last = OptionChainSnapshot.where(instrument: instrument)
                                 .order(snapshot_time: :desc)
                                 .first
        return 0 unless last && last.gamma
        current_gamma.to_d - last.gamma
      end

      def calculate_liquidity_score(option)
        return 0 unless option[:ltp] && option[:ltp] > 0

        spread_pct = ((option[:ask].to_d - option[:bid].to_d) / option[:ltp] * 100).abs
        volume_score = [option[:volume].to_i / 1000.0, 100].min
        spread_score = spread_pct <= 0.3 ? 100 : [100 - (spread_pct - 0.3) * 50, 0].max

        (volume_score * 0.5 + spread_score * 0.5).round(2)
      end

      def classify_oi_flow(option, option_type)
        # Placeholder — full classification on Day 3
        nil
      end
    end
  end
  ```
- **Deliverable:** Complete ingestion pipeline with derived metrics

  ```ruby
  # frozen_string_literal: true

  module Calculations
    class IVAnalyticsCalculator
      def self.rank(current_iv, history)
        return 50 if history.empty? || current_iv.nil?
        history = history.map(&:to_d) + [current_iv.to_d]
        sorted = history.sort
        idx = sorted.index(current_iv.to_d)
        return 50 if idx.nil? || sorted.length <= 1
        (idx.to_f / (sorted.length - 1) * 100).round(2)
      end

      def self.percentile(current_iv, history)
        return 50 if history.empty? || current_iv.nil?
        Calculations::Statistics.percentile(history.map(&:to_d), current_iv.to_d).round(2)
      end

      def self.trend(current_iv, history, lookback: 5)
        return :neutral if history.length < lookback || current_iv.nil?

        recent = history.last(lookback).map(&:to_d)
        slope = Calculations::Statistics.slope(recent, recent.length)

        if slope > 0.05
          :rising
        elsif slope < -0.05
          :falling
        else
          :stable
        end
      end

      def self.is_elevated?(current_iv, history, threshold: 70)
        return false if history.empty? || current_iv.nil?
        percentile(current_iv, history) > threshold
      end

      def self.is_compressed?(current_iv, history, threshold: 30)
        return false if history.empty? || current_iv.nil?
        percentile(current_iv, history) < threshold
      end
    end
  end
  ```
- **Deliverable:** IV Rank, Percentile, Trend, Elevated/Compressed detection

### Task 2.2: Create `GammaAnalyticsCalculator`

  ```ruby
  # frozen_string_literal: true

  module Calculations
    class GammaAnalyticsCalculator
      def self.velocity(current_gamma, previous_gamma)
        return 0 unless current_gamma && previous_gamma
        current_gamma.to_d - previous_gamma.to_d
      end

      def self.acceleration(current_velocity, previous_velocity)
        return 0 unless current_velocity && previous_velocity
        current_velocity.to_d - previous_velocity.to_d
      end

      def self.gamma_exposure(strikes_data)
        # Total gamma exposure across strikes
        strikes_data.sum { |s| s[:gamma].to_d * s[:oi].to_i }
      end

      def self.gamma_flip_point(strikes_data)
        # Strike where gamma exposure flips from positive to negative
        sorted = strikes_data.sort_by { |s| s[:strike].to_d }
        sorted.find { |s| s[:gamma].to_d < 0 }&.dig(:strike)
      end

      def self.gamma_score(gamma, velocity, acceleration)
        return 0 unless gamma
        base = gamma.abs * 10000
        vel_bonus = velocity.abs * 5000
        acc_bonus = acceleration.abs * 2500

        [base + vel_bonus + acc_bonus, 100].min
      end
    end
  end
  ```
- **Deliverable:** Gamma velocity, acceleration, exposure, flip point
- **Acceptance Criteria:** `GammaAnalyticsCalculator.gamma_score(0.005, 0.001, 0.0005)` returns calculated score

## Day 3 — OI Flow Classification

### Task 3.1: Create `OIFlowClassifier`
- **File:** `app/lib/calculations/oi_flow_classifier.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Calculations
    class OIFlowClassifier
      PATTERNS = {
        long_buildup: "Long Build-up",
        short_buildup: "Short Build-up",
        long_unwinding: "Long Unwinding",
        short_covering: "Short Covering",
        neutral: "Neutral"
      }.freeze

      def self.classify(price_change:, oi_change:, volume_change:)
        return PATTERNS[:neutral] if oi_change.nil? || volume_change.nil?

        price_up = price_change.to_f > 0
        price_down = price_change.to_f < 0
        oi_up = oi_change.to_i > 0
        oi_down = oi_change.to_i < 0
        volume_high = volume_change.to_i > 0

        if price_up && oi_up && volume_high
          PATTERNS[:long_buildup]
        elsif price_down && oi_up && volume_high
          PATTERNS[:short_buildup]
        elsif price_down && oi_down
          PATTERNS[:long_unwinding]
        elsif price_up && oi_down
          PATTERNS[:short_covering]
        else
          PATTERNS[:neutral]
        end
      end

      def self.classify_option(option_type:, underlying_change:, oi_change:, volume_change:)
        # For options: interpret based on CE/PE
        if option_type.to_s.upcase == "CE"
          # CE OI up + underlying up = Call writing (bearish)
          # CE OI up + underlying down = Call buying (bearish hedge)
          classify(price_change: underlying_change, oi_change: oi_change, volume_change: volume_change)
        else
          # PE OI up + underlying down = Put writing (bullish)
          # PE OI up + underlying up = Put buying (bullish hedge)
          classify(price_change: -underlying_change, oi_change: oi_change, volume_change: volume_change)
        end
      end

      def self.bullish_flow?(classification)
        classification == PATTERNS[:long_buildup] ||
          classification == PATTERNS[:short_covering]
      end

      def self.bearish_flow?(classification)
        classification == PATTERNS[:short_buildup] ||
          classification == PATTERNS[:long_unwinding]
      end
    end
  end
  ```
- **Deliverable:** OI flow classification for CE/PE
- **Acceptance Criteria:** `OIFlowClassifier.classify(price_change: 1, oi_change: 1000, volume_change: 500)` returns "Long Build-up"

  ```ruby
  # frozen_string_literal: true

  class OIFlowAnalyticsService < BaseService
    def initialize(underlying:)
      @underlying = underlying
    end

    def call(snapshot_time: nil)
      snapshot_time ||= OptionChainSnapshot.for_underlying(@underlying)
                                          .order(snapshot_time: :desc)
                                          .first&.snapshot_time

      return failure(:no_snapshot) unless snapshot_time

      ce_data = OptionChainSnapshot.for_underlying(@underlying)
                                  .calls
                                  .at_time(snapshot_time)
                                  .order(strike: :asc)

      pe_data = OptionChainSnapshot.for_underlying(@underlying)
                                  .puts
                                  .at_time(snapshot_time)
                                  .order(strike: :asc)

      ce_total_oi = ce_data.sum(:oi)
      pe_total_oi = pe_data.sum(:oi)
      pcr = pe_total_oi.to_f / ce_total_oi rescue 0

      bullish_strikes = ce_data.select { |s| s.oi_flow_bullish? } + pe_data.select { |s| s.oi_flow_bullish? }
      bearish_strikes = ce_data.select { |s| !s.oi_flow_bullish? } + pe_data.select { |s| !s.oi_flow_bullish? }

      success(
        snapshot_time: snapshot_time,
        pcr: pcr.round(2),
        ce_total_oi: ce_total_oi,
        pe_total_oi: pe_total_oi,
        bullish_strikes: bullish_strikes.count,
        bearish_strikes: bearish_strikes.count,
        max_pain_strike: calculate_max_pain(ce_data + pe_data),
        oi_concentration: calculate_oi_concentration(ce_data + pe_data)
      )
    end

    private

    def calculate_max_pain(strikes)
      return nil if strikes.empty?

      # Simplified max pain: strike with highest total OI
      strikes.max_by { |s| s.oi.to_i }&.strike
    end

    def calculate_oi_concentration(strikes)
      return 0 if strikes.empty?
      total_oi = strikes.sum(&:oi)
      return 0 if total_oi.zero?

      # Top 3 strikes concentration
      top3 = strikes.sort_by { |s| -s.oi.to_i }.first(3).sum(&:oi)
      (top3.to_f / total_oi * 100).round(2)
    end
  end
  ```
- **Deliverable:** PCR, max pain, OI concentration analytics
- **Acceptance Criteria:** `OIFlowAnalyticsService.new(underlying: nifty).call` returns PCR and max pain

### Task 4.1: Create `OptionIntelligenceEngine`
- **File:** `app/engines/option_intelligence_engine.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    class OptionIntelligenceEngine < BaseEngine
      def initialize(underlying:, spot_price:)
        super()
        @underlying = underlying
        @spot_price = spot_price
      end

      def analyze
        snapshot = latest_snapshot
        return failure(:no_snapshot) unless snapshot

        strikes = fetch_strikes(snapshot.snapshot_time)
        return failure(:no_strikes) if strikes.empty?

        scored_strikes = strikes.map do |strike|
          {
            strike: strike.strike,
            option_type: strike.option_type,
            scores: score_strike(strike),
            recommendation: recommend(strike)
          }
        end

        success(
          underlying: @underlying.trading_symbol,
          spot_price: @spot_price,
          atm_strike: atm_strike(scored_strikes),
          best_call: scored_strikes.select { |s| s[:option_type] == "CE" }.max_by { |s| s[:scores][:total] },
          best_put: scored_strikes.select { |s| s[:option_type] == "PE" }.max_by { |s| s[:scores][:total] },
          all_scores: scored_strikes.sort_by { |s| -s[:scores][:total] }
        )
      end

      private

      def latest_snapshot
        OptionChainSnapshot.for_underlying(@underlying)
                          .order(snapshot_time: :desc)
                          .first
      end

      def fetch_strikes(snapshot_time)
        atm = ((@spot_price / 50).round * 50)  # NIFTY step
        OptionChainSnapshot.for_underlying(@underlying)
                            .at_time(snapshot_time)
                            .where(strike: (atm - 250)..(atm + 250))
                            .order(strike: :asc)
      end

      def score_strike(strike)
        iv_score = iv_score(strike)
        gamma_score = gamma_score(strike)
        theta_score = theta_score(strike)
        liquidity_score = strike.liquidity_score || 0
        oi_score = oi_score(strike)
        flow_score = flow_score(strike)

        {
          iv: iv_score,
          gamma: gamma_score,
          theta: theta_score,
          liquidity: liquidity_score,
          oi: oi_score,
          flow: flow_score,
          total: (iv_score * 0.15 + gamma_score * 0.20 + theta_score * 0.15 +
                  liquidity_score * 0.20 + oi_score * 0.15 + flow_score * 0.15).round(2)
        }
      end

      def iv_score(strike)
        return 50 unless strike.iv_rank
        # Prefer moderate IV (30-70 rank)
        [100 - (strike.iv_rank - 50).abs * 2, 0].max
      end

      def gamma_score(strike)
        return 0 unless strike.gamma
        # Prefer higher gamma for responsiveness
        [strike.gamma * 10000, 100].min
      end

      def theta_score(strike)
        return 0 unless strike.theta
        # Lower theta decay is better (less time risk)
        decay = strike.theta.abs
        [100 - decay * 10, 0].max
      end

      def oi_score(strike)
        return 0 unless strike.oi
        [strike.oi / 5000.0, 100].min
      end

      def flow_score(strike)
        return 50 unless strike.oi_flow_classification
        case strike.oi_flow_classification
        when "Long Build-up" then 90
        when "Short Covering" then 80
        when "Neutral" then 50
        when "Long Unwinding" then 30
        when "Short Build-up" then 20
        else 50
        end
      end

      def recommend(strike)
        return :avoid if strike.liquidity_score && strike.liquidity_score < 50
        return :avoid if strike.iv_rank && strike.iv_rank > 80  # Too expensive

        if strike.delta && strike.delta.between?(0.40, 0.65)
          :ideal
        elsif strike.delta && strike.delta.between?(0.30, 0.75)
          :acceptable
        else
          :suboptimal
        end
      end

      def atm_strike(strikes)
        strikes.min_by { |s| (s[:strike].to_d - @spot_price).abs }&.dig(:strike)
      end
    end
  end
  ```
- **Deliverable:** Composite option intelligence scoring
- **Acceptance Criteria:** `OptionIntelligenceEngine.new(underlying: nifty, spot_price: 22500).analyze` returns best CE/PE

  ```ruby
  # frozen_string_literal: true

  module Engines
    class StrikeSelectionEngine < BaseEngine
      def initialize(underlying:, spot_price:, direction:)
        super()
        @underlying = underlying
        @spot_price = spot_price
        @direction = direction  # :long or :short
      end

      def select
        intelligence = fetch_intelligence
        return failure(:no_intelligence) unless intelligence

        candidates = intelligence[:all_scores]
        return failure(:no_candidates) if candidates.empty?

        # Filter by direction
        if @direction == :long
          candidates = candidates.select { |c| c[:option_type] == "CE" }
        else
          candidates = candidates.select { |c| c[:option_type] == "PE" }
        end

        # Filter by recommendation
        candidates = candidates.reject { |c| c[:recommendation] == :avoid }

        # Sort by total score
        best = candidates.max_by { |c| c[:scores][:total] }

        return failure(:no_suitable_strike) unless best

        success(
          strike: best[:strike],
          option_type: best[:option_type],
          score: best[:scores][:total],
          recommendation: best[:recommendation],
          breakdown: best[:scores]
        )
      end

      private

      def fetch_intelligence
        cached = CacheStore.fetch("option_intelligence:#{@underlying.id}")
        return cached if cached

        # Compute on-demand
        engine = OptionIntelligenceEngine.new(
          underlying: @underlying,
          spot_price: @spot_price
        )
        engine.analyze.value
      end
    end
  end
  ```
- **Deliverable:** Direction-aware strike selection
- **Acceptance Criteria:** `StrikeSelectionEngine.new(underlying: nifty, spot_price: 22500, direction: :long).select` returns best CE

  ```ruby
  # frozen_string_literal: true

  class OptionChainRepository < BaseRepository
    self.model = OptionChainSnapshot

    def self.latest_snapshot(underlying:, as_of: nil)
      as_of ||= Time.current
      snapshot = model.for_underlying(underlying)
                      .where("snapshot_time <= ?", as_of)
                      .order(snapshot_time: :desc)
                      .first

      snapshot ? Result.success(snapshot) : Result.failure(DataError.new(reason: :no_snapshot))
    end

    def self.strikes_at_time(underlying:, time:, range: 5)
      snapshot = latest_snapshot(underlying: underlying, as_of: time).value
      return Result.failure(DataError.new(reason: :no_snapshot)) unless snapshot

      atm_strike = model.for_underlying(underlying)
                       .at_time(snapshot.snapshot_time)
                       .order(Arel.sql("ABS(strike - #{snapshot.strike})"))
                       .first&.strike

      return Result.failure(DataError.new(reason: :no_atm)) unless atm_strike

      step = 50  # Configurable per underlying
      min_strike = atm_strike - range * step
      max_strike = atm_strike + range * step

      strikes = model.for_underlying(underlying)
                     .at_time(snapshot.snapshot_time)
                     .where(strike: min_strike..max_strike)
                     .order(strike: :asc, option_type: :asc)

      Result.success(strikes)
    end

    def self.iv_history(underlying:, strike:, option_type:, from:, to:)
      data = model.for_underlying(underlying)
                 .for_strike(strike)
                 .where(option_type: option_type)
                 .where(snapshot_time: from..to)
                 .order(snapshot_time: :asc)
                 .pluck(:snapshot_time, :iv)

      Result.success(data)
    end

    def self.oi_flow_history(underlying:, from:, to:)
      snapshots = model.for_underlying(underlying)
                      .where(snapshot_time: from..to)
                      .order(snapshot_time: :asc)

      flow = snapshots.group_by(&:snapshot_time).map do |time, group|
        {
          time: time,
          ce_oi: group.select { |s| s.option_type == "CE" }.sum(&:oi),
          pe_oi: group.select { |s| s.option_type == "PE" }.sum(&:oi),
          pcr: group.select { |s| s.option_type == "PE" }.sum(&:oi).to_f /
               group.select { |s| s.option_type == "CE" }.sum(&:oi)
        }
      end

      Result.success(flow)
    end

    def self.greeks_at_strike(underlying:, strike:, option_type:, time: nil)
      time ||= Time.current
      snapshot = latest_snapshot(underlying: underlying, as_of: time).value
      return Result.failure(DataError.new(reason: :no_snapshot)) unless snapshot

      record = model.for_underlying(underlying)
                   .at_time(snapshot.snapshot_time)
                   .where(strike: strike, option_type: option_type)
                   .first

      record ? Result.success(record) : Result.failure(DataError.new(reason: :strike_not_found))
    end

    def self.liquidity_ranking(underlying:, time: nil)
      time ||= Time.current
      snapshot = latest_snapshot(underlying: underlying, as_of: time).value
      return Result.failure(DataError.new(reason: :no_snapshot)) unless snapshot

      strikes = model.for_underlying(underlying)
                     .at_time(snapshot.snapshot_time)
                     .order(liquidity_score: :desc)

      Result.success(strikes)
    end
  end
  ```
- **Deliverable:** Rich option chain query interface
- **Acceptance Criteria:** `OptionChainRepository.oi_flow_history` returns PCR timeseries

  ```ruby
  module Validators
    class OptionChainValidator
      RULES = {
        delta_range: ->(s) { s.delta.nil? || s.delta.between?(-1, 1) },
        gamma_positive: ->(s) { s.gamma.nil? || s.gamma >= 0 },
        theta_negative: ->(s) { s.theta.nil? || s.theta <= 0 },  # Theta is typically negative for long options
        vega_positive: ->(s) { s.vega.nil? || s.vega >= 0 },
        iv_positive: ->(s) { s.iv.nil? || s.iv >= 0 },
        spread_non_negative: ->(s) { s.spread.nil? || s.spread >= 0 },
        oi_non_negative: ->(s) { s.oi.nil? || s.oi >= 0 },
        volume_non_negative: ->(s) { s.volume.nil? || s.volume >= 0 }
      }.freeze

      def self.validate(snapshot)
        errors = []
        RULES.each do |name, rule|
          errors << name unless rule.call(snapshot)
        end

        {
          valid: errors.empty?,
          errors: errors,
          instrument: snapshot.instrument&.trading_symbol
        }
      end

      def self.validate_batch(snapshots)
        snapshots.map { |s| validate(s) }
      end
    end
  end
  ```
- **Deliverable:** Greeks validation rules
- **Acceptance Criteria:** `OptionChainValidator.validate(snapshot)` catches negative gamma or delta > 1

  ```ruby
  # frozen_string_literal: true

  class OptionChainQualityChecker < BaseService
    def initialize(underlying:)
      @underlying = underlying
    end

    def call
      latest = OptionChainSnapshot.for_underlying(@underlying)
                                 .order(snapshot_time: :desc)
                                 .first

      return failure(:no_snapshot) unless latest

      age = Time.current - latest.snapshot_time
      snapshots = OptionChainSnapshot.for_underlying(@underlying)
                                    .at_time(latest.snapshot_time)

      validations = Validators::OptionChainValidator.validate_batch(snapshots)
      invalid = validations.reject { |v| v[:valid] }

      # Check for missing strikes (should have ATM ±5)
      expected_strikes = 11  # ATM ±5 × 2 types = 22, but 11 unique strikes
      actual_strikes = snapshots.pluck(:strike).uniq.count

      success(
        underlying: @underlying.trading_symbol,
        snapshot_age_seconds: age.round,
        snapshot_fresh: age < 2.minutes,
        record_count: snapshots.count,
        expected_strikes: expected_strikes,
        actual_strikes: actual_strikes,
        strike_coverage: (actual_strikes.to_f / expected_strikes * 100).round(2),
        invalid_count: invalid.length,
        quality_score: calculate_score(age, actual_strikes, expected_strikes, invalid.length)
      )
    end

    private

    def calculate_score(age, actual, expected, invalid)
      freshness = age < 2.minutes ? 30 : 0
      coverage = [actual.to_f / expected * 30, 30].min
      validity = invalid.zero? ? 40 : [40 - invalid * 2, 0].max
      (freshness + coverage + validity).round(2)
    end
  end
  ```
- **Deliverable:** Composite option chain quality score
- **Acceptance Criteria:** Score < 90 triggers alert for stale or incomplete chain
