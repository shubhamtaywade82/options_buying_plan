```ruby
# app/engines/execution/execution_engine.rb
# frozen_string_literal: true

module Engines
  module Execution
    class ExecutionEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[
        trade_setup
        risk_validation
        instrument
        direction
        quantity
        entry_price
        stop_loss
        target
      ].freeze

      MAX_RETRIES = 3
      RETRY_BACKOFF = [1, 2, 4].freeze  # seconds

      def initialize(user:)
        @user = user
        @order_placers = {}
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        setup = context[:trade_setup]
        instrument = context[:instrument]
        direction = context[:direction]
        quantity = context[:quantity]
        entry_price = context[:entry_price]
        stop_loss = context[:stop_loss]
        target = context[:target]

        # Determine order type based on setup
        order_type = determine_order_type(context)

        # Place order with retry logic
        order_result = place_with_retry(
          instrument: instrument,
          side: direction == :long ? "buy" : "sell",
          quantity: quantity,
          order_type: order_type,
          price: entry_price,
          trigger_price: stop_loss,
          product_type: determine_product_type(setup),
          setup: setup
        )

        return order_result if order_result.failure?

        order = order_result.value

        # Monitor for fills
        monitor_result = monitor_fill(order, context)
        return monitor_result if monitor_result.failure?

        # Record execution metrics
        record_execution_metrics(order, context)

        emit(EventTypes::ORDER_FILLED, {
          order_id: order.id,
          broker_order_id: order.broker_order_id,
          instrument: instrument.trading_symbol,
          filled_quantity: order.filled_quantity,
          average_price: order.average_price,
          slippage: order.slippage
        })

        success(
          order: order,
          execution_summary: build_execution_summary(order, context),
          metrics: execution_metrics(order)
        )
      end

      private

      def determine_order_type(context)
        if context[:stop_entry]
          "sl"
        elsif context[:limit_price]
          "limit"
        else
          "market"
        end
      end

      def determine_product_type(setup)
        # MIS for intraday, NRML for overnight (but options are always MIS/CNC)
        return "MIS" if setup.instrument.option?

        # Check if expiry is today
        return "MIS" if setup.instrument.expiry == Date.current

        "MIS"  # Default to MIS for scalping
      end

      def place_with_retry(instrument:, side:, quantity:, order_type:, price:, trigger_price:, product_type:, setup:)
        retries = 0

        begin
          placer = get_placer(order_type)
          placer.place(
            instrument: instrument,
            side: side,
            quantity: quantity,
            price: price,
            trigger_price: trigger_price,
            product_type: product_type
          )
        rescue => e
          retries += 1
          if retries <= MAX_RETRIES
            ExecutionMetrics.record(:retry, { attempt: retries, error: e.message })
            sleep(RETRY_BACKOFF[retries - 1])
            retry
          else
            Result.failure(
              ExecutionError.new(
                code: :max_retries_exceeded,
                message: "Order failed after #{MAX_RETRIES} retries: #{e.message}",
                instrument: instrument.trading_symbol,
                setup_id: setup.id
              )
            )
          end
        end
      end

      def get_placer(order_type)
        @order_placers[order_type] ||= case order_type
                                        when "limit" then LimitOrderPlacer.new(user: @user)
                                        when "sl" then SLOrderPlacer.new(user: @user)
                                        when "sl_m" then SLMOrderPlacer.new(user: @user)
                                        else MarketOrderPlacer.new(user: @user)
                                        end
      end

      def monitor_fill(order, context)
        timeout = 30  # seconds
        start_time = Time.current

        loop do
          order.reload

          if order.filled?
            return Result.success(order)
          elsif order.cancelled? || order.rejected?
            return Result.failure(
              ExecutionError.new(
                code: :order_failed,
                message: "Order #{order.status}: #{order.rejection_reason}",
                order_id: order.id
              )
            )
          elsif order.partial?
            handle_partial_fill(order, context)
          end

          if Time.current - start_time > timeout
            # Cancel remaining and return what we have
            cancel_result = OrderService.new(user: @user).cancel(order: order)
            return Result.failure(
              ExecutionError.new(
                code: :fill_timeout,
                message: "Fill timeout after #{timeout}s, #{order.filled_quantity}/#{order.quantity} filled",
                order_id: order.id
              )
            )
          end

          sleep 0.5
        end
      end

      def handle_partial_fill(order, context)
        # Decide whether to continue waiting or adjust
        filled_ratio = order.filled_quantity.to_f / order.quantity

        if filled_ratio >= 0.8
          # Close to complete, wait longer
          return
        elsif filled_ratio >= 0.5
          # Modify to market for remainder
          modify_to_market(order)
        else
          # Cancel and retry
          cancel_and_retry(order, context)
        end
      end

      def modify_to_market(order)
        OrderService.new(user: @user).modify(
          order: order,
          params: { order_type: "market" }
        )
      end

      def cancel_and_retry(order, context)
        OrderService.new(user: @user).cancel(order: order)
        # Retry logic handled by caller
      end

      def record_execution_metrics(order, context)
        expected_price = context[:entry_price].to_d
        actual_price = order.average_price.to_d
        slippage = expected_price.zero? ? 0 : ((actual_price - expected_price).abs / expected_price * 100).round(4)

        order.update!(
          slippage: slippage,
          execution_latency_ms: ((order.executed_at - order.placed_at) * 1000).round(2) if order.executed_at && order.placed_at
        )

        ExecutionMetrics.record(:execution, {
          instrument: order.instrument.trading_symbol,
          slippage: slippage,
          latency_ms: order.execution_latency_ms,
          quantity: order.quantity,
          filled_quantity: order.filled_quantity
        })
      end

      def build_execution_summary(order, context)
        {
          order_id: order.id,
          broker_order_id: order.broker_order_id,
          instrument: order.instrument.trading_symbol,
          side: order.side,
          quantity: order.quantity,
          filled_quantity: order.filled_quantity,
          entry_price: order.average_price,
          expected_price: context[:entry_price],
          slippage: order.slippage,
          stop_loss: context[:stop_loss],
          target: context[:target],
          product_type: order.product_type,
          placed_at: order.placed_at,
          executed_at: order.executed_at,
          execution_latency_ms: order.execution_latency_ms
        }
      end

      def execution_metrics(order)
        {
          fill_rate: (order.filled_quantity.to_f / order.quantity * 100).round(2),
          slippage_bps: (order.slippage * 100).round(2),  # Convert to basis points
          latency_ms: order.execution_latency_ms,
          retry_count: order.execution_log&.count { |log| log["type"] == "retry" } || 0
        }
      end
    end
  end
end
```

```ruby
module Engines
  module Execution
    class BaseOrderPlacer
      attr_reader :user

      def initialize(user:)
        @user = user
      end

      def place(instrument:, side:, quantity:, price:, trigger_price:, product_type:)
        raise NotImplementedError
      end

      protected

      def build_payload(instrument, side, quantity, price, trigger_price, product_type)
        {
          dhanClientId: AppConfig.config.broker.client_id,
          correlationId: SecureRandom.uuid,
          transactionType: side.upcase,
          exchangeSegment: instrument.exchange_segment,
          productType: product_type,
          orderType: order_type_code,
          validity: "DAY",
          tradingSymbol: instrument.trading_symbol,
          securityId: instrument.security_id,
          quantity: quantity,
          price: price.to_f.round(2),
          triggerPrice: trigger_price.to_f.round(2),
          afterMarket: false
        }
      end

      def order_type_code
        raise NotImplementedError
      end

      def execute_place(payload)
        result = Gateway::DhanHQ::RestClient.post("/orders", payload: payload)
        return result if result.failure?

        normalized = Gateway::DhanHQ::ResponseNormalizer.normalize(
          result.value,
          type: :order
        )

        Result.success(normalized)
      end

      def persist_order(order_data, instrument, side, quantity, price, product_type)
        Order.create!(
          user: @user,
          instrument: instrument,
          broker_order_id: order_data[:broker_order_id],
          side: side,
          quantity: quantity,
          price: price,
          order_type: self.class.name.demodulize.downcase.sub("placer", ""),
          product_type: product_type,
          status: order_data[:status],
          placed_at: order_data[:placed_at] || Time.current,
          execution_log: [{ type: "placed", at: Time.current.iso8601 }]
        )
      end
    end
  end
end
```

```ruby
module Engines
  module Execution
    class SLOrderPlacer < BaseOrderPlacer
      def place(instrument:, side:, quantity:, price:, trigger_price:, product_type:)
        # For SL orders: trigger_price is the stop trigger, price is the limit price
        # Ensure limit price is appropriate for the direction

        adjusted_price = adjust_limit_price(side, price, trigger_price)

        payload = build_payload(
          instrument,
          side,
          quantity,
          adjusted_price,
          trigger_price,
          product_type
        )

        result = execute_place(payload)
        return result if result.failure?

        order = persist_order(
          result.value,
          instrument,
          side,
          quantity,
          adjusted_price,
          product_type
        )

        Result.success(order)
      end

      protected

      def order_type_code
        "SL"
      end

      private

      def adjust_limit_price(side, price, trigger_price)
        # For buy SL: limit price should be slightly above trigger
        # For sell SL: limit price should be slightly below trigger
        buffer = (trigger_price * 0.001).round(2)  # 0.1% buffer

        if side == "buy"
          [price, trigger_price + buffer].max
        else
          [price, trigger_price - buffer].min
        end
      end
    end
  end
end
```

**Task 3.1: Create `OrderSlicer`**

```ruby
# app/engines/execution/order_slicer.rb
# frozen_string_literal: true

module Engines
  module Execution
    class OrderSlicer
      MAX_SLICE_SIZE = 1000  # Maximum quantity per slice
      MIN_SLICE_SIZE = 50    # Minimum quantity per slice
      SLICE_DELAY = 0.5      # Seconds between slices

      def initialize(user:)
        @user = user
      end

      def call(instrument:, side:, quantity:, order_type:, price:, trigger_price:, product_type:)
        return single_order(
          instrument: instrument,
          side: side,
          quantity: quantity,
          order_type: order_type,
          price: price,
          trigger_price: trigger_price,
          product_type: product_type
        ) if quantity <= MAX_SLICE_SIZE

        slices = calculate_slices(quantity)
        results = []

        slices.each_with_index do |slice_qty, index|
          # Wait between slices (except first)
          sleep(SLICE_DELAY) if index > 0

          result = place_slice(
            instrument: instrument,
            side: side,
            quantity: slice_qty,
            order_type: order_type,
            price: price,
            trigger_price: trigger_price,
            product_type: product_type,
            slice_number: index + 1,
            total_slices: slices.length
          )

          results << result

          # Stop if critical failure
          break if result.failure? && index < slices.length - 1
        end

        aggregate_results(results)
      end

      private

      def calculate_slices(total_quantity)
        # Optimize slice sizes based on liquidity
        num_slices = (total_quantity.to_f / MAX_SLICE_SIZE).ceil
        base_size = total_quantity / num_slices

        slices = []
        remaining = total_quantity

        num_slices.times do
          slice = [remaining, base_size].min
          slice = [slice, MIN_SLICE_SIZE].max
          slices << slice
          remaining -= slice
        end

        slices
      end

      def single_order(instrument:, side:, quantity:, order_type:, price:, trigger_price:, product_type:)
        placer = get_placer(order_type)
        placer.place(
          instrument: instrument,
          side: side,
          quantity: quantity,
          price: price,
          trigger_price: trigger_price,
          product_type: product_type
        )
      end

      def place_slice(instrument:, side:, quantity:, order_type:, price:, trigger_price:, product_type:, slice_number:, total_slices:)
        placer = get_placer(order_type)
        result = placer.place(
          instrument: instrument,
          side: side,
          quantity: quantity,
          price: adjust_price_for_slice(price, slice_number),
          trigger_price: trigger_price,
          product_type: product_type
        )

        # Tag slice metadata
        if result.success?
          order = result.value
          order.update!(
            execution_log: order.execution_log + [{
              type: "slice",
              slice_number: slice_number,
              total_slices: total_slices,
              at: Time.current.iso8601
            }]
          )
        end

        result
      end

      def adjust_price_for_slice(price, slice_number)
        # Slight price adjustment for each slice to avoid detection
        return price if price.zero? || price.nil?

        # Vary by 0.02% per slice
        adjustment = 1 + (slice_number * 0.0002)
        (price * adjustment).round(2)
      end

      def get_placer(order_type)
        case order_type.to_s
        when "limit" then LimitOrderPlacer.new(user: @user)
        when "sl" then SLOrderPlacer.new(user: @user)
        when "sl_m" then SLMOrderPlacer.new(user: @user)
        else MarketOrderPlacer.new(user: @user)
        end
      end

      def aggregate_results(results)
        successful = results.select(&:success?)
        failed = results.select(&:failure?)

        if successful.empty?
          return Result.failure(
            ExecutionError.new(
              code: :all_slices_failed,
              message: "All #{results.length} order slices failed",
              failures: failed.map { |f| f.error.to_h }
            )
          )
        end

        total_filled = successful.sum { |r| r.value.filled_quantity.to_i }
        total_ordered = successful.sum { |r| r.value.quantity.to_i }

        Result.success(
          orders: successful.map(&:value),
          total_filled: total_filled,
          total_ordered: total_ordered,
          fill_rate: total_ordered.zero? ? 0 : (total_filled.to_f / total_ordered * 100).round(2),
          failed_slices: failed.length
        )
      end
    end
  end
end
```

**Task 3.2: Create `RetryLogic`**

```ruby
# app/engines/execution/retry_logic.rb
# frozen_string_literal: true

module Engines
  module Execution
    class RetryLogic
      MAX_RETRIES = 3
      BACKOFF_BASE = 1.0

      RETRYABLE_ERRORS = [
        :timeout,
        :rate_limit,
        :network_error,
        :broker_unavailable,
        :order_pending
      ].freeze

      NON_RETRYABLE_ERRORS = [
        :insufficient_funds,
        :invalid_order,
        :market_closed,
        :symbol_not_found,
        :rejected_by_rms
      ].freeze

      def initialize(max_retries: MAX_RETRIES, backoff_base: BACKOFF_BASE)
        @max_retries = max_retries
        @backoff_base = backoff_base
        @attempt = 0
      end

      def execute
        @attempt = 0

        loop do
          @attempt += 1
          result = yield

          return result if result.success?
          return result if non_retryable?(result.error)

          break if @attempt >= @max_retries

          wait = calculate_backoff
          ExecutionMetrics.record(:retry, {
            attempt: @attempt,
            next_wait: wait,
            error: result.error.code
          })

          sleep(wait)
        end

        Result.failure(
          ExecutionError.new(
            code: :max_retries_exceeded,
            message: "Failed after #{@max_retries} attempts",
            last_error: yield.error.to_h
          )
        )
      end

      def attempt_count
        @attempt
      end

      private

      def non_retryable?(error)
        return false unless error.respond_to?(:code)
        NON_RETRYABLE_ERRORS.include?(error.code)
      end

      def calculate_backoff
        # Exponential backoff with jitter
        base = @backoff_base * (2 ** (@attempt - 1))
        jitter = rand * 0.5  # 0-50% jitter
        (base + jitter).round(2)
      end
    end
  end
end
```

**Task 4.1: Create `PartialFillHandler`**

```ruby
# app/engines/execution/partial_fill_handler.rb
# frozen_string_literal: true

module Engines
  module Execution
    class PartialFillHandler
      FILL_TIMEOUT = 30  # seconds to wait for complete fill
      MIN_FILL_TO_KEEP = 0.5  # Keep order if >= 50% filled

      def initialize(user:)
        @user = user
      end

      def handle(order, target_quantity:)
        return Result.success(order) if order.filled_quantity.to_i >= target_quantity

        filled_ratio = order.filled_quantity.to_f / target_quantity

        if filled_ratio >= MIN_FILL_TO_KEEP
          # Wait longer or modify to market
          wait_or_modify(order, remaining: target_quantity - order.filled_quantity)
        else
          # Cancel and retry
          cancel_and_retry(order, target_quantity)
        end
      end

      private

      def wait_or_modify(order, remaining:)
        start_time = Time.current

        loop do
          order.reload

          if order.filled?
            return Result.success(order)
          end

          elapsed = Time.current - start_time

          if elapsed > FILL_TIMEOUT
            # Modify remaining to market order
            return modify_to_market(order, remaining)
          end

          sleep 0.5
        end
      end

      def modify_to_market(order, remaining)
        # Cancel existing and place market order for remainder
        cancel_result = OrderService.new(user: @user).cancel(order: order)
        return cancel_result if cancel_result.failure?

        # Place market order for remaining
        placer = MarketOrderPlacer.new(user: @user)
        placer.place(
          instrument: order.instrument,
          side: order.side,
          quantity: remaining,
          price: nil,
          trigger_price: nil,
          product_type: order.product_type
        )
      end

      def cancel_and_retry(order, target_quantity)
        cancel_result = OrderService.new(user: @user).cancel(order: order)
        return cancel_result if cancel_result.failure?

        # Return failure to trigger full retry
        Result.failure(
          ExecutionError.new(
            code: :partial_fill_cancelled,
            message: "Order cancelled due to low fill rate (#{order.filled_quantity}/#{target_quantity}), retrying",
            filled_quantity: order.filled_quantity,
            target_quantity: target_quantity
          )
        )
      end
    end
  end
end
```

**Task 4.2: Create `SlippageMonitor`**

```ruby
# app/engines/execution/slippage_monitor.rb
# frozen_string_literal: true

module Engines
  module Execution
    class SlippageMonitor
      SLIPPAGE_WARN_THRESHOLD = 0.10  # 0.10% warning
      SLIPPAGE_REJECT_THRESHOLD = 0.50  # 0.50% reject

      def initialize(user:)
        @user = user
      end

      def record(order:, expected_price:)
        return { slippage: 0, status: :no_reference } unless expected_price.to_d > 0

        actual = order.average_price.to_d
        expected = expected_price.to_d

        slippage = ((actual - expected).abs / expected * 100).round(4)

        # Update order
        order.update!(
          slippage: slippage,
          slippage_status: slippage_status(slippage)
        )

        # Alert if excessive
        if slippage >= SLIPPAGE_REJECT_THRESHOLD
          AlertService.notify(
            level: :warning,
            message: "Excessive slippage: #{slippage}% on #{order.instrument.trading_symbol}",
            metadata: {
              order_id: order.id,
              expected: expected,
              actual: actual,
              slippage: slippage
            }
          )
        end

        {
          slippage: slippage,
          status: slippage_status(slippage),
          expected: expected,
          actual: actual
        }
      end

      def average_slippage(period: 1.day)
        orders = Order.where(user: @user)
                      .where("placed_at > ?", period.ago)
                      .where.not(slippage: nil)

        return 0 if orders.empty?

        (orders.average(:slippage) || 0).to_f.round(4)
      end

      private

      def slippage_status(slippage)
        case slippage
        when 0..0.05 then :excellent
        when 0.05..0.10 then :good
        when 0.10..0.25 then :acceptable
        when 0.25..0.50 then :high
        else :excessive
        end
      end
    end
  end
end
```

```ruby
        @order = order
      end

      def transition_to(new_status, metadata: {})
        current = @order.status.to_sym

        unless valid_transition?(current, new_status.to_sym)
          return Result.failure(
            ExecutionError.new(
              code: :invalid_state_transition,
              message: "Cannot transition from #{current} to #{new_status}",
              current: current,
              requested: new_status
            )
          )
        end

        @order.update!(
          status: new_status.to_s,
          execution_log: @order.execution_log + [metadata.merge(
            from: current,
            to: new_status,
            at: Time.current.iso8601
          )]
        )

        # Emit state change event
        EventBus.publish(EventTypes::ORDER_STATE_CHANGED, {
          order_id: @order.id,
          broker_order_id: @order.broker_order_id,
          from: current,
          to: new_status,
          metadata: metadata
        })

        Result.success(@order)
      end

      def current_state
        @order.status.to_sym
      end

      def terminal?
        %i[completed cancelled rejected].include?(current_state)
      end

      def can_cancel?
        %i[pending open partial].include?(current_state)
      end

      def time_in_state
        return 0 unless @order.execution_log.last

        last_transition = Time.parse(@order.execution_log.last["at"]) rescue Time.current
        Time.current - last_transition
      end

      private

      def valid_transition?(from, to)
        STATE_TRANSITIONS[from]&.include?(to)
      end
    end
  end
end
```

```ruby
# app/engines/execution/execution_validator.rb
# frozen_string_literal: true

module Engines
  module Execution
    class ExecutionValidator
      VALIDATION_CHECKS = [
        :order_accepted_by_broker,
        :price_within_tolerance,
        :quantity_matches,
        :product_type_correct,
        :no_duplicate_orders
      ].freeze

      def initialize(order:, expected_params:)
        @order = order
        @expected = expected_params
      end

      def call
        VALIDATION_CHECKS.each do |check|
          result = send(check)
          return result if result.failure?
        end

        Result.success(
          valid: true,
          checks: VALIDATION_CHECKS.map { |c| { check: c, passed: true } }
        )
      end

      private

      def order_accepted_by_broker
        if @order.broker_order_id.blank?
          return Result.failure(
            ExecutionError.new(
              code: :no_broker_order_id,
              message: "Order not accepted by broker"
            )
          )
        end

        Result.success(true)
      end

      def price_within_tolerance
        expected = @expected[:price].to_d
        actual = @order.price.to_d

        return Result.success(true) if expected.zero? || actual.zero?  # Market orders

        tolerance = expected * 0.01  # 1% tolerance
        diff = (actual - expected).abs

        if diff > tolerance
          Result.failure(
            ExecutionError.new(
              code: :price_out_of_tolerance,
              message: "Execution price #{actual} differs from expected #{expected} by #{diff}",
              expected: expected,
              actual: actual,
              tolerance: tolerance
            )
          )
        else
          Result.success(true)
        end
      end

      def quantity_matches
        expected = @expected[:quantity].to_i
        actual = @order.quantity.to_i

        if actual != expected
          Result.failure(
            ExecutionError.new(
              code: :quantity_mismatch,
              message: "Order quantity #{actual} does not match expected #{expected}",
              expected: expected,
              actual: actual
            )
          )
        else
          Result.success(true)
        end
      end

      def product_type_correct
        expected = @expected[:product_type]
        actual = @order.product_type

        if actual != expected
          Result.failure(
            ExecutionError.new(
              code: :product_type_mismatch,
              message: "Product type #{actual} does not match expected #{expected}",
              expected: expected,
              actual: actual
            )
          )
        else
          Result.success(true)
        end
      end

      def no_duplicate_orders
        # Check for duplicate broker order IDs
        duplicates = Order.where(broker_order_id: @order.broker_order_id)
                          .where.not(id: @order.id)

        if duplicates.exists?
          Result.failure(
            ExecutionError.new(
              code: :duplicate_order,
              message: "Duplicate broker order ID detected",
              duplicates: duplicates.pluck(:id)
            )
          )
        else
          Result.success(true)
        end
      end
    end
  end
end
```

```ruby
module Engines
  module Execution
    class ExecutionMetrics
      METRIC_TYPES = %i[
        execution
        slippage
        latency
        fill_rate
        retry
        partial_fill
        cancellation
        modification
      ].freeze

      def self.record(metric_type, data)
        key = "execution_metrics:#{metric_type}:#{Date.current}"

        RedisPool.general.with do |redis|
          redis.hincrbyfloat(key, "count", 1)
          redis.hincrbyfloat(key, "total_value", data[:value] || 0) if data[:value]
          redis.hset(key, "last_at", Time.current.iso8601)
          redis.expire(key, 30.days)
        end

        # Also log to time-series if significant
        if significant_event?(metric_type, data)
          log_to_timeseries(metric_type, data)
        end
      end

      def self.summary(period: 1.day)
        METRIC_TYPES.index_with do |type|
          fetch_summary(type, period)
        end
      end

      def self.average_slippage(user: nil, period: 1.day)
        scope = Order.where("placed_at > ?", period.ago).where.not(slippage: nil)
        scope = scope.where(user: user) if user

        (scope.average(:slippage) || 0).to_f.round(4)
      end

      def self.fill_rate(user: nil, period: 1.day)
        scope = Order.where("placed_at > ?", period.ago).where.not(filled_quantity: nil)
        scope = scope.where(user: user) if user

        total_ordered = scope.sum(:quantity).to_f
        total_filled = scope.sum(:filled_quantity).to_f

        total_ordered.zero? ? 0 : (total_filled / total_ordered * 100).round(2)
      end

      def self.average_latency(user: nil, period: 1.day)
        scope = Order.where("placed_at > ?", period.ago).where.not(execution_latency_ms: nil)
        scope = scope.where(user: user) if user

        (scope.average(:execution_latency_ms) || 0).to_f.round(2)
      end

      def self.significant_event?(metric_type, data)
        case metric_type
        when :slippage then data[:slippage].to_f > 0.25
        when :retry then data[:attempt].to_i > 1
        when :partial_fill then data[:fill_rate].to_f < 80
        else false
        end
      end

      def self.log_to_timeseries(metric_type, data)
        # Log to market_ticks or dedicated metrics table
        # Implementation depends on monitoring infrastructure
      end

        key = "execution_metrics:#{metric_type}:#{Date.current}"
        data = RedisPool.general.with { |r| r.hgetall(key) }

        {
          count: data["count"].to_f,
          average_value: data["count"].to_f > 0 ? (data["total_value"].to_f / data["count"].to_f).round(4) : 0,
```

```markdown
## Overview

The Execution Engine handles order placement with automatic retry, partial fill management, slippage monitoring, and complete audit trails. It supports limit orders, market orders, stop-loss entries, and order slicing for large quantities.

## Order Types

| Type | Use Case | Risk |
|------|----------|------|

| Market | Guaranteed fill | Slippage risk |
| SL | Stop-loss entry | Trigger + execution gap |

- Max slice size: 1000 units

- Delay between slices: 0.5s
- Price variation: 0.02% per slice to avoid detection

| Error Type | Retry? | Backoff |
|------------|--------|---------|

| Rate limit | Yes | 1s, 2s, 4s |
```

```ruby
  def perform(trade_setup_id)
    setup = TradeSetup.find(trade_setup_id)
    return unless setup.status == "risk_validated"

        status: "execution_failed",
        rejection_reason: result.error.message
      )

      # Alert on execution failure
      AlertService.notify(
        level: :critical,
        message: "Execution failed: #{result.error.message}",
```
