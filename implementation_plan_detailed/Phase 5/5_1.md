# app/strategies/base_strategy.rb
# frozen_string_literal: true

module Strategies
  class BaseStrategy < Engines::BaseEngine
    VERSION_FORMAT = /\A\d+\.\d+\.\d+\z/.freeze

    class << self
      def strategy_name
        name.demodulize.underscore
      end

      def version
        @version ||= "1.0.0"
      end

      def version=(v)
        raise ArgumentError, "Invalid version format: #{v}" unless v.to_s.match?(VERSION_FORMAT)
        @version = v.to_s
      end

      def required_features
        []
      end

      def optional_features
        []
      end

      def timeframes
        %w[5m]
      end

      def supported_instruments
        %w[NIFTY BANKNIFTY SENSEX FINNIFTY]
      end

      def max_holding_time
        30.minutes
      end

      def description
        "Base strategy interface"
      end
    end

    def initialize(instrument:, context: {})
      @instrument = instrument
      @context = context
    end

    def call
      validation = validate_strategy_readiness
      return validation if validation.failure?

      signal = analyze_entry
      return signal if signal.failure?

      success(
        signal: signal.value,
        strategy: self.class.strategy_name,
        version: self.class.version,
        instrument: @instrument.trading_symbol,
        timestamp: Time.current.iso8601
      )
    end

    def should_enter?(context = {})
      raise NotImplementedError, "#{self.class} must implement #should_enter?"
    end

    def should_exit?(position, context = {})
      raise NotImplementedError, "#{self.class} must implement #should_exit?"
    end

    def confidence(context = {})
      0.0
    end

    def score(context = {})
      Result.success(
        signal: :neutral,
        confidence: 0,
        direction: nil,
        metadata: {}
      )
    end

    def validate_inputs!(inputs)
      missing = self.class.required_features - inputs.keys.map(&:to_s)
      return super if missing.empty?

      Result.failure(
        ValidationError.new(
          strategy: self.class.strategy_name,
          missing_features: missing,
          available_features: inputs.keys
        )
      )
    end

    private

    def validate_strategy_readiness
      errors = []

      unless self.class.timeframes.include?(@context[:timeframe])
        errors << "Unsupported timeframe: #{@context[:timeframe]}"
      end

      unless self.class.supported_instruments.include?(@instrument.trading_symbol)
        errors << "Unsupported instrument: #{@instrument.trading_symbol}"
      end

      unless market_hours_valid?
        errors << "Outside valid trading hours"
      end

      if errors.any?
        return Result.failure(
          ValidationError.new(
            strategy: self.class.strategy_name,
            errors: errors
          )
        )
      end

      Result.success(true)
    end

    def analyze_entry
      enter = should_enter?(@context)
      return Result.failure(:no_signal) unless enter

      confidence_score = confidence(@context)
      direction = determine_direction

      Result.success(
        action: :enter,
        direction: direction,
        confidence: confidence_score,
        setup_type: self.class.strategy_name,
        timestamp: Time.current,
        metadata: entry_metadata
      )
    end

    def determine_direction
      :neutral
    end

    def entry_metadata
      {
        timeframe: @context[:timeframe],
        instrument: @instrument.trading_symbol,
        spot_price: @context[:spot_price],
        features_used: self.class.required_features
      }
    end

    def market_hours_valid?
      calendar = ExchangeCalendar.for_exchange("NSE").find_by(date: Date.current)
      return false unless calendar&.trading_day

      now = Time.current
      open_time = Time.zone.parse(calendar.market_open.to_s)
      close_time = Time.zone.parse(calendar.market_close.to_s)

      now.between?(open_time, close_time)
    end
  end
end

      regime[:type] == :expansion || liquidity[:score].to_i < 40

      regime[:type].to_s.include?('trending')
    end
  end

module Strategies
  class StrategyResult
    attr_reader :signal, :direction, :confidence, :metadata, :strategy, :version, :timestamp

    def initialize(signal:, direction:, confidence:, strategy:, version:, timestamp:, metadata: {})
      @signal = signal
      @direction = direction
      @confidence = [[confidence, 0].max, 100].min
      @strategy = strategy
      @version = version
      @timestamp = timestamp
      @metadata = metadata
    end

    def valid?
      %i[long short].include?(direction) && confidence >= 50
    end

    def long?
      direction == :long
    end

    def short?
      direction == :short
    end

    def to_h
      {
        signal: signal,
        direction: direction,
        confidence: confidence,
        strategy: strategy,
        version: version,
        timestamp: timestamp.iso8601,
        metadata: metadata,
        valid: valid?
      }
    end

    def to_json(*args)
      to_h.to_json(*args)
    end

    def merge(other)
      self.class.new(
        signal: other.signal || signal,
        direction: other.direction || direction,
        confidence: [confidence, other.confidence].max,
        strategy: "#{strategy}+#{other.strategy}",
        version: "#{version}/#{other.version}",
        timestamp: [timestamp, other.timestamp].max,
        metadata: metadata.merge(other.metadata)
      )
    end
  end
end

# app/strategies/strategy_registry.rb
# frozen_string_literal: true

module Strategies
  class StrategyRegistry
    class StrategyNotFound < StandardError; end
    class InvalidStrategy < StandardError; end

    class << self
      def register(strategy_class)
        validate_strategy!(strategy_class)

        registry[strategy_class.strategy_name] = {
          class: strategy_class,
          version: strategy_class.version,
          registered_at: Time.current
        }
      end

      def find(name)
        entry = registry[name.to_s]
        raise StrategyNotFound, "Strategy '#{name}' not found. Available: #{names.join(', ')}" unless entry

        entry[:class]
      end

      def find_version(name, version)
        # For backtesting consistency, find specific version
        entry = registry[name.to_s]
        return nil unless entry

        # In production, would query versioned strategy store
        entry[:class] if entry[:version] == version.to_s
      end

      def all
        registry.values.map { |v| v[:class] }
      end

      def names
        registry.keys
      end

      def details
        registry.transform_values do |v|
          {
            version: v[:version],
            registered_at: v[:registered_at],
            required_features: v[:class].required_features,
            timeframes: v[:class].timeframes,
            instruments: v[:class].supported_instruments
          }
        end
      end

      def load_all!
        strategy_files.each do |file|
          load_strategy(file)
        end

        Rails.logger.info "[StrategyRegistry] Loaded #{registry.count} strategies: #{names.join(', ')}"
      end

      def reload!(name)
        entry = registry[name.to_s]
        return false unless entry

        file = strategy_files.find { |f| f.include?(name.to_s) }
        return false unless file

        load_strategy(file)
        true
      end

      def health_check
        registry.transform_values do |entry|
          klass = entry[:class]
          {
            version: entry[:version],
            interface_valid: klass.ancestors.include?(Strategies::BaseStrategy),
            required_methods: %i[should_enter? should_exit? confidence score].all? { |m| klass.method_defined?(m) },
            registered_at: entry[:registered_at]
          }
        end
      end

      private

      def registry
        @registry ||= {}
      end

      def strategy_files
        Dir[Rails.root.join("app/strategies/**/*.rb")].reject { |f| f.include?("base_") || f.include?("strategy_") }
      end

      def load_strategy(file)
        require_dependency file

        filename = File.basename(file, ".rb")
        klass_name = filename.camelize

        # Handle namespaced strategies
        if filename.include?("/")
          parts = filename.split("/")
          klass_name = parts.map(&:camelize).join("::")
        end

        klass = "Strategies::#{klass_name}".constantize

        if klass < Strategies::BaseStrategy
          register(klass)
          Rails.logger.info "[StrategyRegistry] Registered #{klass.strategy_name} v#{klass.version}"
        end
      rescue => e
        Rails.logger.warn "[StrategyRegistry] Failed to load #{file}: #{e.message}"
      end

      def validate_strategy!(klass)
        unless klass < Strategies::BaseStrategy
          raise InvalidStrategy, "#{klass} must inherit from Strategies::BaseStrategy"
        end

        unless klass.method_defined?(:should_enter?)
          raise InvalidStrategy, "#{klass} must implement #should_enter?"
        end

        unless klass.method_defined?(:should_exit?)
          raise InvalidStrategy, "#{klass} must implement #should_exit?"
        end
      end
    end
  end
end

module Strategies
  class StrategyLoader
    def initialize(watch: false)
      @watch = watch
      @file_hashes = {}
    end

    def load
      StrategyRegistry.load_all!
      start_watching if @watch && Rails.env.development?
    end

    def reload_changed
      changed_files.each do |file|
        StrategyRegistry.reload!(file_to_name(file))
      end
    end

    private

    def start_watching
      @watcher = Concurrent::TimerTask.new(execution_interval: 5) do
        reload_changed
      end
      @watcher.execute
    end

    def changed_files
      strategy_files.select do |file|
        current_hash = Digest::MD5.file(file).hexdigest
        previous_hash = @file_hashes[file]

        if previous_hash && previous_hash != current_hash
          @file_hashes[file] = current_hash
          true
        else
          @file_hashes[file] = current_hash unless previous_hash
          false
        end
      end
    end

    def strategy_files
      Dir[Rails.root.join("app/strategies/**/*.rb")].reject { |f| f.include?("base_") }
    end

    def file_to_name(file)
      File.basename(file, ".rb").split("/").last
    end
  end
end

module Strategies
  class StrategyValidator
    def self.validate!(strategy_class, context: {})
      errors = []

      # Check required features exist in context
      missing = strategy_class.required_features - context.keys.map(&:to_s)
      errors << "Missing required features: #{missing.join(', ')}" if missing.any?

      # Check timeframe support
      if context[:timeframe] && !strategy_class.timeframes.include?(context[:timeframe])
        errors << "Unsupported timeframe: #{context[:timeframe]}"
      end

      # Check instrument support
      if context[:instrument] && !strategy_class.supported_instruments.include?(context[:instrument].trading_symbol)
        errors << "Unsupported instrument: #{context[:instrument].trading_symbol}"
      end

      # Check method signatures
      %i[should_enter? should_exit? confidence].each do |method|
        unless strategy_class.instance_method(method).arity == -1 || strategy_class.instance_method(method).arity == 1
          errors << "##{method} must accept exactly 1 argument (context)"
        end
      end

      raise InvalidStrategy, errors.join("; ") if errors.any?

      true
    end

    def self.validate_all(context: {})
      StrategyRegistry.all.map do |strategy|
        begin
          validate!(strategy, context: context)
          { strategy: strategy.strategy_name, valid: true }
        rescue InvalidStrategy => e
          { strategy: strategy.strategy_name, valid: false, error: e.message }
        end
      end
    end
  end
end

# app/strategies/strategy_orchestrator.rb
# frozen_string_literal: true

module Strategies
  class StrategyOrchestrator < BaseService
    DEFAULT_MODE = :consensus  # :consensus, :best_signal, :weighted

    def initialize(instrument:, context:, strategies: nil, mode: DEFAULT_MODE)
      @instrument = instrument
      @context = context.is_a?(StrategyContext) ? context : StrategyContext.new(context)
      @strategies = strategies || StrategyRegistry.all
      @mode = mode
    end

    def call
      results = run_strategies

      return Result.failure(:no_strategies_run) if results.empty?

      aggregated = aggregate_results(results)

      emit(EventTypes::STRATEGY_AGGREGATED, {
        instrument: @instrument.trading_symbol,
        strategies_run: results.length,
        consensus: aggregated[:consensus],
        best_signal: aggregated[:best_signal]
      })

      Result.success(aggregated)
    end

    private

    def run_strategies
      @strategies.map do |strategy_class|
        begin
          strategy = strategy_class.new(instrument: @instrument, context: @context.to_h)
          result = strategy.call

          if result.success?
            {
              strategy: strategy_class.strategy_name,
              version: strategy_class.version,
              signal: result.value[:signal],
              valid: true
            }
          else
            {
              strategy: strategy_class.strategy_name,
              error: result.error.to_h,
              valid: false
            }
          end
        rescue => e
          {
            strategy: strategy_class.strategy_name,
            error: e.message,
            valid: false
          }
        end
      end
    end

    def aggregate_results(results)
      valid_results = results.select { |r| r[:valid] }

      case @mode
      when :consensus
        consensus_aggregation(valid_results)
      when :best_signal
        best_signal_aggregation(valid_results)
      when :weighted
        weighted_aggregation(valid_results)
      else
        consensus_aggregation(valid_results)
      end
    end

    def consensus_aggregation(results)
      return { consensus: :none, confidence: 0 } if results.empty?

      long_count = results.count { |r| r[:signal][:direction] == :long }
      short_count = results.count { |r| r[:signal][:direction] == :short }
      total = results.length

      long_confidence = results.select { |r| r[:signal][:direction] == :long }.sum { |r| r[:signal][:confidence] } / [long_count, 1].max
      short_confidence = results.select { |r| r[:signal][:direction] == :short }.sum { |r| r[:signal][:confidence] } / [short_count, 1].max

      if long_count > short_count && long_count > total * 0.5
        { consensus: :long, confidence: long_confidence, count: long_count, dissent: short_count }
      elsif short_count > long_count && short_count > total * 0.5
        { consensus: :short, confidence: short_confidence, count: short_count, dissent: long_count }
      else
        { consensus: :mixed, confidence: [long_confidence, short_confidence].max, long: long_count, short: short_count }
      end
    end

    def best_signal_aggregation(results)
      return { consensus: :none, confidence: 0 } if results.empty?

      best = results.max_by { |r| r[:signal][:confidence] }
      {
        consensus: best[:signal][:direction],
        confidence: best[:signal][:confidence],
        strategy: best[:strategy],
        version: best[:version]
      }
    end

    def weighted_aggregation(results)
      return { consensus: :none, confidence: 0 } if results.empty?

      long_weight = results.select { |r| r[:signal][:direction] == :long }.sum { |r| r[:signal][:confidence] }
      short_weight = results.select { |r| r[:signal][:direction] == :short }.sum { |r| r[:signal][:confidence] }

      if long_weight > short_weight
        { consensus: :long, confidence: long_weight / [results.length, 1].max, long_weight: long_weight, short_weight: short_weight }
      else
        { consensus: :short, confidence: short_weight / [results.length, 1].max, long_weight: long_weight, short_weight: short_weight }
      end
    end
  end
end

module Strategies
  class StrategyVersionManager
    VERSION_STORE_KEY = "strategy_versions".freeze

    def initialize
      @store = RedisPool.general
    end

    def current_version(strategy_name)
      @store.with { |r| r.hget(VERSION_STORE_KEY, strategy_name) } || "1.0.0"
    end

    def set_version(strategy_name, version)
      @store.with { |r| r.hset(VERSION_STORE_KEY, strategy_name, version) }
    end

    def record_deployment(strategy_name, version, metadata = {})
      @store.with do |r|
        r.hset("#{VERSION_STORE_KEY}:#{strategy_name}:history", Time.current.to_i, {
          version: version,
          deployed_at: Time.current.iso8601,
          metadata: metadata
        }.to_json)
      end
    end

    def deployment_history(strategy_name)
      @store.with do |r|
        r.hgetall("#{VERSION_STORE_KEY}:#{strategy_name}:history").transform_values do |v|
          JSON.parse(v, symbolize_names: true)
        end
      end
    end

    def rollback!(strategy_name, version)
      set_version(strategy_name, version)
      record_deployment(strategy_name, version, rolled_back_from: current_version(strategy_name))
    end
  end
end

# app/strategies/strategy_configuration.rb

module Strategies
  class StrategyConfiguration
    def initialize(strategy_name)
      @strategy_name = strategy_name
      @config = load_config
    end

    def get(key, default = nil)
      @config.fetch(key.to_s, default)
    end

    def set(key, value)
      @config[key.to_s] = value
      persist!
    end

    def merge!(hash)
      @config.merge!(hash.stringify_keys)
      persist!
    end

    def to_h
      @config.dup
    end

    def valid?
      required_keys.all? { |k| @config.key?(k.to_s) }
    end

    private

    def load_config
      # Load from Redis cache first, then database, then defaults
      cached = CacheStore.fetch(config_key)
      return cached if cached

      db_config = StrategyParam.find_by(strategy_name: @strategy_name)&.params || {}
      defaults = default_config

      merged = defaults.merge(db_config)
      CacheStore.write(config_key, merged, ttl: 300)
      merged
    end

    def persist!
      CacheStore.write(config_key, @config, ttl: 300)

      # Async persist to database
      StrategyParamUpsertJob.perform_later(@strategy_name, @config)
    end

    def config_key
      "strategy_config:#{@strategy_name}:#{Rails.env}"
    end

    def default_config
      case @strategy_name
      when "orb"
        { opening_range_minutes: 15, breakout_threshold: 0.3, volume_multiplier: 1.5 }
      when "trend_following"
        { ema_fast: 9, ema_slow: 21, adx_threshold: 25 }
      when "pullback"
        { vwap_deviation: 1.0, structure_confirmation: true }
      else
        {}
      end
    end

    def required_keys
      case @strategy_name
      when "orb" then %i[opening_range_minutes breakout_threshold]
      when "trend_following" then %i[ema_fast ema_slow adx_threshold]
      else []
      end
    end
  end
end

module Api
  module V1
    class StrategiesController < ApplicationController
      def index
        render json: {
          strategies: StrategyRegistry.details,
          count: StrategyRegistry.names.length
        }
      end

      def show
        strategy = StrategyRegistry.find(params[:id])
        render json: {
          name: strategy.strategy_name,
          version: strategy.version,
          description: strategy.description,
          required_features: strategy.required_features,
          timeframes: strategy.timeframes,
          instruments: strategy.supported_instruments
        }
      rescue StrategyRegistry::StrategyNotFound => e
        render json: { error: e.message }, status: :not_found
      end

      def validate
        strategy = StrategyRegistry.find(params[:id])
        context = Strategies::StrategyContext.new(params[:context] || {})

        result = StrategyValidator.validate!(strategy, context: context.to_h)
        render json: { valid: true, strategy: strategy.strategy_name }
      rescue StrategyRegistry::StrategyNotFound => e
        render json: { error: e.message }, status: :not_found
      rescue Strategies::InvalidStrategy => e
        render json: { valid: false, error: e.message }, status: :unprocessable_entity
      end

      def run
        instrument = Instrument.find(params[:instrument_id])
        strategy = StrategyRegistry.find(params[:id])

        context = Strategies::StrategyContext.new(
          params[:context].merge(spot_price: params[:spot_price])
        )

        instance = strategy.new(instrument: instrument, context: context.to_h)
        result = instance.call

        if result.success?
          render json: result.value
        else
          render json: { error: result.error.to_h }, status: :unprocessable_entity
        end
      end
    end
  end
end

# Strategy System Documentation

## Architecture

The strategy layer is a plugin-based system where each strategy inherits from `Strategies::BaseStrategy` and implements:

- `should_enter?(context)` → Boolean (trade setup detected)
- `should_exit?(position, context)` → Boolean (exit condition met)
- `confidence(context)` → Float 0-100 (setup strength)
- `score(context)` → Result with signal details

## Strategy Context

All strategies receive a `StrategyContext` value object containing:

| Field | Type | Description |
|-------|------|-------------|
| `spot_price` | Decimal | Current underlying price |
| `candles` | Array | Recent OHLCV data |
| `regime` | Hash | Market regime classification |
| `structure` | Hash | Market structure detection |
| `momentum` | Hash | Momentum analysis |
| `liquidity` | Hash | Liquidity assessment |
| `option_intelligence` | Hash | Option chain intelligence |

## Registration

Strategies auto-register on boot via `StrategyRegistry.load_all!`. Place strategy files in `app/strategies/` following the naming convention:

```
app/strategies/
  orb_strategy.rb          → Strategies::OrbStrategy
  trend_following/
    ema_cross_strategy.rb  → Strategies::TrendFollowing::EmaCrossStrategy
```

## Configuration

Per-strategy parameters are stored in `strategy_params` table and cached in Redis. Access via:

```ruby
config = StrategyConfiguration.new("orb")
config.get(:opening_range_minutes)  # => 15
```

## Multi-Strategy Modes

The `StrategyOrchestrator` supports three aggregation modes:

- **:consensus** — Majority vote with confidence weighting
- **:best_signal** — Highest confidence signal wins
- **:weighted** — Confidence-weighted sum

## Versioning

Strategies are versioned (SemVer). Backtesting uses pinned versions via `StrategyVersionManager`.
