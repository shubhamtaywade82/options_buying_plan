module Engines
  class MomentumEngine < BaseEngine
    OUTPUTS = %w[accelerating decelerating neutral dying].freeze

    def self.engine_name
      "momentum"
    end

    def self.required_features
      %w[atr ema vwap roc volume]
    end

    def initialize(instrument:, timeframe: "5m")
      super()
      @instrument = instrument
      @timeframe = timeframe
    end

    def analyze
      validate_inputs!(build_inputs)

      components = {
        atr_expansion: atr_momentum,
        ema_slope: ema_momentum,
        vwap_distance: vwap_momentum,
        roc: roc_momentum,
        volume_acceleration: volume_momentum
      }

      score = MomentumScoreCalculator.new(components: components).call
      classification = classify_momentum(score.value, components)

      emit(EventTypes::REGIME_CHANGED, {
        engine: engine_name,
        instrument_id: @instrument.id,
        momentum: classification[:direction],
        score: score.value,
        components: components,
        timestamp: Time.current
      })

      success(
        instrument: @instrument.trading_symbol,
        timeframe: @timeframe,
        momentum: classification[:direction],
        momentum_score: score.value,
        components: components,
        classification: classification,
        timestamp: Time.current
      )
    end

    private

    def build_inputs
      { instrument: @instrument, timeframe: @timeframe }
    end

    def atr_momentum
      atr = fetch_indicator(:atr)
      return { status: :missing } unless atr

      {
        status: atr[:expansion] ? :expanding : atr[:contraction] ? :contracting : :stable,
        value: atr[:value],
        expansion: atr[:expansion],
        contraction: atr[:contraction],
        score: atr_score(atr)
      }
    end

    def ema_momentum
      ema = fetch_indicator(:ema, period: 21)
      return { status: :missing } unless ema

      {
        status: ema[:slope].to_f > 0.05 ? :rising : ema[:slope].to_f < -0.05 ? :falling : :flat,
        slope: ema[:slope],
        above_price: ema[:above_price],
        below_price: ema[:below_price],
        score: ema_score(ema)
      }
    end

    def vwap_momentum
      vwap = fetch_indicator(:vwap)
      return { status: :missing } unless vwap

      {
        status: vwap[:distance_from_price].to_f > 0.2 ? :above : vwap[:distance_from_price].to_f < -0.2 ? :below : :near,
        distance_percent: vwap[:distance_from_price],
        above: vwap[:above_vwap],
        below: vwap[:below_vwap],
        score: vwap_score(vwap)
      }
    end

    def roc_momentum
      roc = fetch_indicator(:roc)
      return { status: :missing } unless roc

      {
        status: roc[:value].to_f > 1 ? :positive : roc[:value].to_f < -1 ? :negative : :flat,
        value: roc[:value],
        accelerating: roc[:accelerating],
        decelerating: roc[:decelerating],
        score: roc_score(roc)
      }
    end

    def volume_momentum
      rv = fetch_indicator(:relative_volume)
      return { status: :missing } unless rv

      {
        status: rv[:spike] ? :spiking : rv[:value].to_f > 1.2 ? :above_average : :normal,
        relative_volume: rv[:value],
        spike: rv[:spike],
        confirmation: rv[:confirmation],
        score: volume_score(rv)
      }
    end

    def fetch_indicator(name, **kwargs)
      result = Indicators::IndicatorRegistry.compute(
        name,
        instrument: @instrument,
        timeframe: @timeframe,
        **kwargs
      )
      result.success? ? result.value : nil
    end

    def atr_score(atr)
      return 50 unless atr[:value]
      if atr[:expansion]
        [50 + atr[:value] * 2, 100].min
      elsif atr[:contraction]
        [50 - atr[:value] * 2, 0].max
      else
        50
      end.round(2)
    end

    def ema_score(ema)
      return 50 unless ema[:slope]
      slope = ema[:slope].to_f
      if slope > 0.1
        [50 + slope * 200, 100].min
      elsif slope < -0.1
        [50 + slope * 200, 0].max
      else
        50 + slope * 100
      end.round(2)
    end

    def vwap_score(vwap)
      return 50 unless vwap[:distance_from_price]
      dist = vwap[:distance_from_price].to_f
      if dist > 0
        [50 + dist * 100, 100].min
      else
        [50 + dist * 100, 0].max
      end.round(2)
    end

    def roc_score(roc)
      return 50 unless roc[:value]
      val = roc[:value].to_f
      if val > 0
        [50 + val * 5, 100].min
      else
        [50 + val * 5, 0].max
      end.round(2)
    end

    def volume_score(rv)
      return 50 unless rv[:value]
      val = rv[:value].to_f
      if val > 1.5
        90
      elsif val > 1.2
        75
      elsif val > 0.8
        60
      else
        40
      end
    end

    def classify_momentum(score, components)
      bullish_count = components.count { |_, c| c[:score].to_f > 60 }
      bearish_count = components.count { |_, c| c[:score].to_f < 40 }

      if score >= 70 && bullish_count >= 3
        { direction: :accelerating, strength: :strong, bias: :bullish }
      elsif score >= 70
        { direction: :accelerating, strength: :strong, bias: :neutral }
      elsif score <= 30 && bearish_count >= 3
        { direction: :decelerating, strength: :strong, bias: :bearish }
      elsif score <= 30
        { direction: :decelerating, strength: :strong, bias: :neutral }
      elsif score.between?(45, 55)
        { direction: :dying, strength: :weak, bias: :neutral }
      else
        { direction: :neutral, strength: :moderate, bias: :neutral }
      end
    end

module Engines
  module Momentum
    class ATRMomentumAnalyzer
      def initialize(candles:, period: 14)
        @candles = candles
        @period = period
      end

      def call
        return { status: :insufficient_data } if @candles.length < @period * 2

        recent_atr = calculate_atr(@candles.last(@period))
        previous_atr = calculate_atr(@candles.last(@period * 2).first(@period))

        return { status: :missing } if previous_atr.zero?

        ratio = recent_atr / previous_atr
        change_percent = ((recent_atr - previous_atr) / previous_atr * 100).round(4)

        {
          status: ratio > 1.1 ? :expanding : ratio < 0.9 ? :contracting : :stable,
          recent_atr: recent_atr.round(4),
          previous_atr: previous_atr.round(4),
          ratio: ratio.round(4),
          change_percent: change_percent,
          score: score(ratio)
        }
      end

      private

      def calculate_atr(candles)
        tr_values = candles.each_cons(2).map do |prev, curr|
          [
            curr.high.to_d - curr.low.to_d,
            (curr.high.to_d - prev.close.to_d).abs,
            (curr.low.to_d - prev.close.to_d).abs
          ].max
        end
        tr_values.sum.to_f / tr_values.length
      end

      def score(ratio)
        if ratio > 1.2
          90
        elsif ratio > 1.1
          75
        elsif ratio > 0.95
          60
        elsif ratio > 0.9
          40
        else
          25
        end
      end
    end
  end
end

```ruby
module Engines
  module Momentum
    class MomentumEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[atr ema vwap roc volume].freeze

      def initialize(instrument:)
        @instrument = instrument
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        features = result.value

        acceleration = detect_acceleration(features)
        deceleration = detect_deceleration(features)
        divergence = detect_divergence(features)

        momentum_score = calculate_momentum_score(
          acceleration: acceleration,
          deceleration: deceleration,
          divergence: divergence
        )

        emit(EventTypes::MOMENTUM_ANALYZED, {
          instrument_id: @instrument.id,
          acceleration: acceleration,
          deceleration: deceleration,
          divergence: divergence,
          score: momentum_score
        })

        success(
          acceleration: acceleration,
          deceleration: deceleration,
          divergence: divergence,
          momentum_score: momentum_score,
          direction: momentum_direction(momentum_score)
        )
      end

      private

      def detect_acceleration(features)
        atr_expanding = features[:atr][:current] > features[:atr][:sma] * 1.2
        ema_accelerating = features[:ema][:slope] > features[:ema][:previous_slope]
        volume_confirming = features[:volume][:current] > features[:volume][:average] * 1.5

        {
          atr_expanding: atr_expanding,
          ema_accelerating: ema_accelerating,
          volume_confirming: volume_confirming,
          overall: atr_expanding && ema_accelerating
        }
      end

      def detect_deceleration(features)
        atr_contracting = features[:atr][:current] < features[:atr][:sma] * 0.8
        ema_flattening = features[:ema][:slope].abs < features[:ema][:previous_slope].abs * 0.5
        volume_drying = features[:volume][:current] < features[:volume][:average] * 0.5

        {
          atr_contracting: atr_contracting,
          ema_flattening: ema_flattening,
          volume_drying: volume_drying,
          overall: atr_contracting && ema_flattening
        }
      end

      def detect_divergence(features)
        price_making_higher_high = features[:price][:high] > features[:price][:previous_high]
        momentum_making_lower_high = features[:roc][:current] < features[:roc][:previous]

        price_making_lower_low = features[:price][:low] < features[:price][:previous_low]
        momentum_making_higher_low = features[:roc][:current] > features[:roc][:previous]

        {
          bearish_divergence: price_making_higher_high && momentum_making_lower_high,
          bullish_divergence: price_making_lower_low && momentum_making_higher_low
        }
      end

      def calculate_momentum_score(acceleration:, deceleration:, divergence:)
        score = 50  # Neutral baseline

        score += 25 if acceleration[:overall]
        score -= 25 if deceleration[:overall]
        score += 15 if divergence[:bullish_divergence]
        score -= 15 if divergence[:bearish_divergence]

        [[score, 0].max, 100].min
      end

      def momentum_direction(score)
        return :strong_bullish if score >= 80
        return :bullish if score >= 60
        return :neutral if score >= 40
        return :bearish if score >= 20
        :strong_bearish
      end
    end
  end
end
```

```ruby
module Engines
  module Momentum
    class ATRExpansionDetector
      EXPANSION_THRESHOLD = 1.2  # ATR > 20% above SMA
      CONTRACTION_THRESHOLD = 0.8

      def initialize(candles:, period: 14)
        @candles = candles
        @period = period
      end

      def call
        atr_values = calculate_atr_series
        return Result.failure(:insufficient_data) if atr_values.length < @period * 2

        current_atr = atr_values.last
        atr_sma = Calculations::Statistics.sma(atr_values, @period)

        {
          current: current_atr,
          sma: atr_sma,
          expanding: current_atr > atr_sma * EXPANSION_THRESHOLD,
          contracting: current_atr < atr_sma * CONTRACTION_THRESHOLD,
          percent_change: ((current_atr - atr_sma) / atr_sma * 100).round(2)
        }
      end

      private

      def calculate_atr_series
        tr_values = @candles.each_cons(2).map do |prev, curr|
          [
            curr.high - curr.low,
            (curr.high - prev.close).abs,
            (curr.low - prev.close).abs
          ].max
        end

        # Wilder's smoothing
        ema(tr_values, @period)
      end

      def ema(values, period)
        multiplier = 2.0 / (period + 1)
        values.each_with_object([]) do |value, emas|
          emas << if emas.empty?
                    values.first(period).sum.to_f / period
                  else
                    (value - emas.last) * multiplier + emas.last
                  end
        end
      end
    end
  end
end
```

```ruby
      def initialize(candles:, fast_period: 9, slow_period: 21)
        @candles = candles
        @fast_period = fast_period
        @slow_period = slow_period
      end

      def call
        closes = @candles.map(&:close)

        fast_ema = Calculations::Statistics.ema(closes, @fast_period)
        slow_ema = Calculations::Statistics.ema(closes, @slow_period)

        fast_slope = calculate_slope(closes.last(@fast_period * 2))
        slow_slope = calculate_slope(closes.last(@slow_period * 2))

        {
          fast: fast_ema,
          slow: slow_ema,
          slope: fast_slope,
          previous_slope: previous_slope(closes),
          acceleration: fast_slope > slow_slope,
          alignment: fast_ema > slow_ema ? :bullish : :bearish
        }
      end

      private

      def calculate_slope(values)
        Calculations::Statistics.slope(values, [values.length, 10].min)
      end

      def previous_slope(closes)
        return 0 if closes.length < 20
        previous_values = closes[-20..-10]
        Calculations::Statistics.slope(previous_values, previous_values.length)
      end
    end
  end
```

```ruby
module Engines
  module Momentum
    class VWAPDistanceCalculator
      DEVIATION_BANDS = [1.0, 2.0, 3.0].freeze

      def initialize(candles:)
        @candles = candles
      end

      def call
        vwap = calculate_vwap
        current_price = @candles.last.close
        std_dev = calculate_std_dev(vwap)

        distance = ((current_price - vwap) / vwap * 100).round(4)

        {
          vwap: vwap,
          current_price: current_price,
          distance_percent: distance,
          position: position_relative_to_vwap(distance),
          band: determine_band(distance, std_dev),
          mean_reversion_probability: mean_reversion_score(distance, std_dev)
        }
      end

      private

      def calculate_vwap
        total_value = @candles.sum { |c| c.vwap.to_f * c.volume.to_f }
        total_volume = @candles.sum(&:volume)
        total_volume.zero? ? 0 : total_value / total_volume
      end

      def calculate_std_dev(vwap)
        prices = @candles.map(&:close)
        mean = prices.sum.to_f / prices.length
        variance = prices.sum { |p| (p - mean) ** 2 } / prices.length
        Math.sqrt(variance)
      end

      def position_relative_to_vwap(distance)
        return :at_vwap if distance.abs < 0.1
        distance > 0 ? :above_vwap : :below_vwap
      end

      def determine_band(distance, std_dev)
        return :neutral if std_dev.zero?
        deviation = (distance / std_dev).abs

        case deviation
        when 0..1 then :first_band
        when 1..2 then :second_band
        else :third_band
        end
      end

      def mean_reversion_score(distance, std_dev)
        return 50 if std_dev.zero?
        deviation = (distance / std_dev).abs
        [deviation * 33, 100].min  # Higher deviation = higher reversion probability
      end
    end
  end
end
```

```ruby
module Engines
  module Momentum
    class VolumeAccelerationDetector
      THRESHOLD = 1.5  # 50% above average

      def initialize(candles:, lookback: 20)
        @candles = candles
        @lookback = lookback
      end

      def call
        volumes = @candles.last(@lookback).map(&:volume)
        return Result.failure(:insufficient_data) if volumes.length < @lookback

        current = volumes.last
        average = volumes.sum.to_f / volumes.length
        previous = volumes[-2]

        {
          current: current,
          average: average,
          previous: previous,
          ratio: current.zero? ? 0 : (current / average).round(2),
          accelerating: current > average * THRESHOLD,
          decelerating: current < average * 0.5,
          trend: volume_trend(volumes)
        }
      end

      private

      def volume_trend(volumes)
        first_half = volumes.first(volumes.length / 2).sum.to_f
        second_half = volumes.last(volumes.length / 2).sum.to_f
        return :flat if first_half.zero?

        ratio = second_half / first_half
        case ratio
        when 0..0.8 then :declining
        when 0.8..1.2 then :stable
        else :rising
        end
      end
    end
  end
end
```

```ruby
  module Momentum
    class MomentumDivergenceDetector
      def initialize(candles:, indicator: :roc)
        @candles = candles
        @indicator = indicator
      end

      def call
        prices = @candles.map(&:close)
        indicator_values = calculate_indicator

        {
          bearish_divergence: detect_bearish_divergence(prices, indicator_values),
          bullish_divergence: detect_bullish_divergence(prices, indicator_values),
          hidden_bullish: detect_hidden_bullish(prices, indicator_values),
          hidden_bearish: detect_hidden_bearish(prices, indicator_values)
        }
      end

      private

      def calculate_indicator
        case @indicator
        when :roc then calculate_roc
        when :rsi then calculate_rsi
        else calculate_roc
        end
      end

      def calculate_roc
        @candles.each_cons(2).map { |prev, curr| ((curr.close - prev.close) / prev.close * 100).round(2) }
      end

      def calculate_rsi
        # Simplified RSI calculation
        gains = []
        losses = []
        @candles.each_cons(2) do |prev, curr|
          change = curr.close - prev.close
          gains << (change > 0 ? change : 0)
          losses << (change < 0 ? change.abs : 0)
        end
        [Calculations::Statistics.rsi(gains, losses)]
      end

      def detect_bearish_divergence(prices, indicator)
        # Price higher high, indicator lower high
        recent_prices = prices.last(10)
        recent_indicator = indicator.last(10)

        price_hh = recent_prices.max == recent_prices.last
        indicator_lh = recent_indicator.last < recent_indicator.max

        price_hh && indicator_lh
      end

      def detect_bullish_divergence(prices, indicator)
        # Price lower low, indicator higher low
        recent_prices = prices.last(10)
        recent_indicator = indicator.last(10)

        price_ll = recent_prices.min == recent_prices.last
        indicator_hl = recent_indicator.last > recent_indicator.min

        price_ll && indicator_hl
      end

      def detect_hidden_bullish(prices, indicator)
        # Price higher low, indicator lower low (trend continuation)
        false  # Simplified
      end

      def detect_hidden_bearish(prices, indicator)
        # Price lower high, indicator higher high (trend continuation)
        false  # Simplified
      end
    end
```

```ruby
          divergence: score_divergence,

          components: component_scores,

        vol = @features[:volume]
        return 80 if vol[:accelerating] && vol[:trend] == :rising
        return 30 if vol[:decelerating]
        50
      end

      def score_divergence
        return 50 unless @features[:divergence]
        div = @features[:divergence]
        return 30 if div[:bearish_divergence]
        return 70 if div[:bullish_divergence]
        50
      end

      def score_roc
        return 50 unless @features[:roc]
        roc = @features[:roc]
        return 80 if roc[:current] > 2
        return 20 if roc[:current] < -2
        50
      end

      def interpret_direction(score)
        case score
        when 0..30 then :strong_bearish
        when 30..45 then :bearish
        when 45..55 then :neutral
        when 55..70 then :bullish
        else :strong_bullish
        end
      end

      def calculate_confidence(scores)
        variance = scores.values.map { |v| (v - 50).abs }.sum / scores.values.length
        [variance * 2, 100].min  # Higher variance = higher confidence
      end
    end
  end
```

```ruby
        @history_key = "momentum_history:#{@instrument.id}"

          timestamp: Time.current,
          score: reading[:momentum_score],
          direction: reading[:direction]
        }
        history = history.last(20)  # Keep last 20
        CacheStore.write(@history_key, history, ttl: 1.hour)
      end

      def persistent_signal?(direction, min_readings: MIN_PERSISTENCE)
        history = fetch_history
        return false if history.length < min_readings

        recent = history.last(min_readings)
        recent.all? { |h| h[:direction] == direction }
      end

      def momentum_shift?
        history = fetch_history
        return false if history.length < 5

        recent = history.last(5)
        # Detect shift from bullish to bearish or vice versa
        first_half = recent.first(2).map { |h| h[:direction] }
        second_half = recent.last(2).map { |h| h[:direction] }

        bullish_to_bearish = first_half.all? { |d| d.to_s.include?('bullish') } &&
                            second_half.all? { |d| d.to_s.include?('bearish') }
        bearish_to_bullish = first_half.all? { |d| d.to_s.include?('bearish') } &&
                            second_half.all? { |d| d.to_s.include?('bullish') }

        bullish_to_bearish || bearish_to_bullish
      end

      private

      def fetch_history
        CacheStore.fetch(@history_key) || []
      end
    end
  end
```

**Task 1.1: Create `LiquidityEngine`**

```ruby
# app/engines/liquidity/liquidity_engine.rb
# frozen_string_literal: true

module Engines
  module Liquidity
    class LiquidityEngine < Engines::BaseEngine
      SPREAD_THRESHOLD = 0.30  # 0.30% rejection threshold
      MIN_DEPTH_RUPEES = 500_000  # ₹5L minimum depth per side

      def initialize(instrument:)
        @instrument = instrument
      end

      def call(context: {})
        depth = context[:depth]
        oi = context[:open_interest]
        volume = context[:volume]

        return failure(:missing_depth) unless depth

        spread_analysis = analyze_spread(depth)
        imbalance = calculate_imbalance(depth)
        book_pressure = analyze_book_pressure(depth)
        slippage = estimate_slippage(depth, context[:order_size])

        liquidity_score = calculate_liquidity_score(
          spread_analysis: spread_analysis,
          imbalance: imbalance,
          book_pressure: book_pressure,
          oi: oi,
          volume: volume
        )

        emit(EventTypes::LIQUIDITY_ASSESSED, {
          instrument_id: @instrument.id,
          score: liquidity_score,
          spread: spread_analysis[:percent],
          slippage: slippage
        })

        success(
          score: liquidity_score,
          spread: spread_analysis,
          imbalance: imbalance,
          book_pressure: book_pressure,
          slippage: slippage,
          tradable: liquidity_score >= 60 && spread_analysis[:percent] <= SPREAD_THRESHOLD
        )
      end

      private

      def analyze_spread(depth)
        best_bid = depth[:bid_levels].first&.dig(:price).to_d
        best_ask = depth[:ask_levels].first&.dig(:price).to_d
        ltp = depth[:ltp].to_d

        spread = best_ask - best_bid
        spread_percent = ltp.zero? ? 0 : (spread / ltp * 100).round(4)

        {
          bid: best_bid,
          ask: best_ask,
          spread: spread,
          percent: spread_percent,
          acceptable: spread_percent <= SPREAD_THRESHOLD
        }
      end

      def calculate_imbalance(depth)
        total_bid_qty = depth[:bid_levels].sum { |l| l[:quantity].to_i }
        total_ask_qty = depth[:ask_levels].sum { |l| l[:quantity].to_i }
        total = total_bid_qty + total_ask_qty

        return { ratio: 0, direction: :neutral } if total.zero?

        bid_ratio = total_bid_qty.to_f / total
        ask_ratio = total_ask_qty.to_f / total

        {
          bid_quantity: total_bid_qty,
          ask_quantity: total_ask_qty,
          ratio: bid_ratio.round(4),
          direction: bid_ratio > 0.6 ? :bid_heavy : ask_ratio > 0.6 ? :ask_heavy : :neutral,
          pressure: (bid_ratio - 0.5).abs.round(4)
        }
      end

      def analyze_book_pressure(depth)
        bid_walls = detect_walls(depth[:bid_levels])
        ask_walls = detect_walls(depth[:ask_levels])

        {
          bid_walls: bid_walls,
          ask_walls: ask_walls,
          nearest_wall_distance: nearest_wall_distance(depth),
          absorption_zones: detect_absorption(depth)
        }
      end

      def detect_walls(levels)
        return [] if levels.empty?
        avg_qty = levels.sum { |l| l[:quantity].to_i } / levels.length
        levels.select { |l| l[:quantity].to_i > avg_qty * 3 }
      end

      def nearest_wall_distance(depth)
        ltp = depth[:ltp].to_d
        return 0 if ltp.zero?

        all_walls = (depth[:bid_levels] + depth[:ask_levels]).select { |l| l[:quantity].to_i > 1000 }
        return 0 if all_walls.empty?

        distances = all_walls.map { |w| (w[:price].to_d - ltp).abs }
        (distances.min / ltp * 100).round(4)
      end

      def detect_absorption(depth)
        # Large quantity at single level without price movement
        []
      end

      def estimate_slippage(depth, order_size)
        return { estimate: 0, confidence: 0 } unless order_size

        ltp = depth[:ltp].to_d
        return { estimate: 0, confidence: 0 } if ltp.zero?

        # Walk the book
        remaining = order_size
        total_cost = 0
        levels_consumed = 0

        depth[:ask_levels].each do |level|
          qty = level[:quantity].to_i
          price = level[:price].to_d
          take = [remaining, qty].min
          total_cost += take * price
          remaining -= take
          levels_consumed += 1
          break if remaining <= 0
        end

        if remaining > 0
          return { estimate: nil, confidence: 0, reason: :insufficient_depth }
        end

        avg_fill = total_cost / order_size
        slippage = ((avg_fill - ltp) / ltp * 100).round(4)

        {
          estimate: slippage,
          confidence: [100 - levels_consumed * 15, 10].max,
          levels_consumed: levels_consumed,
          avg_fill_price: avg_fill
        }
      end

      def calculate_liquidity_score(spread_analysis:, imbalance:, book_pressure:, oi:, volume:)
        score = 50

        # Spread component (max 25)
        score += 25 if spread_analysis[:acceptable]
        score -= 25 if spread_analysis[:percent] > SPREAD_THRESHOLD * 2

        # Imbalance component (max 15)
        score += 15 if imbalance[:direction] == :neutral
        score += 10 if imbalance[:pressure] < 0.1

        # Book pressure (max 15)
        score += 15 if book_pressure[:bid_walls].any? && book_pressure[:ask_walls].any?

        # OI component (max 20)
        score += 20 if oi && oi[:current].to_i > 100_000

        # Volume component (max 25)
        score += 25 if volume && volume[:current].to_i > volume[:average].to_i * 2

        [[score, 0].max, 100].min
      end
    end
  end
end
```

```ruby
# app/engines/liquidity/slippage_estimator.rb
# frozen_string_literal: true

module Engines
  module Liquidity
    class SlippageEstimator
      def initialize(depth:, order_size:)
        @depth = depth
        @order_size = order_size
      end

      def call
        {
          market_order: estimate_market_order,
          limit_order: estimate_limit_order,
          recommended_order_type: recommended_order_type
        }
      end

      private

      def estimate_market_order
        walk_book(@depth[:ask_levels], @order_size)
      end

      def estimate_limit_order
        best_ask = @depth[:ask_levels].first&.dig(:price).to_d
        {
          fill_probability: fill_probability_at(best_ask),
          time_to_fill: estimated_time(best_ask),
          suggested_price: suggested_limit_price
        }
      end

      def walk_book(levels, quantity)
        remaining = quantity
        total_cost = 0
        levels_used = 0

        levels.each do |level|
          available = level[:quantity].to_i
          price = level[:price].to_d
          take = [remaining, available].min

          total_cost += take * price
          remaining -= take
          levels_used += 1

          break if remaining <= 0
        end

        if remaining > 0
          return { executable: false, reason: :insufficient_liquidity }
        end

        avg_price = total_cost / quantity
        ltp = @depth[:ltp].to_d
        slippage = ltp.zero? ? 0 : ((avg_price - ltp) / ltp * 100).round(4)

        {
          executable: true,
          avg_fill_price: avg_price,
          slippage_percent: slippage,
          levels_used: levels_used,
          market_impact: slippage * 2  # Estimated price impact
        }
      end

      def fill_probability_at(price)
        depth_at_price = @depth[:ask_levels].select { |l| l[:price].to_d <= price }
        total_qty = depth_at_price.sum { |l| l[:quantity].to_i }
        total_qty >= @order_size ? 0.95 : (total_qty.to_f / @order_size * 0.95).round(4)
      end

      def estimated_time(price)
        # Simplified: based on recent trade frequency
        30  # seconds estimate
      end

      def suggested_limit_price
        ltp = @depth[:ltp].to_d
        spread = @depth[:ask_levels].first&.dig(:price).to_d - @depth[:bid_levels].first&.dig(:price).to_d
        ltp + (spread * 0.3)  # Place inside spread
      end

      def recommended_order_type
        market = estimate_market_order
        return :limit if !market[:executable] || market[:slippage_percent] > 0.5
        :market
      end
    end
  end
end
```

```ruby
        )

        emit(EventTypes::OPTION_INTELLIGENCE_ANALYZED, {
          underlying_id: @underlying.id,
          score: composite_score,
          oi_flow: oi_flow[:dominant_flow]
        })

        success(
          oi_flow: oi_flow,
          gamma: gamma_analysis,
          iv: iv_analysis,
          theta: theta_analysis,
          composite_score: composite_score,
          conviction: derive_conviction(composite_score)
        )
      end

      private

      def classify_oi_flow(chain, oi_history)
        ce_oi_change = chain.sum { |s| s[:ce][:oi_change].to_i }
        pe_oi_change = chain.sum { |s| s[:pe][:oi_change].to_i }

        ce_price_change = chain.sum { |s| s[:ce][:ltp].to_d - s[:ce][:prev_ltp].to_d }
        pe_price_change = chain.sum { |s| s[:pe][:ltp].to_d - s[:pe][:prev_ltp].to_d }

        {
          ce_oi_change: ce_oi_change,
          pe_oi_change: pe_oi_change,
          ce_price_change: ce_price_change,
          pe_price_change: pe_price_change,
          dominant_flow: determine_flow(ce_oi_change, pe_oi_change, ce_price_change, pe_price_change),
          sentiment: sentiment_from_flow(ce_oi_change, pe_oi_change)
        }
      end

      def determine_flow(ce_oi, pe_oi, ce_price, pe_price)
        # Long Build-up: OI up, Price up
        # Short Build-up: OI up, Price down
        # Long Unwinding: OI down, Price down
        # Short Covering: OI down, Price up

        if ce_oi > 0 && ce_price > 0
          :ce_long_buildup
        elsif ce_oi > 0 && ce_price < 0
          :ce_short_buildup
        elsif ce_oi < 0 && ce_price < 0
          :ce_long_unwinding
        elsif ce_oi < 0 && ce_price > 0
          :ce_short_covering
        elsif pe_oi > 0 && pe_price > 0
          :pe_long_buildup
        elsif pe_oi > 0 && pe_price < 0
          :pe_short_buildup
        elsif pe_oi < 0 && pe_price < 0
          :pe_long_unwinding
        elsif pe_oi < 0 && pe_price > 0
          :pe_short_covering
        else
          :neutral
        end
      end

      def sentiment_from_flow(ce_oi, pe_oi)
        return :neutral if ce_oi.abs < 1000 && pe_oi.abs < 1000

        if ce_oi > 0 && pe_oi < 0
          :bearish  # Call writing + Put unwinding
        elsif ce_oi < 0 && pe_oi > 0
          :bullish  # Call unwinding + Put writing
        elsif ce_oi > 0 && pe_oi > 0
          :volatile  # Both building
        else
          :consolidating  # Both unwinding
        end
      end

      def analyze_gamma(chain)
        atm_options = chain.select { |s| s[:distance_from_atm].abs <= 1 }

        {
          total_gamma: chain.sum { |s| s[:ce][:gamma].to_d + s[:pe][:gamma].to_d },
          atm_gamma: atm_options.sum { |s| s[:ce][:gamma].to_d + s[:pe][:gamma].to_d },
          gamma_velocity: calculate_gamma_velocity(chain),
          gamma_acceleration: calculate_gamma_acceleration(chain),
          pin_risk: gamma_pin_risk(chain)
        }
      end

      def calculate_gamma_velocity(chain)
        # Change in gamma from previous snapshot
        0.0  # Requires historical comparison
      end

      def calculate_gamma_acceleration(chain)
        # Second derivative of gamma
        0.0
      end

      def gamma_pin_risk(chain)
        # High gamma near ATM at expiry
        atm_gamma = chain.find { |s| s[:distance_from_atm] == 0 }
        return 0 unless atm_gamma

        gamma = atm_gamma[:ce][:gamma].to_d + atm_gamma[:pe][:gamma].to_d
        gamma > 0.01 ? :high : :low
      end

      def analyze_iv(chain, iv_history)
        current_iv = chain.map { |s| s[:ce][:iv].to_d }.sum / chain.length

        {
          current_iv: current_iv,
          iv_rank: calculate_iv_rank(current_iv, iv_history),
          iv_percentile: calculate_iv_percentile(current_iv, iv_history),
          iv_trend: iv_trend(iv_history),
          iv_score: iv_score(current_iv, iv_history)
        }
      end

      def calculate_iv_rank(current, history)
        return 50 if history.empty?
        min = history.min
        max = history.max
        return 50 if max == min
        ((current - min) / (max - min) * 100).round(2)
      end

      def calculate_iv_percentile(current, history)
        return 50 if history.empty?
        below = history.count { |h| h < current }
        (below.to_f / history.length * 100).round(2)
      end

      def iv_trend(history)
        return :unknown if history.length < 5
        recent = history.last(5)
        recent.last > recent.first ? :rising : :falling
      end

      def iv_score(current, history)
        rank = calculate_iv_rank(current, history)
        # Prefer moderate IV (30-70 percentile)
        return 80 if rank.between?(30, 70)
        return 60 if rank.between?(20, 80)
        40
      end

      def analyze_theta(chain)
        total_theta = chain.sum { |s| s[:ce][:theta].to_d.abs + s[:pe][:theta].to_d.abs }

        {
          total_theta: total_theta,
          theta_decay_per_minute: total_theta / 390,  # 6.5 hours
          max_pain_theta: chain.max_by { |s| s[:ce][:theta].to_d.abs + s[:pe][:theta].to_d.abs },
          time_risk: time_risk_assessment(total_theta)
        }
      end

      def time_risk_assessment(total_theta)
        return :low if total_theta < 0.5
        return :moderate if total_theta < 2.0
        :high
      end

      def calculate_composite_score(oi_flow:, gamma:, iv:, theta:)
        score = 50

        # OI Flow (30%)
        score += oi_flow_score(oi_flow) * 0.30

        # Gamma (20%)
        score += gamma_score(gamma) * 0.20

        # IV (25%)
        score += iv[:iv_score] * 0.25

        # Theta (15%)
        score += theta_score(theta) * 0.15

        # Liquidity (10%)
        score += 50 * 0.10  # Placeholder

        [[score.round(2), 0].max, 100].min
      end

      def oi_flow_score(oi_flow)
        case oi_flow[:dominant_flow]
        when :ce_long_buildup, :pe_short_buildup then 80  # Bearish
        when :ce_short_buildup, :pe_long_buildup then 20  # Bullish
        when :ce_long_unwinding, :pe_short_covering then 70  # Bullish unwind
        when :ce_short_covering, :pe_long_unwinding then 30  # Bearish unwind
        else 50
        end
      end

      def gamma_score(gamma)
        return 50 if gamma[:total_gamma].zero?
        return 80 if gamma[:atm_gamma] > 0.005 && gamma[:pin_risk] == :low
        60
      end

      def theta_score(theta)
        return 80 if theta[:time_risk] == :low
        return 50 if theta[:time_risk] == :moderate
        30
      end

      def derive_conviction(score)
        case score
        when 0..30 then :weak
        when 30..50 then :moderate
        when 50..70 then :strong
        else :very_strong
        end
      end
    end
  end
end
```

```ruby
      LOOKBACK_DAYS = 252  # 1 year trading days

      def initialize(iv_history:)

          min_iv: min_iv,
          max_iv: max_iv,
          current_iv: current_iv,
          confidence: [valid_history.length / LOOKBACK_DAYS.to_f * 100, 100].min
```

```ruby
module Engines
  module OptionIntelligence
    class IVPercentileCalculator
      def initialize(iv_history:)
        @history = iv_history
      end

      def call(current_iv:)
        return { percentile: 50, sample_size: 0 } if @history.empty?

        sorted = @history.compact.sort
        index = sorted.bsearch_index { |x| x >= current_iv } || sorted.length

        {
          percentile: (index.to_f / sorted.length * 100).round(2),
          sample_size: sorted.length,
          distribution: distribution_stats(sorted)
        }
      end

      private

      def distribution_stats(sorted)
        {
          mean: (sorted.sum / sorted.length).round(4),
          median: sorted[sorted.length / 2],
          std_dev: calculate_std_dev(sorted)
        }
      end

      def calculate_std_dev(values)
        mean = values.sum / values.length
        variance = values.sum { |v| (v - mean) ** 2 } / values.length
        Math.sqrt(variance).round(4)
      end
    end
  end
end
```

```ruby
module Engines
  module OptionIntelligence
    class GammaVelocityCalculator
      def initialize(snapshots:)
        @snapshots = snapshots.sort_by(&:snapshot_time)
      end

      def call(strike:, option_type:)
        relevant = @snapshots.select { |s| s.strike == strike && s.option_type == option_type }
        return { velocity: 0, acceleration: 0 } if relevant.length < 2

        gammas = relevant.map(&:gamma)
        velocities = gammas.each_cons(2).map { |prev, curr| curr.to_f - prev.to_f }

        {
          velocity: velocities.last.round(6),
          acceleration: velocities.each_cons(2).map { |p, c| c - p }.last&.round(6) || 0,
          trend: gamma_trend(velocities),
          max_gamma: gammas.max&.round(6),
          min_gamma: gammas.min&.round(6)
        }
      end

      private

      def gamma_trend(velocities)
        return :flat if velocities.empty?
        recent = velocities.last(5)
        avg = recent.sum.to_f / recent.length
        return :accelerating if avg > 0.001
        return :decelerating if avg < -0.001
        :stable
      end
    end
  end
```

```ruby
# frozen_string_literal: true

module Engines
  module OptionIntelligence
    class ThetaDecayAnalyzer
      MINUTES_PER_DAY = 390  # 6.5 hours

      def initialize(option_data:)
        @option = option_data
      end

      def call
        theta = @option[:theta].to_d.abs
        ltp = @option[:ltp].to_d

        return { risk: :unknown } if theta.zero? || ltp.zero?

        daily_decay = theta
        hourly_decay = theta / 6.5
        percent_per_day = (daily_decay / ltp * 100).round(4)

        {
          daily_decay: daily_decay,
          hourly_decay: hourly_decay.round(4),
          percent_per_day: percent_per_day,
          risk: decay_risk(percent_per_day),
          breakeven_move: breakeven_move(theta, ltp)
        }
      end

      private

      def decay_risk(percent)
        return :extreme if percent > 10
        return :high if percent > 5
        return :moderate if percent > 2
        :low
      end

      def breakeven_move(theta, ltp)
        # Price move needed to offset theta decay
        (theta / ltp * 100).round(4)
      end
    end
  end
end
```
