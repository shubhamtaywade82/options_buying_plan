## Day 1 — Tick Ingestion Pipeline (Production-Grade)

### Task 1.1: Create `TickIngestionPipeline`
- **File:** `app/pipelines/tick_ingestion_pipeline.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Pipelines
    class TickIngestionPipeline
      include AppInject

      BATCH_SIZE = 500
      FLUSH_INTERVAL = 1        # second
      MAX_BUFFER_AGE = 5        # seconds (force flush stale buffer)
      OUTLIER_THRESHOLD = 5.0   # percent

      def initialize
        @buffer = []
        @mutex = Mutex.new
        @last_flush = Time.current
        @instrument_last_price = {}
      end

      def ingest(raw_tick)
        # 1. Validate
        return Result.failure(:invalid_tick) unless valid_tick?(raw_tick)

        # 2. Outlier rejection
        if outlier?(raw_tick)
          EventBus.publish(EventTypes::SYSTEM_ERROR, {
            type: :tick_outlier,
            instrument_id: raw_tick[:instrument_id],
            price: raw_tick[:last_price]
          })
          return Result.failure(:outlier_rejected)
        end

        # 3. Normalize
        instrument = resolve_instrument(raw_tick[:instrument_id])
        return Result.failure(:instrument_not_found) unless instrument

        normalized = TickNormalizer.normalize(raw_tick, instrument: instrument)

        # 4. Buffer
        @mutex.synchronize do
          @buffer << normalized
          @instrument_last_price[instrument.id] = normalized[:last_price]
        end

        # 5. Flush check
        flush_if_needed

        Result.success(normalized)
      end

      def flush(force: false)
        batch = @mutex.synchronize do
          return Result.success(0) if @buffer.empty?

          should_flush = force ||
                         @buffer.size >= BATCH_SIZE ||
                         (Time.current - @last_flush) >= FLUSH_INTERVAL ||
                         (Time.current - @buffer.first[:timestamp]) >= MAX_BUFFER_AGE

          return Result.success(0) unless should_flush

          @last_flush = Time.current
          @buffer.shift(@buffer.size)
        end

        return Result.success(0) if batch.empty?

        # Bulk insert
        result = TickRepository.insert_batch(batch)

        if result.success?
          # Update cache
          batch.each do |tick|
            CacheStore.write("tick:latest:#{tick[:instrument_id]}", tick, ttl: 5)
          end

          # Emit candle build trigger
          instrument_ids = batch.map { |t| t[:instrument_id] }.uniq
          instrument_ids.each do |id|
            EventBus.publish(EventTypes::CANDLE_CLOSED, {
              instrument_id: id,
              timestamp: batch.last[:timestamp]
            })
          end

          MetricsCollector.record(
            name: "pipeline.ticks_flushed",
            labels: {},
            value: batch.length,
            type: :counter
          )
        end

        result
      end

      def stats
        @mutex.synchronize do
          {
            buffer_size: @buffer.size,
            last_flush: @last_flush,
            instruments_tracked: @instrument_last_price.keys.count
          }
        end
      end

      private

      def valid_tick?(tick)
        tick.is_a?(Hash) &&
          tick[:instrument_id].present? &&
          tick[:last_price].present? &&
          tick[:timestamp].present?
      end

      def outlier?(tick)
        last = @instrument_last_price[tick[:instrument_id]]
        return false unless last && last > 0

        change = ((tick[:last_price] - last) / last * 100).abs
        change > OUTLIER_THRESHOLD
      end

      def resolve_instrument(id)
        @instrument_cache ||= {}
        @instrument_cache[id] ||= Instrument.find_by(id: id)
      end

      def flush_if_needed
        flush if @mutex.synchronize {
          @buffer.size >= BATCH_SIZE ||
          (Time.current - @last_flush) >= FLUSH_INTERVAL
        }
      end
    end
  end
  ```
- **Deliverable:** Thread-safe tick buffering with outlier rejection
- **Acceptance Criteria:** 500 ticks buffered → automatic flush; outlier >5% rejected and logged

## Day 2 — Candle Builder (1m, 5m, 15m, 30m)

### Task 2.1: Create `CandleBuilder`
- **File:** `app/pipelines/candle_builder.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Pipelines
    class CandleBuilder
      TIMEFRAMES = {
        "1m" => 60,
        "5m" => 300,
        "15m" => 900,
        "30m" => 1800
      }.freeze

      def initialize(instrument:, timeframe:)
        @instrument = instrument
        @timeframe = timeframe
        @seconds = TIMEFRAMES[timeframe]
        raise ArgumentError, "Invalid timeframe: #{timeframe}" unless @seconds
      end

      def build(from:, to:)
        ticks = TickRepository.for_range(
          instrument: @instrument,
          from: from,
          to: to
        ).value

        return Result.success([]) if ticks.empty?

        candles = aggregate_ticks(ticks)
        persist_candles(candles)
      end

      def build_from_tick(tick)
        # Incremental build: check if current candle is complete
        bucket = bucket_time(tick[:timestamp])

        # Check for existing incomplete candle
        existing = Candle.for_instrument(@instrument)
                        .for_timeframe(@timeframe)
                        .where(timestamp: bucket)
                        .first

        if existing && !existing.is_complete
          update_candle(existing, tick)
        elsif existing.nil?
          create_candle(bucket, tick)
        end
      end

      def close_candle!(timestamp)
        bucket = bucket_time(timestamp)
        candle = Candle.for_instrument(@instrument)
                      .for_timeframe(@timeframe)
                      .where(timestamp: bucket)
                      .first

        return Result.failure(:no_candle) unless candle

        candle.update!(is_complete: true)

        EventBus.publish(EventTypes::CANDLE_CLOSED, {
          instrument_id: @instrument.id,
          timeframe: @timeframe,
          timestamp: bucket,
          candle: candle.attributes
        })

        Result.success(candle)
      end

      private

      def aggregate_ticks(ticks)
        ticks.group_by { |t| bucket_time(t[:timestamp]) }.map do |bucket, group|
          {
            instrument_id: @instrument.id,
            timeframe: @timeframe,
            timestamp: bucket,
            open: group.first[:last_price],
            high: group.map { |t| t[:last_price] }.compact.max,
            low: group.map { |t| t[:last_price] }.compact.min,
            close: group.last[:last_price],
            volume: group.sum { |t| t[:volume].to_i },
            oi: group.last[:oi],
            vwap: calculate_vwap(group),
            is_complete: true
          }
        end
      end

      def persist_candles(candles)
        return Result.success(0) if candles.empty?

        # Upsert to handle duplicates
        Candle.upsert_all(
          candles,
          unique_by: [:instrument_id, :timeframe, :timestamp],
          returning: false
        )

        Result.success(candles.length)
      rescue => e
        Result.failure(DataError.new(reason: :candle_persist_failed, error: e.message))
      end

      def bucket_time(timestamp)
        epoch = timestamp.to_i
        bucketed = (epoch / @seconds) * @seconds
        Time.at(bucketed).utc
      end

      def calculate_vwap(ticks)
        total_value = ticks.sum { |t| t[:last_price].to_f * t[:volume].to_i }
        total_volume = ticks.sum { |t| t[:volume].to_i }
        total_volume.zero? ? nil : (total_value / total_volume).round(4)
      end

      def create_candle(bucket, tick)
        Candle.create!(
          instrument: @instrument,
          timeframe: @timeframe,
          timestamp: bucket,
          open: tick[:last_price],
          high: tick[:last_price],
          low: tick[:last_price],
          close: tick[:last_price],
          volume: tick[:volume].to_i,
          oi: tick[:oi],
          vwap: tick[:last_price],
          is_complete: false
        )
      end

      def update_candle(candle, tick)
        candle.update!(
          high: [candle.high, tick[:last_price]].compact.max,
          low: [candle.low, tick[:last_price]].compact.min,
          close: tick[:last_price],
          volume: candle.volume + tick[:volume].to_i,
          oi: tick[:oi],
          vwap: recalculate_vwap(candle, tick)
        )
      end

      def recalculate_vwap(candle, tick)
        old_value = candle.vwap.to_f * candle.volume
        new_value = tick[:last_price].to_f * tick[:volume].to_i
        total_volume = candle.volume + tick[:volume].to_i
        total_volume.zero? ? candle.vwap : ((old_value + new_value) / total_volume).round(4)
      end
    end
  end
  ```
- **Deliverable:** Tick → candle aggregation with incremental updates
- **Acceptance Criteria:** 60 ticks in 1 minute → 1 complete 1m candle with correct OHLCV

      return unless instrument

          builder.build_from_tick(latest_tick.attributes) if latest_tick
        end
      end
    end
  end

### Task 2.3: Create `CandleGapDetector`
- **File:** `app/services/candle_gap_detector.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class CandleGapDetector < BaseService
    def initialize(instrument:, timeframe:)
      @instrument = instrument
      @timeframe = timeframe
      @seconds = Pipelines::CandleBuilder::TIMEFRAMES[timeframe]
    end

    def call(from:, to:)
      expected_buckets = generate_expected_buckets(from, to)
      actual_buckets = fetch_actual_buckets(from, to)

      gaps = expected_buckets - actual_buckets

      success(
        instrument: @instrument.trading_symbol,
        timeframe: @timeframe,
        gap_count: gaps.length,
        gaps: gaps.map { |t| t.iso8601 },
        coverage_percent: ((actual_buckets.length.to_f / expected_buckets.length) * 100).round(2)
      )
    end

    private

    def generate_expected_buckets(from, to)
      buckets = []
      current = align_to_bucket(from)
      while current <= to
        buckets << current
        current += @seconds.seconds
      end
      buckets
    end

    def fetch_actual_buckets(from, to)
      Candle.for_instrument(@instrument)
          .for_timeframe(@timeframe)
          .where(timestamp: from..to)
          .where(is_complete: true)
          .pluck(:timestamp)
          .map { |t| t.utc }
    end

    def align_to_bucket(time)
      epoch = time.to_i
      Time.at((epoch / @seconds) * @seconds).utc
    end
  end
  ```
- **Deliverable:** Missing candle detection
- **Acceptance Criteria:** `CandleGapDetector.new(instrument: nifty, timeframe: "5m").call(from: 1.hour.ago, to: Time.current)` returns gap array

### Task 3.1: Create `CandleGapFiller`
- **File:** `app/services/candle_gap_filler.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class CandleGapFiller < BaseService
    def initialize(instrument:, timeframe:)
      @instrument = instrument
      @timeframe = timeframe
    end

    def call(from:, to:)
      gap_result = CandleGapDetector.new(
        instrument: @instrument,
        timeframe: @timeframe
      ).call(from: from, to: to)

      return gap_result if gap_result.failure?

      gaps = gap_result.value[:gaps]
      return success(filled: 0) if gaps.empty?

      filled_count = 0
      gaps.each do |gap_time|
        fill_result = fill_gap(Time.parse(gap_time))
        filled_count += 1 if fill_result.success?
      end

      success(filled: filled_count, gaps_found: gaps.length)
    end

    private

    def fill_gap(gap_time)
      # Fetch from DhanHQ historical API
      from = gap_time
      to = gap_time + Pipelines::CandleBuilder::TIMEFRAMES[@timeframe].seconds

      result = HistoricalDataService.new(instrument: @instrument).call(
        timeframe: @timeframe,
        from: from,
        to: to
      )

      return result if result.failure?

      candles = result.value
      return failure(:no_historical_data) if candles.empty?

      # Persist
      records = candles.map do |c|
        {
          instrument_id: @instrument.id,
          timeframe: @timeframe,
          timestamp: c[:timestamp],
          open: c[:open],
          high: c[:high],
          low: c[:low],
          close: c[:close],
          volume: c[:volume],
          oi: c[:oi],
          vwap: c[:vwap] || ((c[:open] + c[:high] + c[:low] + c[:close]) / 4),
          is_complete: true
        }
      end

      Candle.upsert_all(
        records,
        unique_by: [:instrument_id, :timeframe, :timestamp]
      )

      success(filled: records.length)
    end
  end
  ```
- **Deliverable:** Automatic gap filling from historical API
- **Acceptance Criteria:** Missing 5m candle detected → fetched from DhanHQ → inserted into database

## Day 4 — Option Chain Ingestion Pipeline

### Task 4.1: Create `OptionChainIngestionPipeline`
- **File:** `app/pipelines/option_chain_ingestion_pipeline.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Pipelines
    class OptionChainIngestionPipeline
      def initialize(underlying:)
        @underlying = underlying
      end

      def ingest
        result = OptionChainService.new(underlying: @underlying).call
        return result if result.failure?

        chain = result.value
        snapshot_time = Time.current

        records = []
        chain.each do |strike_data|
          [:ce, :pe].each do |option_type|
            option = strike_data[option_type]
            next unless option

            instrument = Instrument.find_by(
              trading_symbol: option[:trading_symbol],
              exchange_segment: @underlying.exchange_segment
            )
            next unless instrument

            # Calculate derived metrics
            iv_history = fetch_iv_history(instrument, option[:iv])
            iv_rank = Calculations::Statistics.percentile(iv_history, 50) if iv_history.any?

            records << {
              instrument_id: instrument.id,
              underlying_id: @underlying.id,
              snapshot_time: snapshot_time,
              strike: strike_data[:strike],
              option_type: option_type.to_s.upcase,
              ltp: option[:ltp],
              bid: option[:bid],
              ask: option[:ask],
              spread: option[:ask].to_d - option[:bid].to_d,
              volume: option[:volume],
              volume_change: calculate_volume_change(instrument, option[:volume]),
              oi: option[:oi],
              oi_change: option[:oi_change],
              iv: option[:iv],
              iv_change: calculate_iv_change(instrument, option[:iv]),
              iv_rank: iv_rank,
              delta: option[:delta],
              gamma: option[:gamma],
              theta: option[:theta],
              vega: option[:vega],
              gamma_velocity: calculate_gamma_velocity(instrument, option[:gamma]),
              liquidity_score: calculate_liquidity_score(option)
            }
          end
        end

        return Result.failure(:no_records) if records.empty?

        # Bulk upsert
        OptionChainSnapshot.upsert_all(
          records,
          unique_by: [:instrument_id, :snapshot_time],
          returning: false
        )

        # Cache latest
        CacheStore.write(
          "option_chain:latest:#{@underlying.id}",
          { snapshot_time: snapshot_time, records: records.length },
          ttl: 30
        )

        EventBus.publish(EventTypes::OPTION_CHAIN_UPDATED, {
          underlying: @underlying.trading_symbol,
          snapshot_time: snapshot_time,
          strikes_captured: chain.length
        })

        Result.success(records.length)
      end

      private

      def fetch_iv_history(instrument, current_iv)
        return [] unless current_iv

        OptionChainSnapshot.where(instrument: instrument)
                          .where("snapshot_time > ?", 52.weeks.ago)
                          .where.not(iv: nil)
                          .order(snapshot_time: :asc)
                          .pluck(:iv)
      end

      def calculate_volume_change(instrument, current_volume)
        last = OptionChainSnapshot.where(instrument: instrument)
                                 .order(snapshot_time: :desc)
                                 .first
        return 0 unless last && last.volume
        current_volume.to_i - last.volume
      end

      def calculate_iv_change(instrument, current_iv)
        last = OptionChainSnapshot.where(instrument: instrument)
                                 .order(snapshot_time: :desc)
                                 .first
        return 0 unless last && last.iv
        current_iv.to_d - last.iv
      end

      def calculate_gamma_velocity(instrument, current_gamma)
        last = OptionChainSnapshot.where(instrument: instrument)
                                 .order(snapshot_time: :desc)
                                 .first
        return 0 unless last && last.gamma
        current_gamma.to_d - last.gamma
      end

      def calculate_liquidity_score(option)
        return 0 unless option[:ltp] && option[:ltp] > 0

        spread_pct = ((option[:ask].to_d - option[:bid].to_d) / option[:ltp] * 100).abs
        volume_score = [option[:volume].to_i / 1000.0, 100].min
        spread_score = spread_pct <= 0.3 ? 100 : [100 - (spread_pct - 0.3) * 50, 0].max

        (volume_score * 0.5 + spread_score * 0.5).round(2)
      end
    end
  end
  ```
- **Deliverable:** Option chain ingestion with derived metrics
- **Acceptance Criteria:** Ingests 11 strikes × 2 types = 22 records per snapshot; calculates IV rank, gamma velocity, liquidity score

        result = MarketDepthService.new(instrument: @instrument).call
        return result if result.failure?

        depth = result.value
        snapshot_time = Time.current

        # Calculate derived metrics
        bid_imbalance = calculate_imbalance(depth[:bid_levels])
        ask_imbalance = calculate_imbalance(depth[:ask_levels])
        total_bid = depth[:bid_levels]&.sum { |b| b[:quantity] || 0 } || 0
        total_ask = depth[:ask_levels]&.sum { |a| a[:quantity] || 0 } || 0

        # Persist
        snapshot = MarketDepthSnapshot.create!(
          instrument: @instrument,
          snapshot_time: snapshot_time,
          bid_levels: depth[:bid_levels] || [],
          ask_levels: depth[:ask_levels] || [],
          bid_imbalance: bid_imbalance,
          ask_imbalance: ask_imbalance,
          total_bid_quantity: total_bid,
          total_ask_quantity: total_ask,
          spread: depth[:spread],
          liquidity_score: calculate_liquidity_score(depth)
        )

        # Cache
        CacheStore.write(
          "depth:latest:#{@instrument.id}",
          snapshot.attributes,
          ttl: 5
        )

        EventBus.publish(EventTypes::DEPTH_UPDATED, {
          instrument_id: @instrument.id,
          snapshot_time: snapshot_time,
          spread: depth[:spread],
          liquidity_score: snapshot.liquidity_score
        })

        Result.success(snapshot)
      end

      private

      def calculate_imbalance(levels)
        return 0 unless levels && levels.length >= 2
        top = levels[0][:quantity].to_f
        second = levels[1][:quantity].to_f
        return 0 if top + second == 0
        (top - second) / (top + second)
      end

      def calculate_liquidity_score(depth)
        return 0 unless depth[:spread] && depth[:bid_levels] && depth[:ask_levels]

        best_bid = depth[:bid_levels].first&.dig(:price)
        best_ask = depth[:ask_levels].first&.dig(:price)
        return 0 unless best_bid && best_ask && best_bid > 0

        spread_pct = ((best_ask - best_bid) / best_bid * 100).abs
        depth_score = [depth[:bid_levels].sum { |b| b[:quantity] || 0 } / 10000.0, 100].min

        spread_score = spread_pct <= 0.3 ? 100 : [100 - (spread_pct - 0.3) * 50, 0].max
        (spread_score * 0.6 + depth_score * 0.4).round(2)
      end
    end
  end
  ```
- **Deliverable:** Depth snapshot ingestion with liquidity scoring
- **Acceptance Criteria:** `MarketDepthIngestionPipeline.new(instrument: nifty).ingest` creates snapshot with liquidity score

  # frozen_string_literal: true

  class DataQualityGate < BaseService
    CHECKS = %i[
      tick_freshness
      candle_continuity
      option_chain_freshness
      depth_freshness
      sequence_integrity
    ].freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      results = CHECKS.to_h do |check|
        [check, send(check)]
      end

      passed = results.values.all? { |r| r[:passed] }

      if !passed
        EventBus.publish(EventTypes::SYSTEM_ERROR, {
          type: :data_quality_gate_failed,
          instrument: @instrument.trading_symbol,
          failures: results.reject { |_, v| v[:passed] }.keys
        })
      end

      success(
        instrument: @instrument.trading_symbol,
        passed: passed,
        checks: results,
        timestamp: Time.current
      )
    end

    def all_passed?
      call.value[:passed]
    end

    private

    def tick_freshness
      latest = TickRepository.latest(instrument: @instrument, limit: 1).value&.first
      age = latest ? Time.current - latest.timestamp : nil

      {
        passed: age.present? && age < 5.seconds,
        age_seconds: age&.round(3),
        last_tick_at: latest&.timestamp
      }
    end

    def candle_continuity
      gaps = CandleGapDetector.new(instrument: @instrument, timeframe: "5m")
                               .call(from: 15.minutes.ago, to: Time.current)
      gap_count = gaps.success? ? gaps.value[:gap_count] : 0

      {
        passed: gap_count == 0,
        gap_count: gap_count,
        coverage_percent: gaps.success? ? gaps.value[:coverage_percent] : 0
      }
    end

    def option_chain_freshness
      latest = OptionChainSnapshot.for_underlying(@instrument)
                                 .order(snapshot_time: :desc)
                                 .first
      age = latest ? Time.current - latest.snapshot_time : nil

      {
        passed: age.present? && age < 2.minutes,
        age_seconds: age&.round,
        last_snapshot_at: latest&.snapshot_time
      }
    end

    def depth_freshness
      latest = MarketDepthSnapshot.where(instrument: @instrument)
                                 .order(snapshot_time: :desc)
                                 .first
      age = latest ? Time.current - latest.snapshot_time : nil

      {
        passed: age.present? && age < 15.seconds,
        age_seconds: age&.round,
        last_snapshot_at: latest&.snapshot_time
      }
    end

    def sequence_integrity
      # Check for sequence gaps in last 100 ticks
      ticks = TickRepository.latest(instrument: @instrument, limit: 100).value
      return { passed: true, gap_count: 0 } if ticks.size < 2

      gaps = 0
      ticks.each_cons(2) do |a, b|
        gaps += 1 if (b.timestamp - a.timestamp) > 2.seconds
      end

      {
        passed: gaps == 0,
        gap_count: gaps,
        sample_size: ticks.size
      }
    end
  end
  ```
- **Deliverable:** Composite data quality gate for all data types

### Task 6.3: Create `DataRetentionPolicy`
- **File:** `app/services/data_retention_policy.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class DataRetentionPolicy < BaseService
    POLICIES = {
      market_ticks: { compress_after: 7.days, archive_after: 30.days, drop_after: 90.days },
      candles: { compress_after: 30.days, archive_after: 1.year, drop_after: 3.years },
      option_chain_snapshots: { compress_after: 7.days, archive_after: 30.days, drop_after: 90.days },
      market_depth_snapshots: { compress_after: 1.day, archive_after: 7.days, drop_after: 30.days },
      market_regimes: { archive_after: 1.year, drop_after: 3.years },
      market_structures: { archive_after: 1.year, drop_after: 3.years },
      trade_setups: { archive_after: 1.year, drop_after: 3.years },
      trade_scores: { archive_after: 1.year, drop_after: 3.years },
      ai_analyses: { archive_after: 6.months, drop_after: 1.year }
    }.freeze

    def call
      POLICIES.each do |table, policy|
        apply_policy(table, policy)
      end

      success(applied: POLICIES.keys)
    end

    private

    def apply_policy(table, policy)
      model = table.to_s.classify.constantize rescue nil
      return unless model

      if policy[:compress_after]
        compress_old_records(model, older_than: policy[:compress_after])
      end

      if policy[:archive_after]
        archive_old_records(model, older_than: policy[:archive_after])
      end

      if policy[:drop_after]
        drop_old_records(model, older_than: policy[:drop_after])
      end
    end

    def compress_old_records(model, older_than:)
      return unless model.respond_to?(:table_name)

      # TimescaleDB compression
      ActiveRecord::Base.connection.execute(
        "SELECT compress_chunk(i) FROM show_chunks('#{model.table_name}', older_than => INTERVAL '#{older_than.to_i / 86400} days') i"
      )
    rescue => e
      Rails.logger.error "[Retention] Compression failed for #{model.table_name}: #{e.message}"
    end

    def archive_old_records(model, older_than:)
      old_records = model.where("created_at < ?", older_than.ago)
      return if old_records.none?

      # Export to cold storage (S3 or local)
      archive_path = Rails.root.join("storage/archives/#{model.table_name}/#{Date.current}.csv")
      FileUtils.mkdir_p(archive_path.dirname)

      CSV.open(archive_path, "w") do |csv|
        csv << model.column_names
        old_records.find_each do |record|
          csv << record.attributes.values
        end
      end

      Rails.logger.info "[Retention] Archived #{old_records.count} #{model.table_name} records to #{archive_path}"
    end

    def drop_old_records(model, older_than:)
      count = model.where("created_at < ?", older_than.ago).delete_all
      Rails.logger.info "[Retention] Dropped #{count} old #{model.table_name} records"
    end
  end
  ```
- **Deliverable:** Automated data lifecycle management
- **Acceptance Criteria:** `DataRetentionPolicy.new.call` compresses, archives, and drops old records

### Task 8.1: Create `MarketHoursEnforcer`
- **File:** `app/lib/market_hours_enforcer.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class MarketHoursEnforcer
    PRE_MARKET = 15.minutes
    POST_MARKET = 15.minutes

    def self.trading_hours?(exchange: "NSE")
      calendar = ExchangeCalendar.for_exchange(exchange: exchange)
      return false unless calendar

      now = Time.current
      open_time = Time.parse(calendar[:open].to_s)
      close_time = Time.parse(calendar[:close].to_s)

      now.between?(open_time, close_time)
    end

    def self.pre_market?
      calendar = ExchangeCalendar.for_exchange(exchange: "NSE")
      return false unless calendar

      now = Time.current
      open_time = Time.parse(calendar[:open].to_s)
      now.between?(open_time - PRE_MARKET, open_time)
    end

    def self.post_market?
      calendar = ExchangeCalendar.for_exchange(exchange: "NSE")
      return false unless calendar

      now = Time.current
      close_time = Time.parse(calendar[:close].to_s)
      now.between?(close_time, close_time + POST_MARKET)
    end

    def self.market_open?(exchange: "NSE")
      ExchangeCalendar.today_trading_day?(exchange: exchange)
    end

    def self.enforce!
      return if trading_hours?

      raise MarketClosedError, "Market is closed. Trading not allowed."
    end
  end

  class MarketClosedError < StandardError; end
  ```
