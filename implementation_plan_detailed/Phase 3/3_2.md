## Day 1 — Base Option Feature & Delta Scoring

### Task 1.1: Create `BaseOptionFeature`
- **File:** `app/option_features/base_option_feature.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class BaseOptionFeature
      include AppInject

      attr_reader :snapshot

      def initialize(snapshot:)
        @snapshot = snapshot
      end

      def calculate
        raise NotImplementedError, "#{self.class} must implement #calculate"
      end

      def value
        cached = fetch_cache
        return cached if cached

        result = calculate
        write_cache(result) if result.present?
        result
      end

      def score
        # Override in subclasses to return 0-100
        50
      end

      def cache_key
        "option_feature:#{snapshot.instrument_id}:#{snapshot.snapshot_time.to_i}:#{feature_name}"
      end

      def feature_name
        self.class.name.demodulize.underscore.gsub("_feature", "")
      end

      def underlying
        snapshot.underlying
      end

      def instrument
        snapshot.instrument
      end

      private

      def fetch_cache
        CacheStore.fetch(cache_key)
      end

      def write_cache(result)
        CacheStore.write(cache_key, result, ttl: 60)
      end

      def normalize(value, min:, max:, invert: false)
        return 0 if max <= min
        raw = [(value - min) / (max - min), 0].max
        raw = 1 - raw if invert
        (raw * 100).round(2)
      end

      def clamp(value, min: 0, max: 100)
        [[value, min].max, max].min
      end
    end
  end
  ```
- **Deliverable:** Base class for all option features
- **Acceptance Criteria:** `BaseOptionFeature.new(snapshot: snap).value` caches and returns result

### Task 1.2: Create `DeltaFeature`
- **File:** `app/option_features/delta_feature.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class DeltaFeature < BaseOptionFeature
      IDEAL_MIN = 0.40
      IDEAL_MAX = 0.65
      ABSOLUTE_MIN = 0.30
      ABSOLUTE_MAX = 0.80

      def calculate
        {
          raw: snapshot.delta&.to_d,
          ideal: ideal?,
          distance_from_ideal: distance_from_ideal,
          moneyness: moneyness
        }
      end

      def score
        return 0 unless snapshot.delta

        delta = snapshot.delta.abs

        # Perfect score at 0.525 (midpoint of ideal)
        ideal_mid = (IDEAL_MIN + IDEAL_MAX) / 2.0
        distance = (delta - ideal_mid).abs

        if delta < ABSOLUTE_MIN
          # Too far OTM — penalty
          [100 - ((ABSOLUTE_MIN - delta) / 0.10 * 50), 0].max
        elsif delta > ABSOLUTE_MAX
          # Too deep ITM — penalty
          [100 - ((delta - ABSOLUTE_MAX) / 0.20 * 50), 0].max
        else
          # Within acceptable range
          [100 - (distance / 0.125 * 100), 0].max
        end.round(2)
      end

      def ideal?
        return false unless snapshot.delta
        snapshot.delta.abs.between?(IDEAL_MIN, IDEAL_MAX)
      end

      def distance_from_ideal
        return nil unless snapshot.delta
        ideal_mid = (IDEAL_MIN + IDEAL_MAX) / 2.0
        (snapshot.delta.abs - ideal_mid).abs.round(4)
      end

      def moneyness
        return nil unless snapshot.delta
        if snapshot.delta.abs > 0.60
          :deep_itm
        elsif snapshot.delta.abs > 0.40
          :atm
        elsif snapshot.delta.abs > 0.20
          :otm
        else
          :deep_otm
        end
      end
    end
  end
  ```
- **Deliverable:** Delta scoring with ideal range 0.40–0.65
- **Acceptance Criteria:** `DeltaFeature.new(snapshot: snap).score` returns 100 at 0.525, 0 at 0.30 or 0.80

## Day 2 — Gamma Feature (Velocity, Acceleration, Exposure)

### Task 2.1: Create `GammaFeature`
- **File:** `app/option_features/gamma_feature.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class GammaFeature < BaseOptionFeature
      def calculate
        {
          raw: snapshot.gamma&.to_d,
          velocity: snapshot.gamma_velocity&.to_d,
          acceleration: gamma_acceleration,
          exposure: gamma_exposure,
          responsiveness: responsiveness
        }
      end

      def score
        return 0 unless snapshot.gamma

        gamma = snapshot.gamma.abs
        velocity = snapshot.gamma_velocity&.abs || 0
        acceleration = gamma_acceleration&.abs || 0

        # Base score from gamma magnitude (higher = more responsive)
        base = [gamma * 10000, 80].min

        # Bonus for increasing gamma (velocity)
        velocity_bonus = [velocity * 5000, 15].min

        # Bonus for accelerating gamma
        acceleration_bonus = [acceleration * 2500, 5].min

        (base + velocity_bonus + acceleration_bonus).round(2)
      end

      def gamma_acceleration
        return nil unless snapshot.gamma_velocity

        # Need previous velocity
        previous = OptionChainSnapshot.where(instrument: snapshot.instrument)
                                     .where("snapshot_time < ?", snapshot.snapshot_time)
                                     .where(option_type: snapshot.option_type)
                                     .order(snapshot_time: :desc)
                                     .first

        return nil unless previous&.gamma_velocity
        (snapshot.gamma_velocity - previous.gamma_velocity).round(6)
      end

      def gamma_exposure
        return nil unless snapshot.gamma && snapshot.oi
        (snapshot.gamma.to_d * snapshot.oi).round(2)
      end

      def responsiveness
        return nil unless snapshot.gamma
        if snapshot.gamma > 0.008
          :very_high
        elsif snapshot.gamma > 0.005
          :high
        elsif snapshot.gamma > 0.002
          :moderate
        else
          :low
        end
      end
    end
  end
  ```
- **Deliverable:** Gamma with velocity, acceleration, exposure
- **Acceptance Criteria:** `GammaFeature.new(snapshot: snap).score` returns composite 0-100

### Task 2.2: Create `GammaExposureCalculator`
- **File:** `app/lib/calculations/gamma_exposure_calculator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Calculations
    class GammaExposureCalculator
      def self.total_exposure(strikes_data)
        strikes_data.sum { |s| s.gamma.to_d * s.oi.to_i }
      end

      def self.net_exposure(ce_data, pe_data)
        ce_exposure = total_exposure(ce_data)
        pe_exposure = total_exposure(pe_data)
        {
          ce: ce_exposure,
          pe: pe_exposure,
          net: ce_exposure - pe_exposure,
          ratio: pe_exposure.zero? ? 0 : (ce_exposure / pe_exposure).round(2)
        }
      end

      def self.flip_zone(strikes_data, spot_price)
        # Find where gamma exposure changes sign
        sorted = strikes_data.sort_by { |s| s.strike.to_d }
        atm_idx = sorted.index { |s| (s.strike.to_d - spot_price).abs < 50 }

        return nil unless atm_idx

        # Look for sign change around ATM
        window = sorted[[atm_idx - 3, 0].max..[atm_idx + 3, sorted.length - 1].min]
        window.find { |s| s.gamma.to_d < 0 }&.strike
      end

      def self.max_pain_gamma(strikes_data)
        # Strike with highest gamma exposure
        strikes_data.max_by { |s| s.gamma.to_d * s.oi.to_i }&.strike
      end
    end
  end
  ```
- **Deliverable:** Portfolio-level gamma exposure analytics
- **Acceptance Criteria:** `GammaExposureCalculator.total_exposure(snapshots)` returns aggregate exposure

  ```ruby
  class GammaFeatureBatchJob < JobBase

        ttl: 60

  end
  ```

## Day 3 — Theta & Vega Features

### Task 3.1: Create `ThetaFeature`
- **File:** `app/option_features/theta_feature.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class ThetaFeature < BaseOptionFeature
      def calculate
        {
          raw: snapshot.theta&.to_d,
          decay_per_day: decay_per_day,
          decay_vs_expected_move: decay_vs_expected_move,
          time_risk: time_risk,
          expiry_proximity_risk: expiry_proximity_risk
        }
      end

      def score
        return 0 unless snapshot.theta

        theta = snapshot.theta.abs
        expected = expected_move

        # Lower theta is better (less time decay)
        base_score = [100 - theta * 10, 0].max

        # Heavy penalty if theta exceeds expected move
        if expected && expected > 0
          ratio = theta / expected
          if ratio > 0.5
            base_score -= (ratio * 50).round(2)
          end
        end

        # Penalty for near-expiry
        days = days_to_expiry
        if days && days <= 1
          base_score *= 0.3
        elsif days && days <= 3
          base_score *= 0.6
        end

        [base_score, 0].max.round(2)
      end

      def decay_per_day
        return nil unless snapshot.theta
        snapshot.theta.abs.round(4)
      end

      def decay_vs_expected_move
        return nil unless snapshot.theta && expected_move && expected_move > 0
        (snapshot.theta.abs / expected_move * 100).round(2)
      end

      def time_risk
        return :unknown unless snapshot.theta && expected_move
        ratio = snapshot.theta.abs / expected_move
        if ratio > 0.5
          :high
        elsif ratio > 0.3
          :moderate
        else
          :low
        end
      end

      def expiry_proximity_risk
        days = days_to_expiry
        return :unknown unless days
        if days <= 1
          :extreme
        elsif days <= 3
          :high
        elsif days <= 7
          :moderate
        else
          :low
        end
      end

      private

      def expected_move
        # Expected move = IV * spot * sqrt(days/365)
        return nil unless snapshot.iv && underlying_price && days_to_expiry
        snapshot.iv.to_d * underlying_price * Math.sqrt(days_to_expiry / 365.0)
      end

      def underlying_price
        # Fetch from cache or quote service
        CacheStore.fetch("tick:latest:#{underlying.id}")&.dig(:last_price) ||
          MarketQuoteService.new(instrument: underlying).ltp.value
      end

      def days_to_expiry
        return nil unless instrument.expiry
        (instrument.expiry - Date.current).to_i
      end
    end
  end
  ```
- **Deliverable:** Theta decay risk scoring
- **Acceptance Criteria:** `ThetaFeature.new(snapshot: snap).score` penalizes high theta and near-expiry

### Task 4.2: Create `IVTrendFeature`
- **File:** `app/option_features/iv_trend_feature.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class IVTrendFeature < BaseOptionFeature
      LOOKBACK = 5  # snapshots

      def calculate
        history = iv_history
        return { trend: :insufficient_data } if history.length < 2

        slope = Calculations::Statistics.slope(history, history.length)

        {
          trend: trend_direction(slope),
          slope: slope.round(4),
          acceleration: acceleration(history),
          convergence: convergence?(history),
          divergence: divergence?(history)
        }
      end

      def score
        result = calculate
        return 50 if result[:trend] == :insufficient_data

        case result[:trend]
        when :rising then 70
        when :falling then 40
        when :stable then 60
        else 50
        end
      end

      private

      def iv_history
        OptionChainSnapshot.where(instrument: snapshot.instrument)
                          .where(option_type: snapshot.option_type)
                          .where("snapshot_time < ?", snapshot.snapshot_time)
                          .order(snapshot_time: :desc)
                          .limit(LOOKBACK)
                          .pluck(:iv)
                          .compact
                          .reverse
      end

      def trend_direction(slope)
        if slope > 0.1
          :rising
        elsif slope < -0.1
          :falling
        else
          :stable
        end
      end

      def acceleration(history)
        return 0 if history.length < 3
        recent = history.last(3)
        (recent[-1] - recent[-2]) - (recent[-2] - recent[-3])
      end

      def convergence?(history)
        return false if history.length < 5
        # IV converging toward mean
        mean = history.sum.to_f / history.length
        recent = history.last(3)
        recent.all? { |v| (v - mean).abs < (history.first - mean).abs }
      end

      def divergence?(history)
        return false if history.length < 5
        # IV diverging from mean
        mean = history.sum.to_f / history.length
        recent = history.last(3)
        recent.all? { |v| (v - mean).abs > (history.first - mean).abs }
      end
    end
  end
  ```
- **Deliverable:** IV trend with acceleration and convergence/divergence
- **Acceptance Criteria:** `IVTrendFeature.new(snapshot: snap).calculate[:trend]` returns `:rising` for increasing IV

### Task 5.2: Create `VolumeFlowFeature`

  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class VolumeFlowFeature < BaseOptionFeature
      def calculate
        {
          current_volume: snapshot.volume,
          volume_change: snapshot.volume_change,
          relative_volume: relative_volume,
          volume_vs_oi_ratio: volume_vs_oi_ratio,
          activity_level: activity_level
        }
      end

      def score
        return 0 unless snapshot.volume

        rv = relative_volume
        return 50 if rv.nil?

        # Higher relative volume = more activity = better liquidity
        base = [rv * 50, 70].min

        # Bonus for volume > OI (fresh interest)
        if volume_vs_oi_ratio && volume_vs_oi_ratio > 1
          base += 20
        end

        [base, 100].min.round(2)
      end

      def relative_volume
        return nil unless snapshot.volume_change

        historical = OptionChainSnapshot.where(instrument: snapshot.instrument)
                                     .where(option_type: snapshot.option_type)
                                     .where("snapshot_time < ?", snapshot.snapshot_time)
                                     .order(snapshot_time: :desc)
                                     .limit(10)
                                     .pluck(:volume_change)
                                     .compact

        return nil if historical.empty?
        avg = historical.sum.to_f / historical.length
        return nil if avg.zero?
        (snapshot.volume_change.abs / avg).round(2)
      end

      def volume_vs_oi_ratio
        return nil unless snapshot.volume && snapshot.oi
        return nil if snapshot.oi.zero?
        (snapshot.volume.to_f / snapshot.oi).round(4)
      end

      def activity_level
        return :low unless snapshot.volume
        if snapshot.volume > 10000
          :very_high
        elsif snapshot.volume > 5000
          :high
        elsif snapshot.volume > 1000
          :moderate
        else
          :low
        end
      end
    end
  end
  ```
- **Deliverable:** Volume flow with relative volume and activity level
- **Acceptance Criteria:** `VolumeFlowFeature.new(snapshot: snap).score` returns > 70 for high relative volume

- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class SpreadFeature < BaseOptionFeature
      MAX_ACCEPTABLE_SPREAD_PCT = 0.30

      def calculate
        {
          raw_spread: snapshot.spread,
          spread_percent: spread_percent,
          spread_vs_ltp: spread_vs_ltp,
          execution_cost: execution_cost,
          slippage_estimate: slippage_estimate
        }
      end

      def score
        return 0 unless snapshot.spread && snapshot.ltp && snapshot.ltp > 0

        spread_pct = spread_vs_ltp

        if spread_pct <= 0.10
          100
        elsif spread_pct <= 0.20
          90
        elsif spread_pct <= MAX_ACCEPTABLE_SPREAD_PCT
          80 - ((spread_pct - 0.20) / 0.10 * 30)
        elsif spread_pct <= 0.50
          50 - ((spread_pct - 0.30) / 0.20 * 40)
        else
          0
        end.round(2)
      end

      def spread_percent
        return nil unless snapshot.spread && snapshot.ltp && snapshot.ltp > 0
        (snapshot.spread / snapshot.ltp * 100).round(4)
      end

      def spread_vs_ltp
        return nil unless snapshot.spread && snapshot.ltp && snapshot.ltp > 0
        (snapshot.spread / snapshot.ltp).round(4)
      end

      def execution_cost
        return nil unless snapshot.spread
        # Cost of crossing spread (market order)
        (snapshot.spread / 2).round(4)
      end

      def slippage_estimate
        return nil unless snapshot.spread && snapshot.volume
        # Higher volume = lower slippage
        base_slippage = snapshot.spread * 0.5
        volume_factor = [1000.0 / snapshot.volume, 1].min
        (base_slippage * volume_factor).round(4)
      end
    end
  end
  ```
- **Deliverable:** Spread scoring with execution cost estimation
- **Acceptance Criteria:** `SpreadFeature.new(snapshot: snap).score` returns 100 for spread < 0.10%

### Task 6.2: Create `LiquidityFeature`
- **File:** `app/option_features/liquidity_feature.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module OptionFeatures
    class LiquidityFeature < BaseOptionFeature
      def calculate
        {
          liquidity_score: snapshot.liquidity_score,
          spread_score: spread_score,
          depth_score: depth_score,
          oi_score: oi_score,
          volume_score: volume_score,
          overall: overall
        }
      end

      def score
        return snapshot.liquidity_score || 0 if snapshot.liquidity_score

        # Compute if not pre-calculated
        weights = { spread: 0.35, depth: 0.25, oi: 0.20, volume: 0.20 }

        (spread_score * weights[:spread] +
         depth_score * weights[:depth] +
         oi_score * weights[:oi] +
         volume_score * weights[:volume]).round(2)
      end

      def spread_score
        OptionFeatures::SpreadFeature.new(snapshot: snapshot).score
      end

      def depth_score
        return 0 unless snapshot.bid && snapshot.ask
        # Approximate depth from spread tightness
        spread_pct = snapshot.ltp && snapshot.ltp > 0 ? (snapshot.spread / snapshot.ltp) : 1
        [100 - spread_pct * 100, 0].max
      end

      def oi_score
        return 0 unless snapshot.oi
        [snapshot.oi / 5000.0, 100].min
      end

      def volume_score
        return 0 unless snapshot.volume
        [snapshot.volume / 1000.0, 100].min
      end

      def overall
        s = score
        if s >= 80
          :excellent
        elsif s >= 60
          :good
        elsif s >= 40
          :moderate
        else
          :poor
        end
      end
    end
  end
  ```
- **Deliverable:** Composite liquidity scoring
- **Acceptance Criteria:** `LiquidityFeature.new(snapshot: snap).score` returns 0-100 composite

### Task 6.3: Create `LiquidityGate`

  ```ruby
  # frozen_string_literal: true

  class LiquidityGate < BaseService
    MIN_LIQUIDITY_SCORE = 60
    MAX_SPREAD_PCT = 0.30

    def initialize(snapshot:)
      @snapshot = snapshot
    end

    def call
      liquidity = OptionFeatures::LiquidityFeature.new(snapshot: @snapshot)
      spread = OptionFeatures::SpreadFeature.new(snapshot: @snapshot)

      reasons = []
      reasons << :low_liquidity if liquidity.score < MIN_LIQUIDITY_SCORE
      reasons << :wide_spread if spread.spread_vs_ltp && spread.spread_vs_ltp > MAX_SPREAD_PCT
      reasons << :low_volume if @snapshot.volume && @snapshot.volume < 500
      reasons << :low_oi if @snapshot.oi && @snapshot.oi < 5000

      passed = reasons.empty?

      success(
        passed: passed,
        reasons: reasons,
        liquidity_score: liquidity.score,
        spread_percent: spread.spread_vs_ltp
      )
    end

    def self.validate!(snapshot)
      result = new(snapshot: snapshot).call
      raise LiquidityError, result.value[:reasons] unless result.value[:passed]
      result
    end
  end

  class LiquidityError < StandardError; end
  ```
- **Deliverable:** Liquidity validation gate for execution
- **Acceptance Criteria:** `LiquidityGate.new(snapshot: snap).call` returns `passed: false` for wide spread

  ```ruby
  # frozen_string_literal: true

  class OptionFeatureNormalizer < BaseService
    FEATURES = %i[
      delta
      gamma
      theta
      vega
      iv_rank
      iv_trend
      oi_flow
      volume_flow
      spread
      liquidity
    ].freeze

    def initialize(snapshot:)
      @snapshot = snapshot
    end

    def call
      scores = {}

      FEATURES.each do |name|
        scores[name] = compute_score(name)
      end

      scores[:total] = weighted_total(scores)
      scores[:passed] = scores[:total] >= 80

      success(scores)
    end

    private

    def compute_score(name)
      feature = feature_class(name)
      return 0 unless feature

      feature.new(snapshot: @snapshot).score
    end

    def feature_class(name)
      {
        delta: OptionFeatures::DeltaFeature,
        gamma: OptionFeatures::GammaFeature,
        theta: OptionFeatures::ThetaFeature,
        vega: OptionFeatures::VegaFeature,
        iv_rank: OptionFeatures::IVRankFeature,
        iv_trend: OptionFeatures::IVTrendFeature,
        oi_flow: OptionFeatures::OIFlowFeature,
        volume_flow: OptionFeatures::VolumeFlowFeature,
        spread: OptionFeatures::SpreadFeature,
        liquidity: OptionFeatures::LiquidityFeature
      }[name]
    end

    def weighted_total(scores)
      weights = {
        delta: 0.10,
        gamma: 0.10,
        theta: 0.10,
        vega: 0.05,
        iv_rank: 0.10,
        iv_trend: 0.05,
        oi_flow: 0.15,
        volume_flow: 0.10,
        spread: 0.10,
        liquidity: 0.15
      }

      total = scores.sum { |name, score| score * weights[name] }
      total.round(2)
    end
  end
  ```
- **Deliverable:** Normalize all 10 features to 0-100 with weighted total
- **Acceptance Criteria:** `OptionFeatureNormalizer.new(snapshot: snap).call` returns total score and pass/fail

### Task 7.3: Create `OptionFeatureBatchJob`
- **File:** `app/jobs/option_feature_batch_job.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class OptionFeatureBatchJob < JobBase
    def perform(underlying_id, snapshot_time)
      underlying = Instrument.find(underlying_id)
      return unless underlying

      snapshots = OptionChainSnapshot.for_underlying(underlying)
                                    .at_time(Time.parse(snapshot_time))

      snapshots.each do |snap|
        OptionFeatureVector.store(snap)
      end

      # Store aggregate underlying features
      aggregate = aggregate_features(snapshots)
      FeatureStore.write(
        "option_aggregate:#{underlying_id}:#{snapshot_time}",
        aggregate
      )

      EventBus.publish("option_features.computed", {
        underlying_id: underlying_id,
        snapshot_time: snapshot_time,
        count: snapshots.count
      })
    end

    private

    def aggregate_features(snapshots)
      ce = snapshots.calls
      pe = snapshots.puts

      {
        ce_avg_score: ce.average(:liquidity_score)&.round(2),
        pe_avg_score: pe.average(:liquidity_score)&.round(2),
        best_ce: ce.order(liquidity_score: :desc).first&.strike,
        best_pe: pe.order(liquidity_score: :desc).first&.strike,
        pcr: pe.sum(:oi).to_f / ce.sum(:oi)
      }
    end
  end
  ```
- **Deliverable:** Batch feature computation for all strikes
- **Acceptance Criteria:** Job computes and stores feature vectors for all snapshots

  ```ruby
  # frozen_string_literal: true

  module Api
    module V1
      class OptionFeaturesController < ApplicationController
        def show
          snapshot = OptionChainSnapshot.find(params[:id])
          normalizer = OptionFeatureNormalizer.new(snapshot: snapshot)
          result = normalizer.call

          render json: {
            snapshot: serialize_snapshot(snapshot),

        def strike_vector
          underlying = Instrument.find(params[:underlying_id])
          snapshot_time = params[:snapshot_time] ? Time.parse(params[:snapshot_time]) : Time.current

          snapshots = OptionChainSnapshot.for_underlying(underlying)
                                        .at_time(snapshot_time)
                                        .order(strike: :asc)

          vectors = snapshots.map do |snap|
            {
              strike: snap.strike,
              option_type: snap.option_type,
              vector: OptionFeatureVector.build(snap)
            }
          end

          render json: {
            underlying: underlying.trading_symbol,
            snapshot_time: snapshot_time,
            vectors: vectors
          }
        end

        def best_strike
          underlying = Instrument.find(params[:underlying_id])
          direction = params[:direction] || "long"

          snapshot_time = OptionChainSnapshot.for_underlying(underlying)
                                            .order(snapshot_time: :desc)
                                            .first&.snapshot_time

          snapshots = OptionChainSnapshot.for_underlying(underlying)
                                        .at_time(snapshot_time)

          if direction == "long"
            candidates = snapshots.calls
          else
            candidates = snapshots.puts
          end

          scored = candidates.map do |snap|
            [snap, OptionFeatureNormalizer.new(snapshot: snap).call.value[:total]]
          end

          best = scored.max_by { |_, score| score }

          render json: {
            strike: best&.first&.strike,
            score: best&.last,
            option_type: best&.first&.option_type
          }
        end

        private

        def serialize_snapshot(snapshot)
          {
            instrument: snapshot.instrument.trading_symbol,
            strike: snapshot.strike,
            option_type: snapshot.option_type,
            ltp: snapshot.ltp,
            delta: snapshot.delta,
            gamma: snapshot.gamma,
            theta: snapshot.theta,
            vega: snapshot.vega,
            iv: snapshot.iv,
            iv_rank: snapshot.iv_rank
          }
        end
      end
    end
  end
  ```
