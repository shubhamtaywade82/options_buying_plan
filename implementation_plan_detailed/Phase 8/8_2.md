module Engines
  module Performance
    class PerformanceEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[trades].freeze

      def initialize(trades: nil, strategy: nil, instrument: nil, lookback: 30.days)
        @trades = trades || fetch_trades(strategy: strategy, instrument: instrument, lookback: lookback)
        @strategy = strategy
        @instrument = instrument
        @lookback = lookback
      end

      def call(context: {})
        return Result.failure(:no_trades) if @trades.empty?

        report = {
          generated_at: Time.current.iso8601,
          period: @lookback.to_i / 86_400,
          strategy: @strategy,
          instrument: @instrument&.trading_symbol,
          summary: generate_summary,
          win_rate: WinRateCalculator.new(trades: @trades).call,
          profit_factor: ProfitFactorCalculator.new(trades: @trades).call,
          sharpe_ratio: SharpeRatioCalculator.new(trades: @trades).call,
          sortino_ratio: SortinoRatioCalculator.new(trades: @trades).call,
          calmar_ratio: CalmarRatioCalculator.new(trades: @trades).call,
          drawdown: DrawdownAnalyzer.new(trades: @trades).call,
          consecutive_losses: ConsecutiveLossAnalyzer.new(trades: @trades).call,
          expectancy: ExpectancyCalculator.new(trades: @trades).call,
          benchmark: BenchmarkComparison.new(trades: @trades, instrument: @instrument).call
        }

        emit(EventTypes::PERFORMANCE_REPORT_GENERATED, {
          strategy: @strategy,
          trade_count: @trades.length,
          win_rate: report[:win_rate][:win_rate],
          sharpe: report[:sharpe_ratio][:sharpe_ratio]
        })

        success(report)
      end

      private

      def fetch_trades(strategy:, instrument:, lookback:)
        scope = Trade.where("exit_time IS NOT NULL")
                     .where("entry_time > ?", Time.current - lookback)

        scope = scope.where(setup_type: strategy) if strategy
        scope = scope.where(instrument: instrument) if instrument

        scope.order(:entry_time)
      end

      def generate_summary
        pnls = @trades.map { |t| t.gross_pnl.to_f }
        wins = pnls.select { |p| p > 0 }
        losses = pnls.select { |p| p <= 0 }

        {
          total_trades: @trades.length,
          gross_profit: wins.sum.round(2),
          gross_loss: losses.sum.abs.round(2),
          net_pnl: pnls.sum.round(2),
          avg_trade: (pnls.sum / @trades.length).round(2),
          max_win: wins.max || 0,
          max_loss: losses.min || 0,
          win_count: wins.length,
          loss_count: losses.length
        }
      end
    end
  end
end

module Engines
  module Performance
    class WinRateCalculator
      def initialize(trades:)
        @trades = trades
      end

      def call
        return empty_result if @trades.empty?

        total = @trades.length
        wins = @trades.count { |t| t.gross_pnl.to_f > 0 }
        losses = total - wins
        win_rate = wins.to_f / total * 100

        {
          win_rate: win_rate.round(2),
          loss_rate: (100 - win_rate).round(2),
          total_trades: total,
          wins: wins,
          losses: losses,
          statistical_significance: statistical_significance(wins, total),
          confidence_interval: wilson_confidence_interval(wins, total),
          streak_current: current_streak,
          streak_max_win: max_win_streak,
          streak_max_loss: max_loss_streak
        }
      end

      private

      def statistical_significance(wins, total)
        return :insufficient if total < 30

        win_rate = wins.to_f / total
        # Test if win rate is significantly different from 50%
        expected = total * 0.5
        variance = total * 0.5 * 0.5
        z_score = (wins - expected) / Math.sqrt(variance)

        {
          z_score: z_score.round(4),
          p_value: p_value_from_z(z_score.abs).round(6),
          significant_at_5pct: z_score.abs > 1.96,
          significant_at_1pct: z_score.abs > 2.576
        }
      end

      def wilson_confidence_interval(wins, total)
        return { lower: 0, upper: 0 } if total < 10

        p = wins.to_f / total
        z = 1.96  # 95% confidence

        denominator = 1 + (z**2 / total)
        centre = (p + (z**2 / (2 * total))) / denominator
        width = z * Math.sqrt((p * (1 - p) + (z**2 / (4 * total))) / total) / denominator

        {
          lower: [centre - width, 0].max.round(4) * 100,
          upper: [centre + width, 1].min.round(4) * 100
        }
      end

      def p_value_from_z(z)
        # Approximation of standard normal CDF
        b1 = 0.319381530
        b2 = -0.356563782
        b3 = 1.781477937
        b4 = -1.821255978
        b5 = 1.330274429
        p = 0.2316419
        c = 0.39894228

        t = 1 / (1 + p * z)
        phi = c * Math.exp(-(z**2) / 2)
        poly = b1*t + b2*t**2 + b3*t**3 + b4*t**4 + b5*t**5
        1 - phi * poly
      end

      def current_streak
        return { type: :none, count: 0 } if @trades.empty?

        sorted = @trades.sort_by(&:exit_time)
        last = sorted.last
        type = last.gross_pnl.to_f > 0 ? :win : :loss

        count = 0
        sorted.reverse_each do |t|
          break if t.gross_pnl.to_f > 0 && type == :loss
          break if t.gross_pnl.to_f <= 0 && type == :win
          count += 1
        end

        { type: type, count: count }
      end

      def max_win_streak
        max_streak(:win)
      end

      def max_loss_streak
        max_streak(:loss)
      end

      def max_streak(target_type)
        return 0 if @trades.empty?

        sorted = @trades.sort_by(&:exit_time)
        current = 0
        max = 0

        sorted.each do |t|
          is_target = target_type == :win ? t.gross_pnl.to_f > 0 : t.gross_pnl.to_f <= 0
          if is_target
            current += 1
            max = [max, current].max
          else
            current = 0
          end
        end

        max
      end

      def empty_result
        {
          win_rate: 0,
          loss_rate: 0,
          total_trades: 0,
          wins: 0,
          losses: 0,
          statistical_significance: :insufficient,
          confidence_interval: { lower: 0, upper: 0 },
          streak_current: { type: :none, count: 0 },
          streak_max_win: 0,
          streak_max_loss: 0
        }
      end
    end
  end
end

module Engines
  module Performance
    class ProfitFactorCalculator
      def initialize(trades:)
        @trades = trades
      end

      def call
        return empty_result if @trades.empty?

        wins = @trades.select { |t| t.gross_pnl.to_f > 0 }
        losses = @trades.reject { |t| t.gross_pnl.to_f > 0 }

        gross_profit = wins.sum { |t| t.gross_pnl.to_f }
        gross_loss = losses.sum { |t| t.gross_pnl.to_f }.abs

        profit_factor = gross_loss.zero? ? 0 : (gross_profit / gross_loss).round(4)

        {
          profit_factor: profit_factor,
          gross_profit: gross_profit.round(2),
          gross_loss: gross_loss.round(2),
          interpretation: interpret(profit_factor),
          minimum_for_profitability: 1.0,
          grade: grade(profit_factor)
        }
      end

      private

      def interpret(pf)
        case pf
        when 0..1.0    then :unprofitable
        when 1.0..1.5  then :marginal
        when 1.5..2.0  then :decent
        when 2.0..3.0  then :good
        else :excellent
        end
      end

      def grade(pf)
        case pf
        when 0..1.0    then :f
        when 1.0..1.5  then :d
        when 1.5..2.0  then :c
        when 2.0..2.5  then :b
        when 2.5..3.0  then :a
        else :a_plus
        end
      end

      def empty_result
        {
          profit_factor: 0,
          gross_profit: 0,
          gross_loss: 0,
          interpretation: :insufficient_data,
          minimum_for_profitability: 1.0,
          grade: :unknown
        }
      end
    end
  end
end

module Engines
  module Performance
    class SharpeRatioCalculator
      RISK_FREE_RATE = 0.06 / 252  # ~6% annual, daily

      def initialize(trades:)
        @trades = trades
      end

      def call
        return empty_result if @trades.length < 2

        returns = daily_returns
        return empty_result if returns.empty?

        avg_return = returns.sum.to_f / returns.length
        std_dev = standard_deviation(returns)

        return empty_result if std_dev.zero?

        sharpe = ((avg_return - RISK_FREE_RATE) / std_dev) * Math.sqrt(252)

        {
          sharpe_ratio: sharpe.round(4),
          annualized_sharpe: sharpe.round(4),
          daily_returns: returns.length,
          avg_daily_return: (avg_return * 100).round(4),
          daily_volatility: (std_dev * 100).round(4),
          interpretation: interpret(sharpe)
        }
      end

      private

      def daily_returns
        @trades.group_by { |t| t.exit_time&.to_date }
               .transform_values { |day_trades| day_trades.sum { |t| t.gross_pnl.to_f } }
               .values
      end

      def standard_deviation(values)
        mean = values.sum.to_f / values.length
        variance = values.sum { |v| (v - mean)**2 } / values.length
        Math.sqrt(variance)
      end

      def interpret(sharpe)
        case sharpe
        when 0..0.5   then :poor
        when 0.5..1.0 then :below_average
        when 1.0..1.5 then :good
        when 1.5..2.0 then :very_good
        else :excellent
        end
      end

      def empty_result
        {
          sharpe_ratio: 0,
          annualized_sharpe: 0,
          daily_returns: 0,
          avg_daily_return: 0,
          daily_volatility: 0,
          interpretation: :insufficient_data
        }
      end
    end
  end
end

module Engines
  module Performance
    class CalmarRatioCalculator
      def initialize(trades:)
        @trades = trades
      end

      def call
        return empty_result if @trades.empty?

        annual_return = calculate_annual_return
        max_dd = calculate_max_drawdown

        return empty_result if max_dd.zero?

        calmar = annual_return / max_dd

        {
          calmar_ratio: calmar.round(4),
          annual_return_percent: (annual_return * 100).round(2),
          max_drawdown_percent: (max_dd * 100).round(2),
          interpretation: interpret(calmar)
        }
      end

      private

      def calculate_annual_return
        sorted = @trades.sort_by(&:exit_time)
        total_days = days_between(sorted.first.entry_time, sorted.last.exit_time)
        return 0 if total_days.zero?

        total_return = sorted.sum { |t| t.gross_pnl.to_f }
        avg_daily = total_return / total_days
        avg_daily * 252  # Annualize
      end

      def calculate_max_drawdown
        sorted = @trades.sort_by(&:exit_time)
        cumulative = 0
        peak = 0
        max_dd = 0

        sorted.each do |t|
          cumulative += t.gross_pnl.to_f
          peak = [peak, cumulative].max
          dd = peak - cumulative
          max_dd = [max_dd, dd].max
        end

        return 0 if peak.zero?
        max_dd / peak
      end

      def days_between(start_time, end_time)
        return 0 unless start_time && end_time
        ((end_time - start_time) / 86_400).to_i
      end

      def interpret(calmar)
        case calmar
        when 0..0.5   then :poor
        when 0.5..1.0 then :marginal
        when 1.0..2.0 then :good
        when 2.0..3.0 then :very_good
        else :excellent
        end
      end

      def empty_result
        {
          calmar_ratio: 0,
          annual_return_percent: 0,
          max_drawdown_percent: 0,
          interpretation: :insufficient_data
        }
      end
    end
  end
end

module Engines
  module Performance
    class DrawdownAnalyzer
      def initialize(trades:)
        @trades = trades
      end

      def call
        return empty_result if @trades.empty?

        sorted = @trades.sort_by(&:exit_time)
        equity_curve = build_equity_curve(sorted)
        drawdowns = calculate_drawdowns(equity_curve)

        {
          max_drawdown: drawdowns[:max_drawdown].round(2),
          max_drawdown_percent: drawdowns[:max_drawdown_percent].round(2),
          max_drawdown_duration_days: drawdowns[:max_duration_days],
          avg_drawdown: drawdowns[:avg_drawdown].round(2),
          drawdown_count: drawdowns[:count],
          recovery_time_avg_days: drawdowns[:avg_recovery_days],
          current_drawdown: drawdowns[:current].round(2),
          underwater: drawdowns[:underwater],
          equity_curve: equity_curve.last(30), # Last 30 points for charting
          drawdown_series: drawdowns[:series].last(30)
        }
      end

      private

      def build_equity_curve(trades)
        equity = 0
        trades.map do |t|
          equity += t.gross_pnl.to_f
          { date: t.exit_time&.to_date, equity: equity }
        end
      end

      def calculate_drawdowns(equity_curve)
        peak = 0
        max_dd = 0
        max_dd_pct = 0
        dd_start = nil
        max_duration = 0
        current_duration = 0
        drawdowns = []
        series = []
        current_dd = 0

        equity_curve.each_with_index do |point, i|
          equity = point[:equity]
          peak = [peak, equity].max

          if equity < peak
            dd = peak - equity
            dd_pct = peak.zero? ? 0 : (dd / peak * 100)
            current_dd = dd
            current_duration += 1

            if dd > max_dd
              max_dd = dd
              max_dd_pct = dd_pct
              max_duration = current_duration
            end

            dd_start ||= point[:date]
          else
            if current_dd > 0
              drawdowns << {
                amount: current_dd,
                start_date: dd_start,
                end_date: point[:date],
                duration_days: current_duration
              }
            end
            current_dd = 0
            current_duration = 0
            dd_start = nil
          end

          series << { date: point[:date], drawdown: current_dd }
        end

        underwater = current_dd > 0

        {
          max_drawdown: max_dd,
          max_drawdown_percent: max_dd_pct,
          max_duration_days: max_duration,
          avg_drawdown: drawdowns.empty? ? 0 : drawdowns.sum { |d| d[:amount] } / drawdowns.length,
          count: drawdowns.length,
          avg_recovery_days: drawdowns.empty? ? 0 : drawdowns.sum { |d| d[:duration_days] } / drawdowns.length,
          current: current_dd,
          underwater: underwater,
          series: series
        }
      end

      def empty_result
        {
          max_drawdown: 0,
          max_drawdown_percent: 0,
          max_drawdown_duration_days: 0,
          avg_drawdown: 0,
          drawdown_count: 0,
          recovery_time_avg_days: 0,
          current_drawdown: 0,
          underwater: false,
          equity_curve: [],
          drawdown_series: []
        }
      end
    end
  end
end

module Engines
  module Performance
    class ConsecutiveLossAnalyzer
      def initialize(trades:)
        @trades = trades
      end

      def call
        return empty_result if @trades.empty?

        sorted = @trades.sort_by(&:exit_time)
        streaks = extract_streaks(sorted)

        {
          current_streak: streaks[:current],
          max_loss_streak: streaks[:max_loss],
          max_win_streak: streaks[:max_win],
          avg_loss_streak: streaks[:avg_loss_streak].round(2),
          avg_loss_streak_pnl: streaks[:avg_loss_streak_pnl].round(2),
          total_loss_streaks: streaks[:loss_streaks].length,
          total_win_streaks: streaks[:win_streaks].length,
          longest_streak_pnl: streaks[:longest_streak_pnl].round(2),
          streak_risk: streak_risk(streaks),
          streaks: streaks[:loss_streaks].last(5) # Last 5 for detail
        }
      end

      private

      def extract_streaks(trades)
        current_type = nil
        current_count = 0
        current_pnl = 0
        max_loss = 0
        max_win = 0
        loss_streaks = []
        win_streaks = []

        trades.each do |t|
          pnl = t.gross_pnl.to_f
          is_win = pnl > 0

          if is_win && current_type == :win
            current_count += 1
            current_pnl += pnl
          elsif !is_win && current_type == :loss
            current_count += 1
            current_pnl += pnl
          else
            # Streak ended, record it
            if current_type == :loss
              loss_streaks << { count: current_count, pnl: current_pnl }
              max_loss = [max_loss, current_count].max
            elsif current_type == :win
              win_streaks << { count: current_count, pnl: current_pnl }
              max_win = [max_win, current_count].max
            end

            current_type = is_win ? :win : :loss
            current_count = 1
            current_pnl = pnl
          end
        end

        # Record final streak
        if current_type == :loss
          loss_streaks << { count: current_count, pnl: current_pnl }
          max_loss = [max_loss, current_count].max
        elsif current_type == :win
          win_streaks << { count: current_count, pnl: current_pnl }
          max_win = [max_win, current_count].max
        end

        avg_loss = loss_streaks.empty? ? 0 : loss_streaks.sum { |s| s[:count] }.to_f / loss_streaks.length
        avg_loss_pnl = loss_streaks.empty? ? 0 : loss_streaks.sum { |s| s[:pnl] }.to_f / loss_streaks.length
        longest_pnl = (loss_streaks + win_streaks).map { |s| s[:pnl].abs }.max || 0

        {
          current: { type: current_type, count: current_count, pnl: current_pnl },
          max_loss: max_loss,
          max_win: max_win,
          loss_streaks: loss_streaks,
          win_streaks: win_streaks,
          avg_loss_streak: avg_loss,
          avg_loss_streak_pnl: avg_loss_pnl,
          longest_streak_pnl: longest_pnl
        }
      end

      def streak_risk(streaks)
        return :low if streaks[:max_loss] <= 2
        return :moderate if streaks[:max_loss] <= 4
        return :high if streaks[:max_loss] <= 6
        :severe
      end

      def empty_result
        {
          current_streak: { type: :none, count: 0, pnl: 0 },
          max_loss_streak: 0,
          max_win_streak: 0,
          avg_loss_streak: 0,
          avg_loss_streak_pnl: 0,
          total_loss_streaks: 0,
          total_win_streaks: 0,
          longest_streak_pnl: 0,
          streak_risk: :low,
          streaks: []
        }
      end
    end
  end
end

module Engines
  module Performance
    class ExpectancyCalculator
      def initialize(trades:)
        @trades = trades
      end

      def call
        return empty_result if @trades.empty?

        wins = @trades.select { |t| t.gross_pnl.to_f > 0 }
        losses = @trades.reject { |t| t.gross_pnl.to_f > 0 }

        win_rate = wins.length.to_f / @trades.length
        avg_win = wins.empty? ? 0 : wins.sum { |t| t.gross_pnl.to_f } / wins.length
        avg_loss = losses.empty? ? 0 : losses.sum { |t| t.gross_pnl.to_f }.abs / losses.length

        expectancy = (win_rate * avg_win) - ((1 - win_rate) * avg_loss)
        expectancy_per_trade = expectancy
        expectancy_per_rupee_risked = calculate_expectancy_per_risk(expectancy)

        {
          expectancy: expectancy.round(2),
          expectancy_per_trade: expectancy_per_trade.round(2),
          expectancy_per_rupee_risked: expectancy_per_rupee_risked.round(4),
          win_rate: (win_rate * 100).round(2),
          avg_win: avg_win.round(2),
          avg_loss: avg_loss.round(2),
          payoff_ratio: avg_loss.zero? ? 0 : (avg_win / avg_loss).round(2),
          kelly_criterion: calculate_kelly(win_rate, avg_win, avg_loss).round(4),
          half_kelly: (calculate_kelly(win_rate, avg_win, avg_loss) / 2).round(4),
          interpretation: interpret(expectancy)
        }
      end

      private

      def calculate_expectancy_per_risk(expectancy)
        avg_risk = @trades.sum do |t|
          next 0 unless t.stop_loss && t.entry_price
          (t.entry_price.to_d - t.stop_loss.to_d).abs.to_f * t.quantity.to_i
        end / @trades.length

        avg_risk.zero? ? 0 : expectancy / avg_risk
      end

      def calculate_kelly(win_rate, avg_win, avg_loss)
        return 0 if avg_loss.zero?
        b = avg_win / avg_loss  # Odds
        (win_rate * b - (1 - win_rate)) / b
      end

      def interpret(expectancy)
        case expectancy
        when 0..500    then :positive_edge
        when -500..0   then :marginal
        else :negative_edge
        end
      end

      def empty_result
        {
          expectancy: 0,
          expectancy_per_trade: 0,
          expectancy_per_rupee_risked: 0,
          win_rate: 0,
          avg_win: 0,
          avg_loss: 0,
          payoff_ratio: 0,
          kelly_criterion: 0,
          half_kelly: 0,
          interpretation: :insufficient_data
        }
      end
    end
  end
end

module Engines
  module Performance
    class BenchmarkComparison
      BENCHMARK_SYMBOLS = {
        "NIFTY" => "NIFTY 50",
        "BANKNIFTY" => "NIFTY BANK",
        "FINNIFTY" => "NIFTY FIN SERVICE",
        "SENSEX" => "SENSEX"
      }.freeze

      def initialize(trades:, instrument: nil)
        @trades = trades
        @instrument = instrument
      end

      def call
        return empty_result if @trades.empty?

        strategy_return = calculate_strategy_return
        benchmark_return = fetch_benchmark_return
        alpha = benchmark_return ? (strategy_return - benchmark_return) : nil

        {
          strategy_return_percent: (strategy_return * 100).round(2),
          benchmark_return_percent: benchmark_return ? (benchmark_return * 100).round(2) : nil,
          alpha_percent: alpha ? (alpha * 100).round(2) : nil,
          outperformed: alpha && alpha > 0,
          benchmark_symbol: benchmark_symbol,
          period_days: period_days,
          strategy_cagr: calculate_cagr(strategy_return),
          benchmark_cagr: benchmark_return ? calculate_cagr(benchmark_return) : nil
        }
      end

      private

      def calculate_strategy_return
        sorted = @trades.sort_by(&:exit_time)
        total_pnl = sorted.sum { |t| t.gross_pnl.to_f }
        capital = estimated_capital

        return 0 if capital.zero?
        total_pnl / capital
      end

      def fetch_benchmark_return
        return nil unless @instrument

        symbol = benchmark_symbol
        return nil unless symbol

        start_date = @trades.minimum(:entry_time)&.to_date
        end_date = @trades.maximum(:exit_time)&.to_date
        return nil unless start_date && end_date

        # Fetch from CandleRepository or external source
        start_price = fetch_index_price(symbol, start_date)
        end_price = fetch_index_price(symbol, end_date)

        return nil if start_price.nil? || end_price.nil? || start_price.zero?
        (end_price - start_price) / start_price
      end

      def fetch_index_price(symbol, date)
        # Query from database or use cached value
        candle = Candle.where(instrument: Instrument.find_by(trading_symbol: symbol))
                       .where("timestamp::date = ?", date)
                       .order(:timestamp)
                       .first
        candle&.close&.to_f
      rescue
        nil
      end

      def benchmark_symbol
        return nil unless @instrument
        BENCHMARK_SYMBOLS[@instrument.trading_symbol] || "NIFTY 50"
      end

      def estimated_capital
        # Estimate based on max position value or config
        max_pos = @trades.map { |t| t.max_position_value.to_f }.max
        max_pos > 0 ? max_pos : AppConfig.config.risk&.max_capital&.to_f || 500_000
      end

      def period_days
        sorted = @trades.sort_by(&:exit_time)
        return 0 unless sorted.first&.entry_time && sorted.last&.exit_time
        ((sorted.last.exit_time - sorted.first.entry_time) / 86_400).to_i
      end

      def calculate_cagr(total_return)
        years = period_days.to_f / 365
        return 0 if years.zero?
        ((1 + total_return) ** (1 / years) - 1).round(4)
      end

      def empty_result
        {
          strategy_return_percent: 0,
          benchmark_return_percent: nil,
          alpha_percent: nil,
          outperformed: nil,
          benchmark_symbol: benchmark_symbol,
          period_days: 0,
          strategy_cagr: 0,
          benchmark_cagr: nil
        }
      end
    end
  end
end

module Engines
  module Performance
    class PerformanceDashboardData
      def initialize(strategy: nil, instrument: nil, lookback: 30.days)
        @strategy = strategy
        @instrument = instrument
        @lookback = lookback
      end

      def call
        trades = fetch_trades
        return empty_dashboard if trades.empty?

        engine = PerformanceEngine.new(trades: trades, strategy: @strategy, instrument: @instrument, lookback: @lookback)
        result = engine.call

        return empty_dashboard unless result.success?

        report = result.value

        {
          summary: report[:summary],
          win_rate: report[:win_rate],
          profit_factor: report[:profit_factor],
          sharpe_ratio: report[:sharpe_ratio],
          sortino_ratio: report[:sortino_ratio],
          calmar_ratio: report[:calmar_ratio],
          drawdown: report[:drawdown],
          consecutive_losses: report[:consecutive_losses],
          expectancy: report[:expectancy],
          benchmark: report[:benchmark],
          daily_pnl: daily_pnl_series(trades),
          strategy_comparison: strategy_comparison,
          recent_trades: recent_trades(trades)
        }
      end

      private

      def fetch_trades
        scope = Trade.where("exit_time IS NOT NULL")
                     .where("entry_time > ?", Time.current - @lookback)

        scope = scope.where(setup_type: @strategy) if @strategy
        scope = scope.where(instrument: @instrument) if @instrument

        scope.order(:entry_time)
      end

      def daily_pnl_series(trades)
        trades.group_by { |t| t.exit_time&.to_date }
              .transform_values { |day_trades| day_trades.sum { |t| t.gross_pnl.to_f }.round(2) }
              .sort_by { |date, _| date }
              .map { |date, pnl| { date: date, pnl: pnl } }
      end

      def strategy_comparison
        strategies = Trade.where("exit_time IS NOT NULL")
                          .where("entry_time > ?", Time.current - @lookback)
                          .distinct
                          .pluck(:setup_type)

        strategies.map do |strat|
          strat_trades = Trade.where(setup_type: strat)
                              .where("exit_time IS NOT NULL")
                              .where("entry_time > ?", Time.current - @lookback)

          wins = strat_trades.select { |t| t.gross_pnl.to_f > 0 }
          pnls = strat_trades.map { |t| t.gross_pnl.to_f }

          {
            strategy: strat,
            trades: strat_trades.count,
            win_rate: strat_trades.empty? ? 0 : (wins.length.to_f / strat_trades.count * 100).round(2),
            total_pnl: pnls.sum.round(2),
            avg_pnl: strat_trades.empty? ? 0 : (pnls.sum / strat_trades.count).round(2)
          }
        end.sort_by { |s| -s[:total_pnl] }
      end

      def recent_trades(trades)
        trades.last(10).reverse.map do |t|
          {
            id: t.id,
            instrument: t.instrument&.trading_symbol,
            direction: t.direction,
            strategy: t.setup_type,
            pnl: t.gross_pnl.to_f.round(2),
            pnl_percent: t.pnl_percent.to_f.round(2),
            exit_time: t.exit_time&.iso8601
          }
        end
      end

      def empty_dashboard
        {
          message: "No trades in lookback period",
          summary: {},
          daily_pnl: [],
          strategy_comparison: [],
          recent_trades: []
        }
      end
    end
  end
end

class PerformanceReportJob < JobBase
  queue_as :reports

  def perform(report_type = "daily")
    lookback = case report_type
               when "daily" then 1.day
               when "weekly" then 7.days
               when "monthly" then 30.days
               else 1.day
               end

    # Generate overall report
    engine = Engines::Performance::PerformanceEngine.new(lookback: lookback)
    result = engine.call

    return if result.failure?

    report = result.value
    PerformanceReport.create!(
      report_type: report_type,
      summary: report[:summary],
      win_rate_data: report[:win_rate],
      ratios: {
        sharpe: report[:sharpe_ratio],
        sortino: report[:sortino_ratio],
        calmar: report[:calmar_ratio],
        profit_factor: report[:profit_factor]
      },
      drawdown_data: report[:drawdown],
      expectancy_data: report[:expectancy],
      benchmark_data: report[:benchmark],
      period_start: lookback.ago.to_date,
      period_end: Date.current
    )

    # Generate per-strategy reports
    Trade.distinct.pluck(:setup_type).compact.each do |strategy|
      strat_result = Engines::Performance::PerformanceEngine.new(strategy: strategy, lookback: lookback).call
      next if strat_result.failure?

      PerformanceReport.create!(
        report_type: report_type,
        strategy: strategy,
        summary: strat_result.value[:summary],
        win_rate_data: strat_result.value[:win_rate],
        ratios: {
          sharpe: strat_result.value[:sharpe_ratio],
          sortino: strat_result.value[:sortino_ratio],
          calmar: strat_result.value[:calmar_ratio],
          profit_factor: strat_result.value[:profit_factor]
        },
        drawdown_data: strat_result.value[:drawdown],
        expectancy_data: strat_result.value[:expectancy],
        period_start: lookback.ago.to_date,
        period_end: Date.current
      )
    end

    Rails.logger.info "[PerformanceReportJob] Generated #{report_type} performance reports"
  end
end

- **Recovery Time**: Average days to recover
