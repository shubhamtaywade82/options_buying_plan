## Day 1 — Base Indicator Classes, EMA & VWAP

### Task 1.1: Create `BaseIndicator`
- **File:** `app/indicators/base_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class BaseIndicator
      include AppInject

      attr_reader :instrument, :timeframe

      def initialize(instrument:, timeframe: "5m")
        @instrument = instrument
        @timeframe = timeframe
      end

      def calculate
        raise NotImplementedError, "#{self.class} must implement #calculate"
      end

      def value
        cached = fetch_cache
        return cached if cached

        result = calculate
        write_cache(result) if result.present?
        result
      end

      def name
        "#{self.class.name.demodulize.underscore}_#{timeframe}"
      end

      def cache_key
        "indicator:#{instrument.id}:#{timeframe}:#{name}"
      end

      def sufficient_data?(required_bars)
        available = Candle.for_instrument(instrument)
                         .for_timeframe(timeframe)
                         .where(is_complete: true)
                         .count
        available >= required_bars
      end

      def latest_candles(limit:)
        Candle.for_instrument(instrument)
              .for_timeframe(timeframe)
              .where(is_complete: true)
              .order(timestamp: :asc)
              .last(limit)
      end

      def closes(limit:)
        latest_candles(limit: limit).pluck(:close).map(&:to_d)
      end

      def highs(limit:)
        latest_candles(limit: limit).pluck(:high).map(&:to_d)
      end

      def lows(limit:)
        latest_candles(limit: limit).pluck(:low).map(&:to_d)
      end

      def volumes(limit:)
        latest_candles(limit: limit).pluck(:volume).map(&:to_i)
      end

      private

      def fetch_cache
        CacheStore.fetch(cache_key)
      end

      def write_cache(result)
        CacheStore.write(cache_key, result, ttl: 300)  # 5 minutes
      end

      def validate_sufficient_data!(required)
        return if sufficient_data?(required)
        raise InsufficientDataError, "#{name} requires #{required} bars, have #{available_bars}"
      end

      def available_bars
        Candle.for_instrument(instrument)
              .for_timeframe(timeframe)
              .where(is_complete: true)
              .count
      end
    end

    class InsufficientDataError < StandardError; end
  end
  ```
- **Deliverable:** All indicators inherit from `BaseIndicator`
- **Acceptance Criteria:** `TestIndicator.new(instrument: nifty).value` fetches cache or calculates

### Task 1.2: Create `EMAIndicator`
- **File:** `app/indicators/ema_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class EMAIndicator < BaseIndicator
      PERIODS = {
        ema_9: 9,
        ema_21: 21,
        ema_50: 50,
        ema_200: 200
      }.freeze

      attr_reader :period

      def initialize(instrument:, timeframe: "5m", period: 21)
        super(instrument: instrument, timeframe: timeframe)
        @period = period
      end

      def calculate
        validate_sufficient_data!(period * 2)

        closes = closes(limit: period * 2)
        ema = Calculations::Statistics.ema(closes, period)
        { value: ema, period: period, slope: slope(closes, ema) }
      end

      def name
        "ema_#{period}_#{timeframe}"
      end

      def above_price?
        latest = latest_candles(limit: 1).first
        return false unless latest && value
        latest.close > value[:value]
      end

      def below_price?
        latest = latest_candles(limit: 1).first
        return false unless latest && value
        latest.close < value[:value]
      end

      def alignment(other_period:)
        other = EMAIndicator.new(instrument: instrument, timeframe: timeframe, period: other_period).value
        return :unknown unless other && value

        if value[:value] > other[:value]
          :above
        elsif value[:value] < other[:value]
          :below
        else
          :equal
        end
      end

      private

      def slope(closes, current_ema)
        return 0 if closes.length < 2
        previous_close = closes[-2]
        return 0 if previous_close.zero?
        ((current_ema - previous_close) / previous_close * 100).round(4)
      end
    end
  end
  ```
- **Deliverable:** EMA with slope, alignment, and price comparison
- **Acceptance Criteria:** `EMAIndicator.new(instrument: nifty, period: 21).value[:value]` returns float; slope calculated

### Task 1.3: Create `VWAPIndicator`
- **File:** `app/indicators/vwap_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class VWAPIndicator < BaseIndicator
      DEFAULT_BANDS = [1.0, 2.0, 3.0].freeze

      attr_reader :bands

      def initialize(instrument:, timeframe: "5m", bands: DEFAULT_BANDS)
        super(instrument: instrument, timeframe: timeframe)
        @bands = bands
      end

      def calculate
        candles = latest_candles(limit: 1000)  # VWAP typically intraday

        return nil if candles.empty?

        total_value = 0
        total_volume = 0

        candles.each do |c|
          typical_price = (c.high.to_d + c.low.to_d + c.close.to_d) / 3
          total_value += typical_price * c.volume.to_i
          total_volume += c.volume.to_i
        end

        vwap = total_volume.zero? ? 0 : (total_value / total_volume).round(4)

        # Standard deviation
        squared_diffs = candles.map do |c|
          typical_price = (c.high.to_d + c.low.to_d + c.close.to_d) / 3
          (typical_price - vwap) ** 2
        end

        variance = squared_diffs.sum / candles.length
        std_dev = Math.sqrt(variance).round(4)

        band_values = bands.map do |multiplier|
          {
            multiplier: multiplier,
            upper: (vwap + (std_dev * multiplier)).round(4),
            lower: (vwap - (std_dev * multiplier)).round(4)
          }
        end

        {
          value: vwap,
          std_dev: std_dev,
          bands: band_values,
          distance_from_price: distance_from_price(vwap),
          above_vwap: above_vwap?,
          below_vwap: below_vwap?
        }
      end

      def name
        "vwap_#{timeframe}"
      end

      def distance_from_price(vwap)
        latest = latest_candles(limit: 1).first
        return nil unless latest && vwap && vwap > 0
        ((latest.close.to_d - vwap) / vwap * 100).round(4)
      end

      def above_vwap?
        latest = latest_candles(limit: 1).first
        return false unless latest && value
        latest.close > value[:value]
      end

      def below_vwap?
        latest = latest_candles(limit: 1).first
        return false unless latest && value
        latest.close < value[:value]
      end
    end
  end
  ```
- **Deliverable:** VWAP with standard deviation bands
- **Acceptance Criteria:** `VWAPIndicator.new(instrument: nifty).value[:bands]` returns upper/lower for 1x, 2x, 3x

### Task 2.1: Create `ATRIndicator`
- **File:** `app/indicators/atr_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class ATRIndicator < BaseIndicator
      DEFAULT_PERIOD = 14

      attr_reader :period

      def initialize(instrument:, timeframe: "5m", period: DEFAULT_PERIOD)
        super(instrument: instrument, timeframe: timeframe)
        @period = period
      end

      def calculate
        validate_sufficient_data!(period + 1)

        candles = latest_candles(limit: period + 1)
        tr_values = true_ranges(candles)

        atr = tr_values.first(period).sum.to_f / period

        {
          value: atr.round(4),
          period: period,
          expansion: expansion?(atr, candles),
          contraction: contraction?(atr, candles)
        }
      end

      def name
        "atr_#{period}_#{timeframe}"
      end

      def stop_loss_price(direction:, multiplier: 2.0)
        latest = latest_candles(limit: 1).first
        return nil unless latest && value

        if direction == :long
          (latest.close.to_d - (value[:value] * multiplier)).round(4)
        else
          (latest.close.to_d + (value[:value] * multiplier)).round(4)
        end
      end

      private

      def true_ranges(candles)
        ranges = []
        candles.each_cons(2) do |prev, curr|
          tr = [
            curr.high.to_d - curr.low.to_d,
            (curr.high.to_d - prev.close.to_d).abs,
            (curr.low.to_d - prev.close.to_d).abs
          ].max
          ranges << tr
        end
        ranges
      end

      def expansion?(atr, candles)
        return false if candles.length < period * 2
        prev_atr = true_ranges(candles.first(period + 1)).first(period).sum.to_f / period
        atr > prev_atr * 1.1
      end

      def contraction?(atr, candles)
        return false if candles.length < period * 2
        prev_atr = true_ranges(candles.first(period + 1)).first(period).sum.to_f / period
        atr < prev_atr * 0.9
      end
    end
  end
  ```
- **Deliverable:** ATR with expansion/contraction detection
- **Acceptance Criteria:** `ATRIndicator.new(instrument: nifty).value[:value]` returns volatility; stop_loss_price works

### Task 2.2: Create `ADXIndicator`
- **File:** `app/indicators/adx_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class ADXIndicator < BaseIndicator
      DEFAULT_PERIOD = 14

      attr_reader :period

      def initialize(instrument:, timeframe: "5m", period: DEFAULT_PERIOD)
        super(instrument: instrument, timeframe: timeframe)
        @period = period
      end

      def calculate
        validate_sufficient_data!(period * 3)

        candles = latest_candles(limit: period * 3)

        dm_plus = []
        dm_minus = []
        tr_values = []

        candles.each_cons(2) do |prev, curr|
          up_move = curr.high.to_d - prev.high.to_d
          down_move = prev.low.to_d - curr.low.to_d

          dm_plus << (up_move > down_move && up_move > 0 ? up_move : 0)
          dm_minus << (down_move > up_move && down_move > 0 ? down_move : 0)

          tr = [
            curr.high.to_d - curr.low.to_d,
            (curr.high.to_d - prev.close.to_d).abs,
            (curr.low.to_d - prev.close.to_d).abs
          ].max
          tr_values << tr
        end

        # Smooth DM and TR
        di_plus = smooth(dm_plus, period)
        di_minus = smooth(dm_minus, period)
        atr = smooth(tr_values, period)

        return nil if atr.zero?

        dx_values = di_plus.zip(di_minus).map do |plus, minus|
          diff = (plus - minus).abs
          sum = plus + minus
          sum.zero? ? 0 : (diff / sum * 100)
        end

        adx = dx_values.last(period).sum.to_f / period

        {
          value: adx.round(2),
          di_plus: di_plus.last&.round(2),
          di_minus: di_minus.last&.round(2),
          trend_strength: trend_strength(adx),
          trending: adx > 25,
          ranging: adx < 20
        }
      end

      def name
        "adx_#{period}_#{timeframe}"
      end

      def trend_direction
        result = value
        return :unknown unless result

        if result[:di_plus] > result[:di_minus]
          :bullish
        elsif result[:di_minus] > result[:di_plus]
          :bearish
        else
          :neutral
        end
      end

      private

      def smooth(values, period)
        return 0 if values.length < period
        values.last(period).sum.to_f / period
      end

      def trend_strength(adx)
        if adx > 50
          :very_strong
        elsif adx > 25
          :strong
        elsif adx > 20
          :moderate
        else
          :weak
        end
      end
    end
  end
  ```
- **Deliverable:** ADX with DI+ / DI- and trend strength classification
- **Acceptance Criteria:** `ADXIndicator.new(instrument: nifty).value[:trending]` returns `true` when ADX > 25

### Task 2.3: Create `RSIIndicator`
- **File:** `app/indicators/rsi_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class RSIIndicator < BaseIndicator
      DEFAULT_PERIOD = 14

      attr_reader :period

      def initialize(instrument:, timeframe: "5m", period: DEFAULT_PERIOD)
        super(instrument: instrument, timeframe: timeframe)
        @period = period
      end

      def calculate
        validate_sufficient_data!(period + 1)

        closes = closes(limit: period + 1)
        changes = closes.each_cons(2).map { |prev, curr| curr - prev }

        gains = changes.map { |c| c > 0 ? c : 0 }
        losses = changes.map { |c| c < 0 ? c.abs : 0 }

        avg_gain = gains.last(period).sum.to_f / period
        avg_loss = losses.last(period).sum.to_f / period

        return { value: 50, period: period, condition: :neutral } if avg_loss.zero?

        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))

        {
          value: rsi.round(2),
          period: period,
          condition: condition(rsi),
          oversold: rsi < 30,
          overbought: rsi > 70
        }
      end

      def name
        "rsi_#{period}_#{timeframe}"
      end

      def divergence?(lookback: 5)
        rsi_values = (1..lookback).map do |i|
          candle = Candle.for_instrument(instrument)
                        .for_timeframe(timeframe)
                        .where(is_complete: true)
                        .order(timestamp: :desc)
                        .offset(i)
                        .first
          next unless candle
          # Simplified: would need historical RSI values
        end
        false  # Placeholder for full divergence detection
      end

      private

      def condition(rsi)
        if rsi > 70
          :overbought
        elsif rsi < 30
          :oversold
        elsif rsi > 50
          :bullish
        else
          :bearish
        end
      end
    end
  end
  ```

- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class MACDIndicator < BaseIndicator
      FAST_PERIOD = 12
      SLOW_PERIOD = 26
      SIGNAL_PERIOD = 9

      def calculate
        validate_sufficient_data!(SLOW_PERIOD + SIGNAL_PERIOD)

        closes = closes(limit: SLOW_PERIOD + SIGNAL_PERIOD)

        ema_fast = Calculations::Statistics.ema(closes, FAST_PERIOD)
        ema_slow = Calculations::Statistics.ema(closes, SLOW_PERIOD)

        return nil unless ema_fast && ema_slow

        macd_line = ema_fast - ema_slow

        # Calculate signal line (EMA of MACD)
        macd_history = []
        (SLOW_PERIOD...closes.length).each do |i|
          slice = closes[0..i]
          fast = Calculations::Statistics.ema(slice, FAST_PERIOD)
          slow = Calculations::Statistics.ema(slice, SLOW_PERIOD)
          macd_history << (fast - slow) if fast && slow
        end

        signal_line = Calculations::Statistics.ema(macd_history, SIGNAL_PERIOD)
        histogram = signal_line ? (macd_line - signal_line) : nil

        {
          macd: macd_line.round(4),
          signal: signal_line&.round(4),
          histogram: histogram&.round(4),
          bullish_crossover: histogram && histogram > 0 && macd_history.length >= 2 && macd_history[-2] < 0,
          bearish_crossover: histogram && histogram < 0 && macd_history.length >= 2 && macd_history[-2] > 0
        }
      end

      def name
        "macd_#{timeframe}"
      end
    end
  end
  ```
- **Deliverable:** MACD with signal line, histogram, and crossover detection
- **Acceptance Criteria:** `MACDIndicator.new(instrument: nifty).value[:bullish_crossover]` detects MACD crossing above signal

  ```ruby
  module Indicators

        latest_close = candles.last.close.to_d

        trend = if latest_close > upper_band
                  :bullish
                elsif latest_close < lower_band
                  :bearish
                else
                  prev_close && prev_close > upper_band ? :bullish : :bearish
                end

        {
          upper_band: upper_band.round(4),
          lower_band: lower_band.round(4),
          trend: trend,
          value: trend == :bullish ? lower_band.round(4) : upper_band.round(4),
          atr: atr[:value]
        }
      end

      def name
        "supertrend_#{atr_period}_#{multiplier}_#{timeframe}"
      end

      end

      end
    end
  end
  ```

### Task 3.3: Create `BollingerBandsIndicator`
- **File:** `app/indicators/bollinger_bands_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class BollingerBandsIndicator < BaseIndicator
      DEFAULT_PERIOD = 20
      DEFAULT_MULTIPLIER = 2.0

      attr_reader :period, :multiplier

      def initialize(instrument:, timeframe: "5m", period: DEFAULT_PERIOD, multiplier: DEFAULT_MULTIPLIER)
        super(instrument: instrument, timeframe: timeframe)
        @period = period
        @multiplier = multiplier
      end

      def calculate
        validate_sufficient_data!(period)

        closes = closes(limit: period)
        sma = closes.sum.to_f / period
        std_dev = Calculations::Statistics.std_dev(closes, period)

        upper = sma + (multiplier * std_dev)
        lower = sma - (multiplier * std_dev)

        latest = latest_candles(limit: 1).first
        band_width = upper - lower

        {
          middle: sma.round(4),
          upper: upper.round(4),
          lower: lower.round(4),
          bandwidth: band_width.round(4),
          bandwidth_percent: sma.zero? ? 0 : (band_width / sma * 100).round(4),
          squeeze: band_width < (sma * 0.05),  # < 5% of price
          position: position(latest&.close, upper, lower),
          percent_b: percent_b(latest&.close, upper, lower)
        }
      end

      def name
        "bollinger_#{period}_#{multiplier}_#{timeframe}"
      end

      def squeeze?
        value&.dig(:squeeze) == true
      end

      private

      def position(close, upper, lower)
        return nil unless close
        if close > upper
          :above_upper
        elsif close < lower
          :below_lower
        elsif close > (upper + lower) / 2
          :upper_half
        else
          :lower_half
        end
      end

      def percent_b(close, upper, lower)
        return nil unless close && upper && lower
        range = upper - lower
        return 50 if range.zero?
        ((close.to_d - lower) / range * 100).round(2)
      end
    end
  end
  ```
- **Deliverable:** Bollinger Bands with squeeze detection and %B
- **Acceptance Criteria:** `BollingerBandsIndicator.new(instrument: nifty).squeeze?` returns `true` when bandwidth < 5%

### Task 4.1: Create `VolumeProfileIndicator`
- **File:** `app/indicators/volume_profile_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class VolumeProfileIndicator < BaseIndicator
      DEFAULT_BUCKETS = 10

      attr_reader :buckets

      def initialize(instrument:, timeframe: "5m", buckets: DEFAULT_BUCKETS)
        super(instrument: instrument, timeframe: timeframe)
        @buckets = buckets
      end

      def calculate
        candles = latest_candles(limit: 100)  # Last 100 candles for profile

        return nil if candles.empty?

        min_price = candles.minimum(:low).to_d
        max_price = candles.maximum(:high).to_d
        bucket_size = (max_price - min_price) / buckets

        return nil if bucket_size.zero?

        profile = Hash.new(0)

        candles.each do |c|
          typical = (c.high.to_d + c.low.to_d + c.close.to_d) / 3
          idx = ((typical - min_price) / bucket_size).to_i
          idx = [idx, buckets - 1].min
          profile[idx] += c.volume.to_i
        end

        poc_idx = profile.max_by { |_, v| v }&.first
        poc_price = poc_idx ? (min_price + (poc_idx * bucket_size) + (bucket_size / 2)).round(4) : nil

        # Value area (70% of volume)
        total_volume = profile.values.sum
        sorted = profile.sort_by { |_, v| -v }
        va_volume = 0
        va_indices = []

        sorted.each do |idx, vol|
          va_volume += vol
          va_indices << idx
          break if va_volume >= total_volume * 0.7
        end

        va_low = min_price + (va_indices.min * bucket_size)
        va_high = min_price + ((va_indices.max + 1) * bucket_size)

        {
          poc: poc_price,
          value_area_low: va_low.round(4),
          value_area_high: va_high.round(4),
          value_area_width: (va_high - va_low).round(4),
          profile: profile.transform_keys { |k| (min_price + (k * bucket_size)).round(2) },
          total_volume: total_volume
        }
      end

      def name
        "volume_profile_#{buckets}_#{timeframe}"
      end

      def price_in_value_area?
        latest = latest_candles(limit: 1).first
        return false unless latest && value
        latest.close >= value[:value_area_low] && latest.close <= value[:value_area_high]
      end
    end
  end
  ```
- **Deliverable:** Volume Profile with POC and Value Area
- **Acceptance Criteria:** `VolumeProfileIndicator.new(instrument: nifty).value[:poc]` returns point of control

### Task 4.2: Create `RelativeVolumeIndicator`

  ```ruby
  module Indicators
    class RelativeVolumeIndicator < BaseIndicator

                          .where(is_complete: true)
                          .where("timestamp < ?", latest_candles(limit: 1).first&.timestamp)
                          .order(timestamp: :desc)
                          .limit(lookback)
                          .pluck(:volume)

        return nil if historical.empty?

        avg_volume = historical.sum.to_f / historical.length
        rv = avg_volume.zero? ? 0 : (current_volume / avg_volume).round(2)

        {
          value: rv,
          current_volume: current_volume,
          average_volume: avg_volume.round(0),
          lookback: lookback,
          spike: rv > 1.5,
          confirmation: rv > 1.0
        }
      end

      def name
        "rvol_#{lookback}_#{timeframe}"
      end
    end
  end
  ```
- **Deliverable:** Relative volume with spike detection (>1.5x)
- **Acceptance Criteria:** `RelativeVolumeIndicator.new(instrument: nifty).value[:spike]` returns `true` when >1.5x

### Task 4.4: Create `ROCIndicator`
- **File:** `app/indicators/roc_indicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Indicators
    class ROCIndicator < BaseIndicator
      DEFAULT_PERIOD = 12

      attr_reader :period

      def initialize(instrument:, timeframe: "5m", period: DEFAULT_PERIOD)
        super(instrument: instrument, timeframe: timeframe)
        @period = period
      end

      def calculate
        validate_sufficient_data!(period + 1)

        closes = closes(limit: period + 1)
        current = closes.last
        previous = closes[-period - 1]

        return nil if previous.nil? || previous.zero?

        roc = ((current - previous) / previous * 100).round(4)

        {
          value: roc,
          period: period,
          accelerating: roc > 0 && closes[-2] && ((closes[-2] - closes[-period - 2]) / closes[-period - 2] * 100) < roc,
          decelerating: roc < 0 && closes[-2] && ((closes[-2] - closes[-period - 2]) / closes[-period - 2] * 100) > roc
        }
      end

      def name
        "roc_#{period}_#{timeframe}"
      end
    end
  end
  ```
- **Deliverable:** Rate of Change with acceleration/deceleration
- **Acceptance Criteria:** `ROCIndicator.new(instrument: nifty).value[:value]` returns percent change

  ```ruby
    class IndicatorRegistry
      INDICATORS = {

        end

        end

        def compute(name, instrument:, timeframe: "5m", **kwargs)
          klass = find(name)
          return Result.failure(:unknown_indicator) unless klass

          indicator = klass.new(instrument: instrument, timeframe: timeframe, **kwargs)

              indicator = klass.new(instrument: instrument, timeframe: timeframe)
              results[name] = indicator.value
            rescue => e
              results[name] = { error: e.message }
            end
          end

          Result.success(results)
        end
      end
  ```

  ```ruby
  # frozen_string_literal: true

  class CompositeIndicatorService < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def call(timeframe: "5m")
      results = {}

      # Trend indicators
      results[:ema] = ema_bundle(timeframe)
      results[:vwap] = vwap_bundle(timeframe)
      results[:adx] = adx_bundle(timeframe)
      results[:supertrend] = supertrend_bundle(timeframe)

      # Momentum indicators
      results[:rsi] = rsi_bundle(timeframe)
      results[:macd] = macd_bundle(timeframe)
      results[:roc] = roc_bundle(timeframe)

      # Volatility indicators
      results[:atr] = atr_bundle(timeframe)
      results[:bollinger] = bollinger_bundle(timeframe)

      # Volume indicators
      results[:relative_volume] = rv_bundle(timeframe)
      results[:volume_profile] = vp_bundle(timeframe)

      # Session indicators
      results[:opening_range] = or_bundle

      # Composite signals
      results[:trend_alignment] = trend_alignment(results)
      results[:momentum_score] = momentum_score(results)
      results[:volatility_regime] = volatility_regime(results)

      success(results)
    end

    private

    def ema_bundle(timeframe)
      ema_21 = Indicators::IndicatorRegistry.compute(:ema, instrument: @instrument, timeframe: timeframe, period: 21).value
      ema_50 = Indicators::IndicatorRegistry.compute(:ema, instrument: @instrument, timeframe: timeframe, period: 50).value

      {
        ema_21: ema_21,
        ema_50: ema_50,
        alignment: ema_alignment(ema_21, ema_50),
        golden_cross: ema_21 && ema_50 && ema_21[:value] > ema_50[:value],
        death_cross: ema_21 && ema_50 && ema_21[:value] < ema_50[:value]
      }
    end

    def ema_alignment(ema_21, ema_50)
      return :unknown unless ema_21 && ema_50
      ema_21[:value] > ema_50[:value] ? :bullish : :bearish
    end

    def vwap_bundle(timeframe)
      vwap = Indicators::IndicatorRegistry.compute(:vwap, instrument: @instrument, timeframe: timeframe).value
      { value: vwap, above: vwap&.dig(:above_vwap), below: vwap&.dig(:below_vwap) }
    end

    def adx_bundle(timeframe)
      adx = Indicators::IndicatorRegistry.compute(:adx, instrument: @instrument, timeframe: timeframe).value
      { value: adx, trending: adx&.dig(:trending), strength: adx&.dig(:trend_strength) }
    end

    def supertrend_bundle(timeframe)
      st = Indicators::IndicatorRegistry.compute(:supertrend, instrument: @instrument, timeframe: timeframe).value
      { value: st, trend: st&.dig(:trend) }
    end

    def rsi_bundle(timeframe)
      rsi = Indicators::IndicatorRegistry.compute(:rsi, instrument: @instrument, timeframe: timeframe).value
      { value: rsi, condition: rsi&.dig(:condition) }
    end

    def macd_bundle(timeframe)
      macd = Indicators::IndicatorRegistry.compute(:macd, instrument: @instrument, timeframe: timeframe).value
      { value: macd, bullish: macd&.dig(:bullish_crossover), bearish: macd&.dig(:bearish_crossover) }
    end

    def roc_bundle(timeframe)
      roc = Indicators::IndicatorRegistry.compute(:roc, instrument: @instrument, timeframe: timeframe).value
      { value: roc }
    end

    def atr_bundle(timeframe)
      atr = Indicators::IndicatorRegistry.compute(:atr, instrument: @instrument, timeframe: timeframe).value
      { value: atr, expansion: atr&.dig(:expansion) }
    end

    def bollinger_bundle(timeframe)
      bb = Indicators::IndicatorRegistry.compute(:bollinger, instrument: @instrument, timeframe: timeframe).value
      { value: bb, squeeze: bb&.dig(:squeeze), position: bb&.dig(:position) }
    end

    def rv_bundle(timeframe)
      rv = Indicators::IndicatorRegistry.compute(:relative_volume, instrument: @instrument, timeframe: timeframe).value
      { value: rv, spike: rv&.dig(:spike) }
    end

    def vp_bundle(timeframe)
      vp = Indicators::IndicatorRegistry.compute(:volume_profile, instrument: @instrument, timeframe: timeframe).value
      { value: vp, in_value_area: vp&.dig(:poc) }
    end

    def or_bundle
      or_15 = Indicators::IndicatorRegistry.compute(:opening_range, instrument: @instrument, range_minutes: 15).value
      { value: or_15, breakout: or_15&.dig(:breakout) }
    end

    def trend_alignment(results)
      ema = results[:ema][:alignment]
      vwap = results[:vwap][:above] ? :bullish : :bearish
      adx = results[:adx][:trending] ? results[:adx][:strength] : :neutral
      st = results[:supertrend][:trend]

      bullish_count = [ema, vwap, st].count(:bullish)
      bearish_count = [ema, vwap, st].count(:bearish)

      if bullish_count >= 2 && adx != :weak
        :bullish
      elsif bearish_count >= 2 && adx != :weak
        :bearish
      else
        :neutral
      end
    end

    def momentum_score(results)
      rsi_score = case results[:rsi][:value]&.dig(:condition)
                  when :oversold then 0
                  when :bullish then 70
                  when :overbought then 100
                  else 50
                  end

      macd_score = results[:macd][:value]&.dig(:bullish_crossover) ? 80 : 40
      roc_score = results[:roc][:value]&.dig(:value).to_f > 0 ? 70 : 30

      ((rsi_score + macd_score + roc_score) / 3).round(2)
    end

    def volatility_regime(results)
      atr = results[:atr][:value]
      bb = results[:bollinger][:value]

      if atr&.dig(:expansion) || bb&.dig(:squeeze) == false
        :expanding
      elsif atr&.dig(:contraction) || bb&.dig(:squeeze)
        :contracting
      else
        :normal
      end
    end
  end
  ```

### Task 6.2: Create `FeatureStore`
- **File:** `app/lib/feature_store.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class FeatureStore
    def self.write(instrument_id, features)
      RedisPool.general.with do |redis|
        redis.hset("features:#{instrument_id}", features.transform_values(&:to_json))
        redis.expire("features:#{instrument_id}", 300)
      end
    end

    def self.read(instrument_id, keys = nil)
      RedisPool.general.with do |redis|
        if keys
          values = redis.hmget("features:#{instrument_id}", *keys)
          keys.zip(values).to_h.transform_values { |v| v ? JSON.parse(v) : nil }
        else
          redis.hgetall("features:#{instrument_id}").transform_values { |v| JSON.parse(v) }
        end
      end
    end

    def self.fresh?(instrument_id, max_age: 60)
      RedisPool.general.with do |redis|
        redis.ttl("features:#{instrument_id}") > max_age
      end
    end
  end
  ```
- **Deliverable:** Centralized feature storage for engine consumption
- **Acceptance Criteria:** `FeatureStore.write(nifty.id, { ema: 22500 })` stores; `FeatureStore.read(nifty.id, [:ema])` retrieves

  ```ruby
    def initialize(instrument:, indicators:, timeframe: "5m")
      @instrument = instrument
      @indicators = indicators
      @timeframe = timeframe
    end

    def call
      # Use concurrent-ruby for parallel computation
      pool = Concurrent::ThreadPoolExecutor.new(
        min_threads: 1,
        max_threads: 4,
        max_queue: 10
      )

      futures = @indicators.map do |name|
        Concurrent::Future.execute(executor: pool) do
          [name, Indicators::IndicatorRegistry.compute(name, instrument: @instrument, timeframe: @timeframe).value]
        end
      end

      results = futures.map { |f| f.value rescue nil }.compact.to_h

      pool.shutdown
      pool.wait_for_termination(5)

      success(results)
    end
  end
  ```
- **Deliverable:** Parallel indicator computation
- **Acceptance Criteria:** 4 indicators computed in parallel in <2ms total
