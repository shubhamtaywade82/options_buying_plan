```ruby
module Alerting
  class AlertService < BaseService
    SEVERITIES = %w[info warning critical].freeze
    CATEGORIES = %w[trade risk system ai market].freeze

    def initialize
      @classifier = AlertClassifier.new
      @throttler = AlertThrottler.new
      @providers = []
      load_providers
    end

    def notify(alert_type:, message:, metadata: {}, severity: nil, category: nil)
      # Auto-classify if not provided
      severity ||= @classifier.classify(alert_type, message, metadata)
      category ||= @classifier.category(alert_type)

      # Build alert payload
      alert = build_alert(alert_type, message, metadata, severity, category)

      # Throttle check
      if @throttler.throttled?(alert)
        Rails.logger.info "[AlertService] Throttled: #{alert[:type]} (#{alert[:severity]})"
        return Result.success(:throttled)
      end

      # Record to history
      history = record_history(alert)

      # Deliver to providers
      deliver(alert, history)

      emit(EventTypes::ALERT_SENT, {
        type: alert_type,
        severity: severity,
        category: category,
        providers: @providers.map(&:name)
      })

      Result.success(alert)
    end

    private

    def build_alert(type, message, metadata, severity, category)
      {
        id: SecureRandom.uuid,
        type: type,
        message: message,
        severity: severity,
        category: category,
        metadata: metadata,
        timestamp: Time.current.iso8601,
        environment: Rails.env
      }
    end

    def record_history(alert)
      AlertHistory.create!(
        severity: alert[:severity],
        category: alert[:category],
        message: alert[:message],
        metadata: alert[:metadata]
      )
    end

    def deliver(alert, history)
      @providers.each do |provider|
        next unless provider.supports?(alert[:severity])

        AlertDeliveryJob.perform_later(
          provider_name: provider.name,
          alert: alert,
          history_id: history.id
        )
      end
    end

    def load_providers
      if AppConfig.config.alerts&.telegram&.enabled
        @providers << Alerting::Providers::TelegramAlertProvider.new
      end

      if AppConfig.config.alerts&.email&.enabled
        @providers << Alerting::Providers::EmailAlertProvider.new
      end
    end
  end
end
```

```ruby
module Alerting
  class AlertClassifier
    SEVERITY_MAP = {
      # Trade alerts
      trade_entry: "info",
      trade_exit: "info",
      trade_exit_loss: "warning",
      trade_exit_major_loss: "critical",
      trade_partial_exit: "info",

      # Risk alerts
      daily_loss_warning: "warning",
      daily_loss_limit: "critical",
      margin_call_warning: "critical",
      concentration_risk: "warning",
      var_breach: "critical",

      # System alerts
      broker_disconnected: "critical",
      websocket_disconnected: "warning",
      data_stale: "warning",
      database_error: "critical",
      job_failure: "warning",

      # AI alerts
      ai_insight: "info",
      ai_validation_reject: "info",
      ai_low_confidence: "warning",

      # Market alerts
      market_open: "info",
      market_close: "info",
      volatility_spike: "warning"
    }.freeze

    CATEGORY_MAP = {
      trade_entry: "trade",
      trade_exit: "trade",
      trade_exit_loss: "trade",
      trade_exit_major_loss: "trade",
      trade_partial_exit: "trade",
      daily_loss_warning: "risk",
      daily_loss_limit: "risk",
      margin_call_warning: "risk",
      concentration_risk: "risk",
      var_breach: "risk",
      broker_disconnected: "system",
      websocket_disconnected: "system",
      data_stale: "system",
      database_error: "system",
      job_failure: "system",
      ai_insight: "ai",
      ai_validation_reject: "ai",
      ai_low_confidence: "ai",
      market_open: "market",
      market_close: "market",
      volatility_spike: "market"
    }.freeze

    def classify(type, message, metadata = {})
      mapped = SEVERITY_MAP[type.to_sym]
      return mapped if mapped

      # Heuristic classification
      return "critical" if message.downcase.match?(/critical|emergency|stop|breach|failure|disconnected/)
      return "warning" if message.downcase.match?(/warning|caution|stale|degraded|loss/)
      "info"
    end

    def category(type)
      CATEGORY_MAP[type.to_sym] || "system"
    end

    def severity_priority(severity)
      case severity.to_s
      when "critical" then 3
      when "warning" then 2
      when "info" then 1
      else 0
      end
    end
  end
end
```

```ruby
module Alerting
  class AlertThrottler
    PREFIX = "alert_throttle".freeze

    # Throttle rules: max count per window per key
    RULES = {
      "info"    => { count: 10, window: 60 },      # 10 per minute
      "warning" => { count: 5,  window: 60 },      # 5 per minute
      "critical" => { count: 20, window: 60 },      # 20 per minute (never suppress critical)
      "trade_entry" => { count: 5, window: 60 },
      "trade_exit" => { count: 5, window: 60 },
      "ai_insight" => { count: 3, window: 300 }   # 3 per 5 minutes
    }.freeze

    def initialize
      @redis = RedisPool.general
    end

    def throttled?(alert)
      return false if alert[:severity] == "critical" && !RULES.key?(alert[:type].to_s)

      key = throttle_key(alert)
      rule = RULES[alert[:type].to_s] || RULES[alert[:severity].to_s] || { count: 10, window: 60 }

      current = @redis.incr(key)
      @redis.expire(key, rule[:window]) if current == 1

      current > rule[:count]
    end

    def throttle_stats(alert_type = nil)
      keys = @redis.keys("#{PREFIX}:*")
      keys.to_h do |k|
        type = k.sub("#{PREFIX}:", "")
        [type, { current: @redis.get(k).to_i, ttl: @redis.ttl(k) }]
      end
    end

    private

    def throttle_key(alert)
      type = alert[:type]
      severity = alert[:severity]
      "#{PREFIX}:#{type}:#{severity}:#{Time.current.strftime('%Y%m%d%H%M')}"
    end
  end
end
```

```ruby
module Alerting
  module Providers
    class TelegramAlertProvider < BaseProvider
      attr_reader :name

      def initialize
        @name = "telegram"
        @bot_token = AppConfig.config.alerts.telegram.bot_token
        @chat_id = AppConfig.config.alerts.telegram.chat_id
        @client = Telegram::Bot::Client.new(@bot_token) if @bot_token.present?
      end

      def supports?(severity)
        # Telegram gets everything except info-only (unless configured)
        %w[warning critical].include?(severity) || AppConfig.config.alerts.telegram.receive_info
      end

      def deliver(alert)
        return Result.failure(:not_configured) unless @client && @chat_id

        text = format_message(alert)

        @client.api.send_message(
          chat_id: @chat_id,
          text: text,
          parse_mode: "HTML",
          disable_web_page_preview: true
        )

        Result.success(:delivered)
      rescue Telegram::Bot::Exceptions::ResponseError => e
        Result.failure(telegram_error: e.message)
      rescue => e
        Result.failure(error: e.message)
      end

      private

      def format_message(alert)
        emoji = severity_emoji(alert[:severity])
        category = alert[:category]&.upcase

        <<~MSG
          #{emoji} <b>#{category} ALERT</b> #{emoji}

          <b>#{alert[:type]}</b>
          #{escape_html(alert[:message])}

          <pre>#{format_metadata(alert[:metadata])}</pre>

          #{escape_html(alert[:timestamp])}
        MSG
      end

      def severity_emoji(severity)
        case severity.to_s
        when "critical" then "🔴"
        when "warning" then "🟡"
        else "🟢"
        end
      end

      def escape_html(text)
        return "" if text.nil?
        text.gsub(/[<>&]/, "<" => "&lt;", ">" => "&gt;", "&" => "&amp;")
      end

      def format_metadata(metadata)
        metadata.map { |k, v| "#{k}: #{v}" }.join("\n")
      end
    end
  end
end
```

```ruby
class AlertDeliveryJob < JobBase
  queue_as :alerts

  retry_on StandardError, wait: 5.seconds, attempts: 3

  def perform(provider_name:, alert:, history_id:)
    provider = find_provider(provider_name)
    return unless provider

    result = provider.deliver(alert)

    # Update history with delivery status
    history = AlertHistory.find_by(id: history_id)
    return unless history

    if result.success?
      history.update!(
        metadata: history.metadata.merge(
          delivery: {
            provider: provider_name,
            status: "delivered",
            delivered_at: Time.current.iso8601
          }
        )
      )
    else
      history.update!(
        metadata: history.metadata.merge(
          delivery: {
            provider: provider_name,
            status: "failed",
            error: result.error.to_h,
            attempted_at: Time.current.iso8601
          }
        )
      )
      raise "Delivery failed: #{result.error.to_h}" # Trigger retry
    end
  end

  private

  def find_provider(name)
    case name
    when "telegram" then Alerting::Providers::TelegramAlertProvider.new
    when "email" then Alerting::Providers::EmailAlertProvider.new
    else nil
    end
  end
end
```

```ruby
module Risk
  class RiskMonitor
    def self.check_limits
      today = Date.current.beginning_of_day
      realized = Trade.where("exit_time > ?", today).sum { |t| [t.gross_pnl.to_f, 0].min }.abs
      limit = AppConfig.config.risk.daily_loss_limit

      if realized >= limit
        Alerting::Alerts::RiskAlert.daily_limit_breached(realized, limit)
        # Trigger circuit breaker
        CircuitBreaker.trigger!(:daily_loss_limit)
      elsif realized >= limit * 0.8
        Alerting::Alerts::RiskAlert.daily_limit_warning(realized, limit)
      end

      # Check margin
      positions = Trade.where(exit_price: nil)
      calculator = PortfolioRiskCalculator.new(trades: positions)
      margin = calculator.margin_utilization
      if margin > 80
        Alerting::Alerts::RiskAlert.margin_warning(margin, 100)
      end

      # Check concentration
      concentration = calculator.concentration_risk
      concentration.each do |instrument_id, data|
        if data[:percent_of_portfolio] > 50
          instrument = Instrument.find(instrument_id)
          Alerting::Alerts::RiskAlert.concentration_risk(
            instrument.trading_symbol,
            data[:percent_of_portfolio]
          )
        end
      end
    end
  end
end
```

```ruby
module Api
  module V1
    class AlertsController < ApplicationController
      def index
        alerts = AlertHistory.recent
                             .by_category(params[:category])
                             .where(severity: params[:severity]) if params[:severity]

        render json: {
          alerts: alerts.limit(params[:limit]&.to_i || 20).map { |a| alert_payload(a) },
          unacknowledged: AlertHistory.unacknowledged.count,
          by_severity: AlertHistory.where("created_at > ?", 24.hours.ago).group(:severity).count,
          throttle_stats: Alerting::AlertThrottler.new.throttle_stats
        }
      end

      def acknowledge
        alert = AlertHistory.find(params[:id])
        alert.acknowledge!(by: params[:acknowledged_by] || "api")

        render json: { id: alert.id, acknowledged: true, acknowledged_at: alert.acknowledged_at }
      end

      def create_test
        # Admin-only endpoint for testing alert delivery
        result = AlertService.new.notify(
          alert_type: params[:type]&.to_sym || :info,
          message: params[:message] || "Test alert from API",
          severity: params[:severity],
          metadata: params[:metadata] || {}
        )

        if result.success?
          render json: { sent: true, alert: result.value }
        else
          render json: { error: result.error.to_h }, status: :unprocessable_entity
        end
      end

          severity: alert.severity,
          category: alert.category,
          message: alert.message,
          metadata: alert.metadata,
          acknowledged: alert.acknowledged,

    end
```

# Alerting & Notifications

## Architecture

Multi-channel alerting with severity classification, throttling, and full audit trail.

## Channels

| Channel | Severities | Use Case |
|---------|------------|----------|
| Telegram | warning, critical (+info if configured) | Real-time mobile alerts |
| Email | warning, critical | Detailed post-mortem, critical failures |

## Severity Levels

| Level | Emoji | Response Time | Examples |
|-------|-------|---------------|----------|
| info | 🟢 | None | Trade entry, market open |
| warning | 🟡 | 5 min | Daily loss 80%, stale data |
| critical | 🔴 | Immediate | Daily loss limit, broker down |

## Alert Types

### Trade
- `trade_entry` — New position opened
- `trade_exit` — Position closed (profit)
- `trade_exit_loss` — Position closed (loss)
- `trade_exit_major_loss` — Position closed (>₹5k loss)

### Risk
- `daily_loss_warning` — 80% of daily limit used
- `daily_loss_limit` — Limit breached, trading halted
- `margin_call_warning` — Margin utilization >80%
- `concentration_risk` — >50% in single instrument

### System
- `broker_disconnected` — DhanHQ API unreachable
- `websocket_disconnected` — Market data stream down
- `data_stale` — No ticks for >60s
- `job_failure` — Background job failed

### AI
- `ai_insight` — Market narrative update
- `ai_validation_reject` — Setup rejected by AI
- `ai_low_confidence` — Model confidence <50%

## Throttling

| Alert Type | Max | Window |
|------------|-----|--------|
| info | 10 | 60s |
| warning | 5 | 60s |
| critical | 20 | 60s |
| ai_insight | 3 | 300s |

Critical alerts are never suppressed.

## API

```bash
# List recent alerts
GET /api/v1/alerts?category=risk&limit=50

# Acknowledge alert
POST /api/v1/alerts/123/acknowledge

# Test alert (admin)
POST /api/v1/alerts/create_test
{ "type": "trade_entry", "message": "Test", "severity": "info" }
```

```ruby
Rails.application.config.after_initialize do
  if AppConfig.config.alerts&.enabled
    # Verify Telegram config if enabled
    if AppConfig.config.alerts.telegram&.enabled
      token = AppConfig.config.alerts.telegram.bot_token
      chat = AppConfig.config.alerts.telegram.chat_id

      if token.blank? || chat.blank?
        Rails.logger.warn "[Alerting] Telegram enabled but missing bot_token or chat_id"
      else
        Rails.logger.info "[Alerting] Telegram provider ready"
      end
    end

    # Verify email config if enabled
    if AppConfig.config.alerts.email&.enabled
      recipients = AppConfig.config.alerts.email.recipients
      if recipients.blank?
        Rails.logger.warn "[Alerting] Email enabled but no recipients configured"
      else
        Rails.logger.info "[Alerting] Email provider ready (#{recipients.length} recipients)"
      end
    end

    Rails.logger.info "[Alerting] Alert service initialized with #{Alerting::AlertService::SEVERITIES.join(', ')} severities"
  else
    Rails.logger.info "[Alerting] Alerts disabled in configuration"
  end
end
```
