### Task 3.3: Create `TickRepository`
- **File:** `app/repositories/tick_repository.rb`
  ```ruby
  class TickRepository
    def self.insert_batch(ticks)
      # Use activerecord-import for bulk insert
      MarketTick.import(ticks, validate: false, batch_size: 1000)
    end

    def self.for_range(instrument:, from:, to:)
      MarketTick.for_instrument(instrument).between(from, to).order(timestamp: :asc)
    end

    def self.latest(instrument:, limit: 1)
      MarketTick.for_instrument(instrument).order(timestamp: :desc).limit(limit)
    end

    def self.gap_exists?(instrument:, from:, to:, expected_interval: 1.second)
      # Detect missing ticks in range
      ticks = for_range(instrument: instrument, from: from, to: to).pluck(:timestamp)
      return true if ticks.empty?

      ticks.each_cons(2) do |a, b|
        return true if (b - a) > expected_interval * 2
      end
      false
    end
  end
  ```
- **Deliverable:** Optimized tick data access

  ```ruby
      candles = ohlcv(instrument: instrument, timeframe: timeframe, from: from, to: to)
      total_value = candles.sum { |c| c.vwap.to_f * c.volume.to_f }
  ```

### Task 5.2: Create `OptionChainRepository`

  ```ruby
  class OptionChainRepository
    def self.latest_snapshot(underlying:, time: nil)
      time ||= Time.current
      OptionChainSnapshot.for_underlying(underlying)
                          .where("snapshot_time <= ?", time)
                          .order(snapshot_time: :desc)
                          .first
    end

    def self.strikes_at_time(underlying:, time:, range: 5)
      # Get ATM ± range strikes
      snapshot = latest_snapshot(underlying: underlying, time: time)
      return [] unless snapshot

      atm_strike = snapshot.strike
      OptionChainSnapshot.for_underlying(underlying)
                          .at_time(snapshot.snapshot_time)
                          .where(strike: (atm_strike - range * 50)..(atm_strike + range * 50))
                          .order(strike: :asc)
    end

    def self.iv_history(underlying:, strike:, option_type:, from:, to:)
      OptionChainSnapshot.for_underlying(underlying)
                          .for_strike(strike)
                          .where(option_type: option_type)
                          .where(snapshot_time: from..to)
                          .order(snapshot_time: :asc)
                          .pluck(:snapshot_time, :iv)
    end
  end
  ```

  ```ruby
  class Position < ApplicationRecord
    STATUSES = %w[open closed partial_exit].freeze
    SIDES = %w[long short].freeze

    belongs_to :instrument
    belongs_to :user
    belongs_to :order
    has_one :trade, dependent: :destroy

    validates :position_id, presence: true, uniqueness: true
    validates :side, inclusion: { in: SIDES }
    validates :status, inclusion: { in: STATUSES }
    validates :quantity, numericality: { greater_than: 0 }
    validates :entry_price, numericality: { greater_than: 0 }

    scope :open, -> { where(status: "open") }
    scope :closed, -> { where(status: "closed") }

    def pnl_percent
      return 0 if entry_price.nil? || entry_price.zero? || current_price.nil?
      side == "long" ? ((current_price - entry_price) / entry_price * 100) : ((entry_price - current_price) / entry_price * 100)
    end

    def update_mfe_mae!
      return unless current_price.present?

      favorable = side == "long" ? current_price - entry_price : entry_price - current_price
      adverse = side == "long" ? entry_price - current_price : current_price - entry_price

      self.mfe = [mfe.to_f, favorable].max
      self.mae = [mae.to_f, adverse].max
      save!
    end
  end
  ```
