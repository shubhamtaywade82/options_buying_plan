```ruby
class VectorEmbedding < ApplicationRecord
  belongs_to :embeddable, polymorphic: true

  validates :embedding, presence: true
  validates :embedding_type, presence: true

  scope :for_type, ->(type) { where(embedding_type: type) }
  scope :recent, -> { order(created_at: :desc) }

  # Cosine similarity search using pgvector <=> (distance) operator
  # Lower distance = more similar. Cosine similarity = 1 - distance
  def self.nearest_neighbors(query_embedding, limit: 5, threshold: 0.8)
    vector_literal = query_embedding.is_a?(Array) ? "[#{query_embedding.join(',')}]" : query_embedding.to_s

    select(Arel.sql("*, 1 - (embedding <=> '#{vector_literal}'::vector) as cosine_similarity"))
      .where("embedding <=> ?::vector < ?", vector_literal, 1 - threshold)
      .order(Arel.sql("embedding <=> '#{vector_literal}'::vector"))
      .limit(limit)
  end

  def cosine_similarity
    attributes["cosine_similarity"] || 0.0
  end
end

module AiGateway
  class EmbeddingService
    DEFAULT_MODEL = "nomic-embed-text".freeze
    DEFAULT_DIMENSION = 768
    CACHE_TTL = 1.day

    def initialize(model: nil, gateway: nil)
      @model = model || AppConfig.config.ai.embedding_model || DEFAULT_MODEL
      @dimension = AppConfig.config.ai.embedding_dimension || DEFAULT_DIMENSION
      @gateway = gateway || AiGateway::AIGateway.new
    end

    def embed(text)
      return nil if text.blank?

      cache_key = "embedding:#{Digest::MD5.hexdigest(text)}"
      cached = CacheStore.fetch(cache_key)
      return cached if cached

      provider = @gateway.instance_variable_get(:@provider_pool)&.best_available
      return nil unless provider

      result = provider.embed(text: text, model: @model)
      return nil if result.failure?

      vector = result.value
      CacheStore.write(cache_key, vector, ttl: CACHE_TTL)

      vector
    end

    def embed_batch(texts)
      texts.compact_blank.map { |t| embed(t) }.compact
    end

    def dimension
      @dimension
    end
  end
end

    end

    def embed_trade(trade)
      text = build_text(trade)
      vector = @embedding_service.embed(text)

      VectorEmbedding.create!(
        embeddable: trade,
        embedding_type: "trade",
        embedding: vector,
        source_text: text,

    def embed_market_context(context)

      vector = @embedding_service.embed(text)

        embeddable_type: "MarketContext",

        embedding_type: "market_context",

        source_text: text,
        metadata: context.slice(:instrument, :timeframe, :timestamp)

module AiGateway
  class VectorStore
    def initialize(dimension: 768)
      @dimension = dimension
    end

    def store(record:, vector:, type:, metadata: {})
      VectorEmbedding.create!(
        embeddable: record,
        embedding_type: type,
        embedding: vector,
        metadata: metadata
      )
    end

    def search(vector:, type: nil, limit: 5, threshold: 0.8)
      scope = VectorEmbedding.all
      scope = scope.for_type(type) if type

      scope.nearest_neighbors(vector, limit: limit, threshold: threshold)
    end

    def delete_for(record)
      VectorEmbedding.where(embeddable: record).destroy_all
    end

    def count(type: nil)
      scope = VectorEmbedding.all
      scope = scope.for_type(type) if type
      scope.count
    end

    def stats
      {
        total_embeddings: count,
        trade_embeddings: count(type: "trade"),
        market_context_embeddings: count(type: "market_context"),
        oldest: VectorEmbedding.minimum(:created_at),
        newest: VectorEmbedding.maximum(:created_at)
      }
    end
  end
end

module AiGateway
  class SimilarTradeFinder
    DEFAULT_LIMIT = 5
    DEFAULT_THRESHOLD = 0.75

    def initialize(vector_store: nil, embedder: nil)
      @vector_store = vector_store || VectorStore.new
      @embedder = embedder || TradeEmbedder.new
    end

    def find_similar_to_current_context(context, limit: DEFAULT_LIMIT)
      text = context_to_text(context)
      vector = @embedder.instance_variable_get(:@embedding_service).embed(text)
      return [] unless vector

      results = @vector_store.search(
        vector: vector,
        type: "trade",
        limit: limit * 2, # Fetch extra for filtering
        threshold: DEFAULT_THRESHOLD
      )

      filter_and_rank(results, context).first(limit)
    end

    def find_similar_to_trade(trade, limit: DEFAULT_LIMIT)
      existing = VectorEmbedding.find_by(
        embeddable: trade,
        embedding_type: "trade"
      )
      return [] unless existing

      @vector_store.search(
        vector: existing.embedding,
        type: "trade",
        limit: limit,
        threshold: 0.8
      )
    end

    def find_by_strategy_and_regime(strategy:, regime_type:, limit: DEFAULT_LIMIT)
      # Semantic search constrained by metadata
      # First get candidate IDs from metadata, then re-rank by vector similarity
      # Simplified: filter in Ruby for now; can optimize with pgvector + metadata hybrid later
      candidates = VectorEmbedding
        .for_type("trade")
        .where("metadata->>'regime_type' = ?", regime_type.to_s)
        .where("metadata->>'strategy' = ?", strategy.to_s)
        .recent
        .limit(100)

      candidates.map do |c|
        {
          embedding: c,
          metadata: c.metadata
        }
      end.first(limit)
    end

    private

    def context_to_text(context)
      <<~TEXT
        #{context[:instrument]} #{context[:regime]&.dig(:type)} #{context[:regime]&.dig(:direction)}
        Structure #{context[:structure]&.dig(:type)}
        Momentum #{context[:momentum]&.dig(:direction)} #{context[:momentum]&.dig(:score)}
        Liquidity #{context[:liquidity]&.dig(:score)}
      TEXT
    end

    def filter_and_rank(results, context)
      # Prefer same direction, same instrument
      results.sort_by do |r|
        meta = r.metadata || {}
        score = r.cosine_similarity

        # Boost same direction
        score += 0.05 if meta["direction"].to_s == context[:direction].to_s
        # Boost same instrument
        score += 0.05 if meta["instrument_id"].to_s == context[:instrument_id].to_s

        -score # Lower is better for sort_by (descending similarity)
      end
    end
  end
end

```ruby
module AiGateway
  class PatternMatcher
    def initialize(similar_finder: nil)
      @finder = similar_finder || SimilarTradeFinder.new
    end

    def match_current_context(context)
      similar = @finder.find_similar_to_current_context(context, limit: 10)
      return nil if similar.empty?

      aggregate_patterns(similar)
    end

    def match_trade(trade)
      similar = @finder.find_similar_to_trade(trade, limit: 10)
      return nil if similar.empty?

      aggregate_patterns(similar)
    end

    def regime_pattern_summary(strategy:, regime_type:)
      similar = @finder.find_by_strategy_and_regime(
        strategy: strategy,
        regime_type: regime_type,
        limit: 20
      )
      return nil if similar.empty?

      aggregate_patterns(similar.map { |s| s[:embedding] })
    end

    private

    def aggregate_patterns(embeddings)
      wins = embeddings.count { |e| e.metadata&.dig("win") == true }
      total = embeddings.length
      pnls = embeddings.map { |e| e.metadata&.dig("pnl").to_f }

      {
        pattern_match_score: (wins.to_f / total * 100).round(2),
        historical_win_rate: (wins.to_f / total * 100).round(2),
        avg_pnl: (pnls.sum / total).round(2),
        max_pnl: pnls.max,
        min_pnl: pnls.min,
        sample_size: total,
        similar_trades: embeddings.map do |e|
          {
            id: e.embeddable_id,
            pnl: e.metadata&.dig("pnl"),
            win: e.metadata&.dig("win"),
            strategy: e.metadata&.dig("strategy")
          }
        end,
        confidence: confidence_from_sample_size(total),
        direction_bias: direction_bias(embeddings)
      }
    end

    def confidence_from_sample_size(n)
      return 30 if n < 3
      return 60 if n < 5
      return 80 if n < 10
      95
    end

    def direction_bias(embeddings)
      longs = embeddings.count { |e| e.metadata&.dig("direction") == "long" }
      shorts = embeddings.count { |e| e.metadata&.dig("direction") == "short" }

      if longs > shorts * 1.5
        :long_bias
      elsif shorts > longs * 1.5
        :short_bias
      else
        :neutral
      end
    end
  end
end

module AiGateway
  class AIContextEnricher
    def initialize(pattern_matcher: nil)
      @matcher = pattern_matcher || PatternMatcher.new
    end

    def enrich(context)
      pattern = @matcher.match_current_context(context)
      return context unless pattern && pattern[:sample_size] >= 3

      context.merge(
        historical_context: {
          pattern_match_score: pattern[:pattern_match_score],
          historical_win_rate: pattern[:historical_win_rate],
          avg_pnl: pattern[:avg_pnl],
          max_pnl: pattern[:max_pnl],
          min_pnl: pattern[:min_pnl],
          sample_size: pattern[:sample_size],
          direction_bias: pattern[:direction_bias],
          similar_trades_summary: pattern[:similar_trades]
        },
        ai_prompt_addendum: build_prompt_addendum(pattern)
      )
    end

    def enrich_for_agent(agent_type, context)
      enriched = enrich(context)

      case agent_type
      when :setup_validator
        enriched.merge(setup_validator_addendum: enriched[:ai_prompt_addendum])
      when :market_analyst
        enriched.merge(analyst_addendum: build_analyst_addendum(enriched[:historical_context]))
      when :strategy_researcher
        enriched.merge(researcher_context: enriched[:historical_context])
      else
        enriched
      end
    end

    private

    def build_prompt_addendum(pattern)
      return "" if pattern[:sample_size] < 3

      <<~PROMPT
        HISTORICAL PATTERN CONTEXT:
        Found #{pattern[:sample_size]} similar past trades in vector memory.
        Historical win rate: #{pattern[:historical_win_rate]}%
        Average P&L: ₹#{pattern[:avg_pnl].round(2)}
        Direction bias: #{pattern[:direction_bias]}
        Pattern confidence: #{pattern[:confidence]}%
      PROMPT
    end

    def build_analyst_addendum(historical)
      return "" unless historical

      <<~PROMPT
        HISTORICAL SIMILARITY:
        #{historical[:sample_size]} similar contexts found.
        Win rate in similar setups: #{historical[:historical_win_rate]}%.
        Average outcome: ₹#{historical[:avg_pnl].round(2)}.
      PROMPT
    end
  end
end

class TradeEmbeddingJob < JobBase
  queue_as :embeddings

  def perform(trade_id)
    trade = Trade.find(trade_id)
    return unless trade.exit_price.present? # Only closed trades

    # Delete old embedding if exists (for updates)
    VectorEmbedding.where(embeddable: trade, embedding_type: "trade").destroy_all

    embedder = AiGateway::TradeEmbedder.new
    embedding = embedder.embed_trade(trade)

    if embedding
      Rails.logger.info "[TradeEmbeddingJob] Embedded trade #{trade_id} (vector_id: #{embedding.id})"
    else
      Rails.logger.warn "[TradeEmbeddingJob] Failed to embed trade #{trade_id}"
    end
  rescue => e
    Rails.logger.error "[TradeEmbeddingJob] Error for trade #{trade_id}: #{e.message}"
    raise # Let Solid Queue retry
  end
end

module Api
  module V1
    class VectorMemoryController < ApplicationController
      def similar_trades
        context = {
          instrument: params[:instrument],
          direction: params[:direction],
          regime: params[:regime]&.symbolize_keys,
          structure: params[:structure]&.symbolize_keys,
          momentum: params[:momentum]&.symbolize_keys,
          liquidity: params[:liquidity]&.symbolize_keys,
          option_flow: params[:option_flow]&.symbolize_keys,
          instrument_id: params[:instrument_id]
        }.compact

        finder = AiGateway::SimilarTradeFinder.new
        results = finder.find_similar_to_current_context(context, limit: params[:limit]&.to_i || 5)

        render json: {
          query_context: context,
          count: results.length,
          results: results.map do |r|
            {
              trade_id: r.embeddable_id,
              similarity: r.cosine_similarity.round(4),
              metadata: r.metadata,
              source_text_preview: r.source_text&.truncate(200)
            }
          end
        }
      end

          regime: params[:regime]&.symbolize_keys,
          structure: params[:structure]&.symbolize_keys,
          momentum: params[:momentum]&.symbolize_keys,
          liquidity: params[:liquidity]&.symbolize_keys

          render json: pattern
        else
          render json: { message: "No similar patterns found" }, status: :not_found
        end
      end

        store = AiGateway::VectorStore.new
        render json: store.stats

          enriched_keys: enriched.keys,

          prompt_addendum: enriched[:ai_prompt_addendum]

`# docs/vector_memory.md`
```markdown
# AI Memory & Vector Store

## Architecture

Uses PostgreSQL `pgvector` extension for vector storage and similarity search.
Embeddings are generated via Ollama local models (default: `nomic-embed-text`, 768 dimensions).

## Pipeline

1. **Trade Close** → `TradeEmbeddingJob` queues
2. **Embed** → `TradeEmbedder` converts trade features to text, gets vector via `EmbeddingService`
3. **Store** → `VectorStore` persists in `vector_embeddings` table with metadata
4. **Query** → `SimilarTradeFinder` performs cosine similarity search
5. **Match** → `PatternMatcher` aggregates statistics from neighbors
6. **Enrich** → `AIContextEnricher` injects historical context into AI prompts

## Schema

| Column | Type | Description |
|--------|------|-------------|
| embeddable | polymorphic | Link to Trade or other models |
| embedding_type | string | "trade", "market_context" |
| embedding | vector(768) | pgvector embedding |
| source_text | text | Human-readable source |
| metadata | jsonb | Structured trade features |

## Index

- `ivfflat` index on `embedding` using `vector_cosine_ops` for ANN search
- Composite index on `(embeddable_type, embeddable_id)`
- Index on `(embedding_type, created_at)`

## API

```bash
# Find similar trades to current context
POST /api/v1/vector_memory/similar_trades
{
  "instrument": "NIFTY",
  "direction": "long",
  "regime": { "type": "trending_up", "direction": "bullish" },
  "limit": 5
}

# Get pattern match summary
POST /api/v1/vector_memory/pattern_match
{
  "instrument": "NIFTY",
  "direction": "long",
  "regime": { "type": "trending_up" }
}

# Enrich AI context
POST /api/v1/vector_memory/enrich_context
{
  "context": { "instrument": "NIFTY", "direction": "long" }
}
```

## Configuration

```yaml
# config/ai.yml
embedding_model: "nomic-embed-text"
embedding_dimension: 768
similarity_threshold: 0.75
```

## Safety

- Embeddings are only generated for **closed trades** (have exit_price)
- `TradeEmbeddingJob` deletes old embeddings before creating new ones (idempotent)
- Failed embeddings are retried by Solid Queue
- Cache embeddings by MD5 hash to avoid redundant LLM calls

```ruby
def call(agent_type, context: {})
  # Enrich context with vector memory before routing
  enricher = AiGateway::AIContextEnricher.new
  enriched_context = enricher.enrich_for_agent(agent_type, context.symbolize_keys)

  agent_class = AGENT_REGISTRY[agent_type.to_sym]
  return Result.failure(:unknown_agent) unless agent_class

  agent = get_or_create(agent_class)
  result = agent.call(context: enriched_context)

  emit(EventTypes::AI_AGENT_ORCHESTRATED, {
    agent: agent_type,
    success: result.success?,
    confidence: result.success? ? result.value[:ai_confidence] : nil
  })

  result
end
