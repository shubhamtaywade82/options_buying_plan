### Day 1 — ORB Strategy (Opening Range Breakout)

**Task 1.1: Create `Strategies::ORBStrategy`**

```ruby
# app/strategies/orb_strategy.rb
# frozen_string_literal: true

module Strategies
  class ORBStrategy < BaseStrategy
    VERSION = "1.0.0".freeze

    version VERSION
    description "Opening Range Breakout: Trade the breakout of first 15/30 min range with volume confirmation"

    REQUIRED_FEATURES = %w[
      candles
      volume
      regime
    ].freeze

    timeframes %w[5m 15m]
    supported_instruments %w[NIFTY BANKNIFTY FINNIFTY]

    def initialize(instrument:, context: {})
      super
      @config = StrategyConfiguration.new(strategy_name)
      @opening_range_minutes = @config.get(:opening_range_minutes, 15)
      @breakout_threshold = @config.get(:breakout_threshold, 0.1)  # 0.1% minimum breakout
      @volume_multiplier = @config.get(:volume_multiplier, 1.5)
    end

    def should_enter?(context = @context)
      return false unless opening_range_complete?
      return false if breakout_already_triggered_today?

      range = calculate_opening_range
      return false unless range[:high] && range[:low]

      current = context[:spot_price] || latest_close
      volume_confirmed = volume_accelerating?

      # Long breakout: price breaks above opening range high
      if current > range[:high] * (1 + @breakout_threshold / 100) && volume_confirmed
        @direction = :long
        @entry_price = current
        @stop_loss = range[:low]
        @target = calculate_target(range, :long)
        return true
      end

      # Short breakout: price breaks below opening range low
      if current < range[:low] * (1 - @breakout_threshold / 100) && volume_confirmed
        @direction = :short
        @entry_price = current
        @stop_loss = range[:high]
        @target = calculate_target(range, :short)
        return true
      end

      false
    end

    def should_exit?(position, context = @context)
      current = context[:spot_price] || latest_close

      # Target hit
      if position.long? && current >= position.take_profit
        return true
      elsif position.short? && current <= position.take_profit
        return true
      end

      # Stop loss hit
      if position.long? && current <= position.stop_loss
        return true
      elsif position.short? && current >= position.stop_loss
        return true
      end

      # Time-based exit: max holding time exceeded
      if position.holding_time && position.holding_time > self.class.max_holding_time
        return true
      end

      # False breakout: price returns inside opening range
      range = calculate_opening_range
      if position.long? && current < range[:low]
        return true
      elsif position.short? && current > range[:high]
        return true
      end

      false
    end

    def confidence(context = @context)
      return 0 unless opening_range_complete?

      range = calculate_opening_range
      range_size = ((range[:high] - range[:low]) / range[:low] * 100).round(2)

      # Higher confidence for tighter ranges (cleaner breakout)
      range_score = case range_size
                    when 0..0.3 then 90
                    when 0.3..0.6 then 75
                    when 0.6..1.0 then 60
                    else 40
                    end

      # Volume confirmation boost
      volume_score = volume_accelerating? ? 10 : 0

      # Regime alignment
      regime_score = regime_aligned? ? 10 : 0

      [[range_score + volume_score + regime_score, 100].min, 0].max
    end

    def score(context = @context)
      return Result.success(signal: :neutral, confidence: 0, direction: nil) unless should_enter?(context)

      Result.success(
        signal: :enter,
        direction: @direction,
        confidence: confidence(context),
        entry_price: @entry_price,
        stop_loss: @stop_loss,
        target: @target,
        risk_reward: calculate_rr,
        setup: :opening_range_breakout,
        metadata: {
          opening_range_high: calculate_opening_range[:high],
          opening_range_low: calculate_opening_range[:low],
          range_size_percent: ((calculate_opening_range[:high] - calculate_opening_range[:low]) / calculate_opening_range[:low] * 100).round(2)
        }
      )
    end

    private

    def opening_range_complete?
      market_open = ExchangeCalendar.market_hours(exchange: "NSE")[:open]
      return false unless market_open

      range_end = market_open + @opening_range_minutes.minutes
      Time.current >= range_end
    end

    def breakout_already_triggered_today?
      # Check if we already took an ORB trade today
      TradeSetup.where(
        instrument: @instrument,
        setup_type: "orb",
        detected_at: Date.current.all_day
      ).exists?
    end

    def calculate_opening_range
      market_open = ExchangeCalendar.market_hours(exchange: "NSE")[:open]
      return {} unless market_open

      range_candles = @context[:candles].select do |c|
        c_time = c[:timestamp].is_a?(Time) ? c[:timestamp] : Time.parse(c[:timestamp].to_s)
        c_time >= market_open && c_time <= market_open + @opening_range_minutes.minutes
      end

      return {} if range_candles.empty?

      {
        high: range_candles.map { |c| c[:high].to_d }.max,
        low: range_candles.map { |c| c[:low].to_d }.min,
        open: range_candles.first[:open].to_d,
        close: range_candles.last[:close].to_d
      }
    end

    def volume_accelerating?
      return false unless @context[:volume]

      current = @context[:volume][:current].to_i
      average = @context[:volume][:average].to_i

      average > 0 && current > average * @volume_multiplier
    end

    def regime_aligned?
      return true unless @context[:regime]

      if @direction == :long
        @context[:regime][:direction] == :bullish || @context[:regime][:type] == :trending_up
      else
        @context[:regime][:direction] == :bearish || @context[:regime][:type] == :trending_down
      end
    end

    def calculate_target(range, direction)
      range_size = range[:high] - range[:low]

      if direction == :long
        @entry_price + range_size * 2  # 1:2 risk-reward minimum
      else
        @entry_price - range_size * 2
      end
    end

    def calculate_rr
      return 0 unless @entry_price && @stop_loss && @target

      risk = (@entry_price - @stop_loss).abs
      reward = (@target - @entry_price).abs
      return 0 if risk.zero?

      (reward / risk).round(2)
    end

    def latest_close
      @context[:candles].last&.dig(:close).to_d
    end
  end
end
```

```ruby
# app/strategies/trend_following_strategy.rb
# frozen_string_literal: true

module Strategies
  class TrendFollowingStrategy < BaseStrategy
    VERSION = "1.0.0".freeze

    version VERSION
    description "Trend Following: Enter on EMA alignment with ADX confirmation, hold until trend reverses"

    REQUIRED_FEATURES = %w[
      candles
      regime
      momentum
      volume
    ].freeze

    timeframes %w[5m 15m]
    supported_instruments %w[NIFTY BANKNIFTY SENSEX FINNIFTY]

    def initialize(instrument:, context: {})
      super
      @config = StrategyConfiguration.new(strategy_name)
      @ema_fast = @config.get(:ema_fast, 9)
      @ema_slow = @config.get(:ema_slow, 21)
      @adx_threshold = @config.get(:adx_threshold, 25)
      @trend_strength_min = @config.get(:trend_strength_min, 20)
    end

    def should_enter?(context = @context)
      return false unless ema_aligned?
      return false unless adx_strong?
      return false unless volume_confirming?

      @direction = context[:regime][:direction] == :bullish ? :long : :short

      @entry_price = context[:spot_price] || latest_close
      @stop_loss = calculate_stop_loss
      @target = calculate_target

      true
    end

    def should_exit?(position, context = @context)
      current = context[:spot_price] || latest_close

      # EMA cross against position
      if position.long? && ema_bearish_cross?
        return true
      elsif position.short? && ema_bullish_cross?
        return true
      end

      # ADX weakening (trend exhaustion)
      if adx_weakening? && position.unrealized_pnl.to_f > 0
        return true
      end

      # Target/Stop standard checks
      standard_exit_checks(position, current)
    end

    def confidence(context = @context)
      return 0 unless ema_aligned?

      adx = context[:momentum]&.dig(:adx) || 0
      ema_slope = ema_slope_strength

      # ADX score
      adx_score = case adx
                  when 0..20 then 30
                  when 20..30 then 60
                  when 30..40 then 80
                  else 95
                  end

      # EMA slope score
      slope_score = [[ema_slope * 10, 50].min, 0].max

      # Volume score
      vol_score = volume_confirming? ? 10 : 0

      [[adx_score + slope_score + vol_score, 100].min, 0].max
    end

    def score(context = @context)
      return Result.success(signal: :neutral, confidence: 0, direction: nil) unless should_enter?(context)

      Result.success(
        signal: :enter,
        direction: @direction,
        confidence: confidence(context),
        entry_price: @entry_price,
        stop_loss: @stop_loss,
        target: @target,
        risk_reward: calculate_rr,
        setup: :ema_trend_following,
        metadata: {
          ema_fast: @ema_fast,
          ema_slow: @ema_slow,
          adx: context[:momentum]&.dig(:adx),
          trend_strength: context[:momentum]&.dig(:trend_strength)
        }
      )
    end

    private

    def ema_aligned?
      return false unless @context[:candles]&.length.to_i > @ema_slow * 2

      closes = @context[:candles].map { |c| c[:close].to_d }
      fast_ema = Calculations::Statistics.ema(closes, @ema_fast)
      slow_ema = Calculations::Statistics.ema(closes, @ema_slow)

      @ema_fast_value = fast_ema
      @ema_slow_value = slow_ema

      if @context[:regime]&.dig(:direction) == :bullish
        fast_ema > slow_ema
      else
        fast_ema < slow_ema
      end
    end

    def ema_bearish_cross?
      return false unless @ema_fast_value && @ema_slow_value
      @ema_fast_value < @ema_slow_value
    end

    def ema_bullish_cross?
      return false unless @ema_fast_value && @ema_slow_value
      @ema_fast_value > @ema_slow_value
    end

    def adx_strong?
      adx = @context[:momentum]&.dig(:adx) || 0
      adx >= @adx_threshold
    end

    def adx_weakening?
      adx = @context[:momentum]&.dig(:adx) || 0
      adx < @adx_threshold * 0.8
    end

    def volume_confirming?
      return false unless @context[:volume]
      @context[:volume][:current].to_i > @context[:volume][:average].to_i * 1.2
    end

    def ema_slope_strength
      return 0 unless @context[:candles]&.length.to_i > 10
      closes = @context[:candles].last(10).map { |c| c[:close].to_d }
      Calculations::Statistics.slope(closes, 10) || 0
    end

    def calculate_stop_loss
      atr = @context[:momentum]&.dig(:atr, :current) || 20
      multiplier = @direction == :long ? -2 : 2
      @entry_price + (atr * multiplier)
    end

    def calculate_target
      atr = @context[:momentum]&.dig(:atr, :current) || 20
      multiplier = @direction == :long ? 3 : -3
      @entry_price + (atr * multiplier)
    end

    def standard_exit_checks(position, current)
      if position.long?
        return true if current <= position.stop_loss
        return true if current >= position.take_profit
      else
        return true if current >= position.stop_loss
        return true if current <= position.take_profit
      end
      false
    end

    def latest_close
      @context[:candles].last&.dig(:close).to_d
    end
  end
end
```

```ruby
# app/strategies/pullback_strategy.rb
# frozen_string_literal: true

module Strategies
  class PullbackStrategy < BaseStrategy
    VERSION = "1.0.0".freeze

    version VERSION
    description "Pullback: Enter on retracement to VWAP/EMA in established trend with structure confirmation"

    REQUIRED_FEATURES = %w[
      candles
      regime
      structure
      momentum
    ].freeze

    timeframes %w[5m 15m]
    supported_instruments %w[NIFTY BANKNIFTY]

    def initialize(instrument:, context: {})
      super
      @config = StrategyConfiguration.new(strategy_name)
      @vwap_deviation = @config.get(:vwap_deviation, 1.0)  # VWAP standard deviation band
      @structure_confirmation = @config.get(:structure_confirmation, true)
      @min_pullback_percent = @config.get(:min_pullback_percent, 0.3)
    end

    def should_enter?(context = @context)
      return false unless established_trend?
      return false unless pullback_active?
      return false unless @structure_confirmation && structure_supports? unless @structure_confirmation == false

      @direction = @context[:regime][:direction] == :bullish ? :long : :short
      @entry_price = context[:spot_price] || latest_close
      @stop_loss = calculate_stop_loss
      @target = calculate_target

      true
    end

    def should_exit?(position, context = @context)
      current = context[:spot_price] || latest_close

      # Trend reversal
      if position.long? && trend_reversed_to_bearish?
        return true
      elsif position.short? && trend_reversed_to_bullish?
        return true
      end

      # Pullback complete (price returns to trend)
      if position.long? && current >= vwap_value
        return true if position.unrealized_pnl.to_f > 0
      elsif position.short? && current <= vwap_value
        return true if position.unrealized_pnl.to_f > 0
      end

      standard_exit_checks(position, current)
    end

    def confidence(context = @context)
      return 0 unless established_trend?

      # Pullback depth score
      pullback_depth = calculate_pullback_depth
      depth_score = case pullback_depth
                    when 0..0.5 then 40   # Too shallow
                    when 0.5..1.5 then 80 # Ideal
                    when 1.5..3.0 then 60 # Deep but acceptable
                    else 30               # Too deep, trend may be broken
                    end

      # Structure score
      structure_score = structure_supports? ? 15 : 0

      # Momentum divergence score (pullback with weakening momentum = good)
      divergence_score = momentum_divergence? ? 10 : 0

      [[depth_score + structure_score + divergence_score, 100].min, 0].max
    end

    def score(context = @context)
      return Result.success(signal: :neutral, confidence: 0, direction: nil) unless should_enter?(context)

      Result.success(
        signal: :enter,
        direction: @direction,
        confidence: confidence(context),
        entry_price: @entry_price,
        stop_loss: @stop_loss,
        target: @target,
        risk_reward: calculate_rr,
        setup: :pullback_to_vwap,
        metadata: {
          vwap: vwap_value,
          pullback_depth: calculate_pullback_depth,
          structure_type: @context[:structure]&.dig(:type),
          trend_direction: @context[:regime]&.dig(:direction)
        }
      )
    end

    private

    def established_trend?
      return false unless @context[:regime]
      %i[trending_up trending_down].include?(@context[:regime][:type])
    end

    def pullback_active?
      return false unless @context[:candles]&.length.to_i > 5

      closes = @context[:candles].last(5).map { |c| c[:close].to_d }
      vwap = vwap_value

      return false if vwap.zero?

      if @context[:regime][:direction] == :bullish
        # In uptrend, price should be below VWAP (pulled back)
        closes.last < vwap && closes.last < closes.first
      else
        # In downtrend, price should be above VWAP (pulled back)
        closes.last > vwap && closes.last > closes.first
      end
    end

    def structure_supports?
      return true unless @context[:structure]

      structure = @context[:structure]
      if @direction == :long
        structure[:bullish] == true || %w[HL BOS].include?(structure[:type])
      else
        structure[:bearish] == true || %w[LH CHOCH].include?(structure[:type])
      end
    end

    def calculate_pullback_depth
      return 0 unless vwap_value > 0

      current = latest_close
      trend_extreme = @context[:regime][:direction] == :bullish ?
        @context[:candles].map { |c| c[:high].to_d }.max :
        @context[:candles].map { |c| c[:low].to_d }.min

      return 0 if trend_extreme.zero?

      ((trend_extreme - current).abs / trend_extreme * 100).round(2)
    end

    def momentum_divergence?
      return false unless @context[:momentum]
      @context[:momentum][:divergence]&.dig(:bullish_divergence) && @direction == :long ||
        @context[:momentum][:divergence]&.dig(:bearish_divergence) && @direction == :short
    end

    def vwap_value
      @vwap ||= begin
        candles = @context[:candles]
        return 0 if candles.empty?

        total_value = candles.sum { |c| c[:vwap].to_f * c[:volume].to_i }
        total_volume = candles.sum { |c| c[:volume].to_i }
        total_volume.zero? ? 0 : total_value / total_volume
      end
    end

    def trend_reversed_to_bearish?
      @context[:regime]&.dig(:direction) == :bearish
    end

    def trend_reversed_to_bullish?
      @context[:regime]&.dig(:direction) == :bullish
    end

    def calculate_stop_loss
      if @direction == :long
        @context[:candles].last(3).map { |c| c[:low].to_d }.min
      else
        @context[:candles].last(3).map { |c| c[:high].to_d }.max
      end
    end

    def calculate_target
      vwap = vwap_value
      return @entry_price * 1.01 if vwap.zero?

      # Target is VWAP (mean reversion)
      vwap
    end

    def standard_exit_checks(position, current)
      if position.long?
        return true if current <= position.stop_loss
        return true if current >= position.take_profit
      else
        return true if current >= position.stop_loss
        return true if current <= position.take_profit
      end
      false
    end

    def latest_close
      @context[:candles].last&.dig(:close).to_d
    end
  end
end
```

```ruby
# app/strategies/liquidity_sweep_strategy.rb
# frozen_string_literal: true

module Strategies
  class LiquiditySweepStrategy < BaseStrategy
    VERSION = "1.0.0".freeze

    version VERSION
    description "Liquidity Sweep: Trade stop-run reversals after liquidity pool sweep with structure confirmation"

    REQUIRED_FEATURES = %w[
      candles
      structure
      market_depth
      momentum
    ].freeze

    timeframes %w[1m 5m]
    supported_instruments %w[NIFTY BANKNIFTY]

    def initialize(instrument:, context: {})
      super
      @config = StrategyConfiguration.new(strategy_name)
      @sweep_threshold = @config.get(:sweep_threshold, 0.15)  # % beyond structure level
      @reversal_confirmation = @config.get(:reversal_confirmation, true)
      @min_stop_cluster_size = @config.get(:min_stop_cluster_size, 3)
    end

    def should_enter?(context = @context)
      return false unless structure_level_identified?
      return false unless sweep_detected?
      return false unless reversal_confirmed? if @reversal_confirmation

      @direction = determine_sweep_direction
      @entry_price = context[:spot_price] || latest_close
      @stop_loss = calculate_stop_loss
      @target = calculate_target

      true
    end

    def should_exit?(position, context = @context)
      current = context[:spot_price] || latest_close

      # Structure break against position
      if position.long? && structure_broken_down?
        return true
      elsif position.short? && structure_broken_up?
        return true
      end

      # Time decay for sweep trades (quick moves)
      if position.holding_time && position.holding_time > 15.minutes
        return true if position.unrealized_pnl.to_f > 0
      end

      standard_exit_checks(position, current)
    end

    def confidence(context = @context)
      return 0 unless sweep_detected?

      # Sweep quality
      sweep_depth = calculate_sweep_depth
      sweep_score = case sweep_depth
                    when 0..0.2 then 70
                    when 0.2..0.5 then 85
                    when 0.5..1.0 then 75
                    else 50
                    end

      # Reversal confirmation
      confirmation_score = reversal_confirmed? ? 15 : 0

      # Volume on reversal
      volume_score = volume_spike_on_reversal? ? 10 : 0

      [[sweep_score + confirmation_score + volume_score, 100].min, 0].max
    end

    def score(context = @context)
      return Result.success(signal: :neutral, confidence: 0, direction: nil) unless should_enter?(context)

      Result.success(
        signal: :enter,
        direction: @direction,
        confidence: confidence(context),
        entry_price: @entry_price,
        stop_loss: @stop_loss,
        target: @target,
        risk_reward: calculate_rr,
        setup: :liquidity_sweep_reversal,
        metadata: {
          sweep_level: @sweep_level,
          sweep_depth: calculate_sweep_depth,
          structure_type: @context[:structure]&.dig(:type),
          reversal_candle: reversal_candle_features
        }
      )
    end

    private

    def structure_level_identified?
      return false unless @context[:structure]
      %w[HH HL LH LL BOS CHOCH liquidity_sweep].include?(@context[:structure][:type])
    end

    def sweep_detected?
      return false unless @context[:structure]

      structure = @context[:structure]
      current = latest_close

      case structure[:type]
      when "HH", "LH"
        # Bearish sweep: price briefly breaks above structure high then reverses
        @sweep_level = structure[:price_level].to_d
        @sweep_detected = current < @sweep_level && recent_high_above?(@sweep_level)
      when "HL", "LL"
        # Bullish sweep: price briefly breaks below structure low then reverses
        @sweep_level = structure[:price_level].to_d
        @sweep_detected = current > @sweep_level && recent_low_below?(@sweep_level)
      else
        false
      end
    end

    def recent_high_above?(level)
      return false unless @context[:candles]&.length.to_i > 3
      @context[:candles].last(3).any? { |c| c[:high].to_d > level * (1 + @sweep_threshold / 100) }
    end

    def recent_low_below?(level)
      return false unless @context[:candles]&.length.to_i > 3
      @context[:candles].last(3).any? { |c| c[:low].to_d < level * (1 - @sweep_threshold / 100) }

      last_candle = @context[:candles].last
      prev_candle = @context[:candles][-2]

      if @direction == :long
        # Bullish engulfing or strong bullish close after sweep
        last_candle[:close].to_d > last_candle[:open].to_d &&
          last_candle[:close].to_d > prev_candle[:high].to_d
      else
        # Bearish engulfing or strong bearish close after sweep
        last_candle[:close].to_d < last_candle[:open].to_d &&
          last_candle[:close].to_d < prev_candle[:low].to_d
      end

      case @context[:structure][:type]

      else :neutral

      return 0 unless @sweep_level && @sweep_level > 0
      current = latest_close
      ((current - @sweep_level).abs / @sweep_level * 100).round(2)

      @context[:volume][:current].to_i > @context[:volume][:average].to_i * 2

    def structure_broken_down?

    def structure_broken_up?
```

```ruby
      @breakout_threshold = @config.get(:breakout_threshold, 0.2)  # % beyond range
      @volume_multiplier = @config.get(:volume_multiplier, 2.0)

      return false unless breakout_detected?
      return false unless volume_confirmed?
      return false unless momentum_aligned?

      @target = calculate_target

    end

    def should_exit?(position, context = @context)

      # False breakout: return inside range
      if position.long? && current < @consolidation_low
        return true
      elsif position.short? && current > @consolidation_high
        return true
      end

      # Time-based: breakout should move quickly
      if position.holding_time && position.holding_time > 20.minutes
        return true if position.unrealized_pnl.to_f <= 0
      end

      standard_exit_checks(position, current)
    end

    def confidence(context = @context)
      return 0 unless consolidation_identified?

      # Range tightness score
      range_score = consolidation_tightness_score

      # Volume confirmation
      volume_score = volume_confirmed? ? 15 : 0

      # Momentum alignment
      momentum_score = momentum_aligned? ? 10 : 0

      # Structure alignment
      structure_score = structure_aligned? ? 10 : 0

      [[range_score + volume_score + momentum_score + structure_score, 100].min, 0].max
    end

    def score(context = @context)
      return Result.success(signal: :neutral, confidence: 0, direction: nil) unless should_enter?(context)

      Result.success(
        signal: :enter,
        direction: @direction,
        confidence: confidence(context),
        entry_price: @entry_price,
        stop_loss: @stop_loss,
        target: @target,
        risk_reward: calculate_rr,
        setup: :range_breakout,
        metadata: {
          consolidation_high: @consolidation_high,
          consolidation_low: @consolidation_low,
          range_size: @range_size,
          breakout_percent: @breakout_percent,
          volume_ratio: volume_ratio
        }
      )
    end

    private

    def consolidation_identified?
      return false unless @context[:candles]&.length.to_i >= @consolidation_periods

      recent_candles = @context[:candles].last(@consolidation_periods)
      highs = recent_candles.map { |c| c[:high].to_d }
      lows = recent_candles.map { |c| c[:low].to_d }

      @consolidation_high = highs.max
      @consolidation_low = lows.min
      @range_size = @consolidation_high - @consolidation_low

      return false if @consolidation_low.zero?

      range_percent = (@range_size / @consolidation_low * 100).round(2)
      @consolidation_tight = range_percent < 1.0  # Less than 1% range = tight consolidation

      @consolidation_tight
    end

    def breakout_detected?
      return false unless @consolidation_high && @consolidation_low

      current = latest_close

      if current > @consolidation_high * (1 + @breakout_threshold / 100)
        @breakout_direction = :long
        @breakout_percent = ((current - @consolidation_high) / @consolidation_high * 100).round(2)
        true
      elsif current < @consolidation_low * (1 - @breakout_threshold / 100)
        @breakout_direction = :short
        @breakout_percent = ((@consolidation_low - current) / @consolidation_low * 100).round(2)
        true
      else
        false
      end
    end

    def volume_confirmed?
      return false unless @context[:volume]
      @context[:volume][:current].to_i > @context[:volume][:average].to_i * @volume_multiplier
    end

    def momentum_aligned?
      return false unless @context[:momentum]

      if @breakout_direction == :long
        @context[:momentum][:direction].to_s.include?('bullish')
      else
        @context[:momentum][:direction].to_s.include?('bearish')
      end
    end

    def determine_breakout_direction
      @breakout_direction || :neutral
    end

    def consolidation_tightness_score
      return 50 unless @range_size && @consolidation_low > 0
      range_percent = (@range_size / @consolidation_low * 100).round(2)

      case range_percent
      when 0..0.3 then 90
      when 0.3..0.6 then 80
      when 0.6..1.0 then 65
      when 1.0..2.0 then 50
      else 30
      end
    end

    def structure_aligned?
      return true unless @context[:structure]

      if @breakout_direction == :long
        %w[HH HL BOS].include?(@context[:structure][:type])
      else
        %w[LH LL CHOCH].include?(@context[:structure][:type])
      end
    end

    def volume_ratio
      return 0 unless @context[:volume] && @context[:volume][:average].to_i > 0
      (@context[:volume][:current].to_f / @context[:volume][:average].to_f).round(2)
    end

    def calculate_stop_loss
      if @direction == :long
        @consolidation_low
      else
        @consolidation_high
      end
    end

    def calculate_target
      # Measured move: range size projected from breakout point
      if @direction == :long
        @consolidation_high + @range_size
      else
        @consolidation_low - @range_size
      end
    end

        return true if current <= position.stop_loss
        return true if current >= position.take_profit
      else
        return true if current >= position.stop_loss
        return true if current <= position.take_profit
```

```ruby
      return false unless @context[:momentum]
```

```ruby
    end

    def score(context = @context)
      return Result.success(signal: :neutral, confidence: 0, direction: nil) unless should_enter?(context)

      Result.success(
        signal: :enter,
        direction: @direction,
        confidence: confidence(context),
        entry_price: @entry_price,
        stop_loss: @stop_loss,
        target: @target,
        risk_reward: calculate_rr,
        setup: :range_expansion,
        metadata: {
          squeeze_duration: @squeeze_duration,
          expansion_strength: calculate_expansion_strength,
          band_width_before: @band_width_before,
          band_width_after: @band_width_after,
          volume_ratio: volume_ratio
        }
      )
    end

    private

    def squeeze_detected?
      return false unless @context[:candles]&.length.to_i >= @bb_period

      recent = @context[:candles].last(@bb_period)
      closes = recent.map { |c| c[:close].to_d }

      sma = closes.sum.to_f / closes.length
      std_dev = Math.sqrt(closes.sum { |c| (c - sma) ** 2 } / closes.length)

      upper_band = sma + 2 * std_dev
      lower_band = sma - 2 * std_dev
      @band_width_before = upper_band - lower_band

      return false if sma.zero?

      band_width_percent = (@band_width_before / sma * 100).round(2)
      @squeeze_active = band_width_percent < @squeeze_threshold
      @squeeze_duration = count_squeeze_periods if @squeeze_active

      @squeeze_active
    end

    def count_squeeze_periods
      # Count consecutive squeeze periods
      3  # Simplified
    end

    def expansion_triggered?
      return false unless @context[:candles]&.length.to_i > 0

      current = latest_close
      recent = @context[:candles].last(3)
      closes = recent.map { |c| c[:close].to_d }

      # Price breaks out of squeeze band
      sma = closes.sum.to_f / closes.length
      std_dev = Math.sqrt(closes.sum { |c| (c - sma) ** 2 } / closes.length)
      upper = sma + 2 * std_dev
      lower = sma - 2 * std_dev

      @band_width_after = upper - lower

      if current > upper * (1 + @expansion_threshold / 100)
        @expansion_direction = :long
        true
      elsif current < lower * (1 - @expansion_threshold / 100)
        @expansion_direction = :short
        true
      else
        false
      end
    end

    def volume_surge?
      return false unless @context[:volume]
      @context[:volume][:current].to_i > @context[:volume][:average].to_i * @volume_multiplier
    end

    def direction_clear?
      return false unless @context[:momentum]

      if @expansion_direction == :long
        @context[:momentum][:direction].to_s.include?('bullish')
      else
        @context[:momentum][:direction].to_s.include?('bearish')
      end
    end

    def determine_direction
      @expansion_direction || :neutral
    end

    def squeeze_duration_score
      case @squeeze_duration
      when 0..2 then 60
      when 2..5 then 80
      when 5..10 then 90
      else 95
      end
    end

    def calculate_expansion_strength
      return 0 unless @band_width_before && @band_width_after && @band_width_before > 0
      (@band_width_after / @band_width_before).round(2)
    end

    def band_width_normalized?
      return false unless @band_width_after
      # Band width returns to normal levels
      true  # Simplified
    end

    def volume_ratio
      return 0 unless @context[:volume] && @context[:volume][:average].to_i > 0
      (@context[:volume][:current].to_f / @context[:volume][:average].to_f).round(2)
    end

    def calculate_stop_loss
      if @direction == :long
        @context[:candles].last(2).map { |c| c[:low].to_d }.min
      else
        @context[:candles].last(2).map { |c| c[:high].to_d }.max
      end
    end

    def calculate_target
      # Target: 2x the expansion move
      expansion_size = (@entry_price - @band_width_before).abs
      if @direction == :long
        @entry_price + expansion_size * 2
      else
        @entry_price - expansion_size * 2
      end
    end

    def standard_exit_checks(position, current)
      if position.long?
        return true if current <= position.stop_loss
        return true if current >= position.take_profit
      else
        return true if current >= position.stop_loss
        return true if current <= position.take_profit
      end
      false
    end

    def latest_close
      @context[:candles].last&.dig(:close).to_d
    end
  end
end
```

```ruby
# app/models/strategy_performance.rb
# frozen_string_literal: true

class StrategyPerformance < ApplicationRecord
  belongs_to :instrument, optional: true

  validates :strategy_name, presence: true
  validates :timeframe, presence: true, inclusion: { in: %w[1m 5m 15m 30m 1h 1d] }

  scope :for_strategy, ->(name) { where(strategy_name: name) }
  scope :for_timeframe, ->(tf) { where(timeframe: tf) }
  scope :best_performing, -> { order(win_rate: :desc) }
  scope :profitable, -> { where("expectancy > ?", 0) }

  def profitable?
    expectancy.present? && expectancy > 0
  end

  def update_from_trades!(trades)
    wins = trades.select(&:win?)
    losses = trades.reject(&:win?)

    total_trades = trades.length
    win_count = wins.length
    loss_count = losses.length

    gross_profit = wins.sum(&:gross_pnl)
    gross_loss = losses.sum(&:gross_pnl).abs

    update!(
      total_trades: total_trades,
      win_count: win_count,
      loss_count: loss_count,
      win_rate: total_trades.zero? ? 0 : (win_count.to_f / total_trades * 100).round(2),
      profit_factor: gross_loss.zero? ? 0 : (gross_profit / gross_loss).round(2),
      expectancy: total_trades.zero? ? 0 : ((gross_profit - gross_loss) / total_trades).round(2),
      avg_profit: wins.empty? ? 0 : (gross_profit / wins.length).round(2),
      avg_loss: losses.empty? ? 0 : (gross_loss / losses.length).round(2),
      max_drawdown: calculate_max_drawdown(trades),
      avg_holding_time: calculate_avg_holding_time(trades),
      updated_at: Time.current
    )
  end

  private

  def calculate_max_drawdown(trades)
    return 0 if trades.empty?

    equity = 0
    peak = 0
    max_dd = 0

    trades.sort_by(&:entry_time).each do |trade|
      equity += trade.gross_pnl.to_f
      peak = [peak, equity].max
      dd = peak - equity
      max_dd = [max_dd, dd].max
    end

    max_dd.round(2)
  end

  def calculate_avg_holding_time(trades)
    return 0 if trades.empty?

    total_seconds = trades.sum do |t|
      t.exit_time && t.entry_time ? (t.exit_time - t.entry_time).to_i : 0
    end

    (total_seconds / trades.length).round
  end
end
```

```ruby
# app/services/strategy_performance_tracker.rb
# frozen_string_literal: true

class StrategyPerformanceTracker < BaseService
  def initialize(strategy_name:, instrument: nil, timeframe: "5m")
    @strategy_name = strategy_name
    @instrument = instrument
    @timeframe = timeframe
  end

  def call
    trades = fetch_trades
    return Result.success(:no_trades) if trades.empty?

    performance = StrategyPerformance.find_or_initialize_by(
      strategy_name: @strategy_name,
      instrument: @instrument,
      timeframe: @timeframe
    )

    performance.update_from_trades!(trades)

    emit(EventTypes::STRATEGY_PERFORMANCE_UPDATED, {
      strategy: @strategy_name,
      win_rate: performance.win_rate,
      expectancy: performance.expectancy
    })

    Result.success(performance)
  end

  private

  def fetch_trades
    scope = Trade.joins(:trade_setup)
                 .where(trade_setups: { setup_type: @strategy_name })
                 .where(timeframe: @timeframe)

    scope = scope.where(instrument: @instrument) if @instrument
    scope.where(entry_time: 30.days.ago..Time.current).order(:entry_time)
  end
end
```
