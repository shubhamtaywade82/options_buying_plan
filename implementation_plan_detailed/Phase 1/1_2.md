### Task 1.2: Create `DhanHQ::WebSocketClient` Base Class
- **File:** `app/gateway/dhanhq/web_socket_client.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Gateway
    module DhanHQ
      class WebSocketClient
        include AppInject

        DHANHQ_WS_URL = "wss://api-feed.dhan.co"
        HEARTBEAT_INTERVAL = 30  # seconds
        RECONNECT_DELAYS = [1, 2, 5, 10, 30, 60].freeze  # exponential-ish

        attr_reader :connection, :status, :last_message_at, :subscribed_instruments

        def initialize
          @status = :disconnected
          @last_message_at = nil
          @subscribed_instruments = Set.new
          @reconnect_attempts = 0
          @message_sequence = 0
          @mutex = Mutex.new
        end

        def connect
          return if connected?

          @status = :connecting
          Rails.logger.info "[WebSocket] Connecting to #{DHANHQ_WS_URL}"

          @connection = WebSocket::Client::Simple.connect(
            DHANHQ_WS_URL,
            headers: auth_headers
          )

          bind_events
          start_heartbeat
          @status = :connected
          @reconnect_attempts = 0

          EventBus.publish(EventTypes::SYSTEM_HEALTH_CHECK, {
            component: "websocket",
            status: "connected"
          })
        end

        def disconnect
          return unless connected?

          stop_heartbeat
          @connection.close if @connection
          @status = :disconnected
          @subscribed_instruments.clear
        end

        def connected?
          @status == :connected && @connection&.open?
        end

        def subscribe(instruments)
          instruments = Array(instruments)
          return if instruments.empty?

          payload = build_subscription_payload(instruments)
          send(payload)

          instruments.each { |i| @subscribed_instruments << i.id }
          Rails.logger.info "[WebSocket] Subscribed to #{instruments.map(&:trading_symbol).join(', ')}"
        end

        def unsubscribe(instruments)
          instruments = Array(instruments)
          return if instruments.empty?

          payload = build_unsubscription_payload(instruments)
          send(payload)

          instruments.each { |i| @subscribed_instruments.delete(i.id) }
        end

        def send(data)
          return false unless connected?

          @connection.send(data.to_json)
          true
        end

        private

        def auth_headers
          {
            "Authorization" => "Bearer #{Rails.application.credentials.broker[:access_token]}",
            "Client-Id" => AppConfig.config.broker.client_id
          }
        end

        def bind_events
          @connection.on :open do |e|
            handle_open
          end

          @connection.on :message do |msg|
            handle_message(msg)
          end

          @connection.on :close do |e|
            handle_close(e)
          end

          @connection.on :error do |e|
            handle_error(e)
          end
        end

        def handle_open
          Rails.logger.info "[WebSocket] Connection opened"
          @last_message_at = Time.current
          auto_resubscribe if @subscribed_instruments.any?
        end

        def handle_message(msg)
          @last_message_at = Time.current
          @message_sequence += 1

          raw_data = JSON.parse(msg.data)
          normalized = normalize_message(raw_data)

          if normalized
            dispatch(normalized)
            record_metrics(:message_received, normalized[:type])
          end
        rescue JSON::ParserError => e
          Rails.logger.error "[WebSocket] Invalid JSON: #{e.message}"
          record_metrics(:parse_error)
        end

        def handle_close(event)
          Rails.logger.warn "[WebSocket] Connection closed: #{event.code} - #{event.reason}"
          @status = :disconnected
          schedule_reconnect
        end

        def handle_error(error)
          Rails.logger.error "[WebSocket] Error: #{error.message}"
          record_metrics(:connection_error)
          @status = :error
        end

        def schedule_reconnect
          stop_heartbeat
          delay = RECONNECT_DELAYS[[@reconnect_attempts, RECONNECT_DELAYS.length - 1].min]
          @reconnect_attempts += 1

          Rails.logger.info "[WebSocket] Reconnecting in #{delay}s (attempt #{@reconnect_attempts})"

          Concurrent::ScheduledTask.execute(delay) do
            connect
          end
        end

        def start_heartbeat
          @heartbeat_task = Concurrent::TimerTask.new(execution_interval: HEARTBEAT_INTERVAL) do
            check_heartbeat
          end
          @heartbeat_task.execute
        end

        def stop_heartbeat
          @heartbeat_task&.shutdown
          @heartbeat_task = nil
        end

        def check_heartbeat
          return unless connected?

          if @last_message_at && Time.current - @last_message_at > HEARTBEAT_INTERVAL * 2
            Rails.logger.warn "[WebSocket] Heartbeat timeout, forcing reconnect"
            @connection.close
            @status = :disconnected
            schedule_reconnect
          else
            # Send ping if supported
            send({ type: "ping" }) if @connection&.open?
          end
        end

        def auto_resubscribe
          return if @subscribed_instruments.empty?

          instruments = Instrument.where(id: @subscribed_instruments.to_a)
          subscribe(instruments)
          Rails.logger.info "[WebSocket] Auto-resubscribed to #{instruments.count} instruments"
        end

        def build_subscription_payload(instruments)
          {
            action: "subscribe",
            instruments: instruments.map do |i|
              {
                exchangeSegment: i.exchange_segment,
                securityId: i.security_id,
                instrument: map_instrument_type(i.instrument_type)
              }
            end
          }
        end

        def build_unsubscription_payload(instruments)
          {
            action: "unsubscribe",
            instruments: instruments.map do |i|
              {
                exchangeSegment: i.exchange_segment,
                securityId: i.security_id
              }
            end
          }
        end

        def map_instrument_type(type)
          {
            "INDEX" => "INDEX",
            "STOCK" => "EQUITY",
            "OPTION" => "OPTSTK",
            "FUTURE" => "FUTSTK"
          }[type] || "INDEX"
        end

        def normalize_message(raw)
          # DhanHQ WebSocket message format varies by feed type
          case raw["type"]
          when "tick", "ltp"
            normalize_tick(raw)
          when "depth"
            normalize_depth(raw)
          when "order_update"
            normalize_order_update(raw)
          else
            nil
          end
        end

        def normalize_tick(raw)
          {
            type: :tick,
            timestamp: Time.at(raw["timestamp"] / 1000.0),
            instrument_id: resolve_instrument(raw["exchangeSegment"], raw["securityId"]),
            last_price: raw["lastPrice"]&.to_d,
            volume: raw["volume"]&.to_i,
            oi: raw["openInterest"]&.to_i,
            bid: raw["bid"]&.to_d,
            ask: raw["ask"]&.to_d,
            sequence: @message_sequence
          }
        end

        def normalize_depth(raw)
          {
            type: :depth,
            timestamp: Time.current,
            instrument_id: resolve_instrument(raw["exchangeSegment"], raw["securityId"]),
            bid_levels: raw["bids"]&.map { |b| { price: b["price"]&.to_d, quantity: b["quantity"]&.to_i } },
            ask_levels: raw["asks"]&.map { |a| { price: a["price"]&.to_d, quantity: a["quantity"]&.to_i } }
          }
        end

        def normalize_order_update(raw)
          {
            type: :order_update,
            timestamp: Time.current,
            broker_order_id: raw["orderId"],
            status: raw["orderStatus"],
            filled_quantity: raw["filledQuantity"]&.to_i,
            average_price: raw["averagePrice"]&.to_d
          }
        end

        def resolve_instrument(segment, security_id)
          # Cache instrument lookup to avoid DB hit per tick
          @instrument_cache ||= {}
          key = "#{segment}:#{security_id}"
          @instrument_cache[key] ||= Instrument.find_by(exchange_segment: segment, security_id: security_id)&.id
        end

        def dispatch(normalized)
          case normalized[:type]
          when :tick
            EventBus.publish(EventTypes::TICK_RECEIVED, normalized)
          when :depth
            EventBus.publish(EventTypes::DEPTH_UPDATED, normalized)
          when :order_update
            EventBus.publish(EventTypes::ORDER_FILLED, normalized)
          end
        end

        def record_metrics(event, label = nil)
          MetricsCollector.record(
            name: "websocket.#{event}",
            labels: { type: label }.compact,
            value: 1,
            type: :counter
          )
        end
      end
    end
  end
  ```
- **Deliverable:** Complete WebSocket client with reconnection, heartbeat, auto-resubscribe
- **Acceptance Criteria:** `WebSocketClient.new.connect` establishes connection; `subscribe([nifty])` sends subscription payload

## Day 2 — WebSocket Supervisor & Connection Pooling

### Task 2.1: Create `WebSocketSupervisor`
- **File:** `app/gateway/dhanhq/web_socket_supervisor.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Gateway
    module DhanHQ
      class WebSocketSupervisor
        include AppInject

        MAX_CONNECTIONS = 3  # DhanHQ allows multiple connections
        INSTRUMENTS_PER_CONNECTION = 100

        def initialize
          @clients = []
          @mutex = Mutex.new
        end

        def start
          Rails.logger.info "[WebSocketSupervisor] Starting supervisor"
          start_health_monitor
        end

        def stop
          Rails.logger.info "[WebSocketSupervisor] Stopping supervisor"
          stop_health_monitor
          @clients.each(&:disconnect)
          @clients.clear
        end

        def subscribe(instruments)
          instruments = Array(instruments)
          return if instruments.empty?

          # Distribute across connections
          instruments.each_slice(INSTRUMENTS_PER_CONNECTION) do |slice|
            client = find_or_create_client
            client.subscribe(slice)
          end
        end

        def unsubscribe(instruments)
          instruments = Array(instruments)
          @clients.each do |client|
            client.unsubscribe(instruments)
          end
        end

        def status
          @clients.map.with_index do |client, i|
            {
              connection: i + 1,
              status: client.status,
              instruments: client.subscribed_instruments.count,
              last_message_at: client.last_message_at,
              uptime: client.last_message_at ? Time.current - client.last_message_at : nil
            }
          end
        end

        def healthy?
          @clients.any?(&:connected?)
        end

        private

        def find_or_create_client
          @mutex.synchronize do
            available = @clients.find { |c| c.subscribed_instruments.count < INSTRUMENTS_PER_CONNECTION }
            return available if available

            if @clients.length < MAX_CONNECTIONS
              client = WebSocketClient.new
              client.connect
              @clients << client
              client
            else
              # Round-robin to least loaded
              @clients.min_by { |c| c.subscribed_instruments.count }
            end
          end
        end

        def start_health_monitor
          @health_task = Concurrent::TimerTask.new(execution_interval: 10) do
            check_health
          end
          @health_task.execute
        end

        def stop_health_monitor
          @health_task&.shutdown
        end

        def check_health
          @clients.each_with_index do |client, i|
            next if client.connected?

            Rails.logger.warn "[WebSocketSupervisor] Connection #{i + 1} unhealthy, reconnecting"
            client.connect
          end
        end
      end
    end
  end
  ```
- **Deliverable:** Multi-connection supervisor with load distribution
- **Acceptance Criteria:** `WebSocketSupervisor.new.subscribe(150.times.map { create(:instrument) })` creates 2 connections

  ```ruby
  # frozen_string_literal: true

  class TickNormalizer
    def self.normalize(raw_tick, instrument:)
      {
        timestamp: parse_timestamp(raw_tick[:timestamp]),
        instrument_id: instrument.id,
        last_price: raw_tick[:last_price]&.to_d,
        open: raw_tick[:open]&.to_d,
        high: raw_tick[:high]&.to_d,
        low: raw_tick[:low]&.to_d,
        close: raw_tick[:close]&.to_d,
        volume: raw_tick[:volume]&.to_i,
        oi: raw_tick[:oi]&.to_i,
        bid: raw_tick[:bid]&.to_d,
        ask: raw_tick[:ask]&.to_d,
        vwap: calculate_vwap(raw_tick),
        tick_type: "ltp"
      }
    end

    def self.normalize_depth(raw_depth, instrument:)
      {
        timestamp: Time.current,
        instrument_id: instrument.id,
        bid_levels: raw_depth[:bid_levels] || [],
        ask_levels: raw_depth[:ask_levels] || [],
        bid_imbalance: calculate_imbalance(raw_depth[:bid_levels]),
        ask_imbalance: calculate_imbalance(raw_depth[:ask_levels]),
        total_bid_quantity: raw_depth[:bid_levels]&.sum { |b| b[:quantity] || 0 } || 0,
        total_ask_quantity: raw_depth[:ask_levels]&.sum { |a| a[:quantity] || 0 } || 0,
        spread: calculate_spread(raw_depth)
      }
    end

    class << self
      private

      def parse_timestamp(ts)
        return Time.current unless ts
        ts.is_a?(Integer) ? Time.at(ts / 1000.0) : Time.parse(ts.to_s)
      end

      def calculate_vwap(tick)
        return nil unless tick[:vwap] || (tick[:volume] && tick[:volume] > 0)
        tick[:vwap]&.to_d
      end

      def calculate_imbalance(levels)
        return 0 unless levels && levels.length >= 2
        (levels[0][:quantity].to_f - levels[1][:quantity].to_f) / (levels[0][:quantity].to_f + levels[1][:quantity].to_f)
      end

      def calculate_spread(depth)
        best_bid = depth[:bid_levels]&.first&.dig(:price)
        best_ask = depth[:ask_levels]&.first&.dig(:price)
        return nil unless best_bid && best_ask
        best_ask - best_bid
      end
    end
  end
  ```
- **Deliverable:** Raw DhanHQ tick → domain `MarketTick` hash
- **Acceptance Criteria:** `TickNormalizer.normalize({ last_price: 22500.5 }, instrument: nifty)` returns normalized hash

### Task 3.2: Create `MessageDeduplicator`
- **File:** `app/lib/message_deduplicator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class MessageDeduplicator
    DEDUP_WINDOW = 60  # seconds
    MAX_CACHE_SIZE = 10_000

    def initialize
      @seen = {}
      @mutex = Mutex.new
      @cleanup_task = Concurrent::TimerTask.new(execution_interval: DEDUP_WINDOW) do
        cleanup
      end
      @cleanup_task.execute
    end

    def seen?(key)
      @mutex.synchronize do
        return true if @seen.key?(key)
        @seen[key] = Time.current
        false
      end
    end

    def fingerprint(type, data)
      case type
      when :tick
        "#{data[:instrument_id]}:#{data[:timestamp].to_f}:#{data[:last_price]}"
      when :depth
        "#{data[:instrument_id]}:#{data[:timestamp].to_f}:#{data[:bid_levels]&.first&.dig(:price)}"
      when :order_update
        "#{data[:broker_order_id]}:#{data[:status]}:#{data[:filled_quantity]}"
      else
        Digest::SHA256.hexdigest(data.to_json)
      end
    end

    private

    def cleanup
      @mutex.synchronize do
        cutoff = Time.current - DEDUP_WINDOW
        @seen.delete_if { |_, ts| ts < cutoff }

        # Hard limit to prevent memory growth
        if @seen.size > MAX_CACHE_SIZE
          @seen = @seen.sort_by { |_, ts| ts }.last(MAX_CACHE_SIZE / 2).to_h
        end
      end
    end
  end
  ```
- **Deliverable:** Duplicate message detection (same tick delivered twice on reconnect)
- **Acceptance Criteria:** `MessageDeduplicator.new.seen?("NIFTY:1234567890:22500.5")` returns `false` then `true`

### Task 4.1: Create `BackpressureController`
- **File:** `app/lib/backpressure_controller.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class BackpressureController
    HIGH_WATER_MARK = 1000   # Redis queue depth
    LOW_WATER_MARK = 100     # Resume normal processing
    THROTTLE_DELAY = 0.01    # 10ms sleep when throttled

    def initialize(queue_name: "tick_queue")
      @queue_name = queue_name
      @throttled = false
      @mutex = Mutex.new
    end

    def check!
      depth = queue_depth
      return :normal if depth < LOW_WATER_MARK

      if depth >= HIGH_WATER_MARK
        throttle!
        :throttled
      elsif @throttled && depth <= LOW_WATER_MARK
        resume!
        :resumed
      else
        @throttled ? :throttled : :normal
      end
    end

    def throttle!
      @mutex.synchronize { @throttled = true }
      Rails.logger.warn "[Backpressure] Throttling WebSocket, queue depth: #{queue_depth}"
      EventBus.publish(EventTypes::SYSTEM_ERROR, {
        type: :backpressure_throttle,
        queue_depth: queue_depth
      })
    end

    def resume!
      @mutex.synchronize { @throttled = false }
      Rails.logger.info "[Backpressure] Resumed normal processing, queue depth: #{queue_depth}"
    end

    def throttled?
      @mutex.synchronize { @throttled }
    end

    def apply_backpressure
      sleep(THROTTLE_DELAY) while throttled?
    end

    private

    def queue_depth
      RedisPool.general.with { |r| r.llen(@queue_name) }
    end
  end
  ```
- **Deliverable:** Redis queue depth monitoring with throttling
- **Acceptance Criteria:** `BackpressureController.new.check!` returns `:throttled` when Redis queue > 1000

### Task 4.2: Create `TickIngestionJob` (Real-Time)
- **File:** `app/jobs/tick_ingestion_job.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class TickIngestionJob < JobBase
    BATCH_SIZE = 100
    BATCH_TIMEOUT = 1  # second

    def perform(tick_data = nil)
      if tick_data
        # Single tick from EventBus
        buffer_tick(tick_data)
      else
        # Flush buffer (called by scheduled task)
        flush_buffer
      end
    end

    private

    def buffer_tick(tick_data)
      RedisPool.general.with do |redis|
        redis.lpush("tick_buffer", tick_data.to_json)
        redis.expire("tick_buffer", 60)
      end

      # Check if buffer is large enough to flush
      count = RedisPool.general.with { |r| r.llen("tick_buffer") }
      flush_buffer if count >= BATCH_SIZE
    end

    def flush_buffer
      ticks = []

      RedisPool.general.with do |redis|
        BATCH_SIZE.times do
          data = redis.rpop("tick_buffer")
          break unless data
          ticks << JSON.parse(data, symbolize_names: true)
        end
      end

      return if ticks.empty?

      # Apply backpressure
      BackpressureController.new.apply_backpressure

      # Bulk insert
      records = ticks.map do |t|
        instrument = Instrument.find_by(id: t[:instrument_id])
        next unless instrument

        TickNormalizer.normalize(t, instrument: instrument)
      end.compact

      TickRepository.insert_batch(records) if records.any?

      # Publish candle build trigger
      EventBus.publish(EventTypes::CANDLE_CLOSED, {
        instrument_ids: records.map { |r| r[:instrument_id] }.uniq,
        count: records.length
      })
    end
  end
  ```
- **Deliverable:** Buffered tick ingestion with batching
- **Acceptance Criteria:** 100 ticks buffered → automatic flush to TimescaleDB

  ```ruby
              ws.onopen do |handshake|

              end

              end

              ws.onclose do

              end
            end
          end

        sleep 0.5  # Wait for server to start
      end

      def stop
        EventMachine.stop if EventMachine.reactor_running?
        @thread&.join(2)
      end

      def broadcast(data)
        @clients.each { |c| c.send(data.to_json) }
      end

      def simulate_disconnect
        @clients.each(&:close)
        @clients.clear
      end

      private

      def handle_command(ws, msg)
        case msg["action"]
        when "subscribe"
          ws.send({ type: "ack", action: "subscribed" }.to_json)
  ```

### Task 8.1: Create `SequenceValidator`
- **File:** `app/lib/sequence_validator.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class SequenceValidator
    def initialize
      @last_sequence = {}
      @mutex = Mutex.new
    end

    def validate!(instrument_id, sequence)
      @mutex.synchronize do
        last = @last_sequence[instrument_id]

        if last && sequence <= last
          return :duplicate
        elsif last && sequence > last + 1
          gap_size = sequence - last - 1
          @last_sequence[instrument_id] = sequence
          return [:gap, gap_size]
        else
          @last_sequence[instrument_id] = sequence
          return :ok
        end
      end
    end

    def reset!(instrument_id = nil)
      @mutex.synchronize do
        if instrument_id
          @last_sequence.delete(instrument_id)
        else
          @last_sequence.clear
        end
      end
    end
  end
  ```
- **Deliverable:** Sequence gap detection per instrument
- **Acceptance Criteria:** `SequenceValidator.new.validate!("NIFTY", 5)` after `validate!("NIFTY", 3)` returns `[:gap, 1]`

### Task 8.2: Create `DataQualityChecker`
- **File:** `app/services/data_quality_checker.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class DataQualityChecker < BaseService
    def call(instrument:, from:, to:)
      checks = {
        continuity: check_continuity(instrument, from, to),
        outliers: check_outliers(instrument, from, to),
        staleness: check_staleness(instrument),
        sequence_gaps: check_sequence_gaps(instrument, from, to)
      }

      passed = checks.values.all? { |c| c[:passed] }

      success(
        instrument: instrument.trading_symbol,
        checks: checks,
        passed: passed,
        quality_score: calculate_score(checks)
      )
    end

    private

    def check_continuity(instrument, from, to)
      gaps = CandleRepository.gaps(instrument: instrument, timeframe: "1m", from: from, to: to).value
      {
        passed: gaps.empty?,
        gap_count: gaps.length,
        gaps: gaps.first(5)
      }
    end

    def check_outliers(instrument, from, to)
      outliers = TickRepository.outliers(instrument: instrument, from: from, to: to, max_change_percent: 5.0).value
      {
        passed: outliers.empty?,
        outlier_count: outliers.length,
        outliers: outliers.first(5)
      }
    end

    def check_staleness(instrument)
      latest = TickRepository.latest(instrument: instrument, limit: 1).value&.first
      return { passed: false, reason: :no_data } unless latest

      age = Time.current - latest.timestamp
      {
        passed: age < 5.seconds,
        age_seconds: age.round(2),
        last_tick_at: latest.timestamp
      }
    end

    def check_sequence_gaps(instrument, from, to)
      # Check for sequence gaps in raw tick storage
      { passed: true, gap_count: 0 }  # Placeholder for sequence tracking
    end

    def calculate_score(checks)
      weights = { continuity: 0.3, outliers: 0.3, staleness: 0.4 }
      checks.sum do |name, check|
        check[:passed] ? weights[name] * 100 : 0
      end.round
    end
  end
  ```
- **Deliverable:** Automated data quality validation
- **Acceptance Criteria:** `DataQualityChecker.new.call(instrument: nifty, from: 1.minute.ago, to: Time.current)` returns quality score 0-100
