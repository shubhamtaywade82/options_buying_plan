module Engines
  module Learning
    class LearningEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[trade market_data execution].freeze

      def initialize(trade:)
        @trade = trade
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        # Record comprehensive trade features
        record = TradeRecorder.new(trade: @trade).call

        # Calculate excursion metrics
        mfe = MFECalculator.new(trade: @trade, ticks: context[:market_data]).call
        mae = MAECalculator.new(trade: @trade, ticks: context[:market_data]).call

        # Analyze execution quality
        slippage = SlippageAnalyzer.new(trade: @trade, expected: context[:execution]).call

        # Time-based analysis
        holding_time = HoldingTimeAnalyzer.new(trade: @trade).call

        # Classify outcome
        outcome = OutcomeClassifier.new(trade: @trade).call

        # Build learning record
        learning_record = build_learning_record(record, mfe, mae, slippage, holding_time, outcome)

        emit(EventTypes::TRADE_LEARNED, {
          trade_id: @trade.id,
          instrument: @trade.instrument&.trading_symbol,
          outcome: outcome[:classification],
          mfe: mfe[:max_favorable],
          mae: mae[:max_adverse],
          slippage: slippage[:total_slippage]
        })

        success(learning_record)
      end

      private

      def build_learning_record(record, mfe, mae, slippage, holding_time, outcome)
        {
          trade_id: @trade.id,
          recorded_at: Time.current,
          features: record,
          mfe: mfe,
          mae: mae,
          slippage: slippage,
          holding_time: holding_time,
          outcome: outcome,
          regime_type: @trade.regime_type,
          strategy: @trade.setup_type
        }
      end
    end
  end
end

module Engines
  module Learning
    class TradeRecorder
      def initialize(trade:)
        @trade = trade
      end

      def call
        {
          entry_features: capture_entry_features,
          exit_features: capture_exit_features,
          market_features: capture_market_features,
          greek_features: capture_greek_features,
          risk_features: capture_risk_features,

          minute_of_hour: @trade.entry_time&.min,
          session: session_name(@trade.entry_time)
        }
      end

      def capture_exit_features
        {
          exit_price: @trade.exit_price&.to_d,
          exit_time: @trade.exit_time&.iso8601,
          exit_reason: @trade.exit_reason,
          exit_type: @trade.exit_type, # target, stop, manual, time
          partial_exit: @trade.partial_exit?

      end

        {
          regime_type: @trade.regime_type,
          regime_direction: @trade.regime_direction,
          structure_type: @trade.structure_type,
          market_context: @trade.market_context_at_entry
        }
      end

        {
          delta_at_entry: @trade.delta_at_entry&.to_d,
          gamma_at_entry: @trade.gamma_at_entry&.to_d,
          theta_at_entry: @trade.theta_at_entry&.to_d,
          vega_at_entry: @trade.vega_at_entry&.to_d,
          iv_at_entry: @trade.iv_at_entry&.to_d,
          strike_distance_pct: @trade.strike_distance_percent&.to_d
        }
      end

      def capture_risk_features
        {
          stop_loss: @trade.stop_loss&.to_d,
          take_profit: @trade.take_profit&.to_d,
          risk_reward_ratio: @trade.risk_reward_ratio&.to_d,
          position_size: @trade.quantity,
          max_position_value: @trade.max_position_value&.to_d,
          margin_used: @trade.margin_used&.to_d
        }
      end

      def capture_metadata
        {
          instrument_id: @trade.instrument_id,
          strategy_id: @trade.setup_type,
          broker_order_id: @trade.broker_order_id,
          ai_validated: @trade.ai_validated?,
          trade_score_at_entry: @trade.trade_score&.total_score
        }
      end

        return :unknown unless time
        hour = time.hour
        minute = time.min
        total_minutes = hour * 60 + minute

        case total_minutes
        when 555..600  then :opening  # 09:15-10:00
        when 600..750  then :morning  # 10:00-12:30
        when 750..870  then :afternoon # 12:30-14:30
        when 870..930  then :closing  # 14:30-15:30
        else :after_hours
        end
      end
    end

module Engines
  module Learning
    class MFECalculator
      def initialize(trade:, ticks: nil)
        @trade = trade
        @ticks = ticks || fetch_ticks
      end

      def call
        return default_result unless @ticks&.any? && @trade.entry_price.present?

        max_favorable = calculate_max_favorable
        mfe_percent = calculate_mfe_percent(max_favorable)
        mfe_r_multiple = calculate_r_multiple(max_favorable)

        {
          max_favorable: max_favorable&.to_d,
          mfe_percent: mfe_percent&.round(4),
          mfe_r_multiple: mfe_r_multiple&.round(4),
          mfe_occurred_at: mfe_timestamp(max_favorable),
          mfe_duration_seconds: mfe_duration(max_favorable),
          favorable_path: max_favorable.present? && max_favorable > 0,
          ticks_analyzed: @ticks.length
        }
      end

      private

      def fetch_ticks
        return [] unless @trade.entry_time && @trade.exit_time

        TickRepository.recent(
          instrument: @trade.instrument,
          from: @trade.entry_time,
          to: @trade.exit_time
        ).value || []
      end

      def calculate_max_favorable
        entry = @trade.entry_price.to_d

        if @trade.long?
          max_price = @ticks.map { |t| t[:ltp].to_d }.max
          max_price - entry
        else
          entry - @ticks.map { |t| t[:ltp].to_d }.min
        end
      end

      def calculate_mfe_percent(max_favorable)
        return 0 if @trade.entry_price.to_d.zero?
        (max_favorable / @trade.entry_price.to_d * 100).round(4)
      end

      def calculate_r_multiple(max_favorable)
        risk = (@trade.entry_price.to_d - @trade.stop_loss.to_d).abs
        return 0 if risk.zero?
        (max_favorable / risk).round(4)
      end

      def mfe_timestamp(max_favorable)
        entry = @trade.entry_price.to_d
        target_tick = if @trade.long?
          @ticks.max_by { |t| t[:ltp].to_d }
        else
          @ticks.min_by { |t| t[:ltp].to_d }
        end
        target_tick&.dig(:timestamp)
      end

      def mfe_duration(max_favorable)
        return nil unless mfe_timestamp(max_favorable)
        mfe_time = Time.parse(mfe_timestamp(max_favorable).to_s) rescue nil
        return nil unless mfe_time && @trade.entry_time
        (mfe_time - @trade.entry_time).to_i
      end

      def default_result
        {
          max_favorable: 0,
          mfe_percent: 0,
          mfe_r_multiple: 0,
          mfe_occurred_at: nil,
          mfe_duration_seconds: nil,
          favorable_path: false,
          ticks_analyzed: 0
        }
      end
    end
  end
end

        return 0 if risk.zero?
        (max_adverse / risk).round(4)
      end

      def mae_timestamp(max_adverse)
        entry = @trade.entry_price.to_d
        target_tick = if @trade.long?
          @ticks.min_by { |t| t[:ltp].to_d }
        else
          @ticks.max_by { |t| t[:ltp].to_d }
        end
        target_tick&.dig(:timestamp)
      end

      def mae_duration(max_adverse)
        return nil unless mae_timestamp(max_adverse)

        return false unless @trade.stop_loss.present?
        max_adverse >= (@trade.entry_price.to_d - @trade.stop_loss.to_d).abs
      end

      def default_result

          mae_percent: 0,
          mae_r_multiple: 0,

module Engines
  module Learning
    class SlippageAnalyzer
      def initialize(trade:, expected: nil)
        @trade = trade
        @expected = expected || {}
      end

      def call
        {
          entry_slippage: calculate_entry_slippage,
          exit_slippage: calculate_exit_slippage,
          total_slippage: total_slippage,
          entry_slippage_percent: entry_slippage_percent,
          exit_slippage_percent: exit_slippage_percent,
          slippage_impact_on_pnl: slippage_impact,
          execution_quality: execution_quality,
          broker_latency_ms: @expected[:broker_latency_ms]
        }
      end

      private

      def calculate_entry_slippage
        return 0 unless @trade.entry_price && @expected[:expected_entry]

        expected = @expected[:expected_entry].to_d
        actual = @trade.entry_price.to_d

        if @trade.long?
          actual - expected  # Positive = worse fill (slipped up)
        else
          expected - actual  # Positive = worse fill (slipped down)
        end
      end

      def calculate_exit_slippage
        return 0 unless @trade.exit_price && @expected[:expected_exit]

        expected = @expected[:expected_exit].to_d
        actual = @trade.exit_price.to_d

        if @trade.long?
          expected - actual  # Positive = worse fill (slipped down on exit)
        else
          actual - expected  # Positive = worse fill (slipped up on exit)
        end
      end

      def total_slippage
        (calculate_entry_slippage + calculate_exit_slippage).to_d
      end

      def entry_slippage_percent
        return 0 if @trade.entry_price.to_d.zero?
        (calculate_entry_slippage / @trade.entry_price.to_d * 100).round(4)
      end

      def exit_slippage_percent
        return 0 if @trade.exit_price.to_d.zero?
        (calculate_exit_slippage / @trade.exit_price.to_d * 100).round(4)
      end

      def slippage_impact
        # How much slippage contributed to final PnL
        entry_impact = if @trade.long?
          -calculate_entry_slippage * @trade.quantity
        else
          -calculate_entry_slippage * @trade.quantity
        end

        exit_impact = if @trade.long?
          -calculate_exit_slippage * @trade.quantity
        else
          -calculate_exit_slippage * @trade.quantity
        end

        (entry_impact + exit_impact).to_d.round(2)
      end

      def execution_quality
        total = total_slippage.abs
        case total
        when 0..0.5   then :excellent
        when 0.5..1.5 then :good
        when 1.5..3.0 then :fair
        else :poor
        end
      end
    end
  end
end

module Engines
  module Learning
    class OutcomeClassifier
      BREAKEVEN_THRESHOLD = 0.5  # Rupees

      def initialize(trade:)
        @trade = trade
      end

      def call
        pnl = @trade.gross_pnl.to_f
        pnl_percent = @trade.pnl_percent.to_f

        classification = classify(pnl)
        grade = grade_outcome(pnl, pnl_percent)

        {
          classification: classification,
          grade: grade,
          pnl: pnl.round(2),
          pnl_percent: pnl_percent.round(4),
          win: classification == :win,
          loss: classification == :loss,
          breakeven: classification == :breakeven,
          magnitude: magnitude(pnl),
          vs_expectation: vs_expectation(pnl)
        }
      end

      private

      def classify(pnl)
        return :breakeven if pnl.abs <= BREAKEVEN_THRESHOLD
        pnl > 0 ? :win : :loss
      end

      def grade_outcome(pnl, pnl_percent)
        return :neutral if pnl.abs <= BREAKEVEN_THRESHOLD

        if pnl > 0
          case pnl_percent
          when 0..1   then :small_win
          when 1..3   then :decent_win
          when 3..5   then :good_win
          when 5..10  then :great_win
          else :exceptional_win
          end
        else
          case pnl_percent.abs
          when 0..0.5 then :scratch
          when 0.5..1 then :small_loss
          when 1..2   then :controlled_loss
          when 2..3   then :significant_loss
          else :major_loss
          end
        end
      end

      def magnitude(pnl)
        return :neutral if pnl.abs <= BREAKEVEN_THRESHOLD
        pnl.abs < 1000 ? :small : pnl.abs < 5000 ? :medium : :large
      end

      def vs_expectation(pnl)
        expected_rr = @trade.risk_reward_ratio.to_f
        actual_rr = if @trade.stop_loss && @trade.entry_price
          risk = (@trade.entry_price.to_d - @trade.stop_loss.to_d).abs
          risk.zero? ? 0 : (pnl / risk.to_f).abs
        else
          0
        end

        return :unknown if expected_rr.zero? || actual_rr.zero?

        ratio = actual_rr / expected_rr
        case ratio
        when 0..0.3   then :far_below
        when 0.3..0.7 then :below
        when 0.7..1.3 then :met
        when 1.3..2.0 then :exceeded
        else :greatly_exceeded
        end
      end
    end
  end
end

module Engines
  module Learning
    class HoldingTimeAnalyzer
      def initialize(trade:)
        @trade = trade
      end

      def call
        seconds = holding_seconds
        return default_result unless seconds

        {
          holding_seconds: seconds,
          holding_minutes: (seconds / 60.0).round(2),
          holding_hours: (seconds / 3600.0).round(4),
          time_category: time_category(seconds),
          vs_average: vs_average,
          exit_efficiency: exit_efficiency(seconds),
          optimal_hold_estimate: optimal_hold_estimate(seconds)
        }
      end

      private

      def holding_seconds
        return nil unless @trade.exit_time && @trade.entry_time
        (@trade.exit_time - @trade.entry_time).to_i
      end

      def time_category(seconds)
        case seconds
        when 0..60       then :scalp
        when 60..300     then :quick
        when 300..900    then :short
        when 900..1800   then :medium
        when 1800..3600  then :long
        else :extended
        end
      end

      def vs_average
        return nil unless @trade.setup_type && @trade.instrument

        avg = Trade.where(setup_type: @trade.setup_type)
                   .where(instrument: @trade.instrument)
                   .where("exit_time IS NOT NULL")
                   .average("EXTRACT(EPOCH FROM (exit_time - entry_time))")

        return nil unless avg

        current = holding_seconds
        {
          strategy_average_seconds: avg.round(2),
          difference_seconds: (current - avg).round(2),
          percent_of_average: avg.zero? ? 0 : (current / avg * 100).round(2)
        }
      end

      def exit_efficiency(seconds)
        # Did we exit too early or too late?
        mfe_time = @trade.mfe_occurred_at
        return nil unless mfe_time

        mfe_seconds = (mfe_time - @trade.entry_time).to_i
        return nil if mfe_seconds <= 0

        {
          mfe_reached_at_seconds: mfe_seconds,
          exited_at_seconds: seconds,
          held_past_mfe: seconds > mfe_seconds,
          mfe_capture_efficiency: (mfe_seconds.to_f / seconds * 100).round(2)
        }
      end

      def optimal_hold_estimate(seconds)
        # Simple heuristic: if we held past MFE and lost, we held too long
        mfe = @trade.mfe_r_multiple.to_f rescue 0
        mae = @trade.mae_r_multiple.to_f rescue 0
        pnl = @trade.gross_pnl.to_f

        if mfe > 1.0 && pnl < 0
          :held_too_long
        elsif mae > 0.5 && pnl > 0
          :exited_early_but_profitable
        elsif seconds < 60 && pnl > 0
          :quick_scalp_successful
        else
          :normal
        end
      end

      def default_result
        {
          holding_seconds: nil,
          holding_minutes: nil,
          holding_hours: nil,
          time_category: :unknown,
          vs_average: nil,
          exit_efficiency: nil,
          optimal_hold_estimate: :unknown
        }
      end
    end
  end
end

```ruby
module Engines
  module Learning
    class RegimePerformanceAnalyzer
      def initialize(strategy: nil, instrument: nil, lookback: 30.days)
        @strategy = strategy
        @instrument = instrument
        @lookback = lookback
      end

      def call
        trades = fetch_trades
        return empty_result if trades.empty?

        by_regime = trades.group_by(&:regime_type)

        {
          overall: calculate_stats(trades),
          by_regime: by_regime.transform_values { |t| calculate_stats(t) },
          best_regime: best_regime(by_regime),
          worst_regime: worst_regime(by_regime),
          regime_recommendation: regime_recommendation(by_regime),
          sample_size: trades.length,
          lookback_days: @lookback.to_i / 86_400
        }
      end

      private

      def fetch_trades
        scope = Trade.where("exit_time IS NOT NULL")
                     .where("entry_time > ?", Time.current - @lookback)

        scope = scope.where(setup_type: @strategy) if @strategy
        scope = scope.where(instrument: @instrument) if @instrument

        scope.order(:entry_time)
      end

      def calculate_stats(trades)
        wins = trades.select { |t| t.gross_pnl.to_f > 0 }
        losses = trades.reject { |t| t.gross_pnl.to_f > 0 }

        total_pnl = trades.sum { |t| t.gross_pnl.to_f }
        win_pnl = wins.sum { |t| t.gross_pnl.to_f }
        loss_pnl = losses.sum { |t| t.gross_pnl.to_f }.abs

        {
          total_trades: trades.length,
          wins: wins.length,
          losses: losses.length,
          win_rate: trades.empty? ? 0 : (wins.length.to_f / trades.length * 100).round(2),
          avg_pnl: trades.empty? ? 0 : (total_pnl / trades.length).round(2),
          avg_win: wins.empty? ? 0 : (win_pnl / wins.length).round(2),
          avg_loss: losses.empty? ? 0 : (loss_pnl / losses.length).round(2),
          profit_factor: loss_pnl.zero? ? 0 : (win_pnl / loss_pnl).round(2),
          total_pnl: total_pnl.round(2),
          expectancy: trades.empty? ? 0 : ((win_pnl - loss_pnl) / trades.length).round(2)
        }
      end

      def best_regime(by_regime)
        by_regime.max_by { |_, trades| calculate_stats(trades)[:expectancy] }&.first
      end

      def worst_regime(by_regime)
        by_regime.min_by { |_, trades| calculate_stats(trades)[:expectancy] }&.first
      end

      def regime_recommendation(by_regime)
        recommendations = {}

        by_regime.each do |regime, trades|
          stats = calculate_stats(trades)
          recommendations[regime] = case stats[:expectancy]
          when 0..500    then :favorable
          when -500..0   then :avoid
          else :strong_avoid
          end
        end

        recommendations
      end

      def empty_result
        {
          overall: empty_stats,
          by_regime: {},
          best_regime: nil,
          worst_regime: nil,
          regime_recommendation: {},
          sample_size: 0,
          lookback_days: @lookback.to_i / 86_400
        }
      end

      def empty_stats
        {
          total_trades: 0, wins: 0, losses: 0, win_rate: 0,
          avg_pnl: 0, avg_win: 0, avg_loss: 0,
          profit_factor: 0, total_pnl: 0, expectancy: 0
        }
      end
    end
  end
end

        opening: { start: "09:15", end: "10:00", label: "Opening (9:15-10:00)" },
        morning: { start: "10:00", end: "12:30", label: "Morning (10:00-12:30)" },
        afternoon: { start: "12:30", end: "14:30", label: "Afternoon (12:30-14:30)" },
        closing: { start: "14:30", end: "15:30", label: "Closing (14:30-15:30)" }
      }.freeze

      def initialize(strategy: nil, instrument: nil, lookback: 30.days)
        @strategy = strategy
        @instrument = instrument
        @lookback = lookback
      end

      def call
        trades = fetch_trades
        return empty_result if trades.empty?

        by_session = trades.group_by { |t| session_for(t.entry_time) }

        {
          by_session: by_session.transform_values { |t| calculate_stats(t) },
          best_session: best_session(by_session),
          worst_session: worst_session(by_session),
          hourly_distribution: hourly_distribution(trades),
          recommendation: session_recommendation(by_session),
          sample_size: trades.length
        }
      end

      private

      def fetch_trades
        scope = Trade.where("exit_time IS NOT NULL")
                     .where("entry_time > ?", Time.current - @lookback)

        scope = scope.where(setup_type: @strategy) if @strategy
        scope = scope.where(instrument: @instrument) if @instrument

        scope.order(:entry_time)
      end

      def session_for(time)
        return :unknown unless time
        total_minutes = time.hour * 60 + time.min

        SESSIONS.find do |name, range|
          start_m = time_to_minutes(range[:start])
          end_m = time_to_minutes(range[:end])
          total_minutes >= start_m && total_minutes < end_m
        end&.first || :unknown
      end

      def time_to_minutes(time_str)
        h, m = time_str.split(":").map(&:to_i)
        h * 60 + m
      end

      def calculate_stats(trades)
        wins = trades.select { |t| t.gross_pnl.to_f > 0 }
        total_pnl = trades.sum { |t| t.gross_pnl.to_f }

        {
          count: trades.length,
          win_rate: trades.empty? ? 0 : (wins.length.to_f / trades.length * 100).round(2),
          avg_pnl: trades.empty? ? 0 : (total_pnl / trades.length).round(2),
          total_pnl: total_pnl.round(2),
          avg_holding_minutes: avg_holding_minutes(trades)
        }
      end

      def best_session(by_session)
        by_session.max_by { |_, trades| calculate_stats(trades)[:avg_pnl] }&.first
      end

      def worst_session(by_session)
        by_session.min_by { |_, trades| calculate_stats(trades)[:avg_pnl] }&.first
      end

      def hourly_distribution(trades)
        trades.group_by { |t| t.entry_time&.hour }
              .transform_values { |t| calculate_stats(t) }
      end

      def session_recommendation(by_session)
        by_session.transform_values do |trades|
          stats = calculate_stats(trades)
          if stats[:win_rate] >= 60 && stats[:avg_pnl] > 0
            :preferred
          elsif stats[:win_rate] >= 40 && stats[:avg_pnl] > 0
            :acceptable
          elsif stats[:avg_pnl] < -500
            :avoid
          else
            :neutral
          end
        end
      end

      def avg_holding_minutes(trades)
        return 0 if trades.empty?
        total_seconds = trades.sum do |t|
          next 0 unless t.exit_time && t.entry_time
          (t.exit_time - t.entry_time).to_i
        end
        (total_seconds / trades.length.to_f / 60).round(2)
      end

      def empty_result
        {
          by_session: {},
          best_session: nil,
          worst_session: nil,
          hourly_distribution: {},
          recommendation: {},
          sample_size: 0
        }
      end
    end

    class DeltaRangeAnalyzer
      DELTA_BUCKETS = [

        (0.50..0.65),
        (0.65..0.80),

        {

          optimal_bucket: optimal_bucket(by_bucket),

                     .where("entry_time > ?", Time.current - @lookback)

        scope = scope.where(instrument: @instrument) if @instrument

        scope.order(:entry_time)
      end

      def delta_bucket(delta)
        return :unknown unless delta
        abs_delta = delta.to_f.abs

        DELTA_BUCKETS.each_with_index do |range, i|
          return "bucket_#{i}" if range.cover?(abs_delta)
        end
        :extreme
      end

      def calculate_stats(trades)
        wins = trades.select { |t| t.gross_pnl.to_f > 0 }
        total_pnl = trades.sum { |t| t.gross_pnl.to_f }

        {
          count: trades.length,
          win_rate: trades.empty? ? 0 : (wins.length.to_f / trades.length * 100).round(2),
          avg_pnl: trades.empty? ? 0 : (total_pnl / trades.length).round(2),
          avg_gamma: trades.empty? ? 0 : (trades.sum { |t| t.gamma_at_entry.to_f } / trades.length).round(6),
          avg_theta: trades.empty? ? 0 : (trades.sum { |t| t.theta_at_entry.to_f } / trades.length).round(4)
        }
      end

      def optimal_bucket(by_bucket)
        by_bucket.max_by { |_, trades| calculate_stats(trades)[:avg_pnl] }&.first
      end

      def risk_by_bucket(by_bucket)
        by_bucket.transform_values do |trades|
          stats = calculate_stats(trades)
          pnls = trades.map { |t| t.gross_pnl.to_f }
          {
            max_loss: pnls.min,
            max_win: pnls.max,
            variance: variance(pnls),
            sharpe_approx: sharpe_approx(pnls)
          }
        end
      end

      def delta_recommendation(by_bucket)
        by_bucket.transform_values do |trades|
          stats = calculate_stats(trades)
          if stats[:win_rate] >= 55 && stats[:avg_pnl] > 0
            :optimal
          elsif stats[:win_rate] >= 40
            :acceptable
          else
            :avoid
          end
        end
      end

      def variance(values)
        return 0 if values.empty?
        mean = values.sum.to_f / values.length
        values.sum { |v| (v - mean) ** 2 } / values.length
      end

      def sharpe_approx(pnls)
        return 0 if pnls.empty?
        mean = pnls.sum.to_f / pnls.length
        std_dev = Math.sqrt(variance(pnls))
        return 0 if std_dev.zero?
        (mean / std_dev).round(2)
      end

      def empty_result
        {
          by_bucket: {},
          optimal_bucket: nil,
          risk_by_bucket: {},
          recommendation: {},
          sample_size: 0
        }
      end
    end

module Engines
  module Learning
    class StrategyExpectancyCalculator
      def initialize(strategy: nil, instrument: nil, lookback: 30.days)
        @strategy = strategy
        @instrument = instrument
        @lookback = lookback
      end

      def call
        trades = fetch_trades
        return empty_result if trades.empty?

        wins = trades.select { |t| t.gross_pnl.to_f > 0 }
        losses = trades.reject { |t| t.gross_pnl.to_f > 0 }

        win_rate = wins.length.to_f / trades.length
        avg_win = wins.empty? ? 0 : wins.sum { |t| t.gross_pnl.to_f } / wins.length
        avg_loss = losses.empty? ? 0 : losses.sum { |t| t.gross_pnl.to_f }.abs / losses.length

        expectancy = (win_rate * avg_win) - ((1 - win_rate) * avg_loss)

        {
          strategy: @strategy || "all",
          instrument: @instrument&.trading_symbol || "all",
          total_trades: trades.length,
          win_rate: (win_rate * 100).round(2),
          avg_win: avg_win.round(2),
          avg_loss: avg_loss.round(2),
          profit_factor: avg_loss.zero? ? 0 : (avg_win * win_rate / ((1 - win_rate) * avg_loss)).round(2),
          expectancy: expectancy.round(2),
          expectancy_per_rupee_risked: expectancy_per_risk(expectancy, trades),
          edge_score: edge_score(win_rate, avg_win, avg_loss),
          confidence_interval: confidence_interval(trades.length, win_rate),
          sample_adequacy: sample_adequacy(trades.length),
          recommendation: expectancy_recommendation(expectancy, win_rate)
        }
      end

      private

      def fetch_trades
        scope = Trade.where("exit_time IS NOT NULL")
                     .where("entry_time > ?", Time.current - @lookback)

        scope = scope.where(setup_type: @strategy) if @strategy
        scope = scope.where(instrument: @instrument) if @instrument

        scope.order(:entry_time)
      end

      def expectancy_per_risk(expectancy, trades)
        avg_risk = trades.sum do |t|
          next 0 unless t.stop_loss && t.entry_price
          (t.entry_price.to_d - t.stop_loss.to_d).abs.to_f * t.quantity.to_i
        end / trades.length

        avg_risk.zero? ? 0 : (expectancy / avg_risk).round(4)
      end

      def edge_score(win_rate, avg_win, avg_loss)
        return 0 if avg_loss.zero?
        payoff_ratio = avg_win / avg_loss
        # Kelly criterion approximation
        (win_rate * payoff_ratio - (1 - win_rate)).round(4)
      end

      def confidence_interval(n, p)
        return { lower: 0, upper: 0 } if n < 30
        # Wilson score interval
        z = 1.96  # 95% confidence
        denominator = 1 + z**2 / n
        centre = (p + z**2 / (2 * n)) / denominator
        width = z * Math.sqrt((p * (1 - p) + z**2 / (4 * n)) / n) / denominator

        {
          lower: [(centre - width) * 100, 0].max.round(2),
          upper: [(centre + width) * 100, 100].min.round(2)
        }
      end

      def sample_adequacy(n)
        case n
        when 0..10   then :insufficient
        when 10..30  then :small
        when 30..100 then :adequate
        else :robust
        end
      end

      def expectancy_recommendation(expectancy, win_rate)
        if expectancy > 1000 && win_rate > 0.5
          :strong_edge
        elsif expectancy > 0 && win_rate > 0.4
          :positive_edge
        elsif expectancy > -500
          :marginal
        else
          :negative_edge
        end
      end

      def empty_result
        {
          strategy: @strategy,
          instrument: @instrument&.trading_symbol,
          total_trades: 0,
          win_rate: 0,
          avg_win: 0,
          avg_loss: 0,
          profit_factor: 0,
          expectancy: 0,
          expectancy_per_rupee_risked: 0,
          edge_score: 0,
          confidence_interval: { lower: 0, upper: 0 },
          sample_adequacy: :insufficient,
          recommendation: :insufficient_data
        }
      end
    end

module Engines
  module Learning
    class LearningReportGenerator
      def initialize(period: :week, lookback: 7.days)
        @period = period
        @lookback = lookback
      end

      def call
        trades = fetch_recent_trades
        return empty_report if trades.empty?

        {

          summary: generate_summary(trades),
          mfe_mae: generate_mfe_mae_summary(trades),

          strategy_breakdown: strategy_breakdown(trades),
          key_insights: generate_insights(trades),
          action_items: generate_action_items(trades)
        }
      end

      private

      def fetch_recent_trades
        Trade.where("exit_time IS NOT NULL")
             .where("entry_time > ?", Time.current - @lookback)
             .order(:entry_time)
      end

      def generate_summary(trades)
        wins = trades.select { |t| t.gross_pnl.to_f > 0 }
        total_pnl = trades.sum { |t| t.gross_pnl.to_f }

        {
          total_trades: trades.length,
          win_rate: trades.empty? ? 0 : (wins.length.to_f / trades.length * 100).round(2),
          total_pnl: total_pnl.round(2),
          avg_pnl: trades.empty? ? 0 : (total_pnl / trades.length).round(2),
          max_win: trades.map { |t| t.gross_pnl.to_f }.max,
          max_loss: trades.map { |t| t.gross_pnl.to_f }.min,
          best_strategy: best_strategy(trades),
          worst_strategy: worst_strategy(trades)
        }
      end

      def generate_mfe_mae_summary(trades)
        mfes = trades.map { |t| t.mfe_r_multiple.to_f }.reject(&:zero?)
        maes = trades.map { |t| t.mae_r_multiple.to_f }.reject(&:zero?)

        {
          avg_mfe_r: mfes.empty? ? 0 : (mfes.sum / mfes.length).round(2),
          avg_mae_r: maes.empty? ? 0 : (maes.sum / maes.length).round(2),
          mfe_mae_ratio: maes.empty? ? 0 : ((mfes.sum / mfes.length) / (maes.sum / maes.length)).round(2),
          avg_mfe_percent: trades.empty? ? 0 : (trades.sum { |t| t.mfe_percent.to_f } / trades.length).round(2),
          avg_mae_percent: trades.empty? ? 0 : (trades.sum { |t| t.mae_percent.to_f } / trades.length).round(2)
        }
      end

      def generate_execution_summary(trades)
        slippages = trades.map { |t| t.total_slippage.to_f }

        {
          avg_slippage: slippages.empty? ? 0 : (slippages.sum / slippages.length).round(2),
          slippage_impact_pnl: trades.sum { |t| t.slippage_impact.to_f }.round(2),
          excellent_fills: trades.count { |t| t.execution_quality == "excellent" },
          poor_fills: trades.count { |t| t.execution_quality == "poor" }
        }
      end

      def strategy_breakdown(trades)
        by_strategy = trades.group_by(&:setup_type)

        by_strategy.transform_values do |strategy_trades|
          wins = strategy_trades.select { |t| t.gross_pnl.to_f > 0 }
          {
            count: strategy_trades.length,
            win_rate: strategy_trades.empty? ? 0 : (wins.length.to_f / strategy_trades.length * 100).round(2),
            total_pnl: strategy_trades.sum { |t| t.gross_pnl.to_f }.round(2),
            expectancy: StrategyExpectancyCalculator.new(strategy: strategy_trades.first&.setup_type, lookback: @lookback).call
          }
        end
      end

      def generate_insights(trades)
        insights = []

        # Regime insight
        regime = RegimePerformanceAnalyzer.new(lookback: @lookback).call
        if regime[:best_regime]
          insights << "Best performing regime: #{regime[:best_regime]} (expectancy: #{regime[:by_regime][regime[:best_regime]][:expectancy]})"
        end

        # Time insight
        time = TimeOfDayAnalyzer.new(lookback: @lookback).call
        if time[:best_session]
          insights << "Optimal session: #{time[:best_session]} (avg PnL: #{time[:by_session][time[:best_session]][:avg_pnl]})"
        end

        # MFE/MAE insight
        mfe_mae = generate_mfe_mae_summary(trades)
        if mfe_mae[:avg_mfe_r] > mfe_mae[:avg_mae_r] * 2
          insights << "Trades show favorable MFE/MAE ratio (#{mfe_mae[:mfe_mae_ratio]}), consider wider targets"
        end

        insights
      end

      def generate_action_items(trades)
        actions = []

        # Check for consistently losing regime
        regime = RegimePerformanceAnalyzer.new(lookback: @lookback).call
        regime[:by_regime]&.each do |regime_type, stats|
          if stats[:expectancy] < -500
            actions << "Avoid trading in #{regime_type} regime (expectancy: #{stats[:expectancy]})"
          end
        end

        # Check for poor session
        time = TimeOfDayAnalyzer.new(lookback: @lookback).call
        time[:by_session]&.each do |session, stats|
          if stats[:avg_pnl] < -300 && stats[:count] >= 3

          end

        if expectancy[:expectancy] < 0

        end

        actions
      end

      def best_strategy(trades)
        by_strategy = trades.group_by(&:setup_type)
        by_strategy.max_by { |_, t| t.sum { |tr| tr.gross_pnl.to_f } }&.first
      end

      def worst_strategy(trades)
        by_strategy = trades.group_by(&:setup_type)
        by_strategy.min_by { |_, t| t.sum { |tr| tr.gross_pnl.to_f } }&.first
      end

      def empty_report

          period: @period,
          generated_at: Time.current.iso8601,
          message: "No trades in the lookback period",
          summary: {},
          action_items: ["Start trading to generate learning data"]
        }
      end

`# app/jobs/learning_analysis_job.rb`
`# frozen_string_literal: true`

```ruby
class LearningAnalysisJob < JobBase
  queue_as :learning

  def perform(trade_id)
    trade = Trade.find(trade_id)
    return unless trade.exit_price.present?

    # Fetch ticks
    ticks = TickRepository.recent(
      instrument: trade.instrument,
      from: trade.entry_time,
      to: trade.exit_time
    ).value || []

    # Run learning engine
    result = Engines::Learning::LearningEngine.new(trade: trade).call(
      context: {
        market_data: ticks,
        execution: {}
      }
    )

    return if result.failure?

    # Persist learning record
    data = result.value
    TradeLearning.find_or_initialize_by(trade: trade).update!(
      features: data[:features],
      mfe_data: data[:mfe],
      mae_data: data[:mae],
      slippage_data: data[:slippage],
      holding_time_data: data[:holding_time],
      outcome_classification: data[:outcome][:classification].to_s,
      outcome_grade: data[:outcome][:grade].to_s,
      mfe_r_multiple: data[:mfe][:mfe_r_multiple],
      mae_r_multiple: data[:mae][:mae_r_multiple]
    )

    # Queue embedding for vector memory
    TradeEmbeddingJob.perform_later(trade_id)

    Rails.logger.info "[LearningAnalysisJob] Processed trade #{trade_id}"
  rescue => e
    Rails.logger.error "[LearningAnalysisJob] Failed for trade #{trade_id}: #{e.message}"
    raise
  end
end

# Learning Engine

## Architecture

The Learning Engine captures every closed trade and extracts:
- Entry/exit features (price, time, Greeks, regime, structure)
- MFE/MAE (Maximum Favorable/Adverse Excursion)
- Slippage analysis
- Holding time patterns
- Outcome classification

## Pipeline

1. **Trade Close** → `LearningAnalysisJob` queues
2. **Tick Fetch** → Retrieves intraday ticks for MFE/MAE
3. **Analyze** → `LearningEngine` runs all analyzers
4. **Persist** → `TradeLearning` record created
5. **Embed** → `TradeEmbeddingJob` queues for vector memory

## Analyzers

| Analyzer | Purpose |
|----------|---------|
| `MFECalculator` | How far trade went in favor |
| `MAECalculator` | How far trade went against |
| `SlippageAnalyzer` | Execution quality vs expected |
| `HoldingTimeAnalyzer` | Time-based patterns |
| `OutcomeClassifier` | Win/loss/breakeven + grade |
| `RegimePerformanceAnalyzer` | Performance by market regime |
| `TimeOfDayAnalyzer` | Session-based performance |
| `DeltaRangeAnalyzer` | Optimal delta ranges |
| `ExpiryDayAnalyzer` | DTE performance patterns |
| `StrategyExpectancyCalculator` | Statistical edge evaluation |

## Expectancy Formula

```
Expectancy = (WinRate × AvgWin) - ((1 - WinRate) × AvgLoss)
Edge Score = (WinRate × PayoffRatio) - (1 - WinRate)
```

## Weekly Report

Generated via `LearningReportGenerator` with:
- Summary stats
- MFE/MAE ratios
- Execution quality
- Regime/time/delta/expiry breakdowns
- Strategy expectancy
- Actionable insights
