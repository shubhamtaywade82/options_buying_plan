# app/services/monitoring/metrics_collector.rb
# frozen_string_literal: true

module Monitoring
  class MetricsCollector
    def initialize
      @registry = Prometheus::Client.registry
      setup_default_metrics
    end

    def record(metric_name, value, labels: {})
      metric = @registry.get(metric_name)
      return unless metric

      case metric
      when Prometheus::Client::Counter
        metric.increment(labels: labels)
      when Prometheus::Client::Gauge
        metric.set(value, labels: labels)
      when Prometheus::Client::Histogram
        metric.observe(value, labels: labels)
      end
    rescue => e
      Rails.logger.error "[MetricsCollector] Failed to record #{metric_name}: #{e.message}"
    end

    def render
      Prometheus::Client::Formats::Text.marshal(@registry)
    end

    private

    def setup_default_metrics
      # Counters
      Prometheus::Client::Counter.new(:orders_total, docstring: "Total orders placed", labels: [:strategy, :direction]).tap { |m| @registry.register(m) }
      Prometheus::Client::Counter.new(:fills_total, docstring: "Total order fills", labels: [:strategy, :status]).tap { |m| @registry.register(m) }
      Prometheus::Client::Counter.new(:alerts_total, docstring: "Total alerts fired", labels: [:severity, :category]).tap { |m| @registry.register(m) }

      # Gauges
      Prometheus::Client::Gauge.new(:active_positions, docstring: "Current open positions").tap { |m| @registry.register(m) }
      Prometheus::Client::Gauge.new(:unrealized_pnl, docstring: "Unrealized P&L").tap { |m| @registry.register(m) }
      Prometheus::Client::Gauge.new(:daily_pnl, docstring: "Daily realized P&L").tap { |m| @registry.register(m) }
      Prometheus::Client::Gauge.new(:margin_utilization_percent, docstring: "Margin utilization %").tap { |m| @registry.register(m) }

      # Histograms
      Prometheus::Client::Histogram.new(:order_latency_ms, docstring: "Order placement latency", buckets: [10, 50, 100, 250, 500, 1000]).tap { |m| @registry.register(m) }
      Prometheus::Client::Histogram.new(:engine_latency_ms, docstring: "Engine execution latency", labels: [:engine], buckets: [1, 10, 50, 100, 500]).tap { |m| @registry.register(m) }
      Prometheus::Client::Histogram.new(:slippage_amount, docstring: "Slippage per fill", buckets: [0, 0.5, 1, 2, 5, 10]).tap { |m| @registry.register(m) }
    end
  end
end

module Monitoring
  class TradingMetrics
    def initialize(collector: nil)
      @collector = collector || MetricsCollector.new
    end

    def record_order(strategy:, direction:, latency_ms:)
      @collector.record(:orders_total, 1, labels: { strategy: strategy, direction: direction })
      @collector.record(:order_latency_ms, latency_ms)
    end

    def record_fill(strategy:, status:, slippage:)
      @collector.record(:fills_total, 1, labels: { strategy: strategy, status: status })
      @collector.record(:slippage_amount, slippage)
    end

    def record_position_change(count:, unrealized_pnl:)
      @collector.record(:active_positions, count)
      @collector.record(:unrealized_pnl, unrealized_pnl)
    end

    def record_daily_pnl(pnl)
      @collector.record(:daily_pnl, pnl)
    end

    def snapshot
      {
        orders_per_minute: orders_per_minute,
        fills_per_minute: fills_per_minute,
        avg_slippage: avg_slippage,
        avg_latency: avg_latency,
        active_positions: Trade.where(exit_price: nil).count,
        unrealized_pnl: Trade.where(exit_price: nil).sum { |t| t.unrealized_pnl.to_f }.round(2)
      }
    end

    private

    def orders_per_minute
      # Query from metrics or Redis
      Trade.where("created_at > ?", 1.minute.ago).count
    end

    def fills_per_minute
      Trade.where("exit_time > ?", 1.minute.ago).count
    end

    def avg_slippage
      # Would aggregate from slippage records
      0
    end

    def avg_latency
      # Would aggregate from order latency records
      0
    end
  end
end

  class EngineMetrics

    end

      @collector.record(:engine_latency_ms, latency_ms, labels: { engine: engine })
    end

    def record_queue_depth(queue:, depth:)
      @collector.record(:queue_depth, depth, labels: { queue: queue })
    end

    def record_engine_error(engine:, error_type:)
      @collector.record(:engine_errors_total, 1, labels: { engine: engine, type: error_type })
    end

    def snapshot
      {
        engine_latencies: recent_engine_latencies,
        queue_depths: current_queue_depths,
        error_rates: error_rates
      }
    end

    private

    def recent_engine_latencies
      # Aggregate from Prometheus or Redis
      EngineLog.where("created_at > ?", 5.minutes.ago)
               .group(:engine_name)
               .average(:latency_ms)
               .transform_values { |v| v&.round(2) }
    end

    def current_queue_depths
      {
        default: SolidQueue::Job.where(finished_at: nil).count,
        learning: SolidQueue::Job.where(queue_name: "learning", finished_at: nil).count,
        embeddings: SolidQueue::Job.where(queue_name: "embeddings", finished_at: nil).count,
        reports: SolidQueue::Job.where(queue_name: "reports", finished_at: nil).count
      }
    end

    def error_rates
      EngineLog.where("created_at > ?", 5.minutes.ago)
                 .where(success: false)
                 .group(:engine_name)
                 .count
    end
  end
end

module Monitoring
  class DataMetrics
    def initialize(collector: nil)
      @collector = collector || MetricsCollector.new
    end

    def record_tick_lag(instrument:, lag_seconds:)
      @collector.record(:tick_lag_seconds, lag_seconds, labels: { instrument: instrument })
    end

    def record_candle_freshness(instrument:, timeframe:, age_seconds:)
      @collector.record(:candle_age_seconds, age_seconds, labels: { instrument: instrument, timeframe: timeframe })
    end

    def record_gap(instrument:, timeframe:, gap_minutes:)
      @collector.record(:data_gap_minutes, gap_minutes, labels: { instrument: instrument, timeframe: timeframe })
    end

    def snapshot
      {
        tick_lag: latest_tick_lag,
        candle_freshness: candle_freshness,
        gap_count: gap_count,
        ingestion_rate: ingestion_rate
      }
    end

    private

    def latest_tick_lag
      Instrument.limit(10).map do |inst|
        last = inst.ticks.order(:timestamp).last
        lag = last ? (Time.current - last.timestamp).to_i : nil
        { instrument: inst.trading_symbol, lag_seconds: lag }
      end
    end

    def candle_freshness
      Candle.where("timestamp > ?", 5.minutes.ago)
            .group(:instrument_id, :timeframe)
            .maximum(:timestamp)
    end

    def gap_count
      CandleGap.where("detected_at > ?", 1.hour.ago).count
    end

    def ingestion_rate
      ticks = Tick.where("timestamp > ?", 1.minute.ago).count
      (ticks / 60.0).round(2)  # per second
    end
  end
end

    end

    end

    def record_var(confidence:, value:)
      @collector.record(:value_at_risk, value, labels: { confidence: confidence.to_s })
    end

    def snapshot
      positions = Trade.where(exit_price: nil)
      calculator = Risk::PortfolioRiskCalculator.new(trades: positions)

      {
        total_exposure: calculator.total_exposure,
        margin_utilization: calculator.margin_utilization,
        var_95: calculator.value_at_risk(confidence: 0.95),
        var_99: calculator.value_at_risk(confidence: 0.99),
        concentration: calculator.concentration_risk,
        daily_loss_used: daily_loss_used,
        daily_loss_remaining: daily_loss_remaining
      }
    end

    private

    def daily_loss_used
      Trade.where("exit_time > ?", Date.current.beginning_of_day)
           .sum { |t| [t.gross_pnl.to_f, 0].min }
           .abs
           .round(2)
    end

    def daily_loss_remaining
      limit = AppConfig.config.risk&.daily_loss_limit || 50_000
      (limit - daily_loss_used).round(2)
    end
  end
end

module Monitoring
  class AIMetrics
    def initialize(collector: nil)
      @collector = collector || MetricsCollector.new
    end

    def record_request_latency(provider:, latency_ms:)
      @collector.record(:ai_request_latency_ms, latency_ms, labels: { provider: provider })
    end

    def record_request_success(provider:, success:)
      status = success ? "success" : "failure"
      @collector.record(:ai_requests_total, 1, labels: { provider: provider, status: status })
    end

    def record_cost(provider:, tokens:)
      @collector.record(:ai_tokens_total, tokens, labels: { provider: provider })
    end

    def snapshot
      {
        request_latency: recent_latency,
        success_rate: success_rate,
        total_cost: total_cost,
        active_providers: active_providers
      }
    end

    private

    def recent_latency
      AiGatewayLog.where("created_at > ?", 1.hour.ago)
                  .group(:provider)
                  .average(:latency_ms)
                  .transform_values { |v| v&.round(2) }
    end

    def success_rate
      total = AiGatewayLog.where("created_at > ?", 1.hour.ago).count
      successes = AiGatewayLog.where("created_at > ?", 1.hour.ago).where(success: true).count
      total.zero? ? 0 : (successes.to_f / total * 100).round(2)
    end

    def total_cost
      # Approximate cost based on tokens
      AiGatewayLog.where("created_at > ?", 1.day.ago).sum(:estimated_tokens)
    end

    def active_providers
      AiGateway::AIGateway.new.health.select { |_, h| h[:available] }.keys
    end
  end
end

# app/services/monitoring/alerting_rules.rb
# frozen_string_literal: true

module Monitoring
  class AlertingRules
    RULES = [
      {
        name: :high_tick_lag,
        condition: ->(metrics) { metrics[:tick_lag]&.any? { |m| m[:lag_seconds]&.> 30 } },
        severity: :warning,
        message: "Tick lag > 30s detected"
      },
      {
        name: :daily_loss_limit_80pct,
        condition: ->(metrics) {
          used = metrics[:risk]&.dig(:daily_loss_used).to_f
          limit = AppConfig.config.risk&.daily_loss_limit.to_f
          limit > 0 && used / limit > 0.8
        },
        severity: :critical,
        message: "Daily loss at 80% of limit"
      },
      {
        name: :circuit_breaker_open,
        condition: ->(metrics) { metrics[:circuit_breaker] == :open },
        severity: :critical,
        message: "Circuit breaker is OPEN"
      },
      {
        name: :broker_disconnected,
        condition: ->(metrics) { metrics[:broker_status] == :disconnected },
        severity: :critical,
        message: "Broker connection lost"
      },
      {
        name: :ai_gateway_degraded,
        condition: ->(metrics) {
          health = metrics[:ai_gateway_health]
          health.is_a?(Hash) && !health.values.any? { |h| h[:available] }
        },
        severity: :warning,
        message: "All AI providers unavailable"
      },
      {
        name: :queue_backlog,
        condition: ->(metrics) {
          depths = metrics[:queue_depths] || {}
          depths.values.any? { |d| d > 1000 }
        },
        severity: :warning,
        message: "Solid Queue backlog detected"
      },
      {
        name: :emergency_stop_active,
        condition: ->(metrics) { metrics[:emergency_stop] == true },
        severity: :critical,
        message: "Emergency stop is active"
      }
    ].freeze

    def evaluate(snapshot)
      triggered = []

      RULES.each do |rule|
        if rule[:condition].call(snapshot)
          triggered << create_alert(rule, snapshot)
        end
      end

      triggered.each(&:save!)
      triggered
    end

    private

    def create_alert(rule, snapshot)
      AlertHistory.new(
        severity: rule[:severity].to_s,
        category: "system",
        message: rule[:message],
        metadata: { rule: rule[:name], snapshot: snapshot }
      )
    end
  end
end

module Monitoring
  class LogAggregation
    def self.structured_log(level:, event:, payload:)
      log_entry = {
        timestamp: Time.current.iso8601,
        level: level,
        event: event,
        service: "algoscalper",
        environment: Rails.env,
        payload: payload
      }.compact

      Rails.logger.send(level, log_entry.to_json)
    end

    def self.trade_log(trade:, action:)
      structured_log(
        level: :info,
        event: :trade_executed,
        payload: {
          trade_id: trade.id,
          instrument: trade.instrument&.trading_symbol,
          direction: trade.direction,
          strategy: trade.setup_type,
          entry_price: trade.entry_price,
          quantity: trade.quantity,
          action: action
        }
      )
    end

    def self.error_log(error:, context:)
      structured_log(
        level: :error,
        event: :system_error,
        payload: {
          error_class: error.class.name,
          message: error.message,
          backtrace: error.backtrace&.first(5),
          context: context
        }
      )
    end

    def self.metric_log(metric:, value:, labels:)
      structured_log(
        level: :debug,
        event: :metric_recorded,
        payload: {
          metric: metric,
          value: value,
          labels: labels
        }
      )
    end
  end
end

# app/services/monitoring/distributed_tracing.rb
# frozen_string_literal: true

module Monitoring
  class DistributedTracing
    TRACE_HEADER = "X-Request-Trace-Id".freeze

    def self.start_trace
      trace_id = SecureRandom.uuid
      Thread.current[:trace_id] = trace_id
      trace_id
    end

    def self.current_trace
      Thread.current[:trace_id]
    end

    def self.tag_span(name:, metadata: {})
      trace = current_trace
      return unless trace

      Rails.logger.info({
        trace_id: trace,
        span: name,
        timestamp: Time.current.iso8601,
        metadata: metadata
      }.to_json)
    end

    def self.finish_trace
      Thread.current[:trace_id] = nil
    end

    def self.trace_request
      trace_id = start_trace
      begin
        yield trace_id
      ensure
        finish_trace
      end
    end
  end
end

# ApplicationController addition
class ApplicationController < ActionController::API
  before_action :set_trace_id

  private

  def set_trace_id
    trace_id = request.headers["X-Request-Trace-Id"] || Monitoring::DistributedTracing.start_trace
    response.headers["X-Request-Trace-Id"] = trace_id
  end
end

module Monitoring
  class UptimeMonitoring
    HEALTH_ENDPOINTS = [
      { name: :api, url: "/api/v1/dashboard/health" },
      { name: :websocket, url: "/cable" },
      { name: :metrics, url: "/metrics" }
    ].freeze

    def initialize(base_url:)
      @base_url = base_url
    end

    def check_all
      HEALTH_ENDPOINTS.map do |endpoint|
        check_endpoint(endpoint)
      end
    end

    def check_endpoint(endpoint)
      start = Time.current
      response = Net::HTTP.get_response(URI("#{@base_url}#{endpoint[:url]}"))
      latency = ((Time.current - start) * 1000).round(2)

      {
        name: endpoint[:name],
        status: response.is_a?(Net::HTTPSuccess) ? :up : :down,
        status_code: response.code,
        latency_ms: latency,
        checked_at: Time.current.iso8601
      }
    rescue => e
      {
        name: endpoint[:name],
        status: :down,
        error: e.message,
        checked_at: Time.current.iso8601
      }
    end
  end
end
