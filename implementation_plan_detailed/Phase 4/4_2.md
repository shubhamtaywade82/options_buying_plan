## Day 1 — MarketRegimeEngine Base & Classifiers

### Task 1.1: Create `MarketRegimeEngine`
- **File:** `app/engines/market_regime_engine.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    class MarketRegimeEngine < BaseEngine
      REGIME_TYPES = %w[
        trending_up
        trending_down
        range
        expansion
        compression
        reversal
      ].freeze

      def self.engine_name
        "market_regime"
      end

      def self.required_features
        %w[
          adx
          atr
          ema_alignment
          vwap_distance
          bollinger_bandwidth
          rsi
          macd
        ]
      end

      def initialize(instrument:, timeframes: %w[5m 15m 30m])
        super()
        @instrument = instrument
        @timeframes = timeframes
      end

      def classify
        validate_inputs!(build_inputs)

        # Classify each timeframe independently
        timeframe_regimes = @timeframes.to_h do |tf|
          [tf, classify_timeframe(tf)]
        end

        # Aggregate across timeframes
        aggregated = aggregate_regimes(timeframe_regimes)

        # Detect transitions
        transition = detect_transition(aggregated[:regime])

        emit(EventTypes::REGIME_CHANGED, {
          engine: engine_name,
          instrument_id: @instrument.id,
          regime: aggregated[:regime],
          score: aggregated[:score],
          timeframes: timeframe_regimes,
          transition: transition,
          timestamp: Time.current
        })

        success(
          regime: aggregated[:regime],
          regime_score: aggregated[:score],
          timeframes: timeframe_regimes,
          primary_timeframe: aggregated[:primary_timeframe],
          transition: transition,
          confidence: aggregated[:confidence],
          timestamp: Time.current
        )
      end

      private

      def build_inputs
        {
          instrument: @instrument,
          timeframes: @timeframes
        }
      end

      def classify_timeframe(tf)
        adx = fetch_indicator(:adx, tf)
        atr = fetch_indicator(:atr, tf)
        ema = fetch_indicator(:ema, tf)
        vwap = fetch_indicator(:vwap, tf)
        bollinger = fetch_indicator(:bollinger, tf)
        rsi = fetch_indicator(:rsi, tf)
        macd = fetch_indicator(:macd, tf)

        # Weighted scoring for each regime type
        scores = {
          trending_up: trending_up_score(adx, ema, macd, rsi),
          trending_down: trending_down_score(adx, ema, macd, rsi),
          range: range_score(adx, bollinger, ema),
          expansion: expansion_score(atr, bollinger, adx),
          compression: compression_score(atr, bollinger, adx),
          reversal: reversal_score(rsi, macd, ema)
        }

        best = scores.max_by { |_, v| v }

        {
          regime: best.first,
          score: best.last.round(2),
          scores: scores,
          indicators: {
            adx: adx&.dig(:value),
            atr: atr&.dig(:value),
            ema_alignment: ema,
            bollinger_bandwidth: bollinger&.dig(:bandwidth_percent),
            rsi: rsi&.dig(:value),
            macd: macd&.dig(:macd)
          }
        }
      end

      def fetch_indicator(name, tf)
        result = Indicators::IndicatorRegistry.compute(
          name,
          instrument: @instrument,
          timeframe: tf
        )
        result.success? ? result.value : nil
      rescue => e
        Rails.logger.warn "[RegimeEngine] Failed to fetch #{name} #{tf}: #{e.message}"
        nil
      end

      def aggregate_regimes(timeframe_regimes)
        # Weight by timeframe: higher weight for longer timeframes
        weights = { "5m" => 0.2, "15m" => 0.3, "30m" => 0.5 }

        regime_votes = Hash.new(0)
        total_score = 0

        timeframe_regimes.each do |tf, data|
          w = weights[tf] || 0.2
          regime_votes[data[:regime]] += w
          total_score += data[:score] * w
        end

        winner = regime_votes.max_by { |_, v| v }

        {
          regime: winner.first,
          score: (total_score / weights.values.sum).round(2),
          primary_timeframe: timeframe_regimes.max_by { |tf, d| weights[tf] * d[:score] }.first,
          confidence: (winner.last * 100).round(2)
        }
      end

      def detect_transition(current_regime)
        previous = previous_regime
        return nil unless previous

        if previous != current_regime
          { from: previous, to: current_regime, timestamp: Time.current }
        else
          nil
        end
      end

      def previous_regime
        last = MarketRegime.where(instrument: @instrument)
                          .order(detected_at: :desc)
                          .first
        last&.regime_type
      end
    end
  end
  ```
- **Deliverable:** Base regime engine with multi-timeframe aggregation
- **Acceptance Criteria:** `MarketRegimeEngine.new(instrument: nifty).classify` returns regime + score + confidence

  ```ruby
  module Engines
    module MarketRegime
      class TrendingUpClassifier
        def self.score(adx, ema, macd, rsi)
          return 0 unless adx && ema && macd

          s = 0

          # ADX trending and strong
          s += 30 if adx[:trending] && adx[:trend_direction] == :bullish

          s += 15 if rsi && rsi[:value].to_f.between?(50, 70)

          # ADX strength bonus
          s += 10 if adx[:trend_strength] == :strong

          [s, 100].min
        end
      end
    end
  end
  ```

### Task 3.2: Create `RegimePersistenceTracker`
- **File:** `app/engines/market_regime/regime_persistence_tracker.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  module Engines
    module MarketRegime
      class RegimePersistenceTracker
        MIN_PERSISTENCE_BARS = 3

        def initialize(instrument:)
          @instrument = instrument
        end

        def call(current_regime:)
          recent = MarketRegime.where(instrument: @instrument)
                              .order(detected_at: :desc)
                              .limit(MIN_PERSISTENCE_BARS)

          return { persistent: false, bars: 0 } if recent.empty?

          same_regime_count = recent.take_while { |r| r.regime_type == current_regime.to_s }.count

          {
            persistent: same_regime_count >= MIN_PERSISTENCE_BARS,
            bars: same_regime_count,
            whipsaw: whipsaw?(recent),
            stability: stability_score(recent)
          }
        end

        private

        def whipsaw?(recent)
          return false if recent.count < 3
          regimes = recent.map(&:regime_type)
          regimes.each_cons(2).any? { |a, b| a != b }
        end

        def stability_score(recent)
          return 0 if recent.empty?
          regimes = recent.map(&:regime_type)
          unique = regimes.uniq.count
          total = regimes.count
          ((total - unique + 1).to_f / total * 100).round(2)
        end
      end
    end
  end
  ```
- **Deliverable:** Regime persistence and whipsaw detection
- **Acceptance Criteria:** `RegimePersistenceTracker.new(instrument: nifty).call(current_regime: :trending_up)` returns `{ persistent: true }` after 3 consecutive bars

### Task 4.1: Create `RegimeRepository`
- **File:** `app/repositories/regime_repository.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class RegimeRepository < BaseRepository
    self.model = MarketRegime

    def self.record(instrument:, regime_type:, regime_score:, timeframe:, features: {})
      create!(
        instrument: instrument,
        regime_type: regime_type,
        regime_score: regime_score,
        timeframe: timeframe,
        features: features,
        detected_at: Time.current
      )
    end

    def self.latest(instrument:, timeframe: nil)
      scope = model.where(instrument: instrument)
      scope = scope.where(timeframe: timeframe) if timeframe
      scope.order(detected_at: :desc).first
    end

    def self.history(instrument:, timeframe:, limit: 100)
      model.where(instrument: instrument, timeframe: timeframe)
           .order(detected_at: :desc)
           .limit(limit)
    end

    def self.transitions(instrument:, from:, to:)
      all = model.where(instrument: instrument).order(detected_at: :asc)
      transitions = []

      all.each_cons(2) do |prev, curr|
        transitions << curr if prev.regime_type != curr.regime_type
      end

      transitions
    end

    def self.dominant_regime(instrument:, from:, to:)
      regimes = model.where(instrument: instrument)
                    .where(detected_at: from..to)
                    .group(:regime_type)
                    .count

      regimes.max_by { |_, count| count }&.first
    end
  end
  ```
- **Deliverable:** Regime persistence and query interface
- **Acceptance Criteria:** `RegimeRepository.record(...)` creates `MarketRegime` record

  ```ruby
  # frozen_string_literal: true

  class RegimeHistoryAnalyzer < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def call(lookback: 20)
      regimes = RegimeRepository.history(instrument: @instrument, timeframe: "5m", limit: lookback)

      return failure(:no_data) if regimes.empty?

      counts = regimes.group_by(&:regime_type).transform_values(&:count)
      total = regimes.count

      dominant = counts.max_by { |_, v| v }

      # Average regime score
      avg_score = regimes.average(:regime_score)&.round(2)

      # Transition frequency
      transitions = RegimeRepository.transitions(instrument: @instrument, from: lookback.hours.ago, to: Time.current)

      success(
        lookback_bars: total,
        dominant_regime: dominant&.first,
        dominant_percent: ((dominant&.last.to_f / total) * 100).round(2),
        regime_distribution: counts.transform_values { |c| (c.to_f / total * 100).round(2) },
        average_score: avg_score,
        transition_count: transitions.count,
        transition_frequency: total.zero? ? 0 : (transitions.count.to_f / total).round(4),
        whipsaw_prone: transitions.count > (total * 0.3)
      )
    end
  end
  ```
- **Deliverable:** Historical regime analytics
- **Acceptance Criteria:** `RegimeHistoryAnalyzer.new(instrument: nifty).call` returns dominant regime % and transition frequency

  ```ruby
  # frozen_string_literal: true

  class RegimeCache
    TTL = 300  # 5 minutes

    def self.write(instrument_id, regime)
      CacheStore.write("regime:#{instrument_id}", regime, ttl: TTL)
    end

    def self.read(instrument_id)
      CacheStore.fetch("regime:#{instrument_id}")
    end

    def self.fresh?(instrument_id, max_age: 120)
      cached = read(instrument_id)
      return false unless cached && cached[:timestamp]
      Time.current - Time.parse(cached[:timestamp].to_s) < max_age
    end
  end
  ```

### Task 6.1: Create `RegimeStrategyEnabler`
- **File:** `app/services/regime_strategy_enabler.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class RegimeStrategyEnabler < BaseService
    STRATEGY_REGIME_MAP = {
      trend_following: %w[trending_up trending_down],
      pullback: %w[trending_up trending_down],
      breakout: %w[expansion compression],
      reversal: %w[reversal],
      range: %w[range],
      orb: %w[expansion]
    }.freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      current = current_regime
      return failure(:no_regime) unless current

      enabled = {}
      disabled = {}

      STRATEGY_REGIME_MAP.each do |strategy, allowed_regimes|
        if allowed_regimes.include?(current[:regime].to_s)
          enabled[strategy] = {
            confidence: current[:regime_score],
            reason: "Regime #{current[:regime]} supports #{strategy}"
          }
        else
          disabled[strategy] = {
            reason: "Regime #{current[:regime]} incompatible with #{strategy}",
            suggested_regimes: allowed_regimes
          }
        end
      end

      success(
        current_regime: current[:regime],
        regime_score: current[:regime_score],
        enabled_strategies: enabled,
        disabled_strategies: disabled,
        best_strategy: enabled.max_by { |_, v| v[:confidence] }&.first
      )
    end

    def strategy_allowed?(strategy_name)
      result = call
      return false if result.failure?
      result.value[:enabled_strategies].key?(strategy_name.to_sym)
    end

    private

    def current_regime
      RegimeCache.read(@instrument.id) || begin
        engine = Engines::MarketRegimeEngine.new(instrument: @instrument)
        engine.classify.value
      end
    end
  end
  ```
- **Deliverable:** Strategy-regime compatibility matrix
- **Acceptance Criteria:** `RegimeStrategyEnabler.new(instrument: nifty).strategy_allowed?(:trend_following)` returns `true` when trending

  ```ruby
    end

    def call
      regime = current_regime
      return failure(:no_regime) unless regime

      multiplier = REGIME_RISK_MULTIPLIERS[regime[:regime].to_sym] || 0.5

      adjusted = {
        position_size: (@base_risk[:position_size] * multiplier).round,
        max_positions: (@base_risk[:max_positions] * multiplier).to_i,
        stop_loss_multiplier: stop_loss_for_regime(regime[:regime]),
        trailing_stop: regime[:regime].to_s.include?("trending"),
        hold_time_limit: hold_time_for_regime(regime[:regime])
      }

      success(adjusted)
    end

    private

    def current_regime
      RegimeCache.read(@instrument.id)
    end

    def stop_loss_for_regime(regime)
      case regime.to_s
      when "trending_up", "trending_down" then 1.5
      when "expansion" then 2.0
      when "compression" then 1.0
      when "range" then 1.0
      else 1.5
      end
    end

    def hold_time_for_regime(regime)
      case regime.to_s
      when "trending_up", "trending_down" then 30  # minutes
      when "range" then 15
      when "expansion" then 10
      else 20
      end
    end
  end
  ```
- **Deliverable:** Dynamic risk adjustment per regime
- **Acceptance Criteria:** `RegimeBasedRiskAdjuster.new(instrument: nifty, base_risk: { position_size: 100 }).call` halves size in compression

  ```ruby
  # frozen_string_literal: true

  class MultiTimeframeRegimeAggregator < BaseService
    TIMEFRAMES = %w[5m 15m 30m].freeze

    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      regimes = TIMEFRAMES.to_h do |tf|
        engine = Engines::MarketRegimeEngine.new(instrument: @instrument, timeframes: [tf])
        result = engine.classify
        [tf, result.success? ? result.value : nil]
      end

      # Alignment check: all timeframes agree
      regime_types = regimes.values.compact.map { |r| r[:regime] }
      aligned = regime_types.uniq.length == 1

      # Trend strength: higher timeframe dominates if conflict
      dominant = if aligned
                   regime_types.first
                 else
                   # 30m > 15m > 5m
                   regimes["30m"]&.dig(:regime) ||
                     regimes["15m"]&.dig(:regime) ||
                     regimes["5m"]&.dig(:regime)
                 end

      # Composite score: weighted average
      scores = regimes.values.compact.map { |r| r[:regime_score].to_f }
      composite_score = scores.empty? ? 0 : (scores.sum / scores.length).round(2)

      success(
        timeframes: regimes,
        aligned: aligned,
        dominant_regime: dominant,
        composite_score: composite_score,
        recommendation: recommendation(aligned, dominant, composite_score)
      )
    end

    private

    def recommendation(aligned, dominant, score)
      if aligned && score > 75
        :strong_conviction
      elsif aligned && score > 50
        :moderate_conviction
      elsif !aligned && score > 60
        :mixed_signals
      else
        :weak_conviction
      end
    end
  end
  ```
- **Deliverable:** Multi-timeframe regime alignment analysis
- **Acceptance Criteria:** `MultiTimeframeRegimeAggregator.new(instrument: nifty).call` returns aligned/unaligned status

  ```ruby
  # frozen_string_literal: true

  class RegimeConfluenceDetector < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def call
      # Regime + Market Context + Structure confluence
      regime = RegimeCache.read(@instrument.id)
      context = MarketContextCache.read(@instrument.id)

      return failure(:missing_data) unless regime && context

      confluence = 0
      reasons = []

      # Regime and context aligned
      if regime_aligned_with_context?(regime, context)
        confluence += 30
        reasons << "Regime aligned with market context"
      end

      # High score on both
      if regime[:regime_score].to_f > 70 && context[:score].to_f > 70
        confluence += 30
        reasons << "High confidence on both regime and context"
      end

      # No recent whipsaw
      persistence = Engines::MarketRegime::RegimePersistenceTracker.new(instrument: @instrument)
                                                                  .call(current_regime: regime[:regime])
      if persistence[:persistent]
        confluence += 20
        reasons << "Regime persistent (#{persistence[:bars]} bars)"
      end

      # Multi-timeframe aligned
      mtf = MultiTimeframeRegimeAggregator.new(instrument: @instrument).call
      if mtf.success? && mtf.value[:aligned]
        confluence += 20
        reasons << "Multi-timeframe aligned"
      end

      success(
        confluence_score: confluence,
        grade: grade(confluence),
        reasons: reasons,
        trade_quality: trade_quality(confluence)
      )
    end

    private

    def regime_aligned_with_context?(regime, context)
      # Context allows trading and regime is not reversal/compression
      context[:context] == :allow_trading &&
        %w[trending_up trending_down expansion range].include?(regime[:regime].to_s)
    end

    def grade(score)
      if score >= 80 then :A
      elsif score >= 60 then :B
      elsif score >= 40 then :C
      else :D
      end
    end

    def trade_quality(score)
      if score >= 80 then :excellent
      elsif score >= 60 then :good
      elsif score >= 40 then :marginal
      else :avoid
      end
    end
  end
  ```
- **Deliverable:** Confluence scoring across regime, context, structure
- **Acceptance Criteria:** `RegimeConfluenceDetector.new(instrument: nifty).call` returns A-grade when all aligned
