```ruby
# app/engines/replay/historical_replay_engine.rb
# frozen_string_literal: true

module Engines
  module Replay
    class HistoricalReplayEngine < Engines::BaseEngine
      REQUIRED_FEATURES = %w[replay_config tick_data].freeze

      def initialize(strategy_engine:, instrument:, from:, to:)
        @strategy_engine = strategy_engine
        @instrument = instrument
        @from = from
        @to = to
        @speed_controller = ReplaySpeedController.new
        @metrics = ReplayMetricsCollector.new
        @state = :idle
      end

      def call(context: {})
        result = validate_inputs!(context.slice(*REQUIRED_FEATURES))
        return result if result.failure?

        config = context[:replay_config] || default_config
        @speed_controller.configure(config[:speed] || :realtime)

        replay_ticks(config)
      end

      def pause
        @state = :paused
      end

      def resume
        @state = :running if @state == :paused
      end

      def stop
        @state = :stopped
      end

      private

      def default_config
        {
          speed: :realtime,
          enable_signals: true,
          record_equity: true,
          slippage_model: :realistic
        }
      end

      def replay_ticks(config)
        @state = :running
        tick_cursor = load_ticks
        equity = 0.0
        equity_curve = []

        tick_cursor.each do |tick|
          break if @state == :stopped
          sleep(@speed_controller.delay_for(tick)) if @state == :running

          # Simulate market data arrival
          broadcast_tick(tick)

          # Run strategy engine on this tick if configured
          if config[:enable_signals]
            signal_result = @strategy_engine.call(
              context: { tick: tick, instrument: @instrument }
            )

            if signal_result.success? && signal_result.value[:action]
              execute_simulated_signal(signal_result.value, tick, config[:slippage_model])
            end
          end

          # Update equity
          equity = calculate_equity(tick)
          equity_curve << { timestamp: tick[:timestamp], equity: equity.round(2) }

          @metrics.record_tick(tick, equity)
        end

        @state = :completed

        success(
          instrument: @instrument.trading_symbol,
          period: { from: @from, to: @to },
          total_ticks: tick_cursor.count,
          final_equity: equity.round(2),
          equity_curve: equity_curve,
          metrics: @metrics.summary,
          signals_generated: @metrics.signal_count,
          trades_simulated: @metrics.trade_count
        )
      end

      def load_ticks
        TickRepository.recent(
          instrument: @instrument,
          from: @from,
          to: @to
        ).value || []
      end

      def broadcast_tick(tick)
        emit(EventTypes::REPLAY_TICK, {
          instrument: @instrument.trading_symbol,
          tick: tick,
          replay_state: @state
        })
      end

      def execute_simulated_signal(signal, tick, slippage_model)
        slippage = SlippageSimulator.new(model: slippage_model).calculate(signal, tick)
        @metrics.record_signal(signal, tick, slippage)
      end

      def calculate_equity(tick)
        # Simplified: sum of closed PnL + open position mark-to-market
        closed_pnl = @metrics.closed_pnl
        open_pnl = @metrics.open_positions.sum do |pos|
          (tick[:ltp].to_f - pos[:entry_price].to_f) * pos[:quantity].to_i * (pos[:direction] == :long ? 1 : -1)
        end
        closed_pnl + open_pnl
      end
    end
  end
end

module Engines
  module Replay
    class ReplaySpeedController
      SPEEDS = {
        realtime: 1.0,
        fast: 10.0,
        very_fast: 100.0,
        instant: 0.0
      }.freeze

      def initialize
        @speed = :realtime
        @multiplier = 1.0
        @last_tick_time = nil
      end

      def configure(speed)
        @speed = speed
        @multiplier = SPEEDS.fetch(speed, 1.0)
      end

      def delay_for(tick)
        return 0 if @multiplier.zero? || @multiplier >= 1000

        current_time = parse_time(tick[:timestamp])
        return 0 unless current_time

        if @last_tick_time
          elapsed = (current_time - @last_tick_time).to_f
          delay = elapsed / @multiplier
          delay = [delay, 0.001].max  # Min 1ms to prevent CPU spin
          @last_tick_time = current_time
          delay
        else
          @last_tick_time = current_time
          0
        end
      end

      def current_speed
        @speed
      end

      def estimated_duration(total_ticks, avg_tick_interval_ms = 1000)
        total_seconds = (total_ticks * avg_tick_interval_ms) / 1000.0
        (total_seconds / @multiplier).round(2)
      end

      private

      def parse_time(timestamp)
        return timestamp if timestamp.is_a?(Time)
        Time.parse(timestamp.to_s)
      rescue
        nil
      end
    end
  end
end

# app/engines/replay/replay_metrics_collector.rb
# frozen_string_literal: true

module Engines
  module Replay
    class ReplayMetricsCollector
      attr_reader :signal_count, :trade_count, :closed_pnl, :open_positions

      def initialize
        @signal_count = 0
        @trade_count = 0
        @closed_pnl = 0.0
        @open_positions = []
        @trades = []
        @equity_peaks = []
        @max_drawdown = 0.0
        @win_count = 0
        @loss_count = 0
      end

      def record_tick(tick, equity)
        @equity_peaks << equity
        current_peak = @equity_peaks.max
        dd = current_peak - equity
        @max_drawdown = dd if dd > @max_drawdown
      end

      def record_signal(signal, tick, slippage)
        @signal_count += 1

        if signal[:action] == :entry
          @open_positions << {
            entry_time: tick[:timestamp],
            entry_price: tick[:ltp].to_f + slippage,
            direction: signal[:direction],
            quantity: signal[:quantity] || 1,
            stop_loss: signal[:stop_loss],
            take_profit: signal[:take_profit]
          }
          @trade_count += 1
        elsif signal[:action] == :exit && @open_positions.any?
          position = @open_positions.shift
          pnl = calculate_pnl(position, tick[:ltp].to_f + slippage)
          @closed_pnl += pnl
          @trades << { entry: position, exit: tick, pnl: pnl }

          if pnl > 0
            @win_count += 1
          else
            @loss_count += 1
          end
        end
      end

      def summary
        total = @win_count + @loss_count
        win_rate = total.zero? ? 0 : (@win_count.to_f / total * 100).round(2)

        {
          total_trades: @trade_count,
          closed_trades: total,
          win_count: @win_count,
          loss_count: @loss_count,
          win_rate: win_rate,
          gross_pnl: @closed_pnl.round(2),
          avg_pnl: total.zero? ? 0 : (@closed_pnl / total).round(2),
          max_drawdown: @max_drawdown.round(2),
          profit_factor: calculate_profit_factor,
          open_positions: @open_positions.length
        }
      end

      private

      def calculate_pnl(position, exit_price)
        direction_mult = position[:direction] == :long ? 1 : -1
        (exit_price - position[:entry_price]) * position[:quantity] * direction_mult
      end

      def calculate_profit_factor
        wins = @trades.select { |t| t[:pnl] > 0 }.sum { |t| t[:pnl] }
        losses = @trades.select { |t| t[:pnl] <= 0 }.sum { |t| t[:pnl].abs }
        losses.zero? ? 0 : (wins / losses).round(2)
      end
    end
  end
end

```ruby
# app/engines/replay/walk_forward_test_runner.rb
# frozen_string_literal: true

module Engines
  module Replay
    class WalkForwardTestRunner
      DEFAULT_WINDOW_DAYS = 30
      DEFAULT_TEST_DAYS = 7

      def initialize(strategy_engine:, instrument:, from:, to:, window_days: DEFAULT_WINDOW_DAYS, test_days: DEFAULT_TEST_DAYS)
        @strategy_engine = strategy_engine
        @instrument = instrument
        @from = from
        @to = to
        @window_days = window_days
        @test_days = test_days
        @results = []
      end

      def call
        current = @from
        fold = 0

        while current + @window_days.days + @test_days.days <= @to
          fold += 1
          train_start = current
          train_end = current + @window_days.days
          test_start = train_end
          test_end = test_start + @test_days.days

          # Optimize parameters on training window
          optimizer = ParameterOptimization.new(
            strategy_engine: @strategy_engine,
            instrument: @instrument,
            from: train_start,
            to: train_end
          )
          optimal_params = optimizer.call

          # Test on out-of-sample period
          replay = HistoricalReplayEngine.new(
            strategy_engine: @strategy_engine.with_params(optimal_params),
            instrument: @instrument,
            from: test_start,
            to: test_end
          )
          test_result = replay.call(context: { replay_config: { speed: :instant } })

          @results << {
            fold: fold,
            train_period: { from: train_start, to: train_end },
            test_period: { from: test_start, to: test_end },
            optimal_params: optimal_params,
            test_result: test_result.success? ? test_result.value : { error: test_result.error.to_h }
          }

          current = test_end
        end

        aggregate_results
      end

      private

      def aggregate_results
        successful = @results.select { |r| r[:test_result].is_a?(Hash) && r[:test_result][:metrics] }

        return { folds: 0, message: "No completed folds" } if successful.empty?

        pnls = successful.map { |r| r[:test_result][:metrics][:gross_pnl] }
        win_rates = successful.map { |r| r[:test_result][:metrics][:win_rate] }

        {
          total_folds: @results.length,
          completed_folds: successful.length,
          avg_pnl_per_fold: (pnls.sum / successful.length).round(2),
          avg_win_rate: (win_rates.sum / successful.length).round(2),
          consistency: consistency_score(pnls),
          fold_results: @results,
          recommendation: walk_forward_recommendation(pnls)
        }
      end

      def consistency_score(pnls)
        return 0 if pnls.length < 2
        mean = pnls.sum.to_f / pnls.length
        variance = pnls.sum { |p| (p - mean)**2 } / pnls.length
        std_dev = Math.sqrt(variance)
        return 0 if std_dev.zero?
        (mean / std_dev).round(2)  # Sharpe-like consistency
      end

      def walk_forward_recommendation(pnls)
        positive = pnls.count { |p| p > 0 }
        ratio = positive.to_f / pnls.length

        case ratio
        when 0.7..1.0 then :robust
        when 0.5..0.7 then :moderate
        when 0.3..0.5 then :fragile
        else :unreliable
        end
      end
    end
  end
end

# app/engines/replay/monte_carlo_simulator.rb
# frozen_string_literal: true

module Engines
  module Replay
    class MonteCarloSimulator
      DEFAULT_ITERATIONS = 1000

      def initialize(trades:, iterations: DEFAULT_ITERATIONS)
        @trades = trades
        @iterations = iterations
        @returns = trades.map { |t| t.gross_pnl.to_f }
      end

      def call
        return { error: :insufficient_data } if @returns.length < 10

        simulations = @iterations.times.map { simulate_randomized_series }
        equity_curves = simulations.map { |s| build_equity_curve(s) }

        {
          iterations: @iterations,
          original_profit_factor: original_profit_factor,
          original_max_drawdown: original_max_drawdown,
          original_sharpe: original_sharpe,
          simulated_median_profit_factor: percentile(simulations.map(&:profit_factor), 50).round(2),
          simulated_median_max_drawdown: percentile(simulations.map(&:max_drawdown), 50).round(2),
          simulated_median_sharpe: percentile(simulations.map(&:sharpe), 50).round(2),
          worst_case_drawdown: percentile(simulations.map(&:max_drawdown), 95).round(2),
          probability_of_profit: probability_of_profit(simulations),
          confidence_interval_95: confidence_interval(simulations),
          simulation_distribution: distribution_summary(simulations)
        }
      end

      private

      def simulate_randomized_series
        randomized = @returns.shuffle
        cumulative = 0
        peak = 0
        max_dd = 0
        daily_pnls = []

        randomized.each do |pnl|
          cumulative += pnl
          peak = [peak, cumulative].max
          dd = peak - cumulative
          max_dd = dd if dd > max_dd
          daily_pnls << pnl
        end

        wins = randomized.select { |r| r > 0 }
        losses = randomized.reject { |r| r > 0 }

        profit_factor = losses.sum(&:abs).zero? ? 0 : wins.sum / losses.sum(&:abs)
        sharpe = calculate_sharpe(daily_pnls)

        OpenStruct.new(
          profit_factor: profit_factor,
          max_drawdown: max_dd,
          sharpe: sharpe,
          final_equity: cumulative
        )
      end

      def build_equity_curve(returns)
        equity = 0
        returns.map { |r| equity += r; equity }
      end

      def original_profit_factor
        wins = @returns.select { |r| r > 0 }
        losses = @returns.reject { |r| r > 0 }
        losses.sum(&:abs).zero? ? 0 : wins.sum / losses.sum(&:abs)
      end

      def original_max_drawdown
        cumulative = 0
        peak = 0
        max_dd = 0
        @returns.each do |r|
          cumulative += r
          peak = [peak, cumulative].max
          dd = peak - cumulative
          max_dd = dd if dd > max_dd
        end
        max_dd
      end

      def original_sharpe
        calculate_sharpe(@returns)
      end

      def calculate_sharpe(returns)
        return 0 if returns.length < 2
        mean = returns.sum.to_f / returns.length
        std_dev = Math.sqrt(returns.sum { |r| (r - mean)**2 } / returns.length)
        std_dev.zero? ? 0 : (mean / std_dev).round(2)
      end

      def percentile(values, p)
        sorted = values.sort
        index = (sorted.length * (p / 100.0)).ceil - 1
        sorted[[index, 0].max]
      end

      def probability_of_profit(simulations)
        profitable = simulations.count { |s| s.final_equity > 0 }
        (profitable.to_f / simulations.length * 100).round(2)
      end

      def confidence_interval(simulations)
        equities = simulations.map(&:final_equity).sort
        lower_idx = (equities.length * 0.025).floor
        upper_idx = (equities.length * 0.975).ceil
        {
          lower: equities[lower_idx].round(2),
          upper: equities[[upper_idx, equities.length - 1].min].round(2)
        }
      end

      def distribution_summary(simulations)
        equities = simulations.map(&:final_equity).sort
        {
          min: equities.first.round(2),
          max: equities.last.round(2),
          median: percentile(equities, 50).round(2),
          q1: percentile(equities, 25).round(2),
          q3: percentile(equities, 75).round(2)
        }
      end
    end
  end
end

# app/engines/replay/parameter_optimization.rb
# frozen_string_literal: true

module Engines
  module Replay
    class ParameterOptimization
      DEFAULT_GRID = {
        rsi_period: [7, 14, 21],
        atr_multiplier: [1.5, 2.0, 2.5],
        volume_threshold: [1.0, 1.5, 2.0]
      }.freeze

      def initialize(strategy_engine:, instrument:, from:, to:, grid: DEFAULT_GRID)
        @strategy_engine = strategy_engine
        @instrument = instrument
        @from = from
        @to = to
        @grid = grid
      end

      def call
        combinations = generate_combinations(@grid)
        results = []

        combinations.each do |params|
          replay = HistoricalReplayEngine.new(
            strategy_engine: @strategy_engine.with_params(params),
            instrument: @instrument,
            from: @from,
            to: @to
          )

          result = replay.call(context: { replay_config: { speed: :instant } })
          next unless result.success?

          metrics = result.value[:metrics]
          score = optimization_score(metrics)

          results << {
            params: params,
            score: score,
            metrics: metrics,
            equity: result.value[:final_equity]
          }
        end

        best = results.max_by { |r| r[:score] }
        return @grid.transform_values { |v| v.first } if best.nil?

        {
          optimal_params: best[:params],
          optimal_score: best[:score],
          all_results: results.sort_by { |r| -r[:score] }.first(10),
          grid_size: combinations.length,
          evaluated: results.length
        }
      end

      private

      def generate_combinations(grid)
        keys = grid.keys
        values = grid.values
        values[0].product(*values[1..]).map do |combo|
          keys.zip(combo).to_h
        end
      end

      def optimization_score(metrics)
        return -999 if metrics.nil? || metrics[:total_trades] < 5

        # Multi-objective: profit factor * win_rate * sqrt(trade_count) - drawdown_penalty
        pf = [metrics[:profit_factor], 0].max
        wr = metrics[:win_rate] / 100.0
        dd_penalty = metrics[:max_drawdown] / 10_000.0  # Normalize

        (pf * wr * Math.sqrt(metrics[:total_trades]) - dd_penalty).round(2)
      end
    end
  end
end

```ruby
# app/engines/replay/overfitting_detector.rb
# frozen_string_literal: true

module Engines
  module Replay
    class OverfittingDetector
      OVERFIT_THRESHOLD = 0.3  # 30% degradation threshold

      def initialize(train_results:, test_results:)
        @train_results = train_results
        @test_results = test_results
      end

      def call
        metrics = %i[profit_factor win_rate sharpe_ratio avg_pnl]
        comparisons = metrics.map { |m| compare_metric(m) }

        {
          overfitting_detected: comparisons.any? { |c| c[:degraded] },
          severity: overall_severity(comparisons),
          metric_comparisons: comparisons,
          train_test_ratio: train_test_ratio,
          recommendation: recommendation(comparisons)
        }
      end

      private

      def compare_metric(metric)
        train_val = extract_metric(@train_results, metric)
        test_val = extract_metric(@test_results, metric)

        return { metric: metric, degraded: false, train: nil, test: nil } if train_val.nil? || test_val.nil?

        ratio = test_val / train_val rescue 0
        degraded = ratio < (1 - OVERFIT_THRESHOLD)

        {
          metric: metric,
          train: train_val.round(2),
          test: test_val.round(2),
          ratio: ratio.round(2),
          degraded: degraded,
          severity: severity_from_ratio(ratio)
        }
      end

      def extract_metric(results, metric)
        case results
        when Hash
          results.dig(:metrics, metric) || results[metric]
        when OpenStruct, Struct
          results.send(metric) rescue nil
        else
          nil
        end
      end

      def severity_from_ratio(ratio)
        case ratio
        when 0.8..1.0 then :none
        when 0.7..0.8 then :mild
        when 0.5..0.7 then :moderate
        else :severe
        end
      end

      def overall_severity(comparisons)
        severities = comparisons.map { |c| c[:severity] }.compact
        return :none if severities.empty?
        return :severe if severities.include?(:severe)
        return :moderate if severities.include?(:moderate)
        :mild
      end

      def train_test_ratio
        train_trades = extract_metric(@train_results, :total_trades)
        test_trades = extract_metric(@test_results, :total_trades)
        return nil if train_trades.nil? || test_trades.nil? || train_trades.zero?
        (test_trades.to_f / train_trades).round(2)
      end

      def recommendation(comparisons)
        degraded_count = comparisons.count { |c| c[:degraded] }

        if degraded_count == 0
          :no_overfitting
        elsif degraded_count <= 2
          :mild_overfitting_increase_regulation
        else
          :severe_overfitting_simplify_model
        end
      end
    end
  end
end

module Engines
  module Replay
    class ReplayReportGenerator
      def initialize(replay_result:)
        @result = replay_result
      end

      def call
        return { error: :invalid_result } unless @result.is_a?(Hash) && @result[:metrics]

        {
          summary: generate_summary,

          metrics: @result[:metrics],

        {
          instrument: @result[:instrument],
          period: @result[:period],
          total_ticks: @result[:total_ticks],
          final_equity: @result[:final_equity],
          total_trades: m[:total_trades],
          win_rate: m[:win_rate],
          profit_factor: m[:profit_factor],
          max_drawdown: m[:max_drawdown],
          avg_pnl: m[:avg_pnl],
          open_positions: m[:open_positions]
        }
      end

      def generate_chart_data
        {
          equity_curve: @result[:equity_curve].map { |pt| [pt[:timestamp], pt[:equity]] },
          drawdown_series: calculate_drawdown_series,
          trade_distribution: trade_distribution
        }
      end

      def calculate_drawdown_series
        equity = @result[:equity_curve].map { |pt| pt[:equity] }
        peak = 0
        equity.map do |e|
          peak = [peak, e].max
          { equity: e, drawdown: peak - e }
        end
      end

      def trade_distribution
        return {} unless @result[:metrics][:win_count] && @result[:metrics][:loss_count]
        {
          wins: @result[:metrics][:win_count],
          losses: @result[:metrics][:loss_count],
          win_rate: @result[:metrics][:win_rate]
        }
      end

      def benchmark_comparison
        # Compare to buy-and-hold of underlying
        start_price = @result[:equity_curve].first&.[](:equity)
        end_price = @result[:equity_curve].last&.[](:equity)
        return nil if start_price.nil? || end_price.nil? || start_price.zero?

        strategy_return = (end_price - start_price) / start_price
        {
          strategy_return_percent: (strategy_return * 100).round(2),
          buy_hold_return_percent: 0,  # Would need spot prices
          alpha: 0
        }
      end
    end
  end

### Day 5 — ReplayComparison & API Endpoints

**Task 5.1: Create** `ReplayComparison`

```ruby
# app/engines/replay/replay_comparison.rb
# frozen_string_literal: true

module Engines
  module Replay
    class ReplayComparison
      def initialize(replay_results:)
        @results = replay_results
      end

      def call
        return { error: :need_at_least_two } if @results.length < 2

        {
          strategies_compared: @results.length,
          winner: determine_winner,
          comparison_table: comparison_table,
          statistical_significance: statistical_significance,
          recommendation: recommendation
        }
      end

      private

      def determine_winner
        scored = @results.map do |r|
          score = composite_score(r)
          { name: r[:strategy_name] || r[:instrument], score: score, result: r }
        end

        scored.max_by { |s| s[:score] }
      end

      def comparison_table
        metrics = %i[final_equity win_rate profit_factor max_drawdown avg_pnl sharpe_ratio]
        @results.map do |r|
          m = r[:metrics] || {}
          {
            strategy: r[:strategy_name] || r[:instrument],
            final_equity: r[:final_equity],
            trades: m[:total_trades],
            win_rate: m[:win_rate],
            profit_factor: m[:profit_factor],
            max_drawdown: m[:max_drawdown],
            avg_pnl: m[:avg_pnl],
            composite_score: composite_score(r)
          }
        end
      end

      def composite_score(result)
        m = result[:metrics] || {}
        return 0 if m[:total_trades].to_i < 5

        pf = [m[:profit_factor].to_f, 0].max
        wr = m[:win_rate].to_f / 100.0
        dd_penalty = m[:max_drawdown].to_f / 5000.0

        (pf * wr * 10 - dd_penalty).round(2)
      end

      def statistical_significance
        # Simplified: compare best vs second best
        sorted = @results.sort_by { |r| -composite_score(r) }
        best = sorted[0]
        second = sorted[1]

        return nil unless best && second

        best_pnl = best[:metrics][:gross_pnl] || 0
        second_pnl = second[:metrics][:gross_pnl] || 0
        diff = best_pnl - second_pnl

        {
          best_strategy: best[:strategy_name],
          second_best: second[:strategy_name],
          pnl_difference: diff.round(2),
          margin: (diff / [second_pnl.abs, 1].max * 100).round(2)
        }
      end

      def recommendation
        winner = determine_winner
        return :inconclusive unless winner

        if winner[:score] > 0
          :deploy_winner
        else
          :no_viable_strategy
        end
      end
    end
  end
end

module Api
  module V1
    class ReplayController < ApplicationController
      def run_replay
        instrument = Instrument.find(params[:instrument_id])

          }
        })

        if result.success?
          render json: {
            replay: result.value,
            report: Engines::Replay::ReplayReportGenerator.new(replay_result: result.value).call
          }
        else
          render json: { error: result.error.to_h }, status: :unprocessable_entity
        end
      end

      def walk_forward
        instrument = Instrument.find(params[:instrument_id])

        runner = Engines::Replay::WalkForwardTestRunner.new(
          strategy_engine: strategy_engine_for(params[:strategy]),
          instrument: instrument,
          from: parse_time(params[:from]),
          to: parse_time(params[:to]),
          window_days: params[:window_days]&.to_i || 30,
          test_days: params[:test_days]&.to_i || 7
        )

        render json: runner.call
      end

      def monte_carlo
        trades = fetch_trades_for_simulation
        simulator = Engines::Replay::MonteCarloSimulator.new(
          trades: trades,
          iterations: params[:iterations]&.to_i || 1000
        )

        render json: simulator.call
      end

      def optimize
        instrument = Instrument.find(params[:instrument_id])

        optimizer = Engines::Replay::ParameterOptimization.new(
          strategy_engine: strategy_engine_for(params[:strategy]),
          instrument: instrument,
          from: parse_time(params[:from]),
          to: parse_time(params[:to]),
          grid: params[:grid]&.symbolize_keys || {}
        )

        render json: optimizer.call
      end

      def compare
        results = params[:replay_results].map(&:to_unsafe_h)
        comparison = Engines::Replay::ReplayComparison.new(replay_results: results)

        render json: comparison.call
      end

      private

      def parse_time(param)
        Time.parse(param.to_s)
      rescue
        30.days.ago
      end

      def strategy_engine_for(name)
        # Factory for strategy engines
        "Engines::Strategy::#{name.camelize}Engine".constantize.new
      rescue
        Engines::Strategy::BaseStrategyEngine.new
      end

      def fetch_trades_for_simulation
        scope = Trade.where("exit_time IS NOT NULL")
        scope = scope.where(setup_type: params[:strategy]) if params[:strategy]
        scope = scope.where("entry_time > ?", params[:lookback].to_i.days.ago) if params[:lookback]
        scope.order(:entry_time)
      end
    end

# Historical Replay & Walk-Forward Testing

## Architecture

- **HistoricalReplayEngine**: Tick-by-tick replay with configurable speed
- **ReplaySpeedController**: Realtime, 10x, 100x, instant
- **ReplayMetricsCollector**: Tracks equity, drawdown, PnL during replay
- **WalkForwardTestRunner**: Rolling train/test windows to detect overfitting
- **MonteCarloSimulator**: Randomized trade sequence analysis
- **ParameterOptimization**: Grid search over strategy parameters
- **OverfittingDetector**: Train vs test degradation analysis
- **ReplayReportGenerator**: Equity curves and benchmark comparisons
- **ReplayComparison**: A/B strategy comparison with composite scoring

## Speed Modes

| Mode | Multiplier | Use Case |
|------|-----------|----------|
| `:realtime` | 1x | Visual validation |
| `:fast` | 10x | Quick validation |
| `:very_fast` | 100x | Overnight backtests |
| `:instant` | 0x | Parameter optimization |

## Walk-Forward Methodology

1. **Train Window**: Optimize parameters on N days
2. **Test Window**: Evaluate on out-of-sample N+1 days
3. **Roll Forward**: Slide window and repeat
4. **Aggregate**: Consistency score across all folds

## Overfitting Detection

Flagged if test metrics degrade >30% vs training:
- Profit factor collapse
- Win rate drop
- Sharpe ratio deterioration

## API

```bash

  "instrument_id": 1,
  "from": "2024-01-01T09:15:00",
  "to": "2024-01-31T15:30:00",
  "speed": "fast",
  "slippage_model": "realistic"
}

  "from": "2024-01-01",
  "to": "2024-06-01",
  "window_days": 30,

}
