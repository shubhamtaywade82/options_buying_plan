### Task 3.3: Create `CompressionMonitor`
- **File:** `app/services/compression_monitor.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class CompressionMonitor < BaseService
    def call
      result = ActiveRecord::Base.connection.execute(<<-SQL)
        SELECT
          hypertable_name,
          chunk_name,
          compression_status,
          before_compression_total_bytes,
          after_compression_total_bytes,
          pg_size_pretty(before_compression_total_bytes) as before_size,
          pg_size_pretty(after_compression_total_bytes) as after_size
        FROM timescaledb_information.chunks
        WHERE compression_status = 'Compressed'
        ORDER BY before_compression_total_bytes DESC
        LIMIT 20;
      SQL

      stats = result.map do |row|
        ratio = row["before_compression_total_bytes"].to_f / row["after_compression_total_bytes"].to_f
        row.merge("compression_ratio" => ratio.round(2))
      end

      success(stats)
    end

    def compression_ratio(hypertable: "market_ticks")
      result = ActiveRecord::Base.connection.execute(<<-SQL)
        SELECT
          sum(before_compression_total_bytes) as before,
          sum(after_compression_total_bytes) as after
        FROM timescaledb_information.chunks
        WHERE hypertable_name = '#{hypertable}'
          AND compression_status = 'Compressed';
      SQL

      before = result[0]["before"].to_f
      after = result[0]["after"].to_f
      return success(ratio: 0) if after.zero?

      success(ratio: (before / after).round(2))
    end
  end
  ```
- **Deliverable:** Compression statistics and monitoring
- **Acceptance Criteria:** `CompressionMonitor.new.call` returns chunk compression ratios

  # frozen_string_literal: true

  class DataArchiver < BaseService
    ARCHIVE_DIR = Rails.root.join("storage/archives")

    def call(table:, from:, to:)
      model = table.to_s.classify.constantize
      records = model.where("timestamp BETWEEN ? AND ?", from, to)

      return success(archived: 0) if records.none?

      filename = "#{table}_#{from.strftime('%Y%m%d')}_#{to.strftime('%Y%m%d')}.parquet"
      path = ARCHIVE_DIR.join(filename)

      FileUtils.mkdir_p(ARCHIVE_DIR)

      # Export to Parquet for efficient cold storage
      export_to_parquet(records, path)

      success(
        archived: records.count,
        file: path.to_s,
        size_bytes: File.size(path)
      )
    end

    private

    def export_to_parquet(records, path)
      # Use CSV as fallback if parquet gem not available
      # In production, use red-arrow or similar
      CSV.open(path.sub_ext(".csv"), "w") do |csv|
        csv << records.column_names
        records.find_each do |record|
          csv << record.attributes.values
        end
      end
    end
  end
  ```
- **Deliverable:** Pre-drop archival to cold storage
- **Acceptance Criteria:** `DataArchiver.new.call(table: "market_ticks", from: 8.days.ago, to: 7.days.ago)` creates archive file

  # frozen_string_literal: true

  class QueryPlanner
    def self.explain(query)
      result = ActiveRecord::Base.connection.execute("EXPLAIN (ANALYZE, BUFFERS, FORMAT JSON) #{query.to_sql}")
      JSON.parse(result[0]["QUERY PLAN"])
    end

    def self.slow_queries(threshold_ms: 100)
      ActiveRecord::Base.connection.execute(<<-SQL)
        SELECT
          query,
          calls,
          mean_exec_time,
          total_exec_time,
          rows
        FROM pg_stat_statements
        WHERE mean_exec_time > #{threshold_ms}
          AND query NOT LIKE '%pg_stat%'
        ORDER BY mean_exec_time DESC
        LIMIT 10;
      SQL
    end

    def self.missing_indexes
      ActiveRecord::Base.connection.execute(<<-SQL)
        SELECT
          schemaname,
          tablename,
          attname as column,
          n_tup_read,
          n_tup_fetch
        FROM pg_stats
        WHERE n_tup_read > 100000
          AND schemaname = 'public'
        ORDER BY n_tup_read DESC
        LIMIT 20;
      SQL
    end
  end
  ```
- **Deliverable:** Query analysis utilities

### Task 6.1: Create `TickQueryService`
- **File:** `app/services/tick_query_service.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class TickQueryService < BaseService
    def initialize(instrument:)
      @instrument = instrument
    end

    def latest(limit: 1)
      result = TickRepository.latest(instrument: @instrument, limit: limit)
      return result if result.failure?

      success(result.value)
    end

    def range(from:, to:, batch_size: 10_000)
      # Use cursor-based pagination for large ranges
      ticks = []
      cursor = from

      while cursor < to
        batch = MarketTick.for_instrument(@instrument)
                         .where("timestamp >= ? AND timestamp < ?", cursor, [cursor + batch_size.seconds, to].min)
                         .order(timestamp: :asc)
                         .limit(batch_size)

        break if batch.none?

        ticks.concat(batch.to_a)
        cursor = batch.last.timestamp + 0.001
      end

      success(ticks)
    end

    def aggregate(from:, to:, interval: "1 minute")
      # Use continuous aggregate when possible
      view = continuous_view_for(interval)

      if view
        query = ActiveRecord::Base.connection.execute(<<-SQL)
          SELECT * FROM #{view}
          WHERE instrument_id = #{@instrument.id}
            AND bucket BETWEEN '#{from.to_s(:db)}' AND '#{to.to_s(:db)}'
          ORDER BY bucket ASC;
        SQL

        success(query.map { |row| normalize_cagg_row(row) })
      else
        # Fallback to raw aggregation
        fallback_aggregate(from: from, to: to, interval: interval)
      end
    end

    private

    def continuous_view_for(interval)
      {
        "1 minute" => "candles_1m_continuous",
        "5 minutes" => "candles_5m_continuous",
        "15 minutes" => "candles_15m_continuous",
        "30 minutes" => "candles_30m_continuous"
      }[interval]
    end

    def normalize_cagg_row(row)
      {
        timestamp: row["bucket"],
        open: row["open"]&.to_d,
        high: row["high"]&.to_d,
        low: row["low"]&.to_d,
        close: row["close"]&.to_d,
        volume: row["volume"]&.to_i,
        oi: row["oi"]&.to_i,
        vwap: row["vwap"]&.to_d
      }
    end

    def fallback_aggregate(from:, to:, interval:)
      result = ActiveRecord::Base.connection.execute(<<-SQL)
        SELECT
          time_bucket('#{interval}', timestamp) AS bucket,
          first(last_price, timestamp) AS open,
          max(last_price) AS high,
          min(last_price) AS low,
          last(last_price, timestamp) AS close,
          sum(volume) AS volume,
          last(oi, timestamp) AS oi
        FROM market_ticks
        WHERE instrument_id = #{@instrument.id}
          AND timestamp BETWEEN '#{from.to_s(:db)}' AND '#{to.to_s(:db)}'
        GROUP BY bucket
        ORDER BY bucket ASC;
      SQL

      success(result.map { |row| normalize_cagg_row(row) })
    end
  end
  ```
- **Deliverable:** Optimized tick queries with continuous aggregate routing
- **Acceptance Criteria:** `TickQueryService.new(instrument: nifty).aggregate(from: 1.hour.ago, to: Time.current, interval: "5 minutes")` uses `candles_5m_continuous` view

### Task 6.2: Create `TickReplayService`
- **File:** `app/services/tick_replay_service.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class TickReplayService < BaseService
    def initialize(instrument:, from:, to:)
      @instrument = instrument
      @from = from
      @to = to
    end

    def call(speed: 1.0, &block)
      ticks = TickRepository.for_range(
        instrument: @instrument,
        from: @from,
        to: @to
      ).value

      return success(replayed: 0) if ticks.empty?

      baseline = ticks.first.timestamp
      replayed = 0

      ticks.each do |tick|
        target_time = baseline + (tick.timestamp - baseline) / speed

        # Sleep until target time
        sleep_time = target_time - Time.current
        sleep(sleep_time) if sleep_time > 0

        yield(tick) if block_given?
        replayed += 1
      end

      success(replayed: replayed, duration: @to - @from)
    end

    def call_fast(&block)
      # No sleep, just emit all ticks as fast as possible
      ticks = TickRepository.for_range(
        instrument: @instrument,
        from: @from,
        to: @to
      ).value

      ticks.each { |tick| yield(tick) if block_given? }

      success(replayed: ticks.length)
    end
  end
  ```
- **Deliverable:** Historical tick replay for backtesting
- **Acceptance Criteria:** `TickReplayService.new(instrument: nifty, from: 1.hour.ago, to: Time.current).call(speed: 10) { |t| puts t.last_price }` replays at 10x speed

  # frozen_string_literal: true

  class BulkTickImporter < BaseService
    BATCH_SIZE = 10_000

    def initialize(file_path:)
      @file_path = file_path
    end

    def call
      return failure(:file_not_found) unless File.exist?(@file_path)

      total = 0
      batch = []

      CSV.foreach(@file_path, headers: true) do |row|
        batch << {
          timestamp: Time.parse(row["timestamp"]),
          instrument_id: row["instrument_id"].to_i,
          last_price: row["last_price"].to_d,
          volume: row["volume"].to_i,
          oi: row["oi"].to_i,
          bid: row["bid"]&.to_d,
          ask: row["ask"]&.to_d
        }

        if batch.size >= BATCH_SIZE
          TickRepository.insert_batch(batch)
          total += batch.size
          batch = []
        end
      end

      # Final batch
      if batch.any?
        TickRepository.insert_batch(batch)
        total += batch.size
      end

      success(imported: total)
    end
  end
  ```
- **Deliverable:** Historical CSV seeding for backtesting
- **Acceptance Criteria:** `BulkTickImporter.new(file_path: "ticks.csv").call` imports 100K rows in <10s

          continuous_aggregate: benchmark_cagg_query
        }

        results.each do |name, result|
          Rails.logger.info "[Benchmark] #{name}: #{result[:ops_per_sec]} ops/sec | #{result[:latency_ms]} ms/op"
        end

        results
      end

      def benchmark_tick_insertion(count: 10_000)
        instrument = Instrument.first
        ticks = count.times.map do |i|
          {
            timestamp: Time.current - (count - i).seconds,
            instrument_id: instrument.id,
            last_price: 22500 + rand,
            volume: rand(1000..5000),
            oi: 2_500_000
          }
        end

        bm = Benchmark.measure do
          TickRepository.insert_batch(ticks)
        end

        {
          ops_per_sec: (count / bm.real).round,
          latency_ms: (bm.real * 1000 / count).round(3),
          total_ms: (bm.real * 1000).round(2)
        }
      end

      def benchmark_tick_query
        instrument = Instrument.first
        from = 1.hour.ago
        to = Time.current

        bm = Benchmark.measure do
          TickRepository.for_range(instrument: instrument, from: from, to: to)
        end

        count = MarketTick.for_instrument(instrument).between(from, to).count

        {
          ops_per_sec: (count / bm.real).round,
          latency_ms: (bm.real * 1000).round(2),
          rows: count
        }
      end

      def benchmark_candle_query
        instrument = Instrument.first

        bm = Benchmark.measure do
          CandleRepository.ohlcv(
            instrument: instrument,
            timeframe: "5m",
            from: 1.day.ago,
            to: Time.current
          )
        end

        {
          latency_ms: (bm.real * 1000).round(2),
          rows: Candle.for_instrument(instrument).for_timeframe("5m").where("timestamp > ?", 1.day.ago).count
        }
      end

      def benchmark_cagg_query
        instrument = Instrument.first

        bm = Benchmark.measure do
          ActiveRecord::Base.connection.execute(<<-SQL)
            SELECT * FROM candles_5m_continuous
            WHERE instrument_id = #{instrument.id}
              AND bucket > NOW() - INTERVAL '1 day'
          SQL
        end

        {
          latency_ms: (bm.real * 1000).round(2),
          source: "continuous_aggregate"
        }
      end
    end
  end
  ```

## Day 9 — TimescaleDB Hypertable Management

### Task 9.1: Create `HypertableManager`
- **File:** `app/services/hypertable_manager.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class HypertableManager < BaseService
    def list_chunks(hypertable: "market_ticks")
      result = ActiveRecord::Base.connection.execute(<<-SQL)
        SELECT
          chunk_name,
          range_start,
          range_end,
          pg_size_pretty(total_bytes) as size,
          compression_status
        FROM timescaledb_information.chunks
        WHERE hypertable_name = '#{hypertable}'
        ORDER BY range_start DESC
        LIMIT 50;
      SQL

      success(result.map(&:to_h))
    end

    def compress_chunk(chunk_name)
      ActiveRecord::Base.connection.execute(
        "SELECT compress_chunk('#{chunk_name}')"
      )
      success(chunk: chunk_name, compressed: true)
    end

    def decompress_chunk(chunk_name)
      ActiveRecord::Base.connection.execute(
        "SELECT decompress_chunk('#{chunk_name}')"
      )
      success(chunk: chunk_name, decompressed: true)
    end

    def drop_chunk(chunk_name)
      ActiveRecord::Base.connection.execute(
        "SELECT drop_chunks('#{chunk_name}')"
      )
      success(chunk: chunk_name, dropped: true)
    end

    def hypertable_size(hypertable: "market_ticks")
      result = ActiveRecord::Base.connection.execute(<<-SQL)
        SELECT
          hypertable_name,
          pg_size_pretty(total_bytes) as total_size,
          pg_size_pretty(compressed_total_bytes) as compressed_size,
          pg_size_pretty(uncompressed_total_bytes) as uncompressed_size
        FROM timescaledb_information.hypertable_size
        WHERE hypertable_name = '#{hypertable}';
      SQL

      success(result[0]&.to_h)
    end

    def stats
      {
        market_ticks: hypertable_size(hypertable: "market_ticks").value,
        chunks: list_chunks(hypertable: "market_ticks").value
      }
    end
  end
  ```
- **Deliverable:** Programmatic hypertable management
- **Acceptance Criteria:** `HypertableManager.new.list_chunks` returns chunk metadata

### Task 9.2: Create `ChunkMaintenanceJob`
- **File:** `app/jobs/chunk_maintenance_job.rb`
- **Requirements:**
  ```ruby
  # frozen_string_literal: true

  class ChunkMaintenanceJob < JobBase
    def perform
      manager = HypertableManager.new

      # Compress chunks older than 6 hours but not yet compressed
      chunks = manager.list_chunks(hypertable: "market_ticks").value
      chunks.each do |chunk|
        next if chunk["compression_status"] == "Compressed"

        chunk_time = Time.parse(chunk["range_start"])
        if chunk_time < 6.hours.ago
          manager.compress_chunk(chunk["chunk_name"])
          Rails.logger.info "[ChunkMaintenance] Compressed #{chunk['chunk_name']}"
        end
      end

      # Report sizes
      size = manager.hypertable_size(hypertable: "market_ticks").value
      Rails.logger.info "[ChunkMaintenance] market_ticks size: #{size['total_size']}"
    end
  end
  ```
- **Deliverable:** Automated chunk compression
